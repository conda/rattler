42/// Central access point for high level queries about
43/// [`rattler_conda_types::RepoDataRecord`]s from different channels.
44///
45/// The gateway is responsible for fetching and caching repodata. Requests are
46/// deduplicated which means that if multiple requests are made for the same
47/// repodata only the first request will actually fetch the data. All other
48/// requests will wait for the first request to complete and then return the
49/// same data.
50///
51/// The gateway is thread-safe and can be shared between multiple threads. The
52/// gateway struct itself uses internal reference counting and is cheaply
53/// cloneable. There is no need to wrap the gateway in an `Arc`.
54#[derive(Clone)]
55pub struct Gateway {
56    inner: Arc<GatewayInner>,
57}
58
59impl Default for Gateway {
60    fn default() -> Self {
61        Gateway::new()
62    }
63}

66#[derive(Default, Clone, Debug)]
67pub enum SubdirSelection {
68    /// Select all subdirectories
69    #[default]
70    All,
71
72    /// Select these specific subdirectories
73    Some(HashSet<String>),
74}

76impl SubdirSelection {
77    /// Returns `true` if the given subdirectory is part of the selection.
78    pub fn contains(&self, subdir: &str) -> bool {
79        match self {
80            SubdirSelection::All => true,
81            SubdirSelection::Some(subdirs) => subdirs.contains(&subdir.to_string()),
82        }
83    }

111    /// Constructs a new `GatewayQuery` which can be used to query repodata
112    /// records.
113    ///
114    /// # Sources
115    ///
116    /// The `sources` parameter accepts any type that implements `Into<Source>`.
117    /// This includes:
118    /// - `Channel` - traditional conda channels
119    /// - `Arc<dyn RepoDataSource>` - custom repodata sources
120    /// - `Source` - the enum itself
121    ///
122    /// Existing code using channels continues to work unchanged:
123    ///
124    /// ```ignore
125    /// gateway.query(
126    ///     vec![channel1, channel2],
127    ///     vec![Platform::Linux64],
128    ///     vec![spec],
129    /// ).await
130    /// ```
131    ///
132    /// You can also mix channels and custom sources:
133    ///
134    /// ```ignore
135    /// gateway.query(
136    ///     vec![
137    ///         Source::Channel(channel),
138    ///         Source::Custom(my_custom_source),
139    ///     ],
140    ///     vec![Platform::Linux64],
141    ///     vec![spec],
142    /// ).await
143    /// ```
144    pub fn query<AsSource, SourceIter, PlatformIter, PackageNameIter, IntoMatchSpec>(
145        &self,
146        sources: SourceIter,
147        platforms: PlatformIter,
148        specs: PackageNameIter,
149    ) -> RepoDataQuery
150    where
151        AsSource: Into<Source>,
152        SourceIter: IntoIterator<Item = AsSource>,
153        PlatformIter: IntoIterator<Item = Platform>,
154        <PlatformIter as IntoIterator>::IntoIter: Clone,
155        PackageNameIter: IntoIterator<Item = IntoMatchSpec>,
156        IntoMatchSpec: Into<MatchSpec>,
157    {
158        RepoDataQuery::new(
159            self.inner.clone(),
160            sources.into_iter().map(Into::into).collect(),
161            platforms.into_iter().collect(),
162            specs.into_iter().map(Into::into).collect(),
163        )
164    }

185    /// Ensure that given repodata records contain `RunExportsJson`.
186    pub async fn ensure_run_exports(
187        &self,
188        records: impl Iterator<Item = &mut RepoDataRecord>,
189        // We can avoid Arc by cloning, but this requires helper method in the trait definition.
190        progress_reporter: Option<Arc<dyn RunExportsReporter>>,
191    ) -> Result<(), RunExportExtractorError> {
192        let futures = records
193            .filter_map(|record| {
194                if record.package_record.run_exports.is_some() {
195                    // If the package already has run exports, we don't need to do anything.
196                    return None;
197                }
198
199                let extractor = RunExportExtractor::default()
200                    .with_opt_max_concurrent_requests(
201                        self.inner.concurrent_requests_semaphore.clone(),
202                    )
203                    .with_client(self.inner.client.clone())
204                    .with_global_run_exports_cache(self.inner.subdir_run_exports_cache.clone());
205
206                #[cfg(not(target_arch = "wasm32"))]
207                let extractor = extractor.with_package_cache(self.inner.package_cache.clone());
208
209                let progress_reporter = progress_reporter.clone();
210                Some(async move {
211                    extractor
212                        .extract(record, progress_reporter)
213                        .await
214                        .map(|rexp| (record, rexp))
215                })
216            })
217            .collect::<Vec<_>>();
218
219        let results = futures::future::try_join_all(futures).await?;
220
221        for (record, result) in results {
222            record.package_record.run_exports = result;
223        }
224
225        Ok(())
226    }

228    /// Clears any in-memory cache for the given channel.
229    ///
230    /// Any subsequent query will re-fetch any required data from the source.
231    ///
232    /// When `mode` is [`CacheClearMode::InMemoryAndDisk`], this method also
233    /// clears on-disk caches for the specified channel and subdirectories.
234    pub fn clear_repodata_cache(
235        &self,
236        channel: &Channel,
237        subdirs: SubdirSelection,
238        mode: CacheClearMode,
239    ) -> Result<(), std::io::Error> {
240        self.inner.subdirs.retain(|key, _| {
241            key.0.base_url != channel.base_url || !subdirs.contains(key.1.as_str())
242        });
243
244        #[cfg(not(target_arch = "wasm32"))]
245        if mode == CacheClearMode::InMemoryAndDisk {
246            use std::str::FromStr;
247
248            let platforms_to_clear: Vec<Platform> = match &subdirs {
249                SubdirSelection::All => Platform::all().collect(),
250                SubdirSelection::Some(subdirs) => subdirs
251                    .iter()
252                    .filter_map(|s| Platform::from_str(s).ok())
253                    .collect(),
254            };
255
256            let mut errors = Vec::new();
257            for platform in platforms_to_clear {
258                if let Err(e) = remote_subdir::RemoteSubdirClient::clear_cache(
259                    &self.inner.cache,
260                    channel,
261                    platform,
262                ) {
263                    errors.push(e);
264                }
265                if let Err(e) =
266                    sharded_subdir::ShardedSubdir::clear_cache(&self.inner.cache, channel, platform)
267                {
268                    errors.push(e);
269                }
270            }
271            if let Some(first_error) = errors.into_iter().next() {
272                return Err(first_error);
273            }
274        }
275
276        #[cfg(target_arch = "wasm32")]
277        let _ = mode;
278
279        Ok(())
280    }

283struct GatewayInner {
284    /// A map of subdirectories for each channel and platform.
285    subdirs: CoalescedMap<(Channel, Platform), Arc<Subdir>>,
286
287    /// The client to use to fetch repodata.
288    client: LazyClient,
289
290    /// The channel configuration
291    channel_config: ChannelConfig,
292
293    /// The directory to store any cache
294    #[cfg(not(target_arch = "wasm32"))]
295    cache: std::path::PathBuf,
296
297    /// The package cache, stored to reuse memory cache
298    #[cfg(not(target_arch = "wasm32"))]
299    package_cache: PackageCache,
300
301    /// A cache for global run exports.
302    subdir_run_exports_cache: Arc<SubdirRunExportsCache>,
303
304    /// A semaphore to limit the number of concurrent requests.
305    concurrent_requests_semaphore: Option<Arc<tokio::sync::Semaphore>>,
306}
307
308impl GatewayInner {
309    /// Returns the [`Subdir`] for the given channel and platform. This
310    /// function will create the [`Subdir`] if it does not exist yet, otherwise
311    /// it will return the previously created subdir.
312    ///
313    /// If multiple threads request the same subdir their requests will be
314    /// coalesced, and they will all receive the same subdir. If an error
315    /// occurs while creating the subdir all waiting tasks will also return an
316    /// error.
317    #[instrument(skip(self, reporter, channel), fields(channel = %channel.base_url), err(level = Level::INFO))]
318    async fn get_or_create_subdir(
319        &self,
320        channel: &Channel,
321        platform: Platform,
322        reporter: Option<Arc<dyn Reporter>>,
323    ) -> Result<Arc<Subdir>, GatewayError> {
324        let key = (channel.clone(), platform);
325        let channel = channel.clone();
326
327        self.subdirs
328            .get_or_try_init(key, || async move {
329                let subdir = self.create_subdir(&channel, platform, reporter).await?;
330                Ok(Arc::new(subdir))
331            })
332            .await
333            .map_err(|e| match e {
334                CoalescedGetError::Init(gateway_err) => gateway_err,
335                CoalescedGetError::CoalescedRequestFailed => GatewayError::IoError(
336                    "a coalesced request failed".to_string(),
337                    std::io::ErrorKind::Other.into(),
338                ),
339            })
340    }
341
342    async fn create_subdir(
343        &self,
344        channel: &Channel,
345        platform: Platform,
346        reporter: Option<Arc<dyn Reporter>>,
347    ) -> Result<Subdir, GatewayError> {
348        SubdirBuilder::new(self, channel.clone(), platform, reporter)
349            .build()
350            .await
351    }
352}
