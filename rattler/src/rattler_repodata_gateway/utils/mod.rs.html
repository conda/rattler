21/// Convert a URL to a cache filename
22#[cfg(not(target_arch = "wasm32"))]
23pub(crate) fn url_to_cache_filename(url: &::url::Url) -> String {
24    use std::fmt::Write;
25
26    // Start Rant:
27    // This function mimics behavior from Mamba which itself mimics this behavior
28    // from Conda. However, I find this function absolutely ridiculous, it
29    // contains all sort of weird edge cases and returns a hash that could very
30    // easily collide with other files. And why? Why not simply return a little
31    // more descriptive? Like a URL encoded string? End Rant.
32    let mut url_str = url.to_string();
33
34    // Ensure there is a slash if the URL is empty or doesn't refer to json file
35    if url_str.is_empty() || (!url_str.ends_with('/') && !url_str.ends_with(".json")) {
36        url_str.push('/');
37    }
38
39    // Mimicking conda's (weird) behavior by special handling repodata.json
40    let url_str = url_str.strip_suffix("/repodata.json").unwrap_or(&url_str);
41
42    // Compute the MD5 hash of the resulting URL string
43    let hash = rattler_digest::compute_bytes_digest::<rattler_digest::Md5>(url_str);
44
45    // Convert the hash to an MD5 hash.
46    let mut result = String::with_capacity(8);
47    for x in &hash[0..4] {
48        write!(result, "{x:02x}").unwrap();
49    }
50    result
51}
