47/// Defines which operations can be used to link files to disk.
48#[derive(Default)]
49pub struct LinkOptions {
50    /// If true, symbolic links will be allowed to install files to the disk.
51    /// Defaults to `true` if `None`.
52    pub allow_symbolic_links: Option<bool>,
53
54    /// If true, hard links will be allowed to install files to the disk.
55    /// Defaults to `true` if `None`.
56    pub allow_hard_links: Option<bool>,
57
58    /// If true, ref links (or copy-on-write links) will be allowed to install
59    /// files to the disk. Defaults to `true` if `None`.
60    pub allow_ref_links: Option<bool>,
61}

357    /// Install the packages in the given prefix.
358    pub async fn install(
359        self,
360        prefix: impl AsRef<Path>,
361        records: impl IntoIterator<Item = RepoDataRecord>,
362    ) -> Result<InstallationResult, InstallerError> {
363        let prefix = Prefix::create(prefix.as_ref().to_path_buf()).map_err(|err| {
364            InstallerError::FailedToCreatePrefix(prefix.as_ref().to_path_buf(), err)
365        })?;
366
367        // Create a future to determine the currently installed packages. We
368        // can start this in parallel with the other operations and resolve it
369        // when we need it.
370        let installed_provided = self.installed.is_some();
371        let mut installed: Vec<InstallationResultRecord> = if let Some(installed) = self.installed {
372            installed
373                .into_iter()
374                .map(InstallationResultRecord::Max)
375                .collect()
376        } else {
377            let prefix = prefix.clone();
378            // Use sparse collection for much faster reading when checking if packages
379            // changed
380            run_blocking_task(move || {
381                use rattler_conda_types::MinimalPrefixCollection;
382                PrefixRecord::collect_minimal_from_prefix(&prefix)
383                    .map_err(InstallerError::FailedToDetectInstalledPackages)
384            })
385            .await?
386            .into_iter()
387            .map(InstallationResultRecord::Min)
388            .collect()
389        };
390
391        // Construct a transaction from the current and desired situation.
392        let target_platform = self.target_platform.unwrap_or_else(Platform::current);
393        let desired_records: Vec<_> = records.into_iter().collect();
394        let mut transaction = Transaction::from_current_and_desired(
395            installed.iter(),
396            desired_records.iter(),
397            self.reinstall_packages.as_ref(),
398            self.ignored_packages.as_ref(),
399            target_platform,
400        )?;
401        // If transaction is non-empty, we need full prefix records for file operations
402        // Reload them and reconstruct the transaction with full records
403        if !transaction.operations.is_empty() && !installed_provided {
404            let prefix = prefix.clone();
405            installed = run_blocking_task(move || {
406                PrefixRecord::collect_from_prefix(&prefix)
407                    .map_err(InstallerError::FailedToDetectInstalledPackages)
408            })
409            .await?
410            .into_iter()
411            .map(InstallationResultRecord::Max)
412            .collect();
413
414            // Reconstruct transaction with full records to maintain consistency
415            transaction = Transaction::from_current_and_desired(
416                installed.iter(),
417                desired_records.iter(),
418                self.reinstall_packages.as_ref(),
419                self.ignored_packages.as_ref(),
420                target_platform,
421            )?;
422        }
423
424        let transaction = transaction.to_owned();
425
426        // Validate that if the target platform is NoArch, all packages to be installed
427        // must also be noarch (subdir == "noarch")
428        if target_platform == Platform::NoArch {
429            let non_noarch_packages: Vec<String> = transaction
430                .installed_packages()
431                .filter(|record| record.package_record.subdir != "noarch")
432                .map(|record| {
433                    format!(
434                        "{}/{}-{}-{}",
435                        record.package_record.subdir,
436                        record.package_record.name.as_normalized(),
437                        record.package_record.version,
438                        record.package_record.build
439                    )
440                })
441                .collect();
442
443            if !non_noarch_packages.is_empty() {
444                return Err(InstallerError::PlatformSpecificPackagesWithNoarchPlatform(
445                    non_noarch_packages,
446                ));
447            }
448        }
449
450        // Create a mapping from package names to requested specs
451        let spec_mapping = self
452            .requested_specs
453            .as_ref()
454            .map(|specs| create_spec_mapping(specs))
455            .map(Arc::new);
456
457        // Update existing records that weren't modified but have matching requested
458        // specs This needs to happen even if the transaction is empty
459        if let Some(spec_mapping) = &spec_mapping {
460            // We have requested_specs (even if empty), so update/clear as needed
461            update_existing_records(transaction.unchanged_packages(), spec_mapping, &prefix)?;
462        }
463
464        // If the transaction is empty we can short-circuit the installation
465        if transaction.operations.is_empty() {
466            return Ok(InstallationResult {
467                transaction,
468                pre_link_script_result: None,
469                post_link_script_result: None,
470                clobbered_paths: HashMap::default(),
471            });
472        }
473
474        // At this point we can't have any minimal prefix records, so force them to be
475        // prefix records.
476        let transaction = transaction
477            .into_prefix_record(&prefix)
478            .map_err(InstallerError::FailedToDetectInstalledPackages)?;
479
480        let downloader = self.downloader.unwrap_or_default();
481        let package_cache = self.package_cache.unwrap_or_else(|| {
482            PackageCache::new(
483                default_cache_dir()
484                    .expect("failed to determine default cache directory")
485                    .join(rattler_cache::PACKAGE_CACHE_DIR),
486            )
487        });
488
489        // Acquire a global lock on the package cache for the entire installation.
490        // This significantly reduces overhead by avoiding per-package locking.
491        let _global_cache_lock = package_cache
492            .acquire_global_lock()
493            .await
494            .map_err(InstallerError::FailedToAcquireCacheLock)?;
495
496        // Construct a driver.
497        let driver = InstallDriver::builder()
498            .execute_link_scripts(self.execute_link_scripts)
499            .with_io_concurrency_semaphore(
500                self.io_semaphore.unwrap_or(Arc::new(Semaphore::new(100))),
501            )
502            .with_prefix_records(
503                transaction
504                    .unchanged_packages()
505                    .iter()
506                    .chain(transaction.removed_packages()),
507            )
508            .finish();
509
510        // Determine base installer options.
511        let base_install_options = InstallOptions {
512            target_prefix: self.alternative_target_prefix.clone(),
513            platform: Some(target_platform),
514            python_info: transaction.python_info.clone(),
515            apple_codesign_behavior: self.apple_code_sign_behavior,
516            allow_symbolic_links: self.link_options.allow_symbolic_links,
517            allow_hard_links: self.link_options.allow_hard_links,
518            allow_ref_links: self.link_options.allow_ref_links,
519            ..InstallOptions::default()
520        };
521
522        // Preprocess the transaction
523        let pre_process_result = driver
524            .pre_process(&transaction, &prefix, self.reporter.as_deref())
525            .map_err(InstallerError::PreProcessingFailed)?;
526
527        if let Some(reporter) = &self.reporter {
528            reporter.on_transaction_start(&transaction);
529        }
530
531        let mut pending_unlink_futures = FuturesUnordered::new();
532        // Execute the operations (remove) in the transaction.
533        for (operation_idx, operation) in transaction.operations.iter().enumerate() {
534            let reporter = self.reporter.clone();
535            let driver = &driver;
536            let prefix = &prefix;
537
538            let op = async move {
539                // Uninstall the package if it was removed.
540                if let Some(record) = operation.record_to_remove() {
541                    if let Some(reporter) = &reporter {
542                        reporter.on_transaction_operation_start(operation_idx);
543                    }
544
545                    let reporter = reporter
546                        .as_deref()
547                        .map(move |r| (r, r.on_unlink_start(operation_idx, record)));
548                    driver.clobber_registry().unregister_paths(record);
549                    unlink_package(prefix, record).await.map_err(|e| {
550                        InstallerError::UnlinkError(
551                            record.repodata_record.identifier.to_string(),
552                            e,
553                        )
554                    })?;
555                    if let Some((reporter, index)) = reporter {
556                        reporter.on_unlink_complete(index);
557                        if operation.record_to_install().is_none() {
558                            reporter.on_transaction_operation_complete(operation_idx);
559                        }
560                    }
561                }
562                Ok::<(), InstallerError>(())
563            };
564            pending_unlink_futures.push(op);
565        }
566
567        let mut pending_link_futures = FuturesUnordered::new();
568        // Execute the operations (install) in the transaction.
569        for (operation_idx, operation) in transaction
570            .operations
571            .iter()
572            .enumerate()
573            .sorted_by_key(|(_, op)| {
574                op.record_to_install()
575                    .and_then(|r| r.package_record.size)
576                    .unwrap_or(0)
577            })
578            .rev()
579        {
580            let downloader = &downloader;
581            let package_cache = &package_cache;
582            let reporter = self.reporter.clone();
583            let base_install_options = &base_install_options;
584            let driver = &driver;
585            let prefix = &prefix;
586            let spec_mapping_ref = spec_mapping.clone();
587            let operation_future = async move {
588                if let Some(reporter) = &reporter {
589                    if operation.record_to_remove().is_none() {
590                        reporter.on_transaction_operation_start(operation_idx);
591                    }
592                }
593
594                // Start populating the cache with the package if it's not already there.
595                let package_to_install = if let Some(record) = operation.record_to_install() {
596                    let record = record.clone();
597                    let downloader = downloader.clone();
598                    let reporter = reporter.clone();
599                    let package_cache = package_cache.clone();
600                    tokio::spawn(async move {
601                        let populate_cache_report = reporter.clone().map(|r| {
602                            let cache_index = r.on_populate_cache_start(operation_idx, &record);
603                            (r, cache_index)
604                        });
605                        let cache_metadata = populate_cache(
606                            &record,
607                            downloader,
608                            &package_cache,
609                            populate_cache_report.clone(),
610                        )
611                        .await?;
612                        if let Some((reporter, index)) = populate_cache_report {
613                            reporter.on_populate_cache_complete(index);
614                        }
615                        Ok((cache_metadata, record))
616                    })
617                    .map_err(JoinError::try_into_panic)
618                    .map(|res| match res {
619                        Ok(Ok(result)) => Ok(Some(result)),
620                        Ok(Err(e)) => Err(e),
621                        Err(Ok(payload)) => std::panic::resume_unwind(payload),
622                        Err(Err(_err)) => Err(InstallerError::Cancelled),
623                    })
624                    .left_future()
625                } else {
626                    ready(Ok(None)).right_future()
627                };
628
629                // Install the package if it was fetched.
630                if let Some((cache_metadata, record)) = package_to_install.await? {
631                    let reporter = reporter
632                        .as_deref()
633                        .map(|r| (r, r.on_link_start(operation_idx, &record)));
634                    let requested_spec = spec_mapping_ref
635                        .and_then(|mapping| mapping.get(&record.package_record.name).cloned())
636                        .unwrap_or_default();
637
638                    // Reuse index_json and paths_json from cache validation if available
639                    let mut install_options = base_install_options.clone();
640                    if let Some(index_json) = cache_metadata.index_json() {
641                        install_options.index_json = Some(index_json.clone());
642                    }
643                    if let Some(paths_json) = cache_metadata.paths_json() {
644                        install_options.paths_json = Some(paths_json.clone());
645                    }
646
647                    link_package(
648                        &record,
649                        prefix,
650                        cache_metadata.path(),
651                        install_options,
652                        driver,
653                        requested_spec,
654                    )
655                    .await?;
656                    if let Some((reporter, index)) = reporter {
657                        reporter.on_link_complete(index);
658                    }
659                }
660                if let Some(reporter) = &reporter {
661                    if operation.record_to_install().is_some() {
662                        reporter.on_transaction_operation_complete(operation_idx);
663                    }
664                }
665
666                Ok::<_, InstallerError>(())
667            };
668
669            pending_link_futures.push(operation_future);
670        }
671
672        // Wait for all transaction operations to finish
673        while let Some(result) = pending_unlink_futures.next().await {
674            result?;
675        }
676        drop(pending_unlink_futures);
677
678        driver
679            .remove_empty_directories(
680                &transaction.operations,
681                transaction.unchanged_packages(),
682                &prefix,
683            )
684            .map_err(|e| InstallerError::UnlinkError("remove_empty_directories".to_string(), e))?;
685
686        // Wait for all transaction operations to finish
687        while let Some(result) = pending_link_futures.next().await {
688            result?;
689        }
690        drop(pending_link_futures);
691
692        // Post process the transaction
693        let post_process_result =
694            driver.post_process(&transaction, &prefix, self.reporter.as_deref())?;
695
696        if let Some(reporter) = &self.reporter {
697            reporter.on_transaction_complete();
698        }
699
700        let transaction = transaction.into_installation_result_record();
701
702        Ok(InstallationResult {
703            transaction,
704            pre_link_script_result: pre_process_result,
705            post_link_script_result: post_process_result.post_link_result,
706            clobbered_paths: post_process_result.clobbered_paths,
707        })
708    }
709}

711async fn link_package(
712    record: &RepoDataRecord,
713    target_prefix: &Prefix,
714    cached_package_dir: &Path,
715    install_options: InstallOptions,
716    driver: &InstallDriver,
717    requested_specs: Vec<String>,
718) -> Result<(), InstallerError> {
719    let record = record.clone();
720    let target_prefix = target_prefix.clone();
721    let cached_package_dir = cached_package_dir.to_path_buf();
722    let clobber_registry = driver.clobber_registry.clone();
723
724    let (tx, rx) = tokio::sync::oneshot::channel();
725
726    // Since we use the `Prefix` type, the conda-meta folder is guaranteed to exist
727    let conda_meta_path = target_prefix.path().join("conda-meta");
728
729    rayon::spawn_fifo(move || {
730        let inner = move || {
731            // Link the contents of the package into the prefix.
732            let (paths, link_type) = crate::install::link_package_sync(
733                &cached_package_dir,
734                &target_prefix,
735                clobber_registry,
736                install_options,
737            )
738            .map_err(|e| InstallerError::LinkError(record.identifier.to_string(), e))?;
739
740            // Construct a PrefixRecord for the package
741            let prefix_record = PrefixRecord {
742                extracted_package_dir: Some(cached_package_dir.clone()),
743                link: Some(Link {
744                    source: cached_package_dir,
745                    link_type: Some(link_type),
746                }),
747                requested_specs,
748                ..PrefixRecord::from_repodata_record(record.clone(), paths)
749            };
750
751            let pkg_meta_path = prefix_record.file_name();
752            prefix_record
753                .write_to_path(conda_meta_path.join(&pkg_meta_path), true)
754                .map_err(|e| {
755                    InstallerError::IoError(format!("failed to write {pkg_meta_path}"), e)
756                })?;
757
758            Ok(())
759        };
760
761        let _ = tx.send(inner());
762    });
763
764    rx.await.unwrap_or(Err(InstallerError::Cancelled))
765}

767/// Given a repodata record, fetch the package into the cache if its not already
768/// there.
769async fn populate_cache(
770    record: &RepoDataRecord,
771    downloader: LazyClient,
772    cache: &PackageCache,
773    reporter: Option<(Arc<dyn Reporter>, usize)>,
774) -> Result<CacheMetadata, InstallerError> {
775    struct CacheReporterBridge {
776        reporter: Arc<dyn Reporter>,
777        cache_index: usize,
778    }
779
780    impl CacheReporter for CacheReporterBridge {
781        fn on_validate_start(&self) -> usize {
782            self.reporter.on_validate_start(self.cache_index)
783        }
784
785        fn on_validate_complete(&self, index: usize) {
786            self.reporter.on_validate_complete(index);
787        }
788
789        fn on_download_start(&self) -> usize {
790            self.reporter.on_download_start(self.cache_index)
791        }
792
793        fn on_download_progress(&self, index: usize, progress: u64, total: Option<u64>) {
794            self.reporter.on_download_progress(index, progress, total);
795        }
796
797        fn on_download_completed(&self, index: usize) {
798            self.reporter.on_download_completed(index);
799        }
800    }
801
802    cache
803        .get_or_fetch_from_url_with_retry(
804            &record.package_record,
805            record.url.clone(),
806            downloader,
807            default_retry_policy(),
808            reporter.map(|(reporter, cache_index)| {
809                Arc::new(CacheReporterBridge {
810                    reporter,
811                    cache_index,
812                }) as _
813            }),
814        )
815        .await
816        .map_err(|e| InstallerError::FailedToFetch(record.identifier.to_string(), e))
817}

819/// Updates only the `requested_specs` fields in a conda-meta JSON file.
820/// This performs a targeted update without overwriting other
821/// metadata.
822///
823/// This method is needed as we're initially loading
824/// `MinimalPrefixRecord`, which doesn't contain most of the fields.
825/// Therefore direct writing could overwrite data we want to preserve.
826///
827/// Currently we're loading full json, but we could do that in place without
828/// parsing whole file.
829fn update_requested_specs_in_json(
830    path: &Path,
831    requested_specs: &[String],
832    requested_spec: Option<&String>,
833) -> io::Result<()> {
834    use serde_json::Value;
835
836    // Read the existing JSON file
837    let content = fs_err::read_to_string(path)?;
838    let mut json: Value = serde_json::from_str(&content)
839        .map_err(|e| io::Error::new(io::ErrorKind::InvalidData, e))?;
840
841    // Update only the requested_specs fields
842    if let Some(obj) = json.as_object_mut() {
843        // Update requested_specs (plural)
844        obj.insert(
845            "requested_specs".to_string(),
846            Value::Array(
847                requested_specs
848                    .iter()
849                    .map(|s| Value::String(s.clone()))
850                    .collect(),
851            ),
852        );
853
854        // Update or remove requested_spec (singular, deprecated)
855        if let Some(spec) = requested_spec {
856            obj.insert("requested_spec".to_string(), Value::String(spec.clone()));
857        } else {
858            obj.remove("requested_spec");
859        }
860    }
861
862    // Write the updated JSON back to file
863    let updated_content = serde_json::to_string_pretty(&json)
864        .map_err(|e| io::Error::new(io::ErrorKind::InvalidData, e))?;
865    fs_err::write(path, updated_content)?;
866
867    Ok(())
868}

870/// Creates a mapping from package names to their requested spec strings.
871///
872/// This function takes a list of `MatchSpecs` and creates a mapping where:
873/// - The key is the `PackageName` from the `MatchSpec`
874/// - The value is a vector of string representations of all matching
875///   `MatchSpecs`
876///
877/// Only `MatchSpec`s that have a `PackageNameMatcher::Exact` are included.
878/// For multiple `MatchSpec`s with the same package name, all are collected.
879fn create_spec_mapping(specs: &[MatchSpec]) -> std::collections::HashMap<PackageName, Vec<String>> {
880    let mut mapping = std::collections::HashMap::new();
881
882    for spec in specs {
883        if let PackageNameMatcher::Exact(name) = &spec.name {
884            mapping
885                .entry(name.clone())
886                .or_insert_with(Vec::new)
887                .push(spec.to_string());
888        }
889    }
890
891    mapping
892}

894/// Updates existing `PrefixRecord` files with their requested specs.
895///
896/// This function takes existing records that weren't modified by the
897/// transaction and updates their `requested_specs` field based on the spec
898/// mapping or clears it if requested. The updated records are then written back
899/// to disk.
900#[allow(deprecated)]
901fn update_existing_records<'p>(
902    existing_records: impl IntoParallelIterator<Item = &'p InstallationResultRecord>,
903    spec_mapping: &HashMap<PackageName, Vec<String>>,
904    prefix: &Prefix,
905) -> Result<(), InstallerError> {
906    existing_records
907        .into_par_iter()
908        .map(|record| -> Result<(), InstallerError> {
909            let package_name = record.name();
910            let mut updated_record = None;
911
912            // First, check if we need to migrate from deprecated requested_spec to
913            // requested_specs
914            let current_specs = if !record.requested_specs().is_empty() {
915                // Use the new field if it has data
916                record.requested_specs().clone()
917            } else if let Some(spec) = record.requested_spec() {
918                // Migrate from deprecated field
919                vec![spec.clone()]
920            } else {
921                // No specs at all
922                Vec::new()
923            };
924
925            // Check if we need to migrate from the deprecated field
926            let needs_migration = record.requested_spec().is_some();
927
928            // Check if we have requested specs for this package
929            if let Some(requested_specs) = spec_mapping.get(package_name) {
930                // Check if the requested specs are different from what's currently stored
931                if needs_migration || &current_specs != requested_specs {
932                    // Create an updated record with the new requested specs
933                    let mut new_record = record.clone();
934                    *new_record.requested_specs_mut() = requested_specs.clone();
935                    *new_record.requested_spec_mut() = None; // Clear deprecated field
936                    updated_record = Some(new_record);
937                }
938            } else if !current_specs.is_empty() {
939                // Clear the requested_specs if it's not in the mapping
940                let mut new_record = record.clone();
941                *new_record.requested_specs_mut() = Vec::new();
942                *new_record.requested_spec_mut() = None; // Clear deprecated field
943                updated_record = Some(new_record);
944            } else if needs_migration {
945                // Even if current_specs is empty, we still need to clear the deprecated field
946                let mut new_record = record.clone();
947                *new_record.requested_spec_mut() = None;
948                updated_record = Some(new_record);
949            }
950
951            // Write the updated record back to disk if needed
952            if let Some(new_record) = updated_record {
953                let conda_meta_path = prefix.path().join("conda-meta");
954                let pkg_meta_path = format!(
955                    "{}-{}-{}.json",
956                    new_record.name().as_normalized(),
957                    new_record.version(),
958                    new_record.build()
959                );
960                let full_path = conda_meta_path.join(&pkg_meta_path);
961
962                // We need to do a targeted update of just the requested_specs fields
963                // to avoid overwriting other metadata when using minimal records
964                update_requested_specs_in_json(
965                    &full_path,
966                    new_record.requested_specs(),
967                    new_record.requested_spec(),
968                )
969                .map_err(|e| {
970                    InstallerError::IoError(
971                        format!("failed to update requested_specs for {pkg_meta_path}"),
972                        e,
973                    )
974                })?;
975            }
976
977            Ok(())
978        })
979        .collect::<Result<Vec<_>, _>>()?;
980
981    Ok(())
982}
