1//! Sparse remote access to files inside `.conda` archives.
2//!
3//! This module uses HTTP range requests to avoid downloading the full archive.
4//! It opens the outer ZIP container, locates the relevant `info-*.tar.zst` or
5//! `pkg-*.tar.zst` member, and streams only the bytes needed to read a target
6//! path from that inner tarball.
7//!
8//! Only `.conda` archives on servers that support range requests are supported.
9//! For higher-level APIs that fall back to full downloads, see
10//! [`super::fetch::fetch_package_file_from_remote_url`] and
11//! [`super::fetch::fetch_file_from_remote_url`].
12//!
13//! # Example
14//!
15//! ```rust,no_run
16//! # #[tokio::main]
17//! # async fn main() {
18//! use rattler_conda_types::package::IndexJson;
19//! use rattler_package_streaming::reqwest::sparse::fetch_package_file_sparse;
20//! use reqwest::Client;
21//! use reqwest_middleware::ClientWithMiddleware;
22//! use url::Url;
23//!
24//! let client = ClientWithMiddleware::from(Client::new());
25//! let url = Url::parse("https://conda.anaconda.org/conda-forge/linux-64/python-3.10.8-h4a9ceb5_0_cpython.conda").unwrap();
26//!
27//! let index_json: IndexJson = fetch_package_file_sparse(client, url).await.unwrap();
28//! # }
29//! ```

47/// Default number of bytes to fetch from the end of the file.
48/// 64KB should be enough for most packages to include the EOCD, Central Directory,
49/// and often the entire info archive.
50const DEFAULT_TAIL_SIZE: u64 = 64 * 1024;
51
52/// Fetch the raw bytes of a single file from a remote `.conda` package using
53/// HTTP range requests.
54///
55/// Streams the zstd data directly from the server through an async decompressor
56/// and tar reader, stopping as soon as the target file is found. Only the bytes
57/// needed to reach the target file are downloaded and decompressed.
58///
59/// Returns `Ok(None)` if the file is not found in the archive.
60/// Returns an error if the URL does not point to a `.conda` archive or the
61/// server does not support range requests.
62#[instrument(skip_all, fields(url = %redact_known_secrets_from_url(&url, DEFAULT_REDACTION_STR).as_ref().unwrap_or(&url), path = %target_path.display()))]
63pub async fn fetch_file_from_remote_sparse(
64    client: ClientWithMiddleware,
65    url: Url,
66    target_path: &Path,
67) -> Result<Option<Vec<u8>>, ExtractError> {
68    let archive_type = CondaArchiveType::try_from(std::path::Path::new(url.path()))
69        .ok_or(ExtractError::UnsupportedArchiveType)?;
70
71    if archive_type != CondaArchiveType::Conda {
72        return Err(ExtractError::UnsupportedArchiveType);
73    }
74
75    // Create range reader (fetches last 64KB on construction)
76    let (reader, _headers) = AsyncHttpRangeReader::new(
77        client,
78        url,
79        CheckSupportMethod::NegativeRangeRequest(DEFAULT_TAIL_SIZE),
80        HeaderMap::default(),
81    )
82    .await?;
83
84    // Wrap for async_zip (tokio → futures traits + buffering)
85    let buf_reader = futures::io::BufReader::new(reader.compat());
86
87    // Open ZIP (parses EOCD + central directory, data already cached from range request)
88    let mut zip_reader = ZipFileReader::new(buf_reader).await?;
89
90    // Find the tar.zst entry that contains the target path
91    let prefix = conda_entry_prefix(target_path);
92    let (index, _) = zip_reader
93        .file()
94        .entries()
95        .iter()
96        .enumerate()
97        .find(|(_, e)| {
98            e.filename()
99                .as_str()
100                .map(|f| f.starts_with(prefix) && f.ends_with(".tar.zst"))
101                .unwrap_or(false)
102        })
103        .ok_or(ExtractError::MissingComponent)?;
104
105    // Prefetch the entire info entry in a single HTTP request so the
106    // streaming pipeline doesn't trigger many small range requests.
107    let entry = &zip_reader.file().entries()[index];
108    let offset = entry.header_offset();
109    let size = entry.header_size() + entry.compressed_size();
110    zip_reader
111        .inner_mut()
112        .get_mut()
113        .get_mut()
114        .prefetch(offset..offset + size)
115        .await;
116
117    // Get a streaming reader for the ZIP entry (futures::io::AsyncRead).
118    // This does NOT buffer the entire entry — bytes are fetched on demand
119    // via HTTP range requests as the downstream decompressor/tar reader
120    // consumes them.
121    let entry_reader = zip_reader.reader_without_entry(index).await?;
122
123    // Pipeline: async ZIP entry reader -> tokio compat -> buffered -> zstd decoder -> tar
124    let tokio_reader = entry_reader.compat();
125    let buf_reader = tokio::io::BufReader::new(tokio_reader);
126    let zstd_decoder = ZstdDecoder::new(buf_reader);
127    let mut tar = tokio_tar::Archive::new(zstd_decoder);
128
129    let result = get_file_from_tar_archive(&mut tar, target_path).await?;
130
131    debug!(
132        "Requested ranges: {:?}",
133        zip_reader
134            .inner_mut()
135            .get_mut()
136            .get_mut()
137            .requested_ranges()
138            .await
139    );
140
141    Ok(result)
142}

144/// Fetch and parse a typed [`PackageFile`] from a remote `.conda` package
145/// using HTTP range requests.
146///
147/// This is a thin typed wrapper around [`fetch_file_from_remote_sparse`]. It
148/// only works for `.conda` archives on servers that support range requests and
149/// does not perform any full-download fallback.
150///
151/// # Example
152///
153/// ```rust,no_run
154/// # #[tokio::main]
155/// # async fn main() {
156/// use rattler_conda_types::package::IndexJson;
157/// use rattler_package_streaming::reqwest::sparse::fetch_package_file_sparse;
158/// use reqwest::Client;
159/// use reqwest_middleware::ClientWithMiddleware;
160/// use url::Url;
161///
162/// let client = ClientWithMiddleware::from(Client::new());
163/// let url = Url::parse("https://conda.anaconda.org/conda-forge/noarch/tzdata-2024b-hc8b5060_0.conda").unwrap();
164///
165/// let index_json: IndexJson = fetch_package_file_sparse(client, url).await.unwrap();
166/// println!("Package: {}", index_json.name.as_normalized());
167/// # }
168/// ```
169pub async fn fetch_package_file_sparse<P: PackageFile>(
170    client: ClientWithMiddleware,
171    url: Url,
172) -> Result<P, ExtractError> {
173    let bytes = fetch_file_from_remote_sparse(client, url, P::package_path())
174        .await?
175        .ok_or(ExtractError::MissingComponent)?;
176    P::from_slice(&bytes)
177        .map_err(|e| ExtractError::ArchiveMemberParseError(P::package_path().to_owned(), e))
178}

193        use rattler_conda_types::package::{AboutJson, IndexJson};

196        let client = reqwest_middleware::ClientWithMiddleware::from(reqwest::Client::new());

198        let index_json: IndexJson = fetch_package_file_sparse(client.clone(), url.clone())

203        let about_json: AboutJson = fetch_package_file_sparse(client, url).await.unwrap();
204        insta::assert_yaml_snapshot!(about_json);
205    }
