1use std::fmt::{Debug, Formatter};
2
3/// Bitmask indicates if the first component stored in a [`super::Version`] refers to an epoch.
4const EPOCH_MASK: u8 = 0b1;
5const EPOCH_OFFSET: u8 = 0;
6
7/// Bitmask that indicates what the index is of the first segment that belongs to the local version
8/// part. E.g. the part after the '+' sign in `1.2.3+4.5.6`.
9const LOCAL_VERSION_MASK: u8 = (1 << 7) - 1;
10const LOCAL_VERSION_OFFSET: u8 = 1;
11
12/// Encodes several edge cases in a single byte.
13///
14/// The first bit is used to indicate whether or not there is an explicit epoch present in the
15/// version. If the flag is set it means the first entry in the [`Version::components`] array refers
16/// to the epoch instead of to the first component of the first segment.
17///
18/// The remaining bits are used to encode the index of the first segment that belongs to the local
19/// version part instead of to the common part. A value of `0` indicates that there is not local
20/// version part.
21#[derive(Copy, Clone, Eq, PartialEq, Default)]
22#[repr(transparent)]
23pub struct Flags(pub(super) u8);
24
25impl Debug for Flags {
26    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
27        f.debug_struct("Flags")
28            .field("has_epoch", &self.has_epoch())
29            .field("local_segment_index", &self.local_segment_index())
30            .finish()
31    }
32}
33
34impl Flags {
35    /// Sets whether or not the version has an explicit epoch.
36    #[must_use]
37    pub fn with_has_epoch(self, has_epoch: bool) -> Self {
38        let flag = self.0 & !(EPOCH_MASK << EPOCH_OFFSET);
39        Self(
40            flag | if has_epoch {
41                EPOCH_MASK << EPOCH_OFFSET
42            } else {
43                0
44            },
45        )
46    }
47
48    /// Returns true if this instance indicates that an explicit epoch is present.
49    pub fn has_epoch(self) -> bool {
50        (self.0 >> EPOCH_OFFSET) & EPOCH_MASK != 0
51    }
52
53    /// Sets the index where the local segment starts. Returns `None` if the index is too large to
54    /// be stored.
55    #[must_use]
56    pub fn with_local_segment_index(self, index: u8) -> Option<Self> {
57        if index > LOCAL_VERSION_MASK {
58            None
59        } else {
60            Some(Self(
61                (self.0 & !(LOCAL_VERSION_MASK << LOCAL_VERSION_OFFSET))
62                    | (index << LOCAL_VERSION_OFFSET),
63            ))
64        }
65    }
66
67    /// Returns the index of the first segment that belongs to the local part of the version.
68    pub fn local_segment_index(self) -> u8 {
69        (self.0 >> LOCAL_VERSION_OFFSET) & LOCAL_VERSION_MASK
70    }
71}

73#[cfg(test)]
74mod test {
75    use crate::version::flags::Flags;
76
77    #[test]
78    fn test_epoch() {
79        assert!(!Flags::default().has_epoch());
80        assert!(Flags::default().with_has_epoch(true).has_epoch());
81        assert!(!Flags::default()
82            .with_has_epoch(true)
83            .with_has_epoch(false)
84            .has_epoch(),);
85    }
86
87    #[test]
88    fn test_local_segment_idx() {
89        assert_eq!(Flags::default().local_segment_index(), 0);
90        assert_eq!(
91            Flags::default()
92                .with_local_segment_index(42)
93                .unwrap()
94                .local_segment_index(),
95            42
96        );
97        assert_eq!(
98            Flags::default()
99                .with_local_segment_index(127)
100                .unwrap()
101                .local_segment_index(),
102            127
103        );
104        assert_eq!(Flags::default().with_local_segment_index(128), None);
105    }
106
107    #[test]
108    fn test_all_elements() {
109        let flags = Flags::default()
110            .with_has_epoch(true)
111            .with_local_segment_index(101)
112            .unwrap();
113
114        assert!(flags.has_epoch());
115        assert_eq!(flags.local_segment_index(), 101);
116    }
