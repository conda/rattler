1//! Minimal prefix record reading for fast environment change detection.
2//!
3//! This module provides functionality to read only the minimal metadata needed
4//! from conda-meta JSON files to determine if packages have changed, avoiding
5//! the expensive parsing of file lists and other large fields of `PrefixRecord`.
6//!
7//! The most interesting part of this file is custom parser written
8//! using `nom`. It uses `PrefixRecord` structure to parse as little
9//! data as possible, also uses byte parsing on memory map, so we
10//! don't read file entirely. See documentation of
11//! [`MinimalPrefixRecord::from_path`] for more information.
12

30/// We want to use memory map if file size is larger than 1MB to avoid
31/// doing many small reads, as this can be inefficient.
32///
33/// Ideally we should collect statistics to make this choice more
34/// justified, but this is good enough.
35pub const MMAP_HEURISTIC_THRESHOLD: usize = 1024 * 1024;
36
37/// A minimal version of `PrefixRecord` that only contains fields needed for transaction computation.
38/// This is much faster to parse than the full `PrefixRecord`.
39#[derive(Debug, Clone)]
40#[allow(deprecated)]
41pub struct MinimalPrefixRecord {
42    /// The package name
43    pub name: PackageName,
44    /// The package version
45    pub version: VersionWithSource,
46    /// The build string
47    pub build: String,
48
49    /// SHA256 hash of the package
50    pub sha256: Option<Sha256Hash>,
51    /// MD5 hash of the package, only if there is no SHA256 hash.
52    pub md5: Option<Md5Hash>,
53    /// Size of the package in bytes, only if there is no MD5 hash.
54    pub size: Option<u64>,
55
56    /// If this package is independent of architecture this field specifies in
57    /// what way. See [`NoArchType`] for more information.
58    pub noarch: NoArchType,
59    /// Optionally a path within the environment of the site-packages directory.
60    /// This field is only present for python interpreter packages.
61    /// This field was introduced with <https://github.com/conda/ceps/blob/main/cep-17.md>.
62    pub python_site_packages_path: Option<String>,
63
64    /// Deprecated: Old field for requested spec.
65    /// Only used for migration to `requested_specs`.
66    #[deprecated = "Use requested_specs instead"]
67    pub requested_spec: Option<String>,
68
69    /// The list of requested specs that were used to install this package.
70    /// This is used to track which specs requested this package.
71    pub requested_specs: Vec<String>,
72}

74impl MinimalPrefixRecord {
75    // Ideal approach would be to create `SparsePrefixRecord` akin `SpareRepodata`.
76    /// Parse a minimal prefix record from a JSON file using nom with memory mapping.
77    /// This is optimized for performance and stops parsing at the "files" field.
78    ///
79    /// The corner stone of its logic is decision of whether or not we
80    /// want to parse file completely.
81    ///
82    /// If we already met `requested_specs` field and now we're at
83    /// `files`, then we stop as this means new `PrefixRecord` format with
84    /// reordered fields and mandatory serialization of empty
85    /// `requested_spec`.
86    ///
87    /// If we at `files`, but haven't met `requested_specs`, then we have
88    /// legacy file format, so continue parsing either until we find this
89    /// field or until the end of file.
90    pub fn from_path(path: &Path) -> Result<Self, io::Error> {
91        use std::fs::File;
92
93        let filename_without_ext = path
94            .file_stem()
95            .and_then(|stem| stem.to_str())
96            .ok_or_else(|| io::Error::other("Invalid filename"))?;
97        let (build, version, name) = filename_without_ext
98            .rsplitn(3, '-')
99            .next_tuple()
100            .ok_or_else(|| io::Error::other("Invalid conda-meta filename format"))?;
101
102        let mut file = File::open(path)?;
103
104        let initial_metadata = file.metadata()?;
105
106        let file_size = initial_metadata.len() as usize;
107
108        // SAFETY: We create a read-only memory map of a regular file.
109        // We capture the file's metadata to detect if it changes during parsing.
110        let map_or_content = if file_size > MMAP_HEURISTIC_THRESHOLD {
111            let mmap = unsafe { Mmap::map(&file)? };
112            Ok(mmap)
113        } else {
114            let mut content = Vec::with_capacity(file_size);
115            #[allow(clippy::verbose_file_reads)]
116            file.read_to_end(&mut content)?;
117            Err(content)
118        };
119        let input = match &map_or_content {
120            Ok(mmap) => &mmap[..],
121            Err(content) => content.as_slice(),
122        };
123
124        let parsed = parse_minimal_json(input)
125            .map_err(|e| io::Error::other(format!("Failed to parse JSON: {e}")));
126
127        // Verify file hasn't changed during parsing
128        let final_metadata = file.metadata()?;
129        if initial_metadata.modified()? != final_metadata.modified()?
130            || initial_metadata.len() != final_metadata.len()
131        {
132            return Err(io::Error::other("File was modified during parsing"));
133        }
134
135        let parsed = parsed?;
136
137        let version = VersionWithSource::from_str(version)
138            .map_err(|e| io::Error::other(format!("Failed to parse version: {e}")))?;
139
140        let (final_sha256, final_md5, final_size) = if parsed.sha256.is_some() {
141            (parsed.sha256, None, None)
142        } else if parsed.md5.is_some() {
143            (None, parsed.md5, None)
144        } else {
145            (None, None, parsed.size)
146        };
147
148        let final_python_site_packages_path = if name.trim() == "python" {
149            parsed.python_site_packages_path.and_then(|quoted_bytes| {
150                // Use serde_json directly on the quoted byte slice - no allocation!
151                if let Ok(json_str) = std::str::from_utf8(quoted_bytes) {
152                    serde_json::from_str::<String>(json_str).ok()
153                } else {
154                    None
155                }
156            })
157        } else {
158            None
159        };
160
161        // Properly unescape requested_spec if present
162        let final_requested_spec = parsed.requested_spec.and_then(|quoted_bytes| {
163            if let Ok(json_str) = std::str::from_utf8(quoted_bytes) {
164                serde_json::from_str::<String>(json_str).ok()
165            } else {
166                None
167            }
168        });
169
170        // Properly unescape requested_specs
171        let final_requested_specs = parsed
172            .requested_specs
173            .into_iter()
174            .filter_map(|quoted_bytes| {
175                if let Ok(json_str) = std::str::from_utf8(quoted_bytes) {
176                    serde_json::from_str::<String>(json_str).ok()
177                } else {
178                    None
179                }
180            })
181            .collect();
182
183        // Parse noarch field
184        let final_noarch = match parsed.noarch {
185            Some(quoted_bytes) => {
186                if let Ok(json_str) = std::str::from_utf8(quoted_bytes) {
187                    match serde_json::from_str::<String>(json_str) {
188                        Ok(s) => match s.as_str() {
189                            "python" => NoArchType::python(),
190                            "generic" => NoArchType::generic(),
191                            _ => NoArchType::none(),
192                        },
193                        Err(_) => NoArchType::none(),
194                    }
195                } else {
196                    NoArchType::none()
197                }
198            }
199            None => NoArchType::none(),
200        };
201
202        #[allow(deprecated)]
203        Ok(Self {
204            name: name
205                .parse::<PackageName>()
206                .map_err(|e| format!("Could not parse package name: {e:#?}"))
207                .map_err(io::Error::other)?,
208            version,
209            build: build.into(),
210            sha256: final_sha256,
211            md5: final_md5,
212            size: final_size,
213            noarch: final_noarch,
214            python_site_packages_path: final_python_site_packages_path,
215            requested_specs: final_requested_specs,
216            requested_spec: final_requested_spec,
217        })
218    }

252    pub fn from_prefix_record(prefix_record: &PrefixRecord) -> Self {
253        let package_record = &prefix_record.repodata_record.package_record;

255        let (final_sha256, final_md5, final_size) = if package_record.sha256.is_some() {
256            (package_record.sha256, None, None)
257        } else if package_record.md5.is_some() {
258            (None, package_record.md5, None)
259        } else {
260            (None, None, package_record.size)

262

264            package_record.python_site_packages_path.clone()
265        } else {
266            None

268
269        #[allow(deprecated)]
270        Self {
271            name: package_record.name.clone(),
272            version: package_record.version.clone(),
273            build: package_record.build.clone(),
274            sha256: final_sha256,
275            md5: final_md5,
276            size: final_size,

278            python_site_packages_path: final_python_site_packages_path,
279            requested_spec: prefix_record.requested_spec.clone(),
280            requested_specs: prefix_record.requested_specs.clone(),
281        }
282    }

285/// Collect minimal prefix records from a prefix directory.
286/// This is much faster than collecting full `PrefixRecord`s when you only need
287/// to check if packages have changed.
288pub fn collect_minimal_prefix_records(
289    prefix: &Path,
290) -> Result<Vec<MinimalPrefixRecord>, io::Error> {
291    let conda_meta_path = prefix.join("conda-meta");
292
293    if !conda_meta_path.exists() {
294        return Ok(Vec::new());
295    }
296
297    // Collect paths first
298    let json_paths: Vec<_> = fs_err::read_dir(&conda_meta_path)?
299        .filter_map(|entry| {
300            entry.ok().and_then(|e| {
301                if e.file_type().ok()?.is_file()
302                    && e.file_name().to_string_lossy().ends_with(".json")
303                {
304                    Some(e.path())
305                } else {
306                    None
307                }
308            })
309        })
310        .collect();
311
312    // Parse minimal records in parallel if rayon is available
313    #[cfg(feature = "rayon")]
314    {
315        use rayon::prelude::*;
316        json_paths
317            .par_iter()
318            .map(|path| MinimalPrefixRecord::from_path(path))
319            .collect()
320    }
321
322    #[cfg(not(feature = "rayon"))]
323    {
324        json_paths
325            .iter()
326            .map(|path| MinimalPrefixRecord::from_path(path))
327            .collect()
328    }
329}

344/// Struct to hold the parsed fields from JSON
345#[derive(Debug, Default)]
346struct ParsedFields<'a> {
347    sha256: Option<Sha256Hash>,
348    md5: Option<Md5Hash>,
349    size: Option<u64>,
350    python_site_packages_path: Option<&'a [u8]>, // Store quoted JSON string
351    requested_specs: Vec<&'a [u8]>,              // Store quoted JSON strings
352    requested_spec: Option<&'a [u8]>,            // Store quoted JSON string
353    noarch: Option<&'a [u8]>, // Store quoted JSON string or raw for null/bool/other
354}
355
356/// Parsing state to track what fields we've found
357#[derive(Debug, Default)]
358struct ParseState {
359    found_requested_specs: bool,
360    found_files: bool,
361}

363/// Parse whitespace characters
364fn multispace0_bytes(input: &[u8]) -> IResult<&[u8], &[u8]> {
365    take_while(|c| c == b' ' || c == b'\t' || c == b'\n' || c == b'\r')(input)
366}

368/// Parse minimal JSON fields using nom, stopping at "files" field
369fn parse_minimal_json(
370    input: &[u8],
371) -> Result<ParsedFields<'_>, nom::Err<nom::error::Error<&[u8]>>> {
372    let (_, parsed) = parse_json_object(input)?;
373    Ok(parsed)
374}
375
376/// Parse JSON object looking for specific fields
377fn parse_json_object(input: &[u8]) -> IResult<&[u8], ParsedFields<'_>> {
378    let mut fields = ParsedFields::default();
379    let mut state = ParseState::default();
380
381    let (input, _) = preceded(multispace0_bytes, tag(&b"{"[..])).parse(input)?;
382    let (input, _) = multispace0_bytes(input)?;
383
384    let (remaining, _) = parse_fields(input, &mut fields, &mut state)?;
385
386    Ok((remaining, fields))
387}

389/// Parse JSON fields with stateful logic for `requested_specs` and files
390fn parse_fields<'a>(
391    mut input: &'a [u8],
392    fields: &mut ParsedFields<'a>,
393    state: &mut ParseState,
394) -> IResult<&'a [u8], ()> {
395    loop {
396        let (rest, _) = multispace0_bytes(input)?;
397        input = rest;
398
399        if input.starts_with(b"}") {
400            return Ok((input, ()));
401        }
402
403        let (rest, field_name) = parse_json_string_content(input)?;
404        input = rest;
405
406        match field_name {
407            b"files" => {
408                state.found_files = true;
409                // If we've already found requested_specs, we can stop here
410                if state.found_requested_specs {
411                    return Ok((input, ()));
412                }
413                // Otherwise, we need to continue parsing to find requested_specs
414            }
415            b"requested_specs" => {
416                state.found_requested_specs = true;
417            }
418            _ => {}
419        }
420
421        let (rest, _) = preceded(multispace0_bytes, tag(&b":"[..])).parse(input)?;
422        let (rest, _) = multispace0_bytes(rest)?;
423        input = rest;
424
425        let (rest, _) = parse_field_value(input, field_name, fields)?;
426        input = rest;
427
428        // If we just parsed requested_specs and we already found files, we can stop
429        if field_name == b"requested_specs" && state.found_files {
430            return Ok((rest, ()));
431        }
432
433        let (rest, _) = multispace0_bytes(input)?;
434        let (rest, _) = opt(tag(&b","[..])).parse(rest)?;
435        let (rest, _) = multispace0_bytes(rest)?;
436        input = rest;
437    }
438}

440/// Parse JSON string value, returning the content without quotes
441fn parse_json_string_content(input: &[u8]) -> IResult<&[u8], &[u8]> {
442    let (input, _) = tag(&b"\""[..])(input)?;
443
444    let mut i = 0;
445    while i < input.len() {
446        match input[i] {
447            b'"' => {
448                // Found closing quote
449                return Ok((&input[i + 1..], &input[..i]));
450            }
451            b'\\' => {
452                if i + 1 < input.len() {
453                    i += 2;
454                } else {
455                    return Err(nom::Err::Error(nom::error::Error::new(
456                        input,
457                        nom::error::ErrorKind::Escaped,
458                    )));
459                }
460            }
461            _ => {
462                i += 1;
463            }
464        }
465    }
466
467    Err(nom::Err::Error(nom::error::Error::new(
468        input,
469        nom::error::ErrorKind::Tag,
470    )))
471}

473/// Parse JSON string value, returning the full quoted string for `serde_json`.
474fn parse_json_string_quoted(input: &[u8]) -> IResult<&[u8], &[u8]> {
475    let start = input;
476    let (input, _) = tag(&b"\""[..])(input)?;
477
478    let mut i = 0;
479    while i < input.len() {
480        match input[i] {
481            b'"' => {
482                // Found closing quote, return the full quoted string
483                let end_pos = (input.as_ptr() as usize + i + 1) - start.as_ptr() as usize;
484                return Ok((&input[i + 1..], &start[..end_pos]));
485            }
486            b'\\' => {
487                if i + 1 < input.len() {
488                    i += 2;
489                } else {
490                    return Err(nom::Err::Error(nom::error::Error::new(
491                        input,
492                        nom::error::ErrorKind::Escaped,
493                    )));
494                }
495            }
496            _ => {
497                i += 1;
498            }
499        }
500    }
501
502    Err(nom::Err::Error(nom::error::Error::new(
503        input,
504        nom::error::ErrorKind::Tag,
505    )))
506}

508/// Parse field value based on field name
509fn parse_field_value<'a>(
510    input: &'a [u8],
511    field_name: &[u8],
512    fields: &mut ParsedFields<'a>,
513) -> IResult<&'a [u8], ()> {
514    match field_name {
515        b"sha256" => {
516            let (rest, value) = parse_json_string_content(input)?;
517            if let Ok(hex_str) = std::str::from_utf8(value) {
518                if let Ok(bytes) = hex::decode(hex_str) {
519                    if bytes.len() == 32 {
520                        fields.sha256 = Some(Sha256Hash::from(
521                            <[u8; 32]>::try_from(bytes.as_slice()).unwrap(),
522                        ));
523                    }
524                }
525            }
526            Ok((rest, ()))
527        }
528        b"md5" => {
529            let (rest, value) = parse_json_string_content(input)?;
530            if let Ok(hex_str) = std::str::from_utf8(value) {
531                if let Ok(bytes) = hex::decode(hex_str) {
532                    if bytes.len() == 16 {
533                        fields.md5 = Some(Md5Hash::from(
534                            <[u8; 16]>::try_from(bytes.as_slice()).unwrap(),
535                        ));
536                    }
537                }
538            }
539            Ok((rest, ()))
540        }
541        b"size" => {
542            let (rest, value) = take_until_comma_or_brace(input)?;
543            if let Ok(size_str) = std::str::from_utf8(value) {
544                fields.size = size_str.trim().parse().ok();
545            }
546            Ok((rest, ()))
547        }
548        b"python_site_packages_path" => {
549            // Handle both string and null values for python_site_packages_path
550            let (rest, _) = multispace0_bytes(input)?;
551            if rest.starts_with(b"null") {
552                fields.python_site_packages_path = None;
553                Ok((&rest[4..], ()))
554            } else {
555                let (rest, quoted_value) = parse_json_string_quoted(input)?;
556                fields.python_site_packages_path = Some(quoted_value);
557                Ok((rest, ()))
558            }
559        }
560        b"requested_spec" => {
561            // Handle both string and null values for requested_spec
562            let (rest, _) = multispace0_bytes(input)?;
563            if rest.starts_with(b"null") {
564                fields.requested_spec = None;
565                Ok((&rest[4..], ()))
566            } else {
567                let (rest, quoted_value) = parse_json_string_quoted(input)?;
568                fields.requested_spec = Some(quoted_value);
569                Ok((rest, ()))
570            }
571        }
572        b"requested_specs" => {
573            let (rest, specs) = parse_json_array(input)?;
574            fields.requested_specs = specs;
575            Ok((rest, ()))
576        }
577        b"noarch" => {
578            // Handle string values like "python", "generic" or null/false
579            let (rest, _) = multispace0_bytes(input)?;
580            if rest.starts_with(b"null") || rest.starts_with(b"false") {
581                fields.noarch = None;
582                let consumed = if rest.starts_with(b"null") { 4 } else { 5 };
583                Ok((&rest[consumed..], ()))
584            } else if rest.starts_with(b"\"") {
585                let (rest, quoted_value) = parse_json_string_quoted(input)?;
586                fields.noarch = Some(quoted_value);
587                Ok((rest, ()))
588            } else {
589                // Skip unknown noarch values
590                let (rest, _) = skip_json_value(input)?;
591                Ok((rest, ()))
592            }
593        }
594        _ => {
595            // Skip unknown fields
596            let (rest, _) = skip_json_value(input)?;
597            Ok((rest, ()))
598        }
599    }
600}

602/// Parse JSON array of strings, returning quoted byte slices
603fn parse_json_array(input: &[u8]) -> IResult<&[u8], Vec<&[u8]>> {
604    delimited(
605        preceded(multispace0_bytes, tag(&b"["[..])),
606        separated_list0(
607            preceded(multispace0_bytes, tag(&b","[..])),
608            preceded(multispace0_bytes, parse_json_string_quoted),
609        ),
610        preceded(multispace0_bytes, tag(&b"]"[..])),
611    )
612    .parse(input)
613}

615/// Skip any JSON value (string, number, object, array, boolean, null)
616fn skip_json_value(input: &[u8]) -> IResult<&[u8], ()> {
617    let (rest, _) = multispace0_bytes(input)?;
618
619    if rest.starts_with(b"\"") {
620        let (rest, _) = parse_json_string_content(rest)?;
621        Ok((rest, ()))
622    } else if rest.starts_with(b"[") {
623        skip_json_array(rest)
624    } else if rest.starts_with(b"{") {
625        skip_json_object(rest)
626    } else if rest.starts_with(b"true") {
627        Ok((&rest[4..], ()))
628    } else if rest.starts_with(b"false") {
629        Ok((&rest[5..], ()))
630    } else if rest.starts_with(b"null") {
631        Ok((&rest[4..], ()))
632    } else {
633        // Number
634        take_until_comma_or_brace(rest).map(|(rest, _)| (rest, ()))
635    }
636}

638/// Skip JSON array
639fn skip_json_array(input: &[u8]) -> IResult<&[u8], ()> {
640    let mut input = input;
641    let (rest, _) = tag(&b"["[..]).parse(input)?;
642    input = rest;
643
644    let mut depth = 1;
645    while depth > 0 && !input.is_empty() {
646        if input.starts_with(b"[") {
647            depth += 1;
648            input = &input[1..];
649        } else if input.starts_with(b"]") {
650            depth -= 1;
651            input = &input[1..];
652        } else {
653            input = &input[1..];
654        }
655    }
656
657    Ok((input, ()))
658}
659
660/// Skip JSON object
661fn skip_json_object(input: &[u8]) -> IResult<&[u8], ()> {
662    let mut input = input;
663    let (rest, _) = tag(&b"{"[..]).parse(input)?;
664    input = rest;
665
666    let mut depth = 1;
667    while depth > 0 && !input.is_empty() {
668        if input.starts_with(b"{") {
669            depth += 1;
670            input = &input[1..];
671        } else if input.starts_with(b"}") {
672            depth -= 1;
673            input = &input[1..];
674        } else {
675            input = &input[1..];
676        }
677    }
678
679    Ok((input, ()))
680}
681
682/// Take bytes until comma, closing brace, or closing bracket
683fn take_until_comma_or_brace(input: &[u8]) -> IResult<&[u8], &[u8]> {
684    let mut end = 0;
685
686    for &byte in input {
687        if byte == b',' || byte == b'}' || byte == b']' {
688            break;
689        }
690        end += 1;
691    }
692
693    Ok((&input[end..], &input[..end]))
694}
