13/// A representation of the `index.json` file found in package archives.
14///
15/// The `index.json` file contains information about the package build and
16/// dependencies of the package. This data makes up the repodata.json file in
17/// the repository.
18#[serde_as]
19#[sorted]
20#[skip_serializing_none]
21#[derive(Debug, Clone, Deserialize, Serialize, Eq, PartialEq)]
22pub struct IndexJson {
23    /// Optionally, the architecture the package is build for.
24    pub arch: Option<String>,
25
26    /// The build string of the package.
27    pub build: String,
28
29    /// The build number of the package. This is also included in the build
30    /// string.
31    pub build_number: u64,
32
33    /// The package constraints of the package
34    #[serde(default, skip_serializing_if = "Vec::is_empty")]
35    pub constrains: Vec<String>,
36
37    /// The dependencies of the package
38    #[serde(default)]
39    pub depends: Vec<String>,
40
41    /// Extra dependency groups that can be selected using `foobar[extras=["scientific"]]`
42    /// The implementation is specified in this CEP: <https://github.com/conda/ceps/pull/111>
43    #[serde(default, skip_serializing_if = "BTreeMap::is_empty")]
44    #[serde(rename = "extra_depends")]
45    pub experimental_extra_depends: BTreeMap<String, Vec<String>>,
46
47    /// Features are a deprecated way to specify different feature sets for the
48    /// conda solver. This is not supported anymore and should not be used.
49    /// Instead, `mutex` packages should be used to specify
50    /// mutually exclusive features.
51    pub features: Option<String>,
52
53    /// Optionally, the license
54    pub license: Option<String>,
55
56    /// Optionally, the license family
57    pub license_family: Option<String>,
58
59    /// The lowercase name of the package
60    pub name: PackageName,
61
62    /// If this package is independent of architecture this field specifies in
63    /// what way. See [`NoArchType`] for more information.
64    #[serde(skip_serializing_if = "NoArchType::is_none")]
65    pub noarch: NoArchType,
66
67    /// Optionally, the OS the package is build for.
68    pub platform: Option<String>,
69
70    /// A list of Package URLs identifying this package.
71    /// See this CEP: <https://github.com/conda/ceps/pull/63>
72    #[serde(default, skip_serializing_if = "Option::is_none")]
73    pub purls: Option<BTreeSet<PackageUrl>>,
74
75    /// Optionally a path within the environment of the site-packages directory.
76    /// This field is only present for python interpreter packages.
77    /// This field was introduced with <https://github.com/conda/ceps/blob/main/cep-17.md>.
78    pub python_site_packages_path: Option<String>,
79
80    /// The subdirectory that contains this package
81    pub subdir: Option<String>,
82
83    /// The timestamp when this package was created
84    pub timestamp: Option<crate::utils::TimestampMs>,
85
86    /// Track features are nowadays only used to downweight packages (ie. give
87    /// them less priority). To that effect, the number of track features is
88    /// counted (number of commas) and the package is downweighted
89    /// by the number of `track_features`.
90    #[serde(default, skip_serializing_if = "Vec::is_empty")]
91    #[serde_as(as = "crate::utils::serde::Features")]
92    pub track_features: Vec<String>,
93
94    /// The version of the package
95    pub version: VersionWithSource,
96}
97
98impl PackageFile for IndexJson {
99    fn package_path() -> &'static Path {
100        Path::new("info/index.json")
101    }
102
103    fn from_str(str: &str) -> Result<Self, std::io::Error> {
104        serde_json::from_str(str).map_err(Into::into)
105    }
106
107    fn from_slice(slice: &[u8]) -> Result<Self, std::io::Error> {
108        serde_json::from_slice(slice).map_err(Into::into)
109    }
110}
