7/// Container for the conda package format and compression level
8#[derive(Clone, PartialEq, Eq, Debug)]
9pub struct PackageFormatAndCompression {
10    /// The conda archive type that is selected
11    pub archive_type: CondaArchiveType,
12    /// The compression level that is selected
13    pub compression_level: CompressionLevel,
14}

16#[derive(Clone, Debug, Deserialize, Serialize, Default, PartialEq, Eq)]
17#[serde(rename_all = "kebab-case")]
18pub struct BuildConfig {
19    /// package format and compression level

25// deserializer for the package format and compression level
26impl<'de> Deserialize<'de> for PackageFormatAndCompression {
27    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
28    where
29        D: serde::Deserializer<'de>,
30    {
31        let s = String::deserialize(deserializer)?;
32        let s = s.as_str();
33        PackageFormatAndCompression::from_str(s).map_err(D::Error::custom)
34    }
35}

37impl FromStr for PackageFormatAndCompression {
38    type Err = String;
39
40    fn from_str(s: &str) -> Result<Self, Self::Err> {
41        let mut split = s.split(':');
42        let package_format = split.next().ok_or("invalid")?;
43
44        let compression = split.next().unwrap_or("default");
45
46        // remove all non-alphanumeric characters
47        let package_format = package_format
48            .chars()
49            .filter(|c| c.is_alphanumeric())
50            .collect::<String>();
51
52        let archive_type = match package_format.to_lowercase().as_str() {
53            "tarbz2" => CondaArchiveType::TarBz2,
54            "conda" => CondaArchiveType::Conda,
55            _ => return Err(format!("Unknown package format: {package_format}")),
56        };
57
58        let compression_level = match compression {
59            "max" | "highest" => CompressionLevel::Highest,
60            "default" | "normal" => CompressionLevel::Default,
61            "fast" | "lowest" | "min" => CompressionLevel::Lowest,
62            number if number.parse::<i32>().is_ok() => {
63                let number = number.parse::<i32>().unwrap_or_default();
64                match archive_type {
65                    CondaArchiveType::TarBz2 => {
66                        if !(1..=9).contains(&number) {
67                            return Err("Compression level for .tar.bz2 must be between 1 and 9"
68                                .to_string());
69                        }
70                    }
71                    CondaArchiveType::Conda => {
72                        if !(-7..=22).contains(&number) {
73                            return Err(
74                                "Compression level for conda packages (zstd) must be between -7 and 22".to_string()
75                            );
76                        }
77                    }
78                }
79                CompressionLevel::Numeric(number)
80            }
81            _ => return Err(format!("Unknown compression level: {compression}")),
82        };
83
84        Ok(PackageFormatAndCompression {
85            archive_type,
86            compression_level,
87        })
88    }
89}

91impl Serialize for PackageFormatAndCompression {
92    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
93    where
94        S: serde::Serializer,
95    {
96        let package_format = match self.archive_type {
97            CondaArchiveType::TarBz2 => "tarbz2",
98            CondaArchiveType::Conda => "conda",
99        };
100        let compression_level = match self.compression_level {
101            CompressionLevel::Default => "default",
102            CompressionLevel::Highest => "max",
103            CompressionLevel::Lowest => "min",
104            CompressionLevel::Numeric(level) => &level.to_string(),
105        };
106
107        serializer.serialize_str(format!("{package_format}:{compression_level}").as_str())
108    }
109}
