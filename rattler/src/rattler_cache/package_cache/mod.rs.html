41/// A [`PackageCache`] manages a cache of extracted Conda packages on disk.
42///
43/// The store does not provide an implementation to get the data into the store.
44/// Instead, this is left up to the user when the package is requested. If the
45/// package is found in the cache it is returned immediately. However, if the
46/// cache is stale a user defined function is called to populate the cache. This
47/// separates the concerns between caching and fetching of the content.
48#[derive(Clone)]
49pub struct PackageCache {
50    inner: Arc<PackageCacheInner>,
51    cache_origin: bool,
52}
53
54#[derive(Default)]
55struct PackageCacheInner {
56    layers: Vec<PackageCacheLayer>,
57}
58
59pub struct PackageCacheLayer {
60    path: PathBuf,
61    packages: DashMap<BucketKey, Arc<tokio::sync::Mutex<Entry>>>,
62    validation_mode: ValidationMode,
63}

91/// Errors specific to the `PackageCache` interface
92#[derive(Debug, thiserror::Error)]
93#[non_exhaustive]
94pub enum PackageCacheError {
95    /// The operation was cancelled
96    #[error("the operation was cancelled")]
97    Cancelled,
98
99    /// An error occurred in a cache layer
100    #[error("failed to interact with the package cache layer.")]
101    LayerError(#[source] Box<dyn std::error::Error + Send + Sync>), // Wraps layer-specific errors
102
103    /// There are no writable layers to cache package to
104    #[error("no writable layers to cache package to")]
105    NoWritableLayers,
106}
107
108/// Errors specific to individual layers in the `PackageCache`
109#[derive(Debug, thiserror::Error)]
110#[non_exhaustive]
111pub enum PackageCacheLayerError {
112    /// The package is invalid
113    #[error("package is invalid")]
114    InvalidPackage,
115
116    /// The package was not found in this layer
117    #[error("package not found in this layer")]
118    PackageNotFound,
119
120    /// A locking error occurred
121    #[error("{0}")]
122    LockError(String, #[source] std::io::Error),
123
124    /// The operation was cancelled
125    #[error("the operation was cancelled")]
126    Cancelled,
127
128    /// An error occurred while fetching the package.
129    #[error(transparent)]
130    FetchError(#[from] Arc<dyn std::error::Error + Send + Sync + 'static>),
131
132    #[error("package cache layer error: {0}")]
133    OtherError(#[source] Box<dyn std::error::Error + Send + Sync>),
134}

155impl PackageCacheLayer {
156    /// Determine if the layer is read-only in the filesystem
157    pub fn is_readonly(&self) -> bool {
158        self.path
159            .metadata()
160            .is_ok_and(|m| m.permissions().readonly())
161    }

163    /// Validate the packages.
164    pub async fn try_validate(
165        &self,
166        cache_key: &CacheKey,
167    ) -> Result<CacheMetadata, PackageCacheLayerError> {
168        let cache_entry = self
169            .packages
170            .get(&cache_key.clone().into())
171            .ok_or(PackageCacheLayerError::PackageNotFound)?
172            .clone();
173        let mut cache_entry = cache_entry.lock().await;
174        let cache_path = self.path.join(cache_key.to_string());
175
176        match validate_package_common::<
177            fn(PathBuf) -> _,
178            Pin<Box<dyn Future<Output = Result<(), _>> + Send>>,
179            std::io::Error,
180        >(
181            cache_path,
182            cache_entry.last_revision,
183            cache_key.sha256.as_ref(),
184            None,
185            None,
186            self.validation_mode,
187        )
188        .await
189        {
190            Ok(cache_metadata) => {
191                cache_entry.last_revision = Some(cache_metadata.revision);
192                cache_entry.last_sha256 = cache_metadata.sha256;
193                Ok(cache_metadata)
194            }
195            Err(err) => Err(err),
196        }
197    }
198
199    /// Validate the package, and fetch it if invalid.
200    pub async fn validate_or_fetch<F, Fut, E>(
201        &self,
202        fetch: F,
203        cache_key: &CacheKey,
204        reporter: Option<Arc<dyn CacheReporter>>,
205    ) -> Result<CacheMetadata, PackageCacheLayerError>
206    where
207        F: (Fn(PathBuf) -> Fut) + Send + 'static,
208        Fut: Future<Output = Result<(), E>> + Send + 'static,
209        E: std::error::Error + Send + Sync + 'static,
210    {
211        let entry = self
212            .packages
213            .entry(cache_key.clone().into())
214            .or_default()
215            .clone();
216
217        let mut cache_entry = entry.lock().await;
218        let cache_path = self.path.join(cache_key.to_string());
219
220        match validate_package_common(
221            cache_path,
222            cache_entry.last_revision,
223            cache_key.sha256.as_ref(),
224            Some(fetch),
225            reporter,
226            self.validation_mode,
227        )
228        .await
229        {
230            Ok(cache_metadata) => {
231                cache_entry.last_revision = Some(cache_metadata.revision);
232                cache_entry.last_sha256 = cache_metadata.sha256;
233                Ok(cache_metadata)
234            }
235            Err(e) => Err(e),
236        }
237    }
238}

250    /// Adds the origin (url or path) to the cache key to avoid unwanted cache

253        Self {

259    /// Acquires a global lock on the package cache.
260    ///
261    /// This lock can be used to coordinate multiple package operations,
262    /// reducing the overhead of acquiring individual locks for each package.
263    /// The lock is held until the returned `CacheGlobalLock` is dropped.
264    ///
265    /// This is particularly useful when installing many packages at once,
266    /// as it significantly reduces the number of file locking syscalls.
267    pub async fn acquire_global_lock(&self) -> Result<CacheGlobalLock, PackageCacheError> {
268        // Use the first writable layer's path for the global cache lock
269        let (_, writable_layers) = self.split_layers();
270        let cache_layer = writable_layers
271            .first()
272            .ok_or(PackageCacheError::NoWritableLayers)?;
273
274        let lock_file_path = cache_layer.path.join(".cache.lock");
275
276        // Ensure the directory exists
277        tokio_fs::create_dir_all(&cache_layer.path)
278            .await
279            .map_err(|e| {
280                PackageCacheError::LayerError(Box::new(PackageCacheLayerError::LockError(
281                    format!(
282                        "failed to create cache directory: '{}'",
283                        cache_layer.path.display()
284                    ),
285                    e,
286                )))
287            })?;
288
289        CacheGlobalLock::acquire(&lock_file_path)
290            .await
291            .map_err(|e| PackageCacheError::LayerError(Box::new(e)))
292    }

294    /// Constructs a new [`PackageCache`] located at the specified paths.
295    /// Layers are queried in the order they are provided.
296    /// The first writable layer is written to.
297    pub fn new_layered<I>(paths: I, cache_origin: bool, validation_mode: ValidationMode) -> Self
298    where
299        I: IntoIterator,
300        I::Item: Into<PathBuf>,
301    {
302        let layers = paths
303            .into_iter()
304            .map(|path| PackageCacheLayer {
305                path: path.into(),
306                packages: DashMap::default(),
307                validation_mode,
308            })
309            .collect();
310
311        Self {
312            inner: Arc::new(PackageCacheInner { layers }),
313            cache_origin,
314        }
315    }

317    /// Returns a tuple containing two sets of layers:
318    /// - A collection of read-only layers.
319    /// - A collection of writable layers.
320    ///
321    /// The permissions are checked at the time of the function call.
322    pub fn split_layers(&self) -> (Vec<&PackageCacheLayer>, Vec<&PackageCacheLayer>) {
323        self.inner
324            .layers
325            .iter()
326            .partition(|layer| layer.is_readonly())
327    }

329    /// Returns the directory that contains the specified package.
330    ///
331    /// If the package was previously successfully fetched and stored in the
332    /// cache the directory containing the data is returned immediately. If
333    /// the package was not previously fetch the filesystem is checked to
334    /// see if a directory with valid package content exists. Otherwise, the
335    /// user provided `fetch` function is called to populate the cache.
336    ///
337    /// ## Layer Priority
338    ///
339    /// Layers are checked in the order they were provided to [`PackageCache::new_layered`].
340    /// If a valid package is found in any layer, it is returned immediately. If no valid
341    /// package is found in any layer, the package is fetched and written to the first
342    /// writable layer.
343    ///
344    /// If the package is already being fetched by another task/thread the
345    /// request is coalesced. No duplicate fetch is performed.
346    pub async fn get_or_fetch<F, Fut, E>(
347        &self,
348        pkg: impl Into<CacheKey>,
349        fetch: F,
350        reporter: Option<Arc<dyn CacheReporter>>,
351    ) -> Result<CacheMetadata, PackageCacheError>
352    where
353        F: (Fn(PathBuf) -> Fut) + Send + 'static,
354        Fut: Future<Output = Result<(), E>> + Send + 'static,
355        E: std::error::Error + Send + Sync + 'static,
356    {
357        let cache_key = pkg.into();
358        let (_, writable_layers) = self.split_layers();
359
360        for layer in self.inner.layers.iter() {
361            let cache_path = layer.path.join(cache_key.to_string());
362
363            if cache_path.exists() {
364                match layer.try_validate(&cache_key).await {
365                    Ok(lock) => {
366                        return Ok(lock);
367                    }
368                    Err(PackageCacheLayerError::InvalidPackage) => {
369                        // Log and continue to the next layer
370                        tracing::warn!(
371                            "Invalid package in layer at path {:?}, trying next layer.",
372                            layer.path
373                        );
374                    }
375                    Err(PackageCacheLayerError::PackageNotFound) => {
376                        // Log and continue to the next layer
377                        tracing::debug!(
378                            "Package not found in layer at path {:?}, trying next layer.",
379                            layer.path
380                        );
381                    }
382                    Err(err) => return Err(err.into()),
383                }
384            }
385        }
386
387        // No matches in all layers, let's write to the first writable layer
388        tracing::debug!("no matches in all layers. writing to first writable layer");
389        if let Some(layer) = writable_layers.first() {
390            return match layer.validate_or_fetch(fetch, &cache_key, reporter).await {
391                Ok(cache_metadata) => Ok(cache_metadata),
392                Err(e) => Err(e.into()),
393            };
394        }
395
396        Err(PackageCacheError::NoWritableLayers)
397    }

415    /// Returns the directory that contains the specified package.
416    ///
417    /// This is a convenience wrapper around `get_or_fetch` which fetches the
418    /// package from the given path if the package could not be found in the
419    /// cache.
420    pub async fn get_or_fetch_from_path(
421        &self,
422        path: &Path,
423        reporter: Option<Arc<dyn CacheReporter>>,
424    ) -> Result<CacheMetadata, PackageCacheError> {
425        let path_buf = path.to_path_buf();
426        let mut cache_key: CacheKey = CondaArchiveIdentifier::try_from_path(&path_buf)
427            .unwrap()
428            .into();
429        if self.cache_origin {
430            cache_key = cache_key.with_path(path);
431        }
432
433        self.get_or_fetch(
434            cache_key,
435            move |destination| {
436                let path_buf = path_buf.clone();
437                async move {
438                    rattler_package_streaming::tokio::fs::extract(&path_buf, &destination)
439                        .await
440                        .map(|_| ())
441                }
442            },
443            reporter,
444        )
445        .await
446    }

448    /// Returns the directory that contains the specified package.
449    ///
450    /// This is a convenience wrapper around `get_or_fetch` which fetches the
451    /// package from the given URL if the package could not be found in the
452    /// cache.
453    ///
454    /// This function assumes that the `client` is already configured with a
455    /// retry middleware that will retry any request that fails. This function
456    /// uses the passed in `retry_policy` if, after the request has been sent
457    /// and the response is successful, streaming of the package data fails
458    /// and the whole request must be retried.
459    #[instrument(skip_all, fields(url=%url))]
460    pub async fn get_or_fetch_from_url_with_retry(
461        &self,
462        pkg: impl Into<CacheKey>,
463        url: Url,
464        client: LazyClient,
465        retry_policy: impl RetryPolicy + Send + 'static + Clone,
466        reporter: Option<Arc<dyn CacheReporter>>,
467    ) -> Result<CacheMetadata, PackageCacheError> {
468        let request_start = SystemTime::now();
469        // Convert into cache key
470        let mut cache_key = pkg.into();
471        if self.cache_origin {
472            cache_key = cache_key.with_url(url.clone());
473        }
474        // Sha256 of the expected package
475        let sha256 = cache_key.sha256();
476        let md5 = cache_key.md5();
477        let download_reporter = reporter.clone();
478        // Get or fetch the package, using the specified fetch function
479        self.get_or_fetch(cache_key, move |destination| {
480            let url = url.clone();
481            let client = client.clone();
482            let retry_policy = retry_policy.clone();
483            let download_reporter = download_reporter.clone();
484            async move {
485                let mut current_try = 0;
486                // Retry until the retry policy says to stop
487                loop {
488                    current_try += 1;
489                    tracing::debug!("downloading {} to {}", &url, destination.display());
490                    // Extract the package
491                    let result = rattler_package_streaming::reqwest::tokio::extract(
492                        client.client().clone(),
493                        url.clone(),
494                        &destination,
495                        sha256,
496                        download_reporter.clone().map(|reporter| Arc::new(PassthroughReporter {
497                            reporter,
498                            index: Mutex::new(None),
499                        }) as Arc::<dyn DownloadReporter>),
500                    )
501                        .await;
502
503                    let err = match result {
504                        Ok(result) => {
505                            // HACK: Only check one hash. Sometimes it occurs that the server
506                            // reports the wrong md5 hash while the Sha256 hash is valid. We used to
507                            // error on this case. However, the Sha256 hash is already secure enough
508                            // that we can ignore this case.
509                            //
510                            // For context, conda itself only checks one hash.
511                            if let Some(sha256) = sha256 {
512                                if sha256 != result.sha256 {
513                                    // Delete the package if the hash does not match.
514                                    // Failure here is non-fatal: the TempDir guard will
515                                    // clean up on drop; log and continue so the retry
516                                    // loop can proceed rather than panicking.
517                                    if let Err(e) = tokio_fs::remove_dir_all(&destination).await {
518                                        tracing::warn!(
519                                            "failed to remove destination on sha256 mismatch \
520                                             (will be cleaned up on drop): {e}"
521                                        );
522                                    }
523                                    return Err(ExtractError::HashMismatch {
524                                        url: url.clone().redact().to_string(),
525                                        destination: destination.display().to_string(),
526                                        expected: format!("{sha256:x}"),
527                                        actual: format!("{:x}", result.sha256),
528                                        total_size: result.total_size,
529                                    });
530                                }
531                            }  else if let Some(md5) = md5 {
532                                if md5 != result.md5 {
533                                    // Delete the package if the hash does not match.
534                                    // Failure here is non-fatal: the TempDir guard will
535                                    // clean up on drop; log and continue so the retry
536                                    // loop can proceed rather than panicking.
537                                    if let Err(e) = tokio_fs::remove_dir_all(&destination).await {
538                                        tracing::warn!(
539                                            "failed to remove destination on md5 mismatch \
540                                             (will be cleaned up on drop): {e}"
541                                        );
542                                    }
543                                    return Err(ExtractError::HashMismatch {
544                                        url: url.clone().redact().to_string(),
545                                        destination: destination.display().to_string(),
546                                        expected: format!("{md5:x}"),
547                                        actual: format!("{:x}", result.md5),
548                                        total_size: result.total_size,
549                                    });
550                                }
551                            }
552                            return Ok(());
553                        }
554                        Err(err) => err,
555                    };
556
557                    // Check if we should retry this error. We assume that the
558                    // user has middleware installed that handles connection
559                    // retries, but we still need to handle streaming failures
560                    // that occur after a successful response.
561
562                    if !err.should_retry() {
563                        return Err(err);
564                    }
565
566                    // Determine whether to retry based on the retry policy
567                    let execute_after = match retry_policy.should_retry(request_start, current_try) {
568                        RetryDecision::Retry { execute_after } => execute_after,
569                        RetryDecision::DoNotRetry => return Err(err),
570                    };
571                    let duration = execute_after.duration_since(SystemTime::now()).unwrap_or(Duration::ZERO);
572
573                    // Wait for a second to let the remote service restore itself. This increases the
574                    // chance of success.
575                    tracing::warn!(
576                        "failed to download and extract {} to {}: {}. Retry #{}, Sleeping {:?} until the next attempt...",
577                        &url,
578                        destination.display(),
579                        err,
580                        current_try,
581                        duration
582                    );
583                    tokio::time::sleep(duration).await;
584                }
585            }
586        }, reporter)
587            .await
588    }
589}

591/// Shared logic for validating a package.
592async fn validate_package_common<F, Fut, E>(
593    path: PathBuf,
594    known_valid_revision: Option<u64>,
595    given_sha: Option<&Sha256Hash>,
596    fetch: Option<F>,
597    reporter: Option<Arc<dyn CacheReporter>>,
598    validation_mode: ValidationMode,
599) -> Result<CacheMetadata, PackageCacheLayerError>
600where
601    F: Fn(PathBuf) -> Fut + Send,
602    Fut: Future<Output = Result<(), E>> + 'static,
603    E: Error + Send + Sync + 'static,
604{
605    // Open the cache metadata file to read/write revision and hash information.
606    // Concurrent access is coordinated via the global cache lock.
607    let lock_file_path = {
608        // Append the `.lock` extension to the cache path to create the lock file path.
609        let mut path_str = path.as_os_str().to_owned();
610        path_str.push(".lock");
611        PathBuf::from(path_str)
612    };
613
614    // Ensure the directory containing the lock-file exists.
615    if let Some(root_dir) = lock_file_path.parent() {
616        tokio_fs::create_dir_all(root_dir)
617            .map_err(|e| {
618                PackageCacheLayerError::LockError(
619                    format!("failed to create cache directory: '{}'", root_dir.display()),
620                    e,
621                )
622            })
623            .await?;
624    }
625
626    let mut metadata = CacheMetadataFile::acquire(&lock_file_path).await?;
627    let cache_revision = metadata.read_revision()?;
628    let locked_sha256 = metadata.read_sha256()?;
629
630    let hash_mismatch = match (given_sha, &locked_sha256) {
631        (Some(given_hash), Some(locked_sha256)) => given_hash != locked_sha256,
632        _ => false,
633    };
634
635    let cache_dir_exists = path.is_dir();
636    if cache_dir_exists && !hash_mismatch {
637        let path_inner = path.clone();
638
639        let reporter = reporter.as_deref().map(|r| (r, r.on_validate_start()));
640
641        // If we know the revision is already valid we can return immediately.
642        if known_valid_revision == Some(cache_revision) {
643            if let Some((reporter, index)) = reporter {
644                reporter.on_validate_complete(index);
645            }
646            return Ok(CacheMetadata {
647                revision: cache_revision,
648                sha256: locked_sha256,
649                path: path_inner,
650                index_json: None,
651                paths_json: None,
652            });
653        }
654
655        // Validate the package directory.
656        let validation_result = tokio::task::spawn_blocking(move || {
657            validate_package_directory(&path_inner, validation_mode)
658        })
659        .await;
660
661        if let Some((reporter, index)) = reporter {
662            reporter.on_validate_complete(index);
663        }
664
665        match validation_result {
666            Ok(Ok((index_json, paths_json))) => {
667                tracing::debug!("validation succeeded");
668                return Ok(CacheMetadata {
669                    revision: cache_revision,
670                    sha256: locked_sha256,
671                    path,
672                    index_json: Some(index_json),
673                    paths_json: Some(paths_json),
674                });
675            }
676            Ok(Err(e)) => {
677                tracing::warn!("validation for {path:?} failed: {e}");
678                if let Some(cause) = e.source() {
679                    tracing::debug!(
680                        "  Caused by: {}",
681                        std::iter::successors(Some(cause), |e| (*e).source())
682                            .format("\n  Caused by: ")
683                    );
684                }
685            }
686            Err(e) => {
687                if let Ok(panic) = e.try_into_panic() {
688                    std::panic::resume_unwind(panic)
689                }
690            }
691        }
692    } else if !cache_dir_exists {
693        tracing::debug!("cache directory does not exist");
694    } else if hash_mismatch {
695        tracing::warn!(
696            "hash mismatch, wanted a package at location {} with hash {} but the cached package has hash {}, fetching package",
697            path.display(),
698            given_sha.map_or(String::from("<unknown>"), |s| format!("{s:x}")),
699            locked_sha256.map_or(String::from("<unknown>"), |s| format!("{s:x}"))
700        );
701    }
702
703    // If the cache is stale, we need to fetch the package again.
704    // Since we hold the global cache lock, we can safely update the metadata
705    // and fetch the package without worrying about concurrent modifications.
706    if let Some(ref fetch_fn) = fetch {
707        // Write the new revision
708        let new_revision = cache_revision + 1;
709        metadata
710            .write_revision_and_sha(new_revision, given_sha)
711            .await?;
712
713        // Create a temporary directory in the same parent folder for atomic extraction.
714        // Using the same parent ensures the rename operation is atomic (same filesystem).
715        let parent_dir = path.parent().ok_or_else(|| {
716            PackageCacheLayerError::OtherError(Box::new(std::io::Error::other(format!(
717                "cache path '{}' has no parent directory",
718                path.display()
719            ))))
720        })?;
721
722        let prefix = path.file_name().and_then(|n| n.to_str()).unwrap_or("pkg");
723
724        // Use tempfile::Builder to create a temp directory with automatic cleanup on failure
725        let temp_dir = tempfile::Builder::new()
726            .prefix(&format!(".{prefix}"))
727            .tempdir_in(parent_dir)
728            .map_err(|e| {
729                PackageCacheLayerError::OtherError(Box::new(std::io::Error::other(format!(
730                    "failed to create temp directory in '{}': {}",
731                    parent_dir.display(),
732                    e
733                ))))
734            })?;
735
736        // Fetch/extract the package to the temporary directory.
737        let fetch_result = fetch_fn(temp_dir.path().to_path_buf()).await;
738
739        // Handle fetch result
740        match fetch_result {
741            Ok(()) => {
742                // Take ownership of the temp directory path so it won't be auto-deleted
743                let temp_path = temp_dir.keep();
744
745                // Remove any existing (potentially corrupted) directory at the final path
746                if path.is_dir() {
747                    tokio_fs::remove_dir_all(&path).await.map_err(|e| {
748                        PackageCacheLayerError::OtherError(Box::new(std::io::Error::other(
749                            format!(
750                                "failed to remove existing cache directory '{}': {}",
751                                path.display(),
752                                e
753                            ),
754                        )))
755                    })?;
756                }
757
758                // Atomically rename the temp directory to the final destination
759                tokio_fs::rename(&temp_path, &path).await.map_err(|e| {
760                    PackageCacheLayerError::OtherError(Box::new(std::io::Error::other(format!(
761                        "failed to rename temp directory '{}' to '{}': {}",
762                        temp_path.display(),
763                        path.display(),
764                        e
765                    ))))
766                })?;
767
768                // After fetching, return the cache metadata with the new revision.
769                // We don't need to re-validate since we just fetched it.
770                Ok(CacheMetadata {
771                    revision: new_revision,
772                    sha256: given_sha.copied(),
773                    path,
774                    index_json: None,
775                    paths_json: None,
776                })
777            }
778            Err(e) => {
779                // temp_dir is dropped here, automatically cleaning up the directory
780                Err(PackageCacheLayerError::FetchError(Arc::new(e)))
781            }
782        }
783    } else {
784        Err(PackageCacheLayerError::InvalidPackage)
785    }
786}

788struct PassthroughReporter {
789    reporter: Arc<dyn CacheReporter>,
790    index: Mutex<Option<usize>>,
791}

793impl DownloadReporter for PassthroughReporter {
794    fn on_download_start(&self) {
795        let index = self.reporter.on_download_start();
796        assert!(
797            self.index.lock().replace(index).is_none(),
798            "on_download_start was called multiple times"
799        );
800    }
801
802    fn on_download_progress(&self, bytes_downloaded: u64, total_bytes: Option<u64>) {
803        let index = self.index.lock().expect("on_download_start was not called");
804        self.reporter
805            .on_download_progress(index, bytes_downloaded, total_bytes);
806    }
807
808    fn on_download_complete(&self) {
809        let index = self
810            .index
811            .lock()
812            .take()
813            .expect("on_download_start was not called");
814        self.reporter.on_download_completed(index);
815    }

976    /// A helper middleware function that simulates a broken pipe error
977    /// by returning a stream that errors after sending partial data.
978    #[allow(clippy::type_complexity)]
979    async fn fail_with_broken_pipe(
980        State((count, bytes)): State<(Arc<Mutex<i32>>, Arc<Mutex<usize>>)>,
981        req: Request<Body>,
982        next: Next,
983    ) -> Result<Response, StatusCode> {
984        let count = {
985            let mut count = count.lock().await;
986            *count += 1;
987            *count
988        };
989
990        println!(
991            "Running broken pipe middleware for request #{count} for {}",
992            req.uri()
993        );
994        let response = next.run(req).await;
995
996        if count <= 2 {
997            // Get the full response body
998            let body = response.into_body();
999            let mut body = body.into_data_stream();
1000            let mut buffer = Vec::new();
1001            while let Some(Ok(chunk)) = body.next().await {
1002                buffer.extend(chunk);
1003            }
1004
1005            let byte_count = *bytes.lock().await;
1006            // Create a stream that yields partial data then fails with broken pipe
1007            let partial_data: Bytes = buffer
1008                .into_iter()
1009                .take(byte_count)
1010                .collect::<Vec<u8>>()
1011                .into();
1012
1013            let stream = stream::unfold((false, partial_data), |(has_sent, data)| async move {
1014                if has_sent {
1015                    // Return broken pipe error on second iteration
1016                    return Some((
1017                        Err(std::io::Error::new(
1018                            std::io::ErrorKind::BrokenPipe,
1019                            "stream closed because of a broken pipe",
1020                        )),
1021                        (true, data),
1022                    ));
1023                }
1024                // Return the partial data first
1025                Some((Ok::<_, std::io::Error>(data.clone()), (true, data)))
1026            });
1027
1028            let body = Body::from_stream(stream);
1029            return Ok(Response::new(body));
1030        }
1031
1032        Ok(response)
1033    }

1122                None,

1310        pub url: Url,
1311        // is_readonly=true and layer_num=0 means this package will be installed to the first readonly cache layer
1312        pub is_readonly: bool,
1313        pub layer_num: usize,
1314        pub expected_sha: String,
1315    }
1316
1317    /// A helper function to create a layered cache, and install packages to specific layers
1318    async fn create_layered_cache(
1319        readonly_layer_count: usize,
1320        writable_layer_count: usize,
1321        packages: Vec<PackageInstallInfo>, // Use the new struct
1322    ) -> (PackageCache, Vec<TempDir>) {
1323        let mut readonly_dirs = Vec::new();
1324        let mut writable_dirs = Vec::new();
1325
1326        for _ in 0..readonly_layer_count {
1327            readonly_dirs.push(tempdir().unwrap());
1328        }
1329
1330        for _ in 0..writable_layer_count {
1331            writable_dirs.push(tempdir().unwrap());
1332        }
1333
1334        let all_layers_paths: Vec<TempDir> = readonly_dirs
1335            .into_iter()
1336            .chain(writable_dirs.into_iter())
1337            .collect();
1338
1339        let cache = PackageCache::new_layered(
1340            all_layers_paths.iter().map(|dir| dir.path().to_path_buf()),
1341            false,
1342            ValidationMode::default(),
1343        );
1344
1345        let (readonly_layers, writable_layers) = cache.inner.layers.split_at(readonly_layer_count);
1346
1347        // Install the packages to the appropriate layers
1348        for package in packages {
1349            let layer = if package.is_readonly {
1350                &readonly_layers[package.layer_num]
1351            } else {
1352                &writable_layers[package.layer_num]
1353            };
1354            let tar_archive_path =
1355                tools::download_and_cache_file_async(package.url, &package.expected_sha)
1356                    .await
1357                    .unwrap();
1358
1359            let key: CacheKey = CondaArchiveIdentifier::try_from_path(&tar_archive_path)
1360                .unwrap()
1361                .into();
1362            let key =
1363                key.with_sha256(parse_digest_from_hex::<Sha256>(&package.expected_sha).unwrap());
1364
1365            layer
1366                .validate_or_fetch(
1367                    move |destination| {
1368                        let tar_archive_path = tar_archive_path.clone();
1369                        async move {
1370                            rattler_package_streaming::tokio::fs::extract(
1371                                &tar_archive_path,
1372                                &destination,
1373                            )
1374                            .await
1375                            .map(|_| ())
1376                        }
1377                    },
1378                    &key,
1379                    None,
1380                )
1381                .await
1382                .unwrap();
1383        }
1384
1385        for layer in readonly_layers {
1386            #[cfg(unix)]
1387            std::fs::set_permissions(
1388                &layer.path,
1389                std::os::unix::fs::PermissionsExt::from_mode(0o555), // r_x r_x r_x
1390            )
1391            .unwrap();
1392            #[cfg(windows)]
1393            {
1394                let mut perms = std::fs::metadata(&layer.path).unwrap().permissions();
1395                perms.set_readonly(true); // Remove write permissions
1396                std::fs::set_permissions(&layer.path, perms).unwrap();
1397            }
1398        }
1399        (cache, all_layers_paths)
1400    }
