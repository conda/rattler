7#[derive(Debug, Clone, Serialize, Deserialize, Hash, PartialEq, Eq)]

9pub struct PlaceholderString(pub String);

11impl PlaceholderString {
12    pub fn resolve(&self, placeholder: impl AsRef<HashMap<String, String>>) -> String {
13        replace_placeholders(self.0.clone(), placeholder.as_ref())
14    }

17pub struct BaseMenuItemPlaceholders {
18    placeholders: HashMap<String, String>,
19}
20
21impl BaseMenuItemPlaceholders {
22    pub fn new(base_prefix: &Path, prefix: &Path, platform: Platform) -> Self {
23        let dist_name = |p: &Path| {
24            p.file_name()
25                .map_or_else(|| "empty".to_string(), |s| s.to_string_lossy().to_string())
26        };
27
28        let (python, base_python) = if platform.is_windows() {
29            (prefix.join("python.exe"), base_prefix.join("python.exe"))
30        } else {
31            (prefix.join("bin/python"), base_prefix.join("bin/python"))
32        };
33
34        let mut vars = HashMap::from([
35            ("BASE_PREFIX", base_prefix.to_path_buf()),
36            ("PREFIX", prefix.to_path_buf()),
37            ("PYTHON", python),
38            ("BASE_PYTHON", base_python),
39            ("MENU_DIR", prefix.join("Menu")),
40            ("HOME", dirs::home_dir().unwrap_or_default()),
41        ]);
42
43        if platform.is_windows() {
44            vars.insert("BIN_DIR", prefix.join("Library/bin"));
45            vars.insert("SCRIPTS_DIR", prefix.join("Scripts"));
46            vars.insert("BASE_PYTHONW", base_prefix.join("pythonw.exe"));
47            vars.insert("PYTHONW", prefix.join("pythonw.exe"));
48        } else {
49            vars.insert("BIN_DIR", prefix.join("bin"));
50        }
51
52        if platform.is_osx() {
53            vars.insert("PYTHONAPP", prefix.join("python.app/Contents/MacOS/python"));
54        }
55
56        let mut vars: HashMap<String, String> = vars
57            .into_iter()
58            .map(|(k, v)| (k.to_string(), v.to_string_lossy().to_string()))
59            .collect();
60
61        let icon_ext = if platform.is_windows() {
62            "ico"
63        } else if platform.is_osx() {
64            "icns"
65        } else {
66            "png"
67        };
68        vars.insert("ICON_EXT".to_string(), icon_ext.to_string());
69
70        vars.insert("DISTRIBUTION_NAME".to_string(), dist_name(prefix));
71        vars.insert("ENV_NAME".to_string(), dist_name(prefix));
72
73        BaseMenuItemPlaceholders { placeholders: vars }
74    }
75
76    /// Insert the menu item location into the placeholders
77    ///
78    /// - On Linux, this is the path to the `.desktop` file
79    /// - On Windows, this is the path to the start menu `.lnk` file
80    /// - On macOS, this is the path to the `.app` bundle
81    pub fn refine(&self, menu_item_location: &Path) -> MenuItemPlaceholders {
82        let mut vars = self.placeholders.clone();
83        vars.insert(
84            "MENU_ITEM_LOCATION".to_string(),
85            menu_item_location.to_string_lossy().to_string(),
86        );
87        MenuItemPlaceholders { placeholders: vars }
88    }
89}

108/// Replace placeholders in a string with values from a hashmap
109/// This only replaces placeholders in the form of {{ key }} (note: while this looks like a Jinja template, it is not).
110pub fn replace_placeholders(mut text: String, replacements: &HashMap<String, String>) -> String {
111    for (key, value) in replacements {
112        let placeholder = format!("{{{{ {key} }}}}");
113        text = text.replace(&placeholder, value);
114    }
115    text
