76    fn new(mode: MenuMode, menu_name: &str, name: &str) -> Self {
77        let system_config_directory = PathBuf::from("/etc/xdg/");
78        let system_data_directory = PathBuf::from("/usr/share");
79
80        let (config_directory, data_directory) = if mode == MenuMode::System {
81            (
82                system_config_directory.clone(),
83                system_data_directory.clone(),
84            )
85        } else {
86            (
87                dirs::config_dir().expect("could not get config dir"),
88                dirs::data_dir().expect("could not get data dir"),
89            )
90        };
91
92        Directories {
93            menu_name: menu_name.to_string(),
94            name: name.to_string(),
95            data_directory: data_directory.clone(),
96            system_menu_config_location: system_config_directory.join("menus/applications.menu"),
97            menu_config_location: config_directory.join("menus/applications.menu"),
98            config_directory,
99            desktop_entries_location: data_directory.join("applications"),
100            directory_entry_location: data_directory
101                .join(format!("desktop-directories/{}.directory", slugify(name))),
102        }
103    }

105    pub fn ensure_directories_exist(&self) -> Result<(), MenuInstError> {
106        fs::create_dir_all(&self.data_directory)?;
107        fs::create_dir_all(&self.config_directory)?;
108
109        let paths = vec![
110            self.menu_config_location
111                .parent()
112                .ok_or_else(|| MenuInstError::InvalidPath(self.menu_config_location.clone()))?,
113            &self.desktop_entries_location,
114            self.directory_entry_location
115                .parent()
116                .ok_or_else(|| MenuInstError::InvalidPath(self.directory_entry_location.clone()))?,
117        ];
118
119        for path in paths {
120            tracing::debug!("Ensuring path {} exists", path.display());
121            fs::create_dir_all(path)?;
122        }
123
124        Ok(())
125    }

168/// Run `xdg-mime` with the given arguments
169fn xdg_mime(
170    xml_file: &Path,
171    mode: MenuMode,
172    operation: XdgMimeOperation,
173) -> Result<(), XdgMimeError> {
174    let mut command = Command::new("xdg-mime");
175
176    let mode = match mode {
177        MenuMode::System => "system",
178        MenuMode::User => "user",
179    };
180
181    let operation = match operation {
182        XdgMimeOperation::Install => "install",
183        XdgMimeOperation::Uninstall => "uninstall",
184    };
185
186    command
187        .arg(operation)
188        .arg("--mode")
189        .arg(mode)
190        .arg("--novendor")
191        .arg(xml_file);
192
193    let output = command.output()?;
194
195    if !output.status.success() {
196        tracing::warn!(
197            "Could not un/register MIME type with xdg-mime. Writing to '{}' as a fallback.",
198            xml_file.display()
199        );
200        log_output("xdg-mime", output);
201
202        return Err(XdgMimeError::IoError(std::io::Error::other(
203            "xdg-mime failed",
204        )));
205    }
206
207    Ok(())
208}

284    fn command(&self) -> Result<String, MenuInstError> {
285        let envs = self
286            .command
287            .activate
288            .unwrap_or(true)
289            .then(|| {
290                // create a bash activation script and emit it into the script
291                let activator =
292                    Activator::from_path(&self.prefix, shell::Bash, Platform::current())?;
293                let activation_variables = ActivationVariables {
294                    path_modification_behavior: PathModificationBehavior::Prepend,
295                    ..Default::default()
296                };
297                let activation_env = activator.run_activation(activation_variables, None)?;
298
299                let envs = activation_env
300                    .into_iter()
301                    .map(|(k, v)| format!(r#""{k}={v}""#))
302                    .collect();
303                Ok::<Vec<String>, MenuInstError>(envs)
304            })
305            .transpose()?
306            .unwrap_or_default();
307
308        let main_command = self
309            .command
310            .command
311            .iter()
312            .map(|s| s.resolve(&self.placeholders))
313            .map(|part| {
314                if part.starts_with("%") {
315                    Ok(part)
316                } else {
317                    Ok(shlex::try_quote(&part)?.into_owned())
318                }
319            })
320            .collect::<Result<Vec<_>, MenuInstError>>()?
321            .join(" ");
322
323        let main_with_env = if envs.is_empty() {
324            main_command
325        } else {
326            format!("env {} {}", envs.join(" "), main_command)
327        };
328
329        if let Some(pre_command) = &self.command.precommand {
330            let command = [pre_command.0.clone(), main_with_env].join(" && ");
331            Ok(format!("bash -c {}", shlex::try_quote(&command)?))
332        } else {
333            Ok(main_with_env)
334        }
335    }

346    fn create_desktop_entry(&self, tracker: &mut LinuxTracker) -> Result<(), MenuInstError> {
347        let file = self.location();
348        tracing::info!("Creating desktop entry at {}", file.display());
349        let writer = File::create(&file)?;
350        let mut writer = std::io::BufWriter::new(writer);
351
352        tracker.paths.push(file);
353
354        writeln!(writer, "[Desktop Entry]")?;
355        writeln!(writer, "Type=Application")?;
356        writeln!(writer, "Encoding=UTF-8")?;
357        writeln!(writer, "Name={}", self.name)?;
358        writeln!(writer, "Exec={}", self.command()?)?;
359        writeln!(
360            writer,
361            "Terminal={}",
362            self.command.terminal.unwrap_or(false)
363        )?;
364
365        if let Some(icon) = &self.command.icon {
366            let icon = icon.resolve(&self.placeholders);
367            writeln!(writer, "Icon={icon}")?;
368        }
369
370        let description = self.command.description.resolve(&self.placeholders);
371        if !description.is_empty() {
372            writeln!(writer, "Comment={description}")?;
373        }
374
375        if let Some(working_dir) = &self.command.working_dir {
376            let working_dir = working_dir.resolve(&self.placeholders);
377            writeln!(writer, "Path={working_dir}")?;
378        }
379
380        // resolve categories and join them with a semicolon
381        if let Some(categories) = &self.item.categories {
382            writeln!(writer, "Categories={}", self.resolve_and_join(categories))?;
383        }
384
385        if let Some(dbus_activatable) = &self.item.dbus_activatable {
386            writeln!(writer, "DBusActivatable={dbus_activatable}")?;
387        }
388
389        if let Some(generic_name) = &self.item.generic_name {
390            writeln!(
391                writer,
392                "GenericName={}",
393                generic_name.resolve(&self.placeholders)
394            )?;
395        }
396
397        if let Some(hidden) = &self.item.hidden {
398            writeln!(writer, "Hidden={hidden}")?;
399        }
400
401        if let Some(implements) = &self.item.implements {
402            writeln!(writer, "Implements={}", self.resolve_and_join(implements))?;
403        }
404
405        if let Some(keywords) = &self.item.keywords {
406            writeln!(writer, "Keywords={}", self.resolve_and_join(keywords))?;
407        }
408
409        if let Some(mime_types) = &self.item.mime_type {
410            writeln!(writer, "MimeType={}", self.resolve_and_join(mime_types))?;
411        }
412
413        if let Some(no_display) = &self.item.no_display {
414            writeln!(writer, "NoDisplay={no_display}")?;
415        }
416
417        if let Some(not_show_in) = &self.item.not_show_in {
418            writeln!(writer, "NotShowIn={}", self.resolve_and_join(not_show_in))?;
419        }
420
421        if let Some(only_show_in) = &self.item.only_show_in {
422            writeln!(writer, "OnlyShowIn={}", self.resolve_and_join(only_show_in))?;
423        }
424
425        if let Some(single_main_window) = &self.item.single_main_window {
426            writeln!(writer, "SingleMainWindow={single_main_window}")?;
427        }
428
429        if let Some(prefers_non_default_gpu) = &self.item.prefers_non_default_gpu {
430            writeln!(writer, "PrefersNonDefaultGPU={prefers_non_default_gpu}")?;
431        }
432
433        if let Some(startup_notify) = &self.item.startup_notify {
434            writeln!(writer, "StartupNotify={startup_notify}")?;
435        }
436
437        if let Some(startup_wm_class) = &self.item.startup_wm_class {
438            writeln!(
439                writer,
440                "StartupWMClass={}",
441                startup_wm_class.resolve(&self.placeholders)
442            )?;
443        }
444
445        if let Some(try_exec) = &self.item.try_exec {
446            writeln!(writer, "TryExec={}", try_exec.resolve(&self.placeholders))?;
447        }
448
449        Ok(())
450    }

462    pub fn update_desktop_database(&self) {
463        if self.directories.desktop_entries_location.is_dir() {
464            match Command::new("update-desktop-database")
465                .arg(&self.directories.desktop_entries_location)
466                .status()
467            {
468                Ok(status) if status.success() => {
469                    tracing::debug!(
470                        "Updated desktop database at {}",
471                        self.directories.desktop_entries_location.display()
472                    );
473                }
474                Ok(status) => {
475                    tracing::warn!(
476                        "update-desktop-database exited with status {} for {}",
477                        status,
478                        self.directories.desktop_entries_location.display()
479                    );
480                }
481                Err(err) => {
482                    // Command might not exist (desktop-file-utils not installed); don't fail the operation
483                    tracing::warn!("Could not run update-desktop-database: {}", err);
484                }
485            }
486        } else {
487            tracing::warn!(
488                "Desktop entries location does not exist: {} (skipping update-desktop-database)",
489                self.directories.desktop_entries_location.display()
490            );
491        }

494    fn register_mime_types(&self, tracker: &mut LinuxTracker) -> Result<(), MenuInstError> {
495        let Some(mime_types) = self.item.mime_type.as_ref() else {
496            return Ok(());
497        };
498
499        let mime_types = mime_types
500            .iter()
501            .map(|s| s.resolve(&self.placeholders))
502            .collect::<Vec<String>>();
503
504        tracing::info!("Registering mime types {:?}", mime_types);
505        let mut resolved_globs = HashMap::<String, String>::new();
506
507        if let Some(globs) = &self.item.glob_patterns {
508            for (k, v) in globs {
509                resolved_globs.insert(k.resolve(&self.placeholders), v.resolve(&self.placeholders));
510            }
511        }
512
513        for mime_type in &mime_types {
514            if let Some(glob_pattern) = resolved_globs.get(mime_type) {
515                self.glob_pattern_for_mime_type(mime_type, glob_pattern, tracker)?;
516            }
517        }
518
519        let mimeapps = self.directories.config_directory.join("mimeapps.list");
520
521        let mut config = MimeConfig::load(&mimeapps)?;
522        for mime_type in &mime_types {
523            tracing::info!("Registering mime type {} for {}", mime_type, &self.name);
524            config.register_mime_type(mime_type, &self.name);
525        }
526        config.save()?;
527
528        // Store the data so that we can remove it later
529        tracker.mime_types = Some(LinuxRegisteredMimeFile {
530            mime_types: mime_types.clone(),
531            database_path: self.directories.mime_directory(),
532            application: self.name.clone(),
533            config_file: mimeapps,
534        });
535
536        update_mime_database(&self.directories.mime_directory())?;
537
538        Ok(())
539    }

541    fn xml_path_for_mime_type(&self, mime_type: &str) -> Result<(PathBuf, bool), std::io::Error> {
542        let basename = mime_type.replace("/", "-");
543        let mime_directory = self.directories.data_directory.join("mime/packages");
544        if !mime_directory.is_dir() {
545            return Ok((mime_directory.join(format!("{basename}.xml")), false));
546        }
547
548        let xml_files: Vec<PathBuf> = fs::read_dir(&mime_directory)?
549            .filter_map(|entry| {
550                let entry = entry.ok()?;
551                let path = entry.path();
552                let file_name = path.file_name()?.to_str()?;
553                if file_name.contains(&basename) {
554                    Some(path)
555                } else {
556                    None
557                }
558            })
559            .collect();
560
561        if !xml_files.is_empty() {
562            if xml_files.len() > 1 {
563                tracing::debug!(
564                    "Found multiple files for MIME type {}: {:?}. Returning first.",
565                    mime_type,
566                    xml_files
567                );
568            }
569            return Ok((xml_files[0].clone(), true));
570        }
571        Ok((mime_directory.join(format!("{basename}.xml")), false))
572    }

574    fn glob_pattern_for_mime_type(
575        &self,
576        mime_type: &str,
577        glob_pattern: &str,
578        tracker: &mut LinuxTracker,
579    ) -> Result<(), MenuInstError> {
580        let (xml_path, exists) = self.xml_path_for_mime_type(mime_type)?;
581        if exists {
582            // Already registered this mime type
583            return Ok(());
584        }
585
586        // Write the XML that binds our current mime type to the glob pattern
587        let xmlns = "http://www.freedesktop.org/standards/shared-mime-info";
588        let description = format!(
589            "Custom MIME type {mime_type} for '{glob_pattern}' files (registered by menuinst)"
590        );
591
592        let xml = format!(
593            r#"<?xml version="1.0" encoding="UTF-8"?>
594<mime-info xmlns="{xmlns}">
595    <mime-type type="{mime_type}">
596        <glob pattern="{glob_pattern}"/>
597        <comment>{description}</comment>
598    </mime-type>
599</mime-info>"#
600        );
601
602        // Install the XML file and register it as default for our app
603        let file_name = xml_path.file_name().expect("we should have a filename");
604        let mime_dir_in_prefix = self.prefix.join("Menu/registered-mimetypes");
605        fs::create_dir_all(&mime_dir_in_prefix)?;
606
607        // We store the mime file in a temporary dir inside the prefix, but we actually keep
608        // the file + directory around for eventual removal (just use tempdir for unique folder)
609        let temp_dir = TempDir::new_in(&mime_dir_in_prefix)?;
610        let file_path = temp_dir.path().join(file_name);
611        fs::write(&file_path, &xml)?;
612
613        if xdg_mime(&file_path, self.mode, XdgMimeOperation::Install).is_ok() {
614            // keep temp dir in prefix around and the temp file
615            // because we re-use it when unregistering the mime type.
616            let _ = temp_dir.keep();
617            tracker.registered_mime_files.push(file_path);
618        } else {
619            if let Some(parent) = xml_path.parent() {
620                fs::create_dir_all(parent)?;
621            }
622            fs::write(&xml_path, xml)?;
623            tracker.paths.push(xml_path);
624        }
625
626        Ok(())
627    }

631pub fn install_menu_item(

641    menu.install(&mut tracker)?;

643    Ok(tracker)
644}

646/// Remove a menu item on Linux.
647pub fn remove_menu_item(tracker: &LinuxTracker) -> Result<(), MenuInstError> {
648    for path in &tracker.paths {
649        match fs::remove_file(path) {
650            Ok(_) => {}
651            Err(e) => {
652                tracing::warn!("Could not remove file {}: {}", path.display(), e);
653            }
654        }
655    }
656
657    for path in &tracker.registered_mime_files {
658        match xdg_mime(path, tracker.install_mode, XdgMimeOperation::Uninstall) {
659            Ok(_) => {}
660            Err(e) => {
661                tracing::warn!("Could not uninstall mime type: {}", e);
662            }
663        }
664        // Remove the temporary directory we created for the glob file
665        if let Some(parent) = path.parent() {
666            fs::remove_dir_all(parent)?;
667        }
668    }
669
670    if let Some(installed_mime_types) = tracker.mime_types.as_ref() {
671        // load mimetype config
672        let mut config = MimeConfig::load(&installed_mime_types.config_file)?;
673        let application = &installed_mime_types.application;
674        for mime_type in &installed_mime_types.mime_types {
675            config.deregister_mime_type(mime_type, application);
676        }
677        config.save()?;
678
679        update_mime_database(&installed_mime_types.database_path)?;
680    }
681
682    Ok(())
683}

754    struct FakePrefix {
755        _tmp_dir: TempDir,
756        prefix_path: PathBuf,
757        schema: MenuInstSchema,
758    }
759
760    impl FakePrefix {
761        fn new(schema_json: &str) -> Self {
762            let tmp_dir = TempDir::new().unwrap();
763            let prefix_path = tmp_dir.path().join("test-env");
764            let schema_json = test_data().join(schema_json);
765            let menu_folder = prefix_path.join("Menu");
766
767            fs::create_dir_all(&menu_folder).unwrap();
768            fs::copy(
769                &schema_json,
770                menu_folder.join(schema_json.file_name().unwrap()),
771            )
772            .unwrap();
773
774            // Create a icon file for the
775            let schema = std::fs::read_to_string(schema_json).unwrap();
776            let parsed_schema: MenuInstSchema = serde_json::from_str(&schema).unwrap();
777
778            let mut placeholders = HashMap::<String, String>::new();
779            placeholders.insert(
780                "MENU_DIR".to_string(),
781                menu_folder.to_string_lossy().to_string(),
782            );
783
784            for item in &parsed_schema.menu_items {
785                let icon = item.command.icon.as_ref().unwrap();
786                for ext in &["icns", "png", "svg"] {
787                    placeholders.insert("ICON_EXT".to_string(), (*ext).to_string());
788                    let icon_path = icon.resolve(FakePlaceholders {
789                        placeholders: placeholders.clone(),
790                    });
791                    fs::write(&icon_path, []).unwrap();
792                }
793            }
794
795            fs::create_dir_all(prefix_path.join("bin")).unwrap();
796            fs::write(prefix_path.join("bin/python"), []).unwrap();
797
798            Self {
799                _tmp_dir: tmp_dir,
800                prefix_path,
801                schema: parsed_schema,
802            }
803        }
804
805        pub fn prefix(&self) -> &Path {
806            &self.prefix_path
807        }
808    }
