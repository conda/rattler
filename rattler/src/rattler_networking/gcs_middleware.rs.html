14/// The auth headers and the `EntityTag` assigned by the credential library.
15///
16/// The `EntityTag` is an opaque process-local token: the library generates one
17/// per token lifetime and reuses it until the token is refreshed.  Passing it
18/// back to `Credentials::headers()` lets the library confirm our cached copy is
19/// still current (`NotModified`) or hand us a freshly-minted one (`New`).
20struct CachedResource {
21    entity_tag: EntityTag,
22    headers: http::HeaderMap,
23}
24
25/// Token-cache state machine.
26enum CacheState {
27    /// No resource has been fetched yet and no refresh is in flight.
28    Empty,
29    /// One task is fetching a new token; others wait on `GCSInner::refresh_done`.
30    ///
31    /// The [`Weak`] is a liveness token: it points to an [`Arc`] owned by the
32    /// [`RefreshGuard`] that the refreshing task holds.  When the refreshing
33    /// task is cancelled (future dropped), the `Arc` is dropped, the `Weak`
34    /// becomes dead, and any waiter or new caller that observes a dead `Weak`
35    /// knows the refresher is gone and can reset the state to [`Empty`] and
36    /// retry — without requiring the `Drop` impl to acquire the mutex.
37    Refreshing(Weak<()>),
38    /// A valid resource is available.
39    Ready(CachedResource),
40}
41
42/// Shared, ref-counted state owned by every clone of a [`GCSMiddleware`].
43struct GCSInner {
44    /// Credential source built once and reused across all requests.
45    credential: Mutex<Option<Credentials>>,
46    /// Cache state machine guarded by a mutex.
47    cache: Mutex<CacheState>,
48    /// Woken every time a refresh completes (success or failure) so that
49    /// waiters can re-inspect the cache state.
50    refresh_done: Notify,
51}
52
53/// GCS middleware to authenticate requests.
54///
55/// A single [`GCSMiddleware`] instance (or any clone of one) shares one
56/// `OAuth2` credential and one token cache.  At most one token fetch is in
57/// flight at a time regardless of how many concurrent requests are being
58/// processed (singleflight).  Subsequent requests reuse the cached resource
59/// until the credential library signals that the token has changed, at which
60/// point the cache is transparently refreshed.
61#[derive(Clone)]
62pub struct GCSMiddleware {
63    inner: Arc<GCSInner>,
64}
65
66/// Outcome of a single synchronous inspection of the token cache.
67///
68/// Returned by [`GCSMiddleware::poll_cache`], which holds and releases the
69/// `std::sync::Mutex` entirely within a non-`async` context.  This ensures
70/// that no `MutexGuard` ever appears in the state machine of the surrounding
71/// `async fn`, keeping the future `Send`.
72enum PollResult<'a> {
73    /// A refresh is in flight.  The contained future has already been enabled
74    /// (via [`tokio::sync::futures::Notified::enable`]) while the lock was
75    /// held, so awaiting it cannot miss the completion signal.
76    Wait(Pin<Box<dyn std::future::Future<Output = ()> + Send + 'a>>),
77    /// The previous refresher was cancelled; the cache has been reset to
78    /// [`CacheState::Empty`].  The caller should loop immediately.
79    Retry,
80    /// A cached token is available.  The caller should validate it with the
81    /// credential library (no lock held).
82    Validate {
83        entity_tag: EntityTag,
84        headers: http::HeaderMap,
85    },
86    /// The cache was empty and the caller has claimed the refresh slot.  The
87    /// contained `Arc` is the liveness token; its `Weak` counterpart is now
88    /// stored in [`CacheState::Refreshing`].
89    StartRefresh(Arc<()>),
90}
91
92impl Default for GCSMiddleware {
93    fn default() -> Self {
94        Self {
95            inner: Arc::new(GCSInner {
96                credential: Mutex::new(None),
97                cache: Mutex::new(CacheState::Empty),
98                refresh_done: Notify::new(),
99            }),
100        }
101    }
102}

121    /// Create a new authentication middleware for GCS
122    async fn handle(
123        &self,
124        mut req: Request,
125        extensions: &mut http::Extensions,
126        next: Next<'_>,
127    ) -> MiddlewareResult<Response> {
128        if req.url().scheme() == "gcs" {
129            let mut url = req.url().clone();
130            let bucket_name = url.host_str().expect("Host should be present in GCS URL");
131            let new_url = format!(
132                "https://storage.googleapis.com/{}{}",
133                bucket_name,
134                url.path()
135            );
136            url = Url::parse(&new_url).expect("Failed to parse URL");
137            *req.url_mut() = url;
138            req = self.authenticate(req).await?;
139        }
140        next.run(req, extensions).await
141    }
142}

144impl GCSMiddleware {
145    /// Add GCS authentication headers to `req`, drawing from the token cache
146    /// when available and fetching a new token only when necessary.
147    async fn authenticate(&self, mut req: Request) -> MiddlewareResult<Request> {
148        let headers = self.get_or_refresh_token().await?;
149        req.headers_mut().extend(headers);
150        Ok(req)
151    }

153    /// Lazily initialise the `Credentials` object (once per middleware
154    /// lifetime) and return a cheap `Arc`-clone of it.
155    async fn get_credential(&self) -> MiddlewareResult<Credentials> {
156        let mut guard = self.inner.credential.lock().unwrap();
157        if guard.is_none() {
158            let scopes = ["https://www.googleapis.com/auth/devstorage.read_only"];
159            let c = AccessTokenCredentialBuilder::default()
160                .with_scopes(scopes)
161                .build()
162                .map_err(|e| reqwest_middleware::Error::Middleware(anyhow::Error::new(e)))?;
163            *guard = Some(c);
164        }
165        // Credentials is Arc-backed; clone is a cheap refcount bump.
166        Ok(guard.as_ref().unwrap().clone())
167    }
168
169    /// Inspect and (if necessary) update the token cache under the lock, then
170    /// return a [`PollResult`] describing what the caller should do next.
171    ///
172    /// This is a plain (`!async`) function so that the `std::sync::MutexGuard`
173    /// is created and dropped entirely within synchronous code.  The guard
174    /// never appears in an `async` state machine, keeping every future that
175    /// calls this method `Send`.
176    ///
177    /// ## Liveness-token protocol
178    ///
179    /// `mem::replace` takes ownership of the current state before any match
180    /// arm runs, so no pattern binding borrows from `guard`.  This avoids the
181    /// borrow-checker cycle that would otherwise prevent re-assigning `*guard`
182    /// inside the same match.
183    fn poll_cache<'a>(&'a self) -> PollResult<'a> {
184        let mut guard = self.inner.cache.lock().unwrap();
185
186        // Take ownership of the state, leaving a harmless placeholder.
187        // Every arm below restores or updates `*guard` before releasing it.
188        let state = std::mem::replace(&mut *guard, CacheState::Empty);
189
190        match state {
191            // ── Refreshing (live) ────────────────────────────────────────────
192            // Put the state back, subscribe to the completion signal while the
193            // lock is still held (so we cannot miss `notify_waiters()`), then
194            // release the lock.
195            CacheState::Refreshing(weak) if weak.upgrade().is_some() => {
196                *guard = CacheState::Refreshing(weak);
197                let mut notified = Box::pin(self.inner.refresh_done.notified());
198                notified.as_mut().enable();
199                drop(guard);
200                PollResult::Wait(notified)
201            }
202
203            // ── Refreshing (cancelled) ───────────────────────────────────────
204            // The Arc that backs the Weak is gone.  Leave the cache as Empty
205            // (the placeholder set by `mem::replace`) so the caller can retry.
206            CacheState::Refreshing(_dead) => PollResult::Retry,
207
208            // ── Ready ────────────────────────────────────────────────────────
209            // Clone the cached values, restore the state, then tell the caller
210            // to validate the token without the lock.
211            CacheState::Ready(r) => {
212                let entity_tag = r.entity_tag.clone();
213                let headers = r.headers.clone();
214                *guard = CacheState::Ready(r);
215                PollResult::Validate {
216                    entity_tag,
217                    headers,
218                }
219            }
220
221            // ── Empty ────────────────────────────────────────────────────────
222            // Claim the refresh slot atomically under the lock.
223            CacheState::Empty => {
224                let token = Arc::new(());
225                *guard = CacheState::Refreshing(Arc::downgrade(&token));
226                PollResult::StartRefresh(token)
227            }
228        }
229    }
230
231    /// Return cached auth headers, refreshing exactly once when necessary.
232    ///
233    /// ## Concurrency model
234    ///
235    /// The cache cycles through three states stored in `GCSInner::cache`:
236    ///
237    /// * **`Empty`** – the first caller atomically transitions to `Refreshing`
238    ///   (under the lock) and performs the network fetch; all other concurrent
239    ///   callers see `Refreshing` and wait on `refresh_done`.
240    /// * **`Refreshing(weak)`** – callers subscribe to `refresh_done` *before*
241    ///   releasing the lock (via `Notified::enable`), guaranteeing that the
242    ///   wake-up from `notify_waiters` cannot arrive between the lock release
243    ///   and the subscription.  They then loop to re-read the updated state.
244    /// * **`Ready`** – callers clone the cached `EntityTag` + headers, drop
245    ///   the lock, and ask the credential library whether the token is still
246    ///   current.  The library returns `NotModified` (fast path) or `New`
247    ///   (token was silently refreshed); in the latter case the cache is
248    ///   overwritten.  No lock is ever held across an `await`.
249    ///
250    /// ## Cancellation safety
251    ///
252    /// The `Refreshing` variant carries a `Weak<()>` whose corresponding
253    /// strong `Arc<()>` is owned by the [`RefreshGuard`] that the active
254    /// refreshing task holds.  If that task is dropped (cancelled) at any
255    /// `.await` point, the `Arc` is released, making the `Weak` dead.
256    ///
257    /// Any waiter or new caller that subsequently observes `Refreshing(dead)`
258    /// resets the state to `Empty` itself and retries.  The `Drop` impl of
259    /// [`RefreshGuard`] calls `notify_waiters()` to wake existing waiters
260    /// without needing to acquire the mutex.
261    async fn get_or_refresh_token(&self) -> MiddlewareResult<http::HeaderMap> {
262        loop {
263            // `poll_cache` is synchronous: the MutexGuard is acquired and
264            // dropped entirely inside it, so it never appears in this async
265            // state machine.  All variants of `PollResult` are `Send`.
266            match self.poll_cache() {
267                // ── Wait ─────────────────────────────────────────────────────
268                // The contained future was already enabled while the lock was
269                // held; awaiting it is race-free.
270                PollResult::Wait(notified) => {
271                    notified.await;
272                }
273
274                // ── Retry ────────────────────────────────────────────────────
275                // Refresher was cancelled; loop to claim the slot ourselves.
276                PollResult::Retry => {}
277
278                // ── Validate ─────────────────────────────────────────────────
279                // Check whether the cached token is still current.
280                PollResult::Validate {
281                    entity_tag,
282                    headers,
283                } => {
284                    let cred = self.get_credential().await?;
285                    let mut ext = http::Extensions::new();
286                    ext.insert(entity_tag);
287
288                    return match cred
289                        .headers(ext)
290                        .await
291                        .map_err(|e| reqwest_middleware::Error::Middleware(anyhow::Error::new(e)))?
292                    {
293                        CacheableResource::NotModified => Ok(headers),
294                        CacheableResource::New { entity_tag, data } => {
295                            *self.inner.cache.lock().unwrap() = CacheState::Ready(CachedResource {
296                                entity_tag,
297                                headers: data.clone(),
298                            });
299                            Ok(data)
300                        }
301                    };
302                }
303
304                // ── StartRefresh ──────────────────────────────────────────────
305                // We claimed the slot; perform the fetch and update the cache.
306                PollResult::StartRefresh(token) => {
307                    let mut refresh_guard = RefreshGuard {
308                        inner: Arc::clone(&self.inner),
309                        _token: token,
310                        defused: false,
311                    };
312
313                    let cred = self.get_credential().await?;
314                    let fetch = cred
315                        .headers(http::Extensions::new())
316                        .await
317                        .map_err(|e| reqwest_middleware::Error::Middleware(anyhow::Error::new(e)));
318
319                    match fetch {
320                        Ok(CacheableResource::New { entity_tag, data }) => {
321                            let out = data.clone();
322                            *self.inner.cache.lock().unwrap() = CacheState::Ready(CachedResource {
323                                entity_tag,
324                                headers: data,
325                            });
326                            refresh_guard.defused = true;
327                            self.inner.refresh_done.notify_waiters();
328                            return Ok(out);
329                        }
330                        // We passed no ETag, so `NotModified` is impossible.
331                        Ok(CacheableResource::NotModified) => unreachable!(
332                            "no entity tag was provided in extensions, \
333                             so NotModified cannot be returned"
334                        ),
335                        Err(e) => {
336                            *self.inner.cache.lock().unwrap() = CacheState::Empty;
337                            refresh_guard.defused = true;
338                            self.inner.refresh_done.notify_waiters();
339                            return Err(e);
340                        }
341                    }
342                }
343            }
344        }
345    }
346}

348/// RAII guard that signals cancellation to waiters when dropped before being
349/// explicitly defused.
350///
351/// The guard owns the strong [`Arc<()>`] liveness token that is paired with
352/// the [`Weak<()>`] stored in [`CacheState::Refreshing`].  When the guard is
353/// dropped without defusing (i.e. the refreshing future is cancelled), the
354/// strong count drops to zero, making the `Weak` dead.  Any waiter or new
355/// caller that observes a dead `Weak` in `CacheState::Refreshing` resets the
356/// state to [`CacheState::Empty`] and retries — no mutex acquisition is needed
357/// inside this `Drop` impl.
358///
359/// The guard also calls [`Notify::notify_waiters`] on drop so that any tasks
360/// already suspended on `refresh_done` wake up and discover the dead token.
361struct RefreshGuard {
362    inner: Arc<GCSInner>,
363    /// Owning strong reference to the liveness token.  Dropped when the guard
364    /// is dropped, making the corresponding `Weak` in the cache dead.
365    _token: Arc<()>,
366    defused: bool,
367}
368
369impl Drop for RefreshGuard {
370    fn drop(&mut self) {
371        if !self.defused {
372            // `_token` is dropped automatically after this block, making the
373            // `Weak` in `CacheState::Refreshing` dead.  Wake all current
374            // waiters so they can re-inspect the state and detect the
375            // dead token.  New callers will also detect it on their first
376            // lock acquisition without needing to be explicitly woken.
377            self.inner.refresh_done.notify_waiters();
378        }
379    }
380}

404    /// * Keeps an internal "current" `EntityTag` (shared so tests can rotate it).

442            &self,
443            extensions: http::Extensions,
444        ) -> Result<CacheableResource<http::HeaderMap>, CredentialsError> {
445            let current = self.current_etag.lock().unwrap().clone();
446
447            // Simulate the library's ETag protocol.
448            if let Some(caller_tag) = extensions.get::<EntityTag>() {

464                http::header::AUTHORIZATION,
465                "Bearer mock-token".parse().unwrap(),
466            );
467            Ok(CacheableResource::New {
468                entity_tag: current,
469                data: map,
470            })
471        }

560    /// Cancelling the task that owns the `Refreshing` state must not leave
561    /// subsequent callers blocked forever on `refresh_done`.

564    /// refreshing future mid-flight left `CacheState::Refreshing` permanently

566    /// to `get_or_refresh_token` would hang indefinitely on `notified.await`.
567    #[tokio::test]
568    async fn test_cancellation_during_refresh_does_not_deadlock() {
569        // A 2-party barrier: the test driver + the one task that enters
570        // `headers()`.  This lets the driver confirm the refresh is in-flight
571        // (i.e. `Refreshing` is set and the mutex has been released) before
572        // aborting the task.
573        let barrier = Arc::new(Barrier::new(2));
574        let (provider, _count, _etag) = MockProvider::with_barrier(Arc::clone(&barrier));
575        let mw = GCSMiddleware::with_credentials(Credentials::from(provider));

578        let handle = tokio::spawn(async move { mw_clone.get_or_refresh_token().await });
579
580        // Wait until the spawned task is inside `headers()` (i.e. the state is
581        // `Refreshing` and the mutex has been released).
582        barrier.wait().await;
583
584        // Cancel the refreshing task – this is the scenario that previously
585        // deadlocked every subsequent caller.
586        handle.abort();

588
589        // The `RefreshGuard` Drop impl must have called `notify_waiters()`,
590        // waking any current waiters, and the dead `Weak` token in
591        // `CacheState::Refreshing` lets the next caller reset state to `Empty`.
592        // A fresh call must succeed (or at least not hang forever).
593        tokio::time::timeout(std::time::Duration::from_secs(5), mw.get_or_refresh_token())
594            .await
595            .expect("timed out – RefreshGuard did not unblock callers on cancellation")
596            .expect("token fetch failed after cancellation recovery");
597    }

612        // We have to set GOOGLE_APPLICATION_CREDENTIALS to the path of the JSON key

615        std::fs::write(&key_file, credentials).unwrap();
616
617        let prev_value = std::env::var("GOOGLE_APPLICATION_CREDENTIALS").ok();
618        std::env::set_var("GOOGLE_APPLICATION_CREDENTIALS", key_file.path());
619
620        let client = reqwest_middleware::ClientBuilder::new(Client::new())
621            .with(GCSMiddleware::default())
622            .build();
623
624        let url = "gcs://test-channel/noarch/repodata.json";
625        let response = client.get(url).send().await.unwrap();
626        assert!(response.status().is_success());

629        let response = client.get(url).send().await.unwrap();
630        assert!(response.status().is_client_error());

632        if let Some(value) = prev_value {
