1//! Backend to store credentials in the operating system's keyring
2
3use keyring::Entry;
4use std::str::FromStr;
5
6use crate::{
7    authentication_storage::{AuthenticationStorageError, StorageBackend},
8    Authentication,
9};
10
11#[derive(Clone, Debug)]
12/// A storage backend that stores credentials in the operating system's keyring
13pub struct KeyringAuthenticationStorage {
14    /// The `store_key` needs to be unique per program as it is stored
15    /// in a global dictionary in the operating system
16    pub store_key: String,
17}
18
19impl KeyringAuthenticationStorage {
20    /// Create a new authentication storage with the given store key
21    pub fn from_key(store_key: &str) -> Self {
22        Self {
23            store_key: store_key.to_string(),
24        }
25    }
26}
27
28/// An error that can occur when accessing the authentication storage
29#[derive(thiserror::Error, Debug)]
30pub enum KeyringAuthenticationStorageError {
31    // TODO: make this more fine-grained
32    /// An error occurred when accessing the authentication storage
33    #[error("Could not retrieve credentials from authentication storage: {0}")]
34    StorageError(#[from] keyring::Error),
35
36    /// An error occurred when serializing the credentials
37    #[error("Could not serialize credentials {0}")]
38    SerializeCredentialsError(#[from] serde_json::Error),
39
40    /// An error occurred when parsing the credentials
41    #[error("Could not parse credentials stored for {host}")]
42    ParseCredentialsError {
43        /// The host for which the credentials could not be parsed
44        host: String,
45    },
46}
47
48impl Default for KeyringAuthenticationStorage {
49    fn default() -> Self {
50        Self::from_key("rattler")
51    }
52}
53
54impl StorageBackend for KeyringAuthenticationStorage {
55    fn store(
56        &self,
57        host: &str,
58        authentication: &Authentication,
59    ) -> Result<(), AuthenticationStorageError> {
60        let password = serde_json::to_string(authentication)
61            .map_err(KeyringAuthenticationStorageError::from)?;
62        let entry =
63            Entry::new(&self.store_key, host).map_err(KeyringAuthenticationStorageError::from)?;
64
65        entry
66            .set_password(&password)
67            .map_err(KeyringAuthenticationStorageError::from)?;
68
69        Ok(())
70    }
71
72    fn get(&self, host: &str) -> Result<Option<Authentication>, AuthenticationStorageError> {
73        let entry =
74            Entry::new(&self.store_key, host).map_err(KeyringAuthenticationStorageError::from)?;
75        let password = entry.get_password();
76
77        let p_string = match password {
78            Ok(password) => password,
79            Err(keyring::Error::NoEntry) => return Ok(None),
80            Err(e) => return Err(KeyringAuthenticationStorageError::from(e))?,
81        };
82
83        match Authentication::from_str(&p_string) {
84            Ok(auth) => Ok(Some(auth)),
85            Err(err) => {
86                tracing::warn!("Error parsing credentials for {}: {:?}", host, err);
87                Err(KeyringAuthenticationStorageError::ParseCredentialsError {
88                    host: host.to_string(),
89                }
90                .into())
91            }
92        }
93    }
94
95    fn delete(&self, host: &str) -> Result<(), AuthenticationStorageError> {
96        let entry =
97            Entry::new(&self.store_key, host).map_err(KeyringAuthenticationStorageError::from)?;
98        entry
99            .delete_credential()
100            .map_err(KeyringAuthenticationStorageError::from)?;
101
102        Ok(())
103    }
104}
