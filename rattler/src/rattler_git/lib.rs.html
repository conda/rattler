24/// A URL reference to a Git repository.
25#[derive(Debug, Clone, PartialEq, PartialOrd, Eq, Hash, Ord)]
26pub struct GitUrl {
27    /// The URL of the Git repository, with any query parameters, fragments, and leading `git+`
28    /// removed.
29    repository: Url,
30    /// The reference to the commit to use, which could be a branch, tag or revision.
31    reference: GitReference,
32    /// The precise commit to use, if known.
33    precise: Option<GitSha>,
34}
35
36impl GitUrl {
37    /// Create a new [`GitUrl`] from a repository URL and a reference.
38    pub fn from_reference(repository: Url, reference: GitReference) -> Self {
39        let precise = reference.as_sha();
40        Self {
41            repository,
42            reference,
43            precise,
44        }
45    }
46
47    /// Create a new [`GitUrl`] from a repository URL and a precise commit.
48    pub fn from_commit(repository: Url, reference: GitReference, precise: GitSha) -> Self {
49        Self {
50            repository,
51            reference,
52            precise: Some(precise),
53        }
54    }
55
56    /// Set the precise [`GitSha`] to use for this Git URL.
57    #[must_use]
58    pub fn with_precise(mut self, precise: GitSha) -> Self {
59        self.precise = Some(precise);
60        self
61    }
62
63    /// Set the [`GitReference`] to use for this Git URL.
64    #[must_use]
65    pub fn with_reference(mut self, reference: GitReference) -> Self {
66        self.reference = reference;
67        self
68    }
69
70    /// Return the [`Url`] of the Git repository.
71    pub fn repository(&self) -> &Url {
72        &self.repository
73    }
74
75    /// Return the reference to the commit to use, which could be a branch, tag or revision.
76    pub fn reference(&self) -> &GitReference {
77        &self.reference
78    }
79
80    /// Return the precise commit, if known.
81    pub fn precise(&self) -> Option<GitSha> {
82        self.precise
83    }
84}

86impl TryFrom<Url> for GitUrl {
87    type Error = OidParseError;
88
89    /// Initialize a [`GitUrl`] source from a URL.
90    fn try_from(url: Url) -> Result<Self, Self::Error> {
91        // remove the `git+` prefix if it exists
92        let mut url = if url.scheme().starts_with("git+") {
93            let url_as_str = &url.as_str()[4..];
94            Url::parse(url_as_str).expect("url should be valid")
95        } else {
96            url
97        };
98
99        // Remove any query parameters and fragments.
100        url.set_fragment(None);
101
102        // If the URL ends with a reference, like `https://git.example.com/MyProject.git@v1.0`,
103        // extract it.
104        // The URL can also be enriched with the reference type, like `https://git.example.com/MyProject.git@v1.0?rev_type=tag`.
105        // so we can extract the reference and the reference type.
106        let mut reference = GitReference::DefaultBranch;
107        if let Some((prefix, suffix)) = url
108            .path()
109            .rsplit_once('@')
110            .map(|(prefix, suffix)| (prefix.to_string(), suffix.to_string()))
111        {
112            if let Some((_, rev_type)) = url
113                .query_pairs()
114                .find(|(key, _)| key == GIT_URL_QUERY_REV_TYPE)
115            {
116                match rev_type.into_owned().as_str() {
117                    "tag" => reference = GitReference::Tag(suffix),
118                    "branch" => reference = GitReference::Branch(suffix),
119                    "rev" => reference = GitReference::from_rev(suffix),
120                    // a custom reference type is not supported
121                    _ => return Err(OidParseError::UrlParse(url.to_string())),
122                }
123            } else {
124                // try to guess it
125                reference = GitReference::from_rev(suffix);
126            }
127
128            url.set_path(&prefix);
129        }
130        url.set_query(None);
131
132        Ok(Self::from_reference(url, reference))
133    }
134}

136impl From<GitUrl> for Url {
137    fn from(git: GitUrl) -> Self {
138        let mut url = git.repository;
139
140        // If we have a precise commit, add `@` and the commit hash to the URL.
141        if let Some(precise) = git.precise {
142            url.set_path(&format!("{}@{}", url.path(), precise));
143        } else {
144            // Otherwise, add the branch or tag name.
145            match git.reference {
146                GitReference::Branch(rev)
147                | GitReference::Tag(rev)
148                | GitReference::ShortCommit(rev)
149                | GitReference::BranchOrTag(rev)
150                | GitReference::NamedRef(rev)
151                | GitReference::FullCommit(rev)
152                | GitReference::BranchOrTagOrCommit(rev) => {
153                    url.set_path(&format!("{}@{}", url.path(), rev));
154                }
155                GitReference::DefaultBranch => {}
156            }
157        }
158
159        url
160    }
161}

177#[derive(Debug, thiserror::Error)]
178pub enum GitError {
179    #[error(transparent)]
180    GitBinary(#[from] GitBinaryError),
181
182    #[error(transparent)]
183    Io(#[from] std::io::Error),
184
185    #[error(transparent)]
186    FromUtf8(#[from] std::string::FromUtf8Error),
187
188    #[error(transparent)]
189    OidParse(#[from] OidParseError),
190
191    #[error("failed to fetch {0}: {1}")]
192    Fetch(String, String),
193
194    #[error(transparent)]
195    UrlParse(#[from] ::url::ParseError),
196
197    #[error("could not transform original url {0} into a git url: {1}")]
198    GitUrlFormat(String, String),
199
200    #[error(transparent)]
201    ReqwestMiddleware(#[from] reqwest_middleware::Error),
202
203    #[error(transparent)]
204    Reqwest(#[from] reqwest::Error),
205
206    #[error(transparent)]
207    Join(#[from] tokio::task::JoinError),
208
209    #[error("corrupted or invalid git repository at {0}")]
210    InvalidRepository(std::path::PathBuf),
211}
