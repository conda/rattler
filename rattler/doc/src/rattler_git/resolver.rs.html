32/// [`GitResolver`] is responsible for managing and resolving Git repository references.
33/// It maintains a mapping between [`RepositoryReference`] (e.g., a Git URL with branch or tag) and the precise commit [`GitSha`]
34/// and it also provides methods to fetch Git repositories from given [`GitUrl`] and cache it.
35#[derive(Default, Clone)]
36pub struct GitResolver(Arc<DashMap<RepositoryReference, GitSha>>);
37
38impl GitResolver {
39    /// Inserts a new [`GitSha`] for the given [`RepositoryReference`].
40    pub fn insert(&self, reference: RepositoryReference, sha: GitSha) {
41        self.0.insert(reference, sha);
42    }
43
44    /// Returns the [`GitSha`] for the given [`RepositoryReference`], if it exists.
45    fn get(&self, reference: &RepositoryReference) -> Option<Ref<'_, RepositoryReference, GitSha>> {
46        self.0.get(reference)
47    }
48
49    /// Fetch a remote Git repository.
50    pub async fn fetch(
51        &self,
52        url: GitUrl,
53        client: ClientWithMiddleware,
54        cache: PathBuf,
55        reporter: Option<Arc<dyn Reporter>>,
56    ) -> Result<Fetch, GitError> {
57        debug!("Fetching source distribution from Git: {url}");
58
59        let reference = RepositoryReference::from(&url);
60
61        // If we know the precise commit already, reuse it, to ensure that all fetches within a
62        // single process are consistent.
63        let url = {
64            if let Some(precise) = self.get(&reference) {
65                url.with_precise(*precise)
66            } else {
67                url
68            }
69        };
70
71        // Avoid races between different processes, too.
72        let lock_dir = cache.join("locks");
73        let repository_url = RepositoryUrl::new(url.repository());
74
75        let write_guard_path = lock_dir.join(cache_digest(&repository_url));
76        let guard = AsyncPrefixGuard::new(&write_guard_path).await?;
77        let mut write_guard = guard.write().await?;
78
79        // Update the prefix to indicate that we are installing it.
80        write_guard.begin().await?;
81
82        // Fetch the Git repository.
83        let source = GitSource::new(url.clone(), client, cache);
84        let source = if let Some(reporter) = reporter {
85            source.with_reporter(reporter)
86        } else {
87            source
88        };
89
90        let fetch = tokio::task::spawn_blocking(move || source.fetch())
91            .await?
92            .inspect_err(|err| tracing::error!("Error fetching Git repository: {err}"))?;
93
94        // Insert the resolved URL into the in-memory cache. This ensures that subsequent fetches
95        // resolve to the same precise commit.
96        self.insert(reference, fetch.commit());
97
98        write_guard.finish().await?;
99
100        tracing::debug!("Fetched source distribution from Git: {url}");
101        Ok(fetch)
102    }
103
104    /// Given a remote source distribution, return a precise variant, if possible.
105    ///
106    /// For example, given a Git dependency with a reference to a branch or tag, return a URL
107    /// with a precise reference to the current commit of that branch or tag.
108    ///
109    /// This method takes into account various normalizations that are independent from the Git
110    /// layer. For example: removing `#subdirectory=pkg_dir`-like fragments, and removing `git+`
111    /// prefix kinds.
112    ///
113    /// This method will only return precise URLs for URLs that have already been resolved via
114    /// `resolve_precise`, and will return `None` for URLs that have not been resolved _or_
115    /// already have a precise reference.
116    pub fn precise(&self, url: GitUrl) -> Option<GitUrl> {
117        let reference = RepositoryReference::from(&url);
118        let precise = self.get(&reference)?;
119        Some(url.with_precise(*precise))
120    }
121
122    /// Returns `true` if the two Git URLs refer to the same precise commit.
123    pub fn same_ref(&self, a: &GitUrl, b: &GitUrl) -> bool {
124        // Convert `a` to a repository URL.
125        let a_ref = RepositoryReference::from(a);
126
127        // Convert `b` to a repository URL.
128        let b_ref = RepositoryReference::from(b);
129
130        // The URLs must refer to the same repository.
131        if a_ref.url != b_ref.url {
132            return false;
133        }
134
135        // If the URLs have the same tag, they refer to the same commit.
136        if a_ref.reference == b_ref.reference {
137            return true;
138        }
139
140        // Otherwise, the URLs must resolve to the same precise commit.
141        let Some(a_precise) = a.precise().or_else(|| self.get(&a_ref).map(|sha| *sha)) else {
142            return false;
143        };
144
145        let Some(b_precise) = b.precise().or_else(|| self.get(&b_ref).map(|sha| *sha)) else {
146            return false;
147        };
148
149        a_precise == b_precise
150    }
151}

153#[derive(Debug, Clone, PartialEq, Eq, Hash)]
154pub struct ResolvedRepositoryReference {
155    /// An abstract reference to a Git repository, including the URL and the commit (e.g., a branch,
156    /// tag, or revision).
157    pub reference: RepositoryReference,
158    /// The precise commit SHA of the reference.
159    pub sha: GitSha,
160}
161
162#[derive(Debug, Clone, PartialEq, Eq, Hash, Serialize)]
163pub struct RepositoryReference {
164    /// The URL of the Git repository, with any query parameters and fragments removed.
165    pub url: RepositoryUrl,
166    /// The reference to the commit to use, which could be a branch, tag, or revision.
167    #[serde(skip_serializing_if = "GitReference::is_default")]
168    pub reference: GitReference,
169}
170
171impl From<&GitUrl> for RepositoryReference {
172    fn from(git: &GitUrl) -> Self {
173        Self {
174            url: RepositoryUrl::new(git.repository()),
175            reference: git.reference().clone(),
176        }
177    }
178}
