9/// A builder to construct an [`IndicatifReporter`].
10#[derive(Clone)]
11pub struct IndicatifReporterBuilder {
12    multi_progress: Option<MultiProgress>,
13    clear_when_done: bool,
14    prefix: Cow<'static, str>,
15}
16
17impl Default for IndicatifReporterBuilder {
18    fn default() -> Self {
19        Self::new()
20    }
21}

23impl IndicatifReporterBuilder {
24    /// Create a new builder.
25    pub fn new() -> Self {
26        Self {
27            multi_progress: None,
28            clear_when_done: false,
29            prefix: Cow::Borrowed(""),
30        }
31    }
32
33    /// Sets the [`MultiProgress`] to use for the progress bars.
34    pub fn with_multi_progress(self, multi_progress: MultiProgress) -> Self {
35        Self {
36            multi_progress: Some(multi_progress),
37            ..self
38        }
39    }
40
41    /// Sets whether the progress bars are cleared when done.
42    pub fn clear_when_done(self, clear_when_done: bool) -> Self {
43        Self {
44            clear_when_done,
45            ..self
46        }
47    }
48
49    /// Sets the prefix for all progress bars.
50    pub fn with_prefix(self, prefix: impl Into<Cow<'static, str>>) -> Self {
51        Self {
52            prefix: prefix.into(),
53            ..self
54        }
55    }
56
57    /// Finish building [`IndicatifReporter`].
58    pub fn finish(self) -> IndicatifReporter {
59        let multi_progress = self.multi_progress.unwrap_or_default();
60
61        IndicatifReporter {
62            inner: Arc::new(Mutex::new(IndicatifReporterInner {
63                multi_progress,
64                downloads: HashMap::new(),
65                clear_when_done: self.clear_when_done,
66                prefix: self.prefix,
67            })),
68        }
69    }
70}

104impl DownloadReporter for IndicatifReporter {
105    fn on_download_start(&self, url: &Url) -> usize {
106        let mut inner = self.inner.lock();
107        let index = inner.downloads.len();
108
109        let pb = inner.multi_progress.add(ProgressBar::new(0));
110
111        pb.set_style(
112            ProgressStyle::with_template(&format!(
113                "{prefix}{{spinner:.green}} {{msg:.dim}}",
114                prefix = if inner.prefix.is_empty() {
115                    String::new()
116                } else {
117                    format!("{} ", inner.prefix)
118                }
119            ))
120            .expect("failed to create progress style")
121            .tick_chars("⠁⠂⠄⡀⢀⠠⠐⠈ "),
122        );
123
124        pb.set_message(format!("Fetching {}", simplify_url(url)));
125        pb.enable_steady_tick(Duration::from_millis(100));
126
127        inner.downloads.insert(index, pb);
128        index
129    }
130
131    fn on_download_progress(
132        &self,
133        _url: &Url,
134        index: usize,
135        bytes_downloaded: usize,
136        total_bytes: Option<usize>,
137    ) {
138        let inner = self.inner.lock();
139        if let Some(pb) = inner.downloads.get(&index) {
140            if let Some(total) = total_bytes {
141                pb.set_length(total as u64);
142                pb.set_position(bytes_downloaded as u64);
143                pb.set_style(
144                    ProgressStyle::with_template(&format!(
145                        "{prefix}{{spinner:.green}} [{{bar:20.cyan/blue}}] {{bytes}}/{{total_bytes}} {{msg:.dim}}",
146                        prefix = if inner.prefix.is_empty() {
147                            String::new()
148                        } else {
149                            format!("{} ", inner.prefix)
150                        }
151                    ))
152                    .expect("failed to create progress style")
153                    .progress_chars("━━╾─")
154                    .tick_chars("⠁⠂⠄⡀⢀⠠⠐⠈ "),
155                );
156            } else {
157                pb.set_position(bytes_downloaded as u64);
158            }
159        }
160    }
161
162    fn on_download_complete(&self, _url: &Url, index: usize) {
163        let mut inner = self.inner.lock();
164        if let Some(pb) = inner.downloads.remove(&index) {
165            if inner.clear_when_done {
166                pb.finish_and_clear();
167            } else {
168                pb.finish_with_message("Done");
169            }
170        }
171    }
172}

174fn simplify_url(url: &Url) -> String {
175    if let Some(domain) = url.domain() {
176        let path = url.path();
177        if path.len() > 50 {
178            let segments: Vec<&str> = path.split('/').collect();
179            if let Some(last) = segments.last() {
180                return format!("{domain}/.../{last}");
181            }
182        }
183        format!("{domain}{path}")
184    } else {
185        url.to_string()
186    }
