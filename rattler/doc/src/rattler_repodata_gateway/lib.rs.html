3//! `rattler_repodata_gateway` is a crate that provides functionality to
4//! interact with Conda repodata. It currently provides functionality to
5//! download and cache `repodata.json` files through the
6//! [`fetch::fetch_repo_data`] function.
7//!
8//! In the future this crate will also provide more high-level functionality to
9//! query information about specific packages from different sources.
10//!
11//! # Install
12//! Add the following to your *Cargo.toml*:
13//!
14//! ```toml
15//! [dependencies]
16//! rattler_repodata_gateway = "0.2.0"
17//! ```
18//!
19//! or run
20//!
21//! ```bash
22//! cargo add rattler_repodata_gateway
23//! ```
24//!
25//! # Examples
26//! Below is a basic example that shows how to retrieve and cache the repodata
27//! for a conda channel using the [`fetch::fetch_repo_data`] function:
28//!
29//! ```no_run
30//! use std::{default::Default, path::PathBuf};
31//! use reqwest::Client;
32//! use reqwest_middleware::ClientWithMiddleware;
33//! use url::Url;
34//! use rattler_repodata_gateway::fetch;
35//! use rattler_networking::LazyClient;
36//!
37//! #[tokio::main]
38//! async fn main() {
39//!     let repodata_url = Url::parse("https://conda.anaconda.org/conda-forge/osx-64/").unwrap();
40//!     let cache = PathBuf::from("./cache");
41//!
42//!     let result = fetch::fetch_repo_data(
43//!         repodata_url,
44//!         LazyClient::default(),
45//!         cache,
46//!         fetch::FetchRepoDataOptions { ..Default::default() },
47//!         None,
48//!     ).await;
49//!
50//!     let result = match result {
51//!         Err(err) => {
52//!             panic!("{:?}", err);
53//!         }
54//!         Ok(result) => result
55//!     };
56//!
57//!     println!("{:?}", result.cache_state);
58//!     println!("{:?}", result.cache_result);
59//!     println!("{:?}", result.lock_file);
60//!     println!("{:?}", result.repo_data_json_path);
61//! }
62//! ```
