16/// An enum to direct how to update a generated file.
17#[derive(Clone, Copy, PartialEq, Eq)]
18pub enum Mode {
19    /// Overwrite the file on disk if it changed.
20    Overwrite,
21
22    /// Verify that the file on disk contains the same content as the generated content. Returns an
23    /// error if that is not the case.
24    Verify,
25}
26
27/// A helper to update file on disk if it has changed or verify the contents of the file on disk
28/// depending on the `mode`.
29fn update(path: &Path, contents: &str, mode: Mode) -> anyhow::Result<()> {
30    let old_contents = fs::read_to_string(path)?;
31    let old_contents = old_contents.replace("\r\n", "\n");
32    let contents = contents.replace("\r\n", "\n");
33    if old_contents == contents {
34        return Ok(());
35    }
36
37    if mode == Mode::Verify {
38        let changes = difference::Changeset::new(&old_contents, &contents, "\n");
39        anyhow::bail!("==================================================\n`{}` is not up-to-date\n==================================================\n{}", path.display(), changes,);
40    }
41    eprintln!("updating {}", path.display());
42    fs::write(path, contents)?;
43    Ok(())
44}

46/// Re-formats the given input with `rustfmt`.
47fn reformat(text: impl std::fmt::Display) -> anyhow::Result<String> {
48    let mut rustfmt = Command::new("rustfmt")
49        //.arg("--config-path")
50        //.arg(project_root().join("rustfmt.toml"))
51        .stdin(Stdio::piped())
52        .stdout(Stdio::piped())
53        .spawn()?;
54    write!(rustfmt.stdin.take().unwrap(), "{text}")?;
55    let output = rustfmt.wait_with_output()?;
56    let stdout = String::from_utf8(output.stdout)?;
57    let preamble = "Generated file, do not edit by hand, see `crate/tools/src`";
58    Ok(format!("//! {preamble}\n\n{stdout}"))
59}
