43/// Different platforms generate different bindings for enum representations. To not have to
44/// generate completely different files based on the platform we patch these enum representations
45/// inline.
46fn patch_enum_representation<'a>(input: &'a str, enum_name: &str) -> Cow<'a, str> {
47    let replacement = format!(
48        concat!(
49            "\ncfg_if::cfg_if! {{\n",
50            "    if #[cfg(all(target_os = \"windows\", target_env = \"msvc\"))] {{\n",
51            "        pub type {} = libc::c_int;\n",
52            "    }} else {{\n",
53            "        pub type {} = libc::c_uint;\n",
54            "    }}\n",
55            "}}"
56        ),
57        enum_name, enum_name
58    );
59
60    let c_int_repr = format!("\npub type {enum_name} = libc::c_int;");
61    let c_uint_repr = format!("\npub type {enum_name} = libc::c_uint;");
62
63    if let Some((start, str)) = input
64        .match_indices(&c_int_repr)
65        .next()
66        .or_else(|| input.match_indices(&c_uint_repr).next())
67    {
68        Cow::Owned(format!(
69            "{}{replacement}{}",
70            &input[..start],
71            &input[start + str.len()..]
72        ))
73    } else {
74        Cow::Borrowed(input)
75    }
76}

78/// Generate or verify the libsolv bindings.
79pub fn generate(mode: Mode) -> anyhow::Result<()> {
80    let libsolv_path = project_root().join("crates/rattler_libsolv_c/libsolv");
81
82    // Normally the `solvversion.h` is generated from the `solverversion.h.in` by CMake when
83    // building libsolv. However, for the bindings we don't need that much information from that
84    // file. So to be able to generate proper bindings we use a drop-in-replacement for this file
85    // that contains just the macros that are needed for the bindings.
86    //
87    // The behavior of this file might change when we update libsolv so its important to check this
88    // with every upgrade.
89    let temp_include_dir = TempDir::with_prefix("libsolv")?;
90    std::fs::write(
91        temp_include_dir.path().join("solvversion.h"),
92        r#"#ifndef LIBSOLV_SOLVVERSION_H
93#define LIBSOLV_SOLVVERSION_H
94
95#define SOLV_API
96
97#define LIBSOLV_FEATURE_MULTI_SEMANTICS
98#define LIBSOLV_FEATURE_CONDA
99
100#define LIBSOLVEXT_FEATURE_ZLIB_COMPRESSION
101
102#endif
103"#,
104    )?;
105
106    // Define the contents of the bindings and how they are generated
107    let bindings = bindgen::Builder::default()
108        .rust_target("1.88.0".parse().unwrap())
109        .clang_arg(format!("-I{}", temp_include_dir.path().display()))
110        .clang_arg(format!("-I{}", libsolv_path.join("src").display()))
111        .ctypes_prefix("libc")
112        .header(libsolv_path.join("src/solver.h").to_str().unwrap())
113        .header(libsolv_path.join("src/solverdebug.h").to_str().unwrap())
114        .header(libsolv_path.join("src/queue.h").to_str().unwrap())
115        .header(libsolv_path.join("src/pool.h").to_str().unwrap())
116        .header(libsolv_path.join("src/selection.h").to_str().unwrap())
117        .header(libsolv_path.join("src/knownid.h").to_str().unwrap())
118        .header(libsolv_path.join("src/conda.h").to_str().unwrap())
119        .header(libsolv_path.join("src/repo_solv.h").to_str().unwrap())
120        .header(libsolv_path.join("src/repo_write.h").to_str().unwrap())
121        .header(libsolv_path.join("ext/repo_conda.h").to_str().unwrap())
122        .allowlist_type("Id")
123        .allowlist_var(format!("({}).*", ALLOWED_VAR_PREFIX.join("|")))
124        .allowlist_function(format!("({}).*", ALLOWED_FUNC_PREFIX.join("|")))
125        .blocklist_type(DISALLOWED_TYPES.join("|"))
126        .disable_header_comment()
127        .layout_tests(false)
128        .generate()?;
129
130    // Generate the actual bindings and format them
131    let mut libsolv_bindings = Vec::new();
132    bindings.write(Box::new(&mut libsolv_bindings))?;
133
134    // Add a preamble to the bindings to ensure clippy also passes.
135    let libsolv_bindings = reformat(format!("#![allow(non_upper_case_globals, non_camel_case_types, non_snake_case, dead_code, clippy::upper_case_acronyms)]\n\npub use libc::FILE;\n\n{}", String::from_utf8(libsolv_bindings).unwrap()))?;
136
137    // Patch out some platform weirdness
138    let libsolv_bindings = patch_enum_representation(&libsolv_bindings, "SolverRuleinfo");
139
140    // Write (or check) the bindings
141    update(
142        &project_root().join("crates/rattler_libsolv_c/src/lib.rs"),
143        &libsolv_bindings,
144        mode,
145    )
146}
