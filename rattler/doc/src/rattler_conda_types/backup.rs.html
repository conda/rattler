1//! Utilities for excluding directories from backups.
2//!
3//! This module provides cross-platform utilities to mark directories as excluded
4//! from backups. It supports:
5//!
6//! - **CACHEDIR.TAG**: A standard file recognized by many backup tools (borg, restic,
7//!   duplicity, tar --exclude-caches, etc.). See <https://bford.info/cachedir/>.
8//!
9//! - **macOS Time Machine**: Uses the `NSURLIsExcludedFromBackupKey` attribute to
10//!   exclude directories from Time Machine backups.
11
12use std::io::Write;
13use std::path::Path;
14
15/// The standard CACHEDIR.TAG header that identifies a cache directory.
16/// See <https://bford.info/cachedir/> for the specification.
17const CACHEDIR_TAG: &str = "Signature: 8a477f597d28d172789f06886806bc55
18# This file is a cache directory tag created by rattler.
19# For information about cache directory tags, see:
20#\thttps://bford.info/cachedir/
21";
22
23/// Creates a `CACHEDIR.TAG` file in the specified directory.
24///
25/// This file signals to backup tools that this directory contains
26/// cache/derived data that doesn't need to be backed up.
27/// See <https://bford.info/cachedir/> for the specification.
28///
29/// If the file already exists, this function does nothing.
30pub fn create_cachedir_tag(path: &Path) -> Result<(), std::io::Error> {
31    let tag_path = path.join("CACHEDIR.TAG");
32    if tag_path.exists() {
33        return Ok(());
34    }
35
36    let mut file = fs_err::File::create(&tag_path)?;
37    file.write_all(CACHEDIR_TAG.as_bytes())?;
38    Ok(())
39}

41/// Marks a directory as excluded from Time Machine backups on macOS.
42///
43/// This is recommended to prevent derived/temporary files from bloating backups.
44/// Based on the approach used by Cargo: <https://github.com/rust-lang/cargo/pull/7192>
45///
46/// Errors are silently ignored since backup exclusion is an optional feature
47/// and failure shouldn't prevent the application from working.
48#[cfg(target_os = "macos")]
49pub fn exclude_from_time_machine(path: &Path) {
50    use core_foundation::base::TCFType;
51    use core_foundation::{number, string, url};
52    use std::ptr;
53
54    // For compatibility with 10.7 a string is used instead of global kCFURLIsExcludedFromBackupKey
55    let is_excluded_key: Result<string::CFString, _> = "NSURLIsExcludedFromBackupKey".parse();
56    let path = url::CFURL::from_path(path, false);
57    if let (Some(path), Ok(is_excluded_key)) = (path, is_excluded_key) {
58        unsafe {
59            url::CFURLSetResourcePropertyForKey(
60                path.as_concrete_TypeRef(),
61                is_excluded_key.as_concrete_TypeRef(),
62                number::kCFBooleanTrue.cast(),
63                ptr::null_mut(),
64            );
65        }
66    }
67    // Errors are ignored, since it's an optional feature and failure
68    // doesn't prevent the application from working
69}
70
71/// Excludes a directory from backups using all available methods.
72///
73/// This function:
74/// 1. Creates a `CACHEDIR.TAG` file (cross-platform)
75/// 2. Marks the directory as excluded from Time Machine on macOS
76///
77/// The `CACHEDIR.TAG` creation propagates errors, while Time Machine exclusion
78/// silently ignores errors (since it's platform-specific and optional).
79pub fn exclude_from_backups(path: &Path) -> Result<(), std::io::Error> {
80    create_cachedir_tag(path)?;
81    #[cfg(target_os = "macos")]
82    exclude_from_time_machine(path);
83    Ok(())
84}
