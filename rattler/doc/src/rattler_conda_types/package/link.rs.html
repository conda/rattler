8/// Describes python noarch specific entry points
9#[derive(Serialize, Clone, Debug, Deserialize)]
10pub struct PythonEntryPoints {
11    /// A list of commands that should execute certain python commands.
12    #[serde(default, skip_serializing_if = "Vec::is_empty")]
13    pub entry_points: Vec<EntryPoint>,
14}
15
16/// Links for specific types of noarch packages.
17#[derive(Serialize, Clone, Debug, Deserialize)]
18#[serde(tag = "type", rename_all = "lowercase")]
19pub enum NoArchLinks {
20    /// Python noarch specific entry points.
21    Python(PythonEntryPoints),
22
23    /// Generic variant (doesn't have any special entry points)
24    Generic,
25}
26
27/// A representation of the `link.json` file found in noarch package archives.
28///
29/// The `link.json` file contains information about entrypoints that need to be installed for the package.
30#[sorted]
31#[derive(Clone, Debug, Serialize, Deserialize)]
32pub struct LinkJson {
33    /// Links for specific noarch packages
34    pub noarch: NoArchLinks,
35
36    /// The version of the package metadata file
37    pub package_metadata_version: u64,
38}
39
40impl PackageFile for LinkJson {
41    fn package_path() -> &'static Path {
42        Path::new("info/link.json")
43    }
44
45    fn from_str(str: &str) -> Result<Self, std::io::Error> {
46        serde_json::from_str(str).map_err(Into::into)
47    }
48
49    fn from_slice(slice: &[u8]) -> Result<Self, std::io::Error> {
50        serde_json::from_slice(slice).map_err(Into::into)
51    }
52}
