3use std::ptr::NonNull;

5/// Represents an owned byte slice that was allocated using [`libc::malloc`] and is deallocated upon
6/// drop using [`libc::free`].
7pub struct LibcByteSlice {
8    ptr: NonNull<libc::c_void>,
9    len: usize,
10}
11
12// We can safely implement `Send` because LibcByteSlice is immutable
13unsafe impl Send for LibcByteSlice {}
14
15// We can safely implement `Send` because LibcByteSlice is immutable
16unsafe impl Sync for LibcByteSlice {}
17
18impl LibcByteSlice {
19    /// Constructs a `LibcByteSlice` from its raw parts
20    ///
21    /// # Safety
22    ///
23    /// `ptr` should have been allocated using [`libc::malloc`] and `len` should be the size
24    /// in bytes of the allocated chunk of memory
25    pub unsafe fn from_raw_parts(ptr: NonNull<libc::c_void>, len: usize) -> LibcByteSlice {
26        LibcByteSlice { ptr, len }
27    }
28
29    pub(super) fn as_ptr(&self) -> *mut libc::c_void {
30        self.ptr.as_ptr()
31    }
32
33    pub(super) fn len(&self) -> usize {
34        self.len
35    }
36}
37
38impl Drop for LibcByteSlice {
39    fn drop(&mut self) {
40        unsafe { libc::free(self.ptr.as_ptr().cast()) }
41    }
42}
