8/// Wrapper for libsolv solver, which is used to drive dependency resolution
9///
10/// The wrapper functions as an owned pointer, guaranteed to be non-null and freed
11/// when the Solver is dropped
12pub struct Solver<'pool>(NonNull<ffi::Solver>, PhantomData<&'pool Pool>);
13
14impl Drop for Solver<'_> {
15    fn drop(&mut self) {
16        unsafe { ffi::solver_free(self.0.as_mut()) }
17    }
18}
19
20impl Solver<'_> {
21    /// Constructs a new Solver from the provided libsolv pointer. It is the responsibility of the
22    /// caller to ensure the pointer is actually valid.
23    pub(super) unsafe fn new(_pool: &Pool, ptr: NonNull<ffi::Solver>) -> Solver<'_> {
24        Solver(ptr, PhantomData)
25    }
26
27    /// Returns a raw pointer to the wrapped `ffi::Solver`, to be used for calling ffi functions
28    /// that require access to the pool (and for nothing else)
29    fn raw_ptr(&self) -> *mut ffi::Solver {
30        self.0.as_ptr()
31    }
32
33    /// Returns the amount of problems that are yet to be solved
34    fn problem_count(&self) -> u32 {
35        unsafe { ffi::solver_problem_count(self.raw_ptr()) }
36    }
37
38    /// Returns a user-friendly representation of a problem
39    ///
40    /// Safety: the caller must ensure the id is valid
41    unsafe fn problem2str(&self, id: ffi::Id) -> &CStr {
42        let problem = ffi::solver_problem2str(self.raw_ptr(), id);
43        CStr::from_ptr(problem)
44    }
45
46    /// Creates a string for each 'problem' that the solver still has which it encountered while
47    /// solving the matchspecs. Use this function to print the existing problems to string.
48    fn solver_problems(&self) -> Vec<String> {
49        let mut output = Vec::new();
50
51        let count = self.problem_count();
52        for i in 1..=count {
53            // Safe because the id valid (between [1, count])
54            let problem = unsafe { self.problem2str(i as ffi::Id) };
55
56            output.push(
57                problem
58                    .to_str()
59                    .expect("string is invalid UTF8")
60                    .to_string(),
61            );
62        }
63        output
64    }
65
66    pub fn all_solver_problems(&self) -> Vec<SolveProblem> {
67        let mut problems = Vec::new();
68        let mut problem_rules = Queue::<ffi::Id>::default();
69
70        let count = self.problem_count();
71        for i in 1..=count {
72            unsafe {
73                ffi::solver_findallproblemrules(
74                    self.raw_ptr(),
75                    i.try_into().unwrap(),
76                    problem_rules.raw_ptr(),
77                );
78            };
79            for r in problem_rules.id_iter() {
80                if r != 0 {
81                    let mut source_id = 0;
82                    let mut target_id = 0;
83                    let mut dep_id = 0;
84
85                    let problem_type = unsafe {
86                        ffi::solver_ruleinfo(
87                            self.raw_ptr(),
88                            r,
89                            &mut source_id,
90                            &mut target_id,
91                            &mut dep_id,
92                        )
93                    };
94
95                    let pool: *mut ffi::Pool = unsafe { (*self.0.as_ptr()).pool.cast() };
96
97                    let nsolvables = unsafe { (*pool).nsolvables };
98
99                    let target = if target_id < 0 || target_id >= nsolvables {
100                        None
101                    } else {
102                        Some(SolvableId(target_id))
103                    };
104
105                    let source = if source_id < 0 || source_id >= nsolvables {
106                        None
107                    } else {
108                        Some(SolvableId(target_id))
109                    };
110
111                    let dep = if dep_id == 0 {
112                        None
113                    } else {
114                        let dep = unsafe { ffi::pool_dep2str(pool, dep_id) };
115                        let dep = unsafe { CStr::from_ptr(dep) };
116                        let dep = dep.to_str().expect("Invalid UTF8 value").to_string();
117                        Some(dep)
118                    };
119
120                    problems.push(SolveProblem::from_raw(problem_type, dep, source, target));
121                }
122            }
123        }
124        problems
125    }
126
127    /// Sets a solver flag
128    pub fn set_flag(&self, flag: SolverFlag, value: bool) {
129        unsafe { ffi::solver_set_flag(self.raw_ptr(), flag.inner(), i32::from(value)) };
130    }
131
132    /// Solves all the problems in the `queue` and returns a transaction from the found solution.
133    /// Returns an error if problems remain unsolved.
134    pub fn solve(&mut self, queue: &mut SolveGoal) -> Result<Transaction<'_>, Vec<String>> {
135        let result = unsafe {
136            // Run the solve method
137            ffi::solver_solve(self.raw_ptr(), queue.raw_ptr());
138            // If there are no problems left then the solver is done
139            ffi::solver_problem_count(self.raw_ptr()) == 0
140        };
141        if result {
142            let transaction =
143                NonNull::new(unsafe { ffi::solver_create_transaction(self.raw_ptr()) })
144                    .expect("solver_create_transaction returned a nullptr");
145
146            // Safe because we know the `transaction` ptr is valid
147            Ok(unsafe { Transaction::new(self, transaction) })
148        } else {
149            Err(self.solver_problems())
150        }
151    }
