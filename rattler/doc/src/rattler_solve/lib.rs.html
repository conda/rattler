20/// Represents a solver implementation, capable of solving [`SolverTask`]s
21pub trait SolverImpl {
22    /// The repo data associated to a channel and platform combination
23    type RepoData<'a>: SolverRepoData<'a>;
24
25    /// Resolve the dependencies and return the [`RepoDataRecord`]s that should
26    /// be present in the environment.
27    fn solve<
28        'a,
29        R: IntoRepoData<'a, Self::RepoData<'a>>,
30        TAvailablePackagesIterator: IntoIterator<Item = R>,
31    >(
32        &mut self,
33        task: SolverTask<TAvailablePackagesIterator>,
34    ) -> Result<SolverResult, SolveError>;
35}

37/// Represents an error when solving the dependencies for a given environment
38#[derive(thiserror::Error, Debug)]
39pub enum SolveError {
40    /// There is no set of dependencies that satisfies the requirements
41    Unsolvable(Vec<String>),
42
43    /// The solver backend returned operations that we dont know how to install.
44    /// Each string is a somewhat user-friendly representation of which
45    /// operation was not recognized and can be used for error reporting
46    UnsupportedOperations(Vec<String>),
47
48    /// Error when converting matchspec
49    #[error(transparent)]
50    ParseMatchSpecError(#[from] rattler_conda_types::ParseMatchSpecError),
51
52    /// Encountered duplicate records in the available packages.
53    DuplicateRecords(String),
54
55    /// To support Resolvo cancellation
56    Cancelled,
57}
58
59impl fmt::Display for SolveError {
60    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
61        match self {
62            SolveError::Unsolvable(operations) => {
63                write!(
64                    f,
65                    "Cannot solve the request because of: {}",
66                    operations.join(", ")
67                )
68            }
69            SolveError::UnsupportedOperations(operations) => {
70                write!(f, "Unsupported operations: {}", operations.join(", "))
71            }
72            SolveError::ParseMatchSpecError(e) => {
73                write!(f, "Error parsing match spec: {e}")
74            }
75            SolveError::Cancelled => {
76                write!(f, "Solve operation has been cancelled")
77            }
78            SolveError::DuplicateRecords(filename) => {
79                write!(f, "encountered duplicate records for {filename}")
80            }
81        }
82    }
83}

85/// Configuration for filtering packages based on their minimum age.
86///
87/// This feature helps reduce the risk of installing compromised packages by
88/// delaying the installation of newly published versions. In most cases,
89/// malicious releases are discovered and removed from channels within a short
90/// time window (often within an hour). By requiring packages to have been
91/// published for a minimum duration, you give the community time to identify
92/// and report malicious packages before they can be installed.
93///
94/// This is similar to pnpm's `minimumReleaseAge` feature.
95///
96/// # Example
97///
98/// ```
99/// use std::time::Duration;
100/// use rattler_solve::MinimumAgeConfig;
101///
102/// // Only allow packages that have been published for at least 1 hour
103/// let config = MinimumAgeConfig::new(Duration::from_secs(60 * 60))
104///     // But allow "my-internal-package" to bypass this check
105///     .with_exempt_package("my-internal-package".parse().unwrap());
106/// ```
107#[derive(Debug, Clone, PartialEq, Eq)]
108pub struct MinimumAgeConfig {
109    /// The minimum age a package must have before it can be considered for
110    /// installation. Packages published more recently than this duration ago
111    /// will be excluded from the solve.
112    ///
113    /// For example, if set to 7 days, only packages that were published at
114    /// least 7 days ago will be considered.
115    pub min_age: std::time::Duration,
116
117    /// The reference time to use when calculating the cutoff date.
118    /// Packages published after `now - min_age` will be excluded.
119    ///
120    /// Defaults to the current time when [`MinimumAgeConfig::new`] is called.
121    pub now: DateTime<Utc>,
122
123    /// Packages that are exempt from the minimum release age requirement.
124    ///
125    /// This is useful for packages that you trust or that need to be updated
126    /// frequently, even if they were recently published.
127    pub exempt_packages: HashSet<PackageName>,
128
129    /// Whether to include packages that don't have a timestamp.
130    ///
131    /// By default, packages without a timestamp are excluded when a minimum age
132    /// filter is active. Set this to `true` to include them anyway.
133    pub include_unknown_timestamp: bool,
134}
135
136impl Default for MinimumAgeConfig {
137    fn default() -> Self {
138        Self {
139            min_age: std::time::Duration::default(),
140            now: Utc::now(),
141            exempt_packages: HashSet::new(),
142            include_unknown_timestamp: false,
143        }
144    }
145}
146
147impl MinimumAgeConfig {
148    /// Creates a new `MinimumAgeConfig` with the specified minimum age.
149    /// The reference time (`now`) is set to the current time.
150    pub fn new(min_age: std::time::Duration) -> Self {
151        Self {
152            min_age,
153            now: Utc::now(),
154            exempt_packages: HashSet::new(),
155            include_unknown_timestamp: false,
156        }
157    }
158
159    /// Sets the reference time to use when calculating the cutoff date.
160    pub fn with_now(mut self, now: DateTime<Utc>) -> Self {
161        self.now = now;
162        self
163    }
164
165    /// Adds a package to the set of exempt packages.
166    pub fn with_exempt_package(mut self, package: PackageName) -> Self {
167        self.exempt_packages.insert(package);
168        self
169    }
170
171    /// Sets the set of exempt packages.
172    pub fn with_exempt_packages(mut self, packages: impl IntoIterator<Item = PackageName>) -> Self {
173        self.exempt_packages = packages.into_iter().collect();
174        self
175    }
176
177    /// Sets whether packages without a timestamp should be included.
178    ///
179    /// By default, packages without a timestamp are excluded. Call this with
180    /// `true` to include them.
181    pub fn with_include_unknown_timestamp(mut self, include: bool) -> Self {
182        self.include_unknown_timestamp = include;
183        self
184    }
185
186    /// Returns `true` if the given package is exempt from the minimum release
187    /// age check.
188    pub fn is_exempt(&self, package: &PackageName) -> bool {
189        self.exempt_packages.contains(package)
190    }
191
192    /// Computes the cutoff time. Packages published after this time will be
193    /// excluded (unless exempt).
194    pub fn cutoff(&self) -> DateTime<Utc> {
195        let duration = chrono::Duration::from_std(self.min_age)
196            .expect("min_release_age duration is too large");
197        self.now - duration
198    }
199}

201/// Represents the channel priority option to use during solves.
202#[derive(Debug, Clone, Copy, PartialEq, Eq, Default)]
203#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
204#[cfg_attr(feature = "serde", serde(rename_all = "kebab-case"))]
205pub enum ChannelPriority {
206    /// The channel that the package is first found in will be used as the only
207    /// channel for that package.
208    #[default]
209    Strict,
210
211    // Conda also has "Flexible" as an option, where packages present in multiple channels
212    // are only taken from lower-priority channels when this prevents unsatisfiable environment
213    // errors, but this would need implementation in the solvers.
214    // Flexible,
215    /// Packages can be retrieved from any channel as package version takes
216    /// precedence.
217    Disabled,
218}
219#[derive(Debug, Clone, PartialEq, Eq)]
220/// Represents a dependency resolution task, to be solved by one of the backends
221pub struct SolverTask<TAvailablePackagesIterator> {
222    /// An iterator over all available packages
223    pub available_packages: TAvailablePackagesIterator,
224
225    /// Records of packages that are previously selected.
226    ///
227    /// If the solver encounters multiple variants of a single package
228    /// (identified by its name), it will sort the records and select the
229    /// best possible version. However, if there exists a locked version it
230    /// will prefer that variant instead. This is useful to reduce the number of
231    /// packages that are updated when installing new packages.
232    ///
233    /// Usually you add the currently installed packages or packages from a
234    /// lock-file here.
235    pub locked_packages: Vec<RepoDataRecord>,
236
237    /// Records of packages that are previously selected and CANNOT be changed.
238    ///
239    /// If the solver encounters multiple variants of a single package
240    /// (identified by its name), it will sort the records and select the
241    /// best possible version. However, if there is a variant available in
242    /// the `pinned_packages` field it will always select that version no matter
243    /// what even if that means other packages have to be downgraded.
244    pub pinned_packages: Vec<RepoDataRecord>,
245
246    /// Virtual packages considered active
247    pub virtual_packages: Vec<GenericVirtualPackage>,
248
249    /// The specs we want to solve
250    pub specs: Vec<MatchSpec>,
251
252    /// Additional constraints that should be satisfied by the solver.
253    /// Packages included in the `constraints` are not necessarily
254    /// installed, but they must be satisfied by the solution.
255    pub constraints: Vec<MatchSpec>,
256
257    /// The timeout after which the solver should stop
258    pub timeout: Option<std::time::Duration>,
259
260    /// The channel priority to solve with, either [`ChannelPriority::Strict`]
261    /// or [`ChannelPriority::Disabled`]
262    pub channel_priority: ChannelPriority,
263
264    /// Exclude any package that has a timestamp newer than the specified
265    /// timestamp.
266    pub exclude_newer: Option<DateTime<Utc>>,
267
268    /// Only consider packages that have been published for at least the
269    /// specified duration.
270    ///
271    /// This helps reduce the risk of installing compromised packages, as
272    /// malicious releases are typically discovered and removed from channels
273    /// within a short time window. By requiring a minimum age, you give the
274    /// community time to identify and report malicious packages.
275    ///
276    /// Some packages can be exempted from this check using the
277    /// [`MinimumAgeConfig::exempt_packages`] field.
278    pub min_age: Option<MinimumAgeConfig>,
279
280    /// The solve strategy.
281    pub strategy: SolveStrategy,
282
283    /// Dependency overrides that replace dependencies of matching packages.
284    pub dependency_overrides: Vec<(MatchSpec, MatchSpec)>,
285}

287impl<'r, I: IntoIterator<Item = &'r RepoDataRecord>> FromIterator<I>
288    for SolverTask<Vec<RepoDataIter<I>>>
289{
290    fn from_iter<T: IntoIterator<Item = I>>(iter: T) -> Self {
291        Self {
292            available_packages: iter.into_iter().map(|iter| RepoDataIter(iter)).collect(),
293            locked_packages: Vec::new(),
294            pinned_packages: Vec::new(),
295            virtual_packages: Vec::new(),
296            specs: Vec::new(),
297            constraints: Vec::new(),
298            timeout: None,
299            channel_priority: ChannelPriority::default(),
300            exclude_newer: None,
301            min_age: None,
302            strategy: SolveStrategy::default(),
303            dependency_overrides: Vec::new(),
304        }
305    }
306}

308/// Represents the strategy to use when solving dependencies
309#[derive(Debug, Clone, Copy, Default, Eq, PartialEq)]
310#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
311#[cfg_attr(feature = "serde", serde(rename_all = "kebab-case"))]
312pub enum SolveStrategy {
313    /// Resolve the highest version of each package.
314    #[default]
315    Highest,
316
317    /// Resolve the lowest compatible version for each package.
318    ///
319    /// All candidates with the same version are still ordered the same as
320    /// with `Default`. This ensures that the candidate with the highest build
321    /// number is used and down-prioritization still works.
322    LowestVersion,
323
324    /// Resolve the lowest compatible version for direct dependencies but the
325    /// highest for transitive dependencies. This is similar to `LowestVersion`
326    /// but only for direct dependencies.
327    LowestVersionDirect,
328}

330/// A representation of a collection of [`RepoDataRecord`] usable by a
331/// [`SolverImpl`] implementation.
332///
333/// Some solvers might be able to cache the collection between different runs of
334/// the solver which could potentially eliminate some overhead. This trait
335/// enables creating a representation of the repodata that is most suitable for
336/// a specific backend.
337///
338/// Some solvers may add additional functionality to their specific
339/// implementation that enables caching the repodata to disk in an efficient way
340/// (see [`crate::libsolv_c::RepoData`] for an example).
341pub trait SolverRepoData<'a>: FromIterator<&'a RepoDataRecord> {}
342
343/// Defines the ability to convert a type into [`SolverRepoData`].
344pub trait IntoRepoData<'a, S: SolverRepoData<'a>> {
345    /// Converts this instance into an instance of [`SolverRepoData`] which is
346    /// consumable by a specific [`SolverImpl`] implementation.
347    fn into(self) -> S;
348}
349
350impl<'a, S: SolverRepoData<'a>> IntoRepoData<'a, S> for &'a Vec<RepoDataRecord> {
351    fn into(self) -> S {
352        self.iter().collect()
353    }
354}
355
356impl<'a, S: SolverRepoData<'a>> IntoRepoData<'a, S> for &'a [RepoDataRecord] {
357    fn into(self) -> S {
358        self.iter().collect()
359    }
360}
361
362impl<'a, S: SolverRepoData<'a>> IntoRepoData<'a, S> for S {
363    fn into(self) -> S {
364        self
365    }
366}
