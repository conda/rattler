10/// How to address S3 buckets.
11#[derive(Default, Debug, Copy, Clone, Eq, PartialEq)]
12#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
13#[cfg_attr(feature = "serde", serde(rename_all = "kebab-case"))]
14pub enum S3AddressingStyle {
15    /// Address the bucket as a virtual host. E.g. <https://bucket_name.s3.us-east-1.amazonaws.com>.
16    #[default]
17    VirtualHost,
18
19    /// Address the bucket using a path. E.g. <https://s3.us-east-1.amazonaws.com/bucket_name>.
20    Path,
21}

90/// Returns the default HTTP client to use by the aws SDK.
91fn default_http_client() -> SharedHttpClient {
92    use aws_smithy_http_client::{
93        tls::{self, rustls_provider::CryptoMode},
94        Builder,
95    };
96
97    static CLIENT: std::sync::OnceLock<SharedHttpClient> = std::sync::OnceLock::new();
98    CLIENT
99        .get_or_init(|| {
100            Builder::new()
101                .tls_provider(tls::Provider::Rustls(CryptoMode::Ring))
102                .build_https()
103        })
104        .clone()
105}

107impl ResolvedS3Credentials {
108    pub async fn from_sdk() -> Result<Self, FromSDKError> {
109        let config = aws_config::defaults(BehaviorVersion::latest())
110            .http_client(default_http_client())
111            .load()
112            .await;
113        let s3_config = aws_sdk_s3::config::Builder::from(&config).build();
114
115        let region = s3_config
116            .region()
117            .map(Region::to_string)
118            .ok_or(FromSDKError::MissingRegion)?;
119        let endpoint_url_str = config.endpoint_url().unwrap_or("https://s3.amazonaws.com");
120        let endpoint_url = Url::parse(endpoint_url_str).map_err(FromSDKError::InvalidEndpoint)?;
121
122        let Some(credentials_provider) = config.credentials_provider() else {
123            return Err(FromSDKError::NoCredentialsProvider);
124        };
125        let credentials: Credentials = credentials_provider
126            .provide_credentials()
127            .await
128            .map_err(FromSDKError::CredentialsError)?;
129        let access_key_id = credentials.access_key_id().to_string();
130        let secret_access_key = credentials.secret_access_key().to_string();
131        let session_token = credentials.session_token().map(ToString::to_string);
132
133        // Address style is not exposed in the AWS SDK config, so we use the default.
134        // See: <https://github.com/awslabs/aws-sdk-rust/issues/1230>
135        let addressing_style = S3AddressingStyle::default();
136
137        Ok(Self {
138            endpoint_url,
139            region,
140            addressing_style,
141            access_key_id,
142            secret_access_key,
143            session_token,
144        })
145    }
146}

148impl S3Credentials {
149    /// Try to resolve the S3 credentials using the provided authentication
150    /// storage.
151    pub fn resolve(
152        self,
153        bucket_url: &Url,
154        auth_storage: &AuthenticationStorage,
155    ) -> Option<ResolvedS3Credentials> {
156        let (access_key_id, secret_access_key, session_token) =
157            if let (Some(access_key_id), Some(secret_access_key)) =
158                (self.access_key_id, self.secret_access_key)
159            {
160                (access_key_id, secret_access_key, self.session_token)
161            } else if let Some((access_key_id, secret_access_key, session_token)) =
162                load_s3_credentials_from_auth_storage(auth_storage, bucket_url.clone())
163            {
164                // Use the credentials from the authentication storage if they are available.
165                (access_key_id, secret_access_key, session_token)
166            } else {
167                return None;
168            };
169
170        Some(ResolvedS3Credentials {
171            endpoint_url: self.endpoint_url,
172            region: self.region,
173            access_key_id,
174            secret_access_key,
175            session_token,
176            addressing_style: self.addressing_style,
177        })
178    }
179}
180
181fn load_s3_credentials_from_auth_storage(
182    auth_storage: &AuthenticationStorage,
183    channel: Url,
184) -> Option<(String, String, Option<String>)> {
185    let auth = auth_storage.get_by_url(channel).ok()?;
186    if let (
187        _,
188        Some(Authentication::S3Credentials {
189            access_key_id,
190            secret_access_key,
191            session_token,
192        }),
193    ) = auth
194    {
195        Some((access_key_id, secret_access_key, session_token))
196    } else {
197        None
198    }
