7#[derive(ValueEnum, Clone, Default, Debug, PartialEq)]
8#[clap(rename_all = "kebab_case")]
9pub enum S3AddressingStyleOpts {
10    #[default]
11    VirtualHost,
12    Path,
13}
14
15impl From<S3AddressingStyleOpts> for crate::S3AddressingStyle {
16    fn from(value: S3AddressingStyleOpts) -> Self {
17        match value {
18            S3AddressingStyleOpts::VirtualHost => crate::S3AddressingStyle::VirtualHost,
19            S3AddressingStyleOpts::Path => crate::S3AddressingStyle::Path,
20        }
21    }
22}

24impl Display for S3AddressingStyleOpts {
25    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {

28            S3AddressingStyleOpts::Path => write!(f, "path"),
29        }

33/// Manually specified S3 credentials, when these are used no credentials are
34/// read through the AWS SDK.
35///
36/// See [`super::S3Credentials`] for details on how these credentials are used.
37#[derive(Clone, Debug, PartialEq, Parser)]
38pub struct S3CredentialsOpts {
39    /// The endpoint URL of the S3 backend
40    #[arg(long, env = "S3_ENDPOINT_URL", requires_all = ["region"], help_heading = "S3 Credentials")]
41    pub endpoint_url: Option<Url>,
42
43    /// The region of the S3 backend
44    #[arg(long, env = "S3_REGION", requires_all = ["endpoint_url"], help_heading = "S3 Credentials")]
45    pub region: Option<String>,
46
47    /// The access key ID for the S3 bucket.
48    #[arg(long, env = "S3_ACCESS_KEY_ID", requires_all = ["secret_access_key", "endpoint_url", "region"], help_heading = "S3 Credentials")]
49    pub access_key_id: Option<String>,
50
51    /// The secret access key for the S3 bucket.
52    #[arg(long, env = "S3_SECRET_ACCESS_KEY", requires_all = ["access_key_id", "endpoint_url", "region"], help_heading = "S3 Credentials")]
53    pub secret_access_key: Option<String>,
54
55    /// The session token for the S3 bucket.
56    #[arg(long, env = "S3_SESSION_TOKEN", requires_all = ["access_key_id", "secret_access_key", "endpoint_url", "region"], help_heading = "S3 Credentials")]
57    pub session_token: Option<String>,
58
59    /// How to address the bucket
60    #[arg(long, env = "S3_ADDRESSING_STYLE", requires_all = ["region", "endpoint_url"], help_heading = "S3 Credentials", conflicts_with="force_path_style", default_value_t=S3AddressingStyleOpts::default())]
61    pub addressing_style: S3AddressingStyleOpts,
62
63    /// Whether to use path-style S3 URLs
64    #[arg(long, env = "S3_FORCE_PATH_STYLE", requires_all = ["region", "endpoint_url"], help_heading = "S3 Credentials", conflicts_with="addressing_style", hide = true, help = "[deprecated] Whether to use path-style S3 URLs")]
65    pub force_path_style: Option<bool>,
66}
67
68impl From<S3CredentialsOpts> for Option<S3Credentials> {
69    fn from(mut value: S3CredentialsOpts) -> Self {
70        if value.force_path_style.is_some() {
71            tracing::warn!("The `--force-path-style` option is deprecated, please use `--addressing-style=path` instead.");
72            value.addressing_style = S3AddressingStyleOpts::Path;
73        }
74
75        if let (Some(endpoint_url), Some(region)) = (value.endpoint_url, value.region) {
76            Some(S3Credentials {
77                endpoint_url,
78                region,
79                access_key_id: value.access_key_id,
80                secret_access_key: value.secret_access_key,
81                session_token: value.session_token,
82                addressing_style: value.addressing_style.into(),
83            })
84        } else {
85            None
86        }
87    }
