12/// Trait for progress bars
13pub trait ProgressBar {
14    /// Set the current progress and progress message
15    fn set_progress(&mut self, progress: u64, message: &str);
16    /// Set the total amount of bytes
17    fn set_total(&mut self, total: u64);
18}
19
20/// A wrapper for a reader that updates a progress bar
21struct ProgressBarReader {
22    reader: Option<File>,
23    progress_bar: Option<Box<dyn ProgressBar>>,
24    progress: u64,
25    total: u64,
26    message: String,
27}
28
29impl ProgressBarReader {
30    fn new(progress_bar: Option<Box<dyn ProgressBar>>) -> Self {
31        Self {
32            reader: None,
33            progress_bar,
34            progress: 0,
35            total: 0,
36            message: String::new(),
37        }
38    }
39
40    fn set_file(&mut self, file: File) {
41        self.reader = Some(file);
42    }
43
44    fn reset_position(&mut self) {
45        self.progress = 0;
46        if let Some(progress_bar) = &mut self.progress_bar {
47            progress_bar.set_progress(0, &self.message);
48        }
49    }
50
51    fn set_total(&mut self, total_size: u64) {
52        self.total = total_size;
53        if let Some(progress_bar) = &mut self.progress_bar {
54            progress_bar.set_total(total_size);
55        }
56    }
57}
58
59impl Read for ProgressBarReader {
60    fn read(&mut self, buf: &mut [u8]) -> io::Result<usize> {
61        let n = self.reader.as_ref().expect("No reader set!").read(buf)?;
62        self.progress += n as u64;
63        if let Some(progress_bar) = &mut self.progress_bar {
64            progress_bar.set_progress(self.progress, &self.message);
65        }
66        Ok(n)
67    }
68}

70/// a function that sorts paths into two iterators, one that starts with `info/`
71/// and one that does not both iterators are sorted alphabetically for
72/// reproducibility
73fn sort_paths<'a>(paths: &'a [PathBuf], base_path: &'a Path) -> (Vec<PathBuf>, Vec<PathBuf>) {
74    let info = Path::new("info/");
75    let (mut info_paths, mut other_paths): (Vec<_>, Vec<_>) = paths
76        .iter()
77        .map(|p| p.strip_prefix(base_path).unwrap())
78        .map(Path::to_path_buf)
79        .partition(|path| path.starts_with(info));
80
81    info_paths.sort();
82    other_paths.sort();
83
84    (info_paths, other_paths)
85}
86
87fn total_size(base_path: &Path, paths: &[PathBuf]) -> u64 {
88    paths
89        .iter()
90        .map(|p| base_path.join(p).metadata().map_or(0, |m| m.len()))
91        .sum()
92}

94/// Write the contents of a list of paths to a tar.bz2 package
95/// The paths are sorted alphabetically, and paths beginning with `info/` come
96/// first.
97///
98/// # Arguments
99///
100/// * `writer` - the writer to write the package to
101/// * `base_path` - the base path of the package. All paths in `paths` are
102///   relative to this path
103/// * `paths` - a list of paths to include in the package
104/// * `compression_level` - the compression level to use for the inner bzip2
105///   encoded files
106/// * `timestamp` - optional a timestamp to use for all archive files (useful
107///   for reproducible builds)
108///
109/// # Errors
110///
111/// This function will return an error if the writer returns an error, or if the
112/// paths are not relative to the base path.
113///
114/// # Examples
115///
116/// ```no_run
117/// use std::path::PathBuf;
118/// use std::fs::File;
119/// use rattler_package_streaming::write::{write_tar_bz2_package};
120/// use rattler_conda_types::compression_level::CompressionLevel;
121///
122/// let paths = vec![PathBuf::from("info/recipe/meta.yaml"), PathBuf::from("info/recipe/conda_build_config.yaml")];
123/// let mut file = File::create("test.tar.bz2").unwrap();
124/// write_tar_bz2_package(&mut file, &PathBuf::from("test"), &paths, CompressionLevel::Default, None, None).unwrap();
125/// ```
126///
127/// # See also
128///
129/// * [`write_conda_package`]
130pub fn write_tar_bz2_package<W: Write>(
131    writer: W,
132    base_path: &Path,
133    paths: &[PathBuf],
134    compression_level: CompressionLevel,
135    timestamp: Option<&chrono::DateTime<chrono::Utc>>,
136    progress_bar: Option<Box<dyn ProgressBar>>,
137) -> Result<(), std::io::Error> {
138    let mut archive = tar::Builder::new(bzip2::write::BzEncoder::new(
139        writer,
140        bzip2::Compression::new(compression_level.to_bzip2_level()?),
141    ));
142    archive.follow_symlinks(false);
143
144    let total_size = total_size(base_path, paths);
145    let mut progress_bar_wrapper = ProgressBarReader::new(progress_bar);
146    progress_bar_wrapper.set_total(total_size);
147
148    // sort paths alphabetically, and sort paths beginning with `info/` first
149    let (info_paths, other_paths) = sort_paths(paths, base_path);
150    for path in info_paths.iter().chain(other_paths.iter()) {
151        append_path_to_archive(
152            &mut archive,
153            base_path,
154            path,
155            timestamp,
156            &mut progress_bar_wrapper,
157        )?;
158    }
159
160    archive.into_inner()?.finish()?;
161
162    Ok(())
163}

165/// Write the contents of a list of paths to a tar zst archive
166fn write_zst_archive<W: Write>(
167    writer: W,
168    base_path: &Path,
169    paths: &Vec<PathBuf>,
170    compression_level: CompressionLevel,
171    num_threads: Option<u32>,
172    timestamp: Option<&chrono::DateTime<chrono::Utc>>,
173    progress_bar: Option<Box<dyn ProgressBar>>,
174) -> Result<(), std::io::Error> {
175    // Create a temporary tar file
176    let tar_path = tempfile::Builder::new().tempfile_in(base_path)?;
177    let mut archive = tar::Builder::new(&tar_path);
178    archive.follow_symlinks(false);
179
180    let total_size = total_size(base_path, paths);
181    let mut progress_bar_wrapper = ProgressBarReader::new(progress_bar);
182    progress_bar_wrapper.set_total(total_size);
183    for path in paths {
184        append_path_to_archive(
185            &mut archive,
186            base_path,
187            path,
188            timestamp,
189            &mut progress_bar_wrapper,
190        )?;
191    }
192    archive.finish()?;
193
194    // Compress it as tar.zst
195    let tar_file = File::open(&tar_path)?;
196    let compression_level = compression_level.to_zstd_level()?;
197    let mut zst_encoder = zstd::Encoder::new(writer, compression_level)?;
198    #[cfg(not(target_arch = "wasm32"))]
199    zst_encoder.multithread(num_threads.unwrap_or_else(|| num_cpus::get() as u32))?;
200    // mark as "used" to avoid "unused" warning
201    #[cfg(target_arch = "wasm32")]
202    let _ = num_threads;
203
204    progress_bar_wrapper.reset_position();
205    if let Ok(tar_total_size) = tar_file.metadata().map(|v| v.len()) {
206        zst_encoder.set_pledged_src_size(Some(tar_total_size))?;
207        progress_bar_wrapper.set_total(tar_total_size);
208    };
209    zst_encoder.include_contentsize(true)?;
210
211    // Append tar.zst to the archive
212    progress_bar_wrapper.set_file(tar_file);
213    io::copy(&mut progress_bar_wrapper, &mut zst_encoder)?;
214    zst_encoder.finish()?;
215
216    Ok(())
217}

219/// Write a `.conda` package to a writer
220/// A `.conda` package is an outer uncompressed zip archive that contains a
221/// `metadata.json` file, a `pkg-archive.tar.zst` file, and a
222/// `info-archive.tar.zst` file. The inner zstd encoded files are sorted
223/// alphabetically. The `info-archive.tar.zst` file comes last in the outer zip
224/// archive.
225///
226/// # Arguments
227///
228/// * `writer` - the writer to write the package to
229/// * `base_path` - the base path of the package. All paths in `paths` are
230///   relative to this path
231/// * `paths` - a list of paths to include in the package
232/// * `compression_level` - the compression level to use for the inner zstd
233///   encoded files
234/// * `compression_num_threads` - the number of threads to use for zstd
235///   compression (defaults to the number of CPU cores if `None`)
236/// * `timestamp` - optional a timestamp to use for all archive files (useful
237///   for reproducible builds)
238///
239/// # Errors
240///
241/// This function will return an error if the writer returns an error, or if the
242/// paths are not relative to the base path.
243#[allow(clippy::too_many_arguments)]
244pub fn write_conda_package<W: Write + Seek>(
245    writer: W,
246    base_path: &Path,
247    paths: &[PathBuf],
248    compression_level: CompressionLevel,
249    compression_num_threads: Option<u32>,
250    out_name: &str,
251    timestamp: Option<&chrono::DateTime<chrono::Utc>>,
252    progress_bar: Option<Box<dyn ProgressBar>>,
253) -> Result<(), std::io::Error> {
254    // first create the outer zip archive that uses no compression
255    let mut outer_archive = zip::ZipWriter::new(writer);
256
257    let last_modified_time = if let Some(time) = timestamp {
258        DateTime::from_date_and_time(
259            time.year() as u16,
260            time.month() as u8,
261            time.day() as u8,
262            time.hour() as u8,
263            time.minute() as u8,
264            time.second() as u8,
265        )
266        .expect("time should be in correct range")
267    } else {
268        // 1-1-2023 00:00:00 (Fixed date in the past for reproducible builds)
269        DateTime::from_date_and_time(2023, 1, 1, 0, 0, 0)
270            .expect("1-1-2023 00:00:00 should convert into datetime")
271    };
272
273    let options = zip::write::SimpleFileOptions::default()
274        .compression_method(zip::CompressionMethod::Stored)
275        .last_modified_time(last_modified_time)
276        .large_file(true);
277
278    // write the metadata as first file in the zip archive
279    let package_metadata = PackageMetadata::default();
280    let package_metadata = serde_json::to_string(&package_metadata).unwrap();
281    outer_archive.start_file("metadata.json", options)?;
282    outer_archive.write_all(package_metadata.as_bytes())?;
283
284    let (info_paths, other_paths) = sort_paths(paths, base_path);
285
286    let archive_path = format!("pkg-{out_name}.tar.zst");
287
288    outer_archive.start_file(archive_path, options)?;
289    write_zst_archive(
290        &mut outer_archive,
291        base_path,
292        &other_paths,
293        compression_level,
294        compression_num_threads,
295        timestamp,
296        progress_bar,
297    )?;
298
299    // info paths come last
300    let archive_path = format!("info-{out_name}.tar.zst");
301    outer_archive.start_file(archive_path, options)?;
302    write_zst_archive(
303        &mut outer_archive,
304        base_path,
305        &info_paths,
306        compression_level,
307        compression_num_threads,
308        timestamp,
309        None,
310    )?;
311
312    outer_archive.finish()?;
313
314    Ok(())
315}

317fn prepare_header(
318    path: &Path,
319    timestamp: Option<&chrono::DateTime<chrono::Utc>>,
320) -> Result<tar::Header, std::io::Error> {
321    let mut header = tar::Header::new_gnu();
322    let name = b"././@LongLink";
323    header.as_gnu_mut().unwrap().name[..name.len()].clone_from_slice(&name[..]);
324
325    let stat = fs::symlink_metadata(path)?;
326    header.set_metadata_in_mode(&stat, tar::HeaderMode::Deterministic);
327
328    if let Some(timestamp) = timestamp {
329        header.set_mtime(timestamp.timestamp().unsigned_abs());
330    } else {
331        // 1-1-2023 00:00:00 (Fixed date in the past for reproducible builds)
332        header.set_mtime(1672531200);
333    }
334
335    Ok(header)
336}

338fn trace_file_error(path: &Path, err: std::io::Error) -> std::io::Error {
339    println!("{}: {}", path.display(), err);
340    std::io::Error::new(err.kind(), format!("{}: {}", path.display(), err))
341}

343fn append_path_to_archive(
344    archive: &mut tar::Builder<impl Write>,
345    base_path: &Path,
346    path: &Path,
347    timestamp: Option<&chrono::DateTime<chrono::Utc>>,
348    progress_bar: &mut ProgressBarReader,
349) -> Result<(), std::io::Error> {
350    // create a tar header
351    let mut header = prepare_header(&base_path.join(path), timestamp)
352        .map_err(|err| trace_file_error(&base_path.join(path), err))?;
353
354    if header.entry_type().is_file() {
355        let file = fs::File::open(base_path.join(path))
356            .map_err(|err| trace_file_error(&base_path.join(path), err))?;
357        // wrap the file reader in a progress bar reader
358        progress_bar.set_file(file);
359        archive.append_data(&mut header, path, progress_bar)?;
360    } else if header.entry_type().is_symlink() || header.entry_type().is_hard_link() {
361        let target = fs::read_link(base_path.join(path))
362            .map_err(|err| trace_file_error(&base_path.join(path), err))?;
363
364        archive.append_link(&mut header, path, target)?;
365    } else if header.entry_type().is_dir() {
366        archive.append_data(&mut header, path, std::io::empty())?;
367    } else {
368        return Err(std::io::Error::new(
369            std::io::ErrorKind::InvalidInput,
370            "unsupported file type",
371        ));
372    }
373
374    Ok(())
