1//! Functions to extracting or stream a Conda package from a file on disk.

3use crate::{ExtractError, ExtractResult};
4use rattler_conda_types::package::CondaArchiveType;
5use std::path::Path;

7/// Extracts the contents a `.tar.bz2` package archive at the specified path to a directory.
8///
9/// ```rust,no_run
10/// # #[tokio::main]
11/// # async fn main() {
12/// # use std::path::Path;
13/// use rattler_package_streaming::tokio::fs::extract_tar_bz2;
14/// let _ = extract_tar_bz2(
15///     Path::new("conda-forge/win-64/python-3.11.0-hcf16a7b_0_cpython.tar.bz2"),
16///     Path::new("/tmp"))
17///     .await
18///     .unwrap();
19/// # }
20/// ```
21pub async fn extract_tar_bz2(
22    archive: &Path,
23    destination: &Path,
24) -> Result<ExtractResult, ExtractError> {
25    // Open the file for reading using async I/O
26    let file = tokio::fs::File::open(archive)
27        .await
28        .map_err(ExtractError::IoError)?;
29
30    // Use the fully async extraction implementation
31    crate::tokio::async_read::extract_tar_bz2(file, destination).await
32}

34/// Extracts the contents a `.conda` package archive at the specified path to a directory.
35///
36/// ```rust,no_run
37/// # use std::path::Path;
38/// # #[tokio::main]
39/// # async fn main() {
40/// use rattler_package_streaming::tokio::fs::extract_conda;
41/// let _ = extract_conda(
42///     Path::new("conda-forge/win-64/python-3.11.0-hcf16a7b_0_cpython.conda"),
43///     Path::new("/tmp"))
44///     .await
45///     .unwrap();
46/// # }
47/// ```
48pub async fn extract_conda(
49    archive: &Path,
50    destination: &Path,
51) -> Result<ExtractResult, ExtractError> {
52    // Spawn a block task to perform the extraction
53    let destination = destination.to_owned();
54    let archive = archive.to_owned();
55    match tokio::task::spawn_blocking(move || crate::fs::extract_conda(&archive, &destination))
56        .await
57    {
58        Ok(result) => result,
59        Err(err) => {
60            if let Ok(reason) = err.try_into_panic() {
61                std::panic::resume_unwind(reason);
62            }
63            Err(ExtractError::Cancelled)
64        }
65    }
66}

68/// Extracts the contents a package archive at the specified path to a directory. The type of
69/// package is determined based on the file extension of the archive path.
70///
71/// ```rust,no_run
72/// # #[tokio::main]
73/// # async fn main() {
74/// # use std::path::Path;
75/// use rattler_package_streaming::tokio::fs::extract;
76/// let _ = extract(
77///     Path::new("conda-forge/win-64/python-3.11.0-hcf16a7b_0_cpython.conda"),
78///     Path::new("/tmp"))
79///     .await
80///     .unwrap();
81/// # }
82/// ```
83pub async fn extract(archive: &Path, destination: &Path) -> Result<ExtractResult, ExtractError> {
84    match CondaArchiveType::try_from(archive).ok_or(ExtractError::UnsupportedArchiveType)? {
85        CondaArchiveType::TarBz2 => extract_tar_bz2(archive, destination).await,
86        CondaArchiveType::Conda => extract_conda(archive, destination).await,
87    }
