89impl NameField {

91        match self {
92            NameField::Simple(name) => name.resolve(placeholders),
93            NameField::Complex(complex_name) => match env {
94                Environment::Base => complex_name
95                    .target_environment_is_base
96                    .resolve(placeholders),
97                Environment::NotBase => complex_name
98                    .target_environment_is_not_base
99                    .resolve(placeholders),
100            },
101        }
102    }
103}

364/// macOS version number.
365#[derive(Debug, Clone)]
366pub struct MacOSVersion(pub(crate) Vec<u32>);
367
368impl Serialize for MacOSVersion {
369    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
370    where
371        S: serde::Serializer,
372    {
373        self.0
374            .iter()
375            .map(ToString::to_string)
376            .collect::<Vec<String>>()
377            .join(".")
378            .serialize(serializer)
379    }
380}
381
382impl<'de> Deserialize<'de> for MacOSVersion {
383    fn deserialize<D>(deserializer: D) -> Result<MacOSVersion, D::Error>
384    where
385        D: serde::Deserializer<'de>,
386    {
387        let s = String::deserialize(deserializer)?;
388        let version = s
389            .split('.')
390            .map(str::parse)
391            .collect::<Result<Vec<u32>, _>>()
392            .map_err(serde::de::Error::custom)?;
393
394        if version.len() > 3 {
395            return Err(serde::de::Error::custom(
396                "Version number must have at most 3 components",
397            ));
398        }
399
400        Ok(MacOSVersion(version))
401    }
402}

578impl MenuItemCommand {
579    /// Merge the generic `MenuItemCommand` with a platform-specific `BasePlatformSpecific`.
580    pub fn merge(&self, platform: BasePlatformSpecific) -> MenuItemCommand {
581        MenuItemCommand {
582            name: platform.name.unwrap_or_else(|| self.name.clone()),
583            description: platform
584                .description
585                .unwrap_or_else(|| self.description.clone()),
586            command: platform.command.unwrap_or_else(|| self.command.clone()),
587            icon: platform.icon.as_ref().or(self.icon.as_ref()).cloned(),
588            precommand: platform.precommand.or_else(|| self.precommand.clone()),
589            precreate: platform.precreate.or_else(|| self.precreate.clone()),
590            working_dir: platform.working_dir.or_else(|| self.working_dir.clone()),
591            activate: platform.activate.or(self.activate),
592            terminal: platform.terminal.or(self.terminal),
593        }
594    }
