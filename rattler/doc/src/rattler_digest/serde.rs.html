1//! Provides custom serialization/deserialization functions for [`Output`] of a [`Digest`].
2//!
3//! Use the struct [`SerializableHash`] to easily serialize digest outputs. This type
4//! automatically handles both human-readable formats (like JSON) and binary formats
5//! (like MessagePack). See [`SerializableHash`] for detailed documentation on the
6//! serialization behavior.
7
8#![allow(clippy::doc_markdown)]

10use digest::{Digest, Output, OutputSizeUser};
11use serde::de::Error;
12use serde::{Deserialize, Deserializer, Serialize, Serializer};
13use serde_with::{DeserializeAs, SerializeAs};
14use std::borrow::Cow;
15use std::fmt::LowerHex;
16use std::ops::Deref;

18/// Deserialize the [`Output`] of a [`Digest`].
19///
20/// If the deserializer is human-readable, it will parse the digest from a hex
21/// string. Otherwise, it will deserialize raw bytes.
22pub fn deserialize<'de, D, Dig: Digest>(deserializer: D) -> Result<Output<Dig>, D::Error>
23where
24    D: Deserializer<'de>,
25{
26    if deserializer.is_human_readable() {
27        let str = Cow::<'de, str>::deserialize(deserializer)?;
28        super::parse_digest_from_hex::<Dig>(str.as_ref())
29            .ok_or_else(|| Error::custom("failed to parse digest"))
30    } else {
31        let bytes = Cow::<'de, [u8]>::deserialize(deserializer)?;
32        let len = <Dig as OutputSizeUser>::output_size();
33        if bytes.len() != len {
34            return Err(Error::custom(format!(
35                "invalid length, expected {}, got {}",
36                len,
37                bytes.len()
38            )));
39        }
40        Ok(Output::<Dig>::clone_from_slice(&bytes))
41    }
42}

44/// Serializes the [`Output`] of a [`Digest`].
45///
46/// If the serializer is human-readable, it will write the digest as a hex
47/// string. Otherwise, it will deserialize raw bytes.
48pub fn serialize<'a, S: Serializer, Dig: Digest>(
49    digest: &'a Output<Dig>,
50    s: S,
51) -> Result<S::Ok, S::Error>
52where
53    &'a Output<Dig>: LowerHex,
54{
55    if s.is_human_readable() {
56        format!("{digest:x}").serialize(s)
57    } else {
58        // Without specialization, Rust forces Serde to treat &[u8] just like any other slice and
59        // Vec<u8> just like any other vector. In reality this particular slice and vector can
60        // often be serialized and deserialized in a more efficient, compact representation in many
61        // formats.
62        //
63        // Using the `serde_bytes` crate, we can opt into specialized handling.
64        serde_bytes::Bytes::new(digest.as_ref()).serialize(s)
65    }
66}

68/// Wrapper type for easily serializing and deserializing cryptographic hash digests.
69///
70/// This type provides automatic format-aware serialization:
71/// - **Human-readable formats** (JSON, YAML, TOML, etc.): Serializes as a lowercase hexadecimal string
72/// - **Binary formats** (MessagePack, Bincode, etc.): Serializes as raw bytes using `serde_bytes`
73///
74/// # Deserialization Behavior
75///
76/// ## Human-Readable Formats
77/// Accepts a hexadecimal string (case-insensitive) and parses it into the digest output.
78/// Returns an error if the string is not valid hexadecimal or has incorrect length.
79///
80/// ## Binary Formats
81/// Accepts raw bytes in multiple MessagePack encoding formats:
82/// - **Binary format** (`bin 8/16/32`): Most efficient, preferred format
83/// - **Array format** (`fixarray/array 16/32`): Array of integers, for compatibility
84///
85/// Returns an error if the byte length doesn't match the expected digest size.
86///
87/// # Examples
88///
89/// ## JSON (Human-Readable)
90/// ```
91/// use rattler_digest::serde::SerializableHash;
92///
93/// let hash = SerializableHash::<sha2::Sha256>(
94///     rattler_digest::parse_digest_from_hex::<sha2::Sha256>(
95///         "fe51de6107f9edc7aa4f786a70f4a883943bc9d39b3bb7307c04c41410990726"
96///     ).unwrap()
97/// );
98///
99/// // Serializes as: "fe51de6107f9edc7aa4f786a70f4a883943bc9d39b3bb7307c04c41410990726"
100/// let json = serde_json::to_string(&hash).unwrap();
101/// let deserialized: SerializableHash<sha2::Sha256> = serde_json::from_str(&json).unwrap();
102/// ```
103///
104/// ## MessagePack (Binary)
105/// ```
106/// use rattler_digest::serde::SerializableHash;
107///
108/// let hash = SerializableHash::<sha2::Sha256>(
109///     rattler_digest::parse_digest_from_hex::<sha2::Sha256>(
110///         "fe51de6107f9edc7aa4f786a70f4a883943bc9d39b3bb7307c04c41410990726"
111///     ).unwrap()
112/// );
113///
114/// // Serializes as raw 32 bytes using MessagePack binary format
115/// let bytes = rmp_serde::to_vec(&hash).unwrap();
116/// let deserialized: SerializableHash<sha2::Sha256> = rmp_serde::from_slice(&bytes).unwrap();
117/// ```
118#[derive(Debug, Clone)]
119pub struct SerializableHash<T: Digest>(pub Output<T>);
120
121impl<T: Digest> Serialize for SerializableHash<T>
122where
123    Output<T>: LowerHex,
124{
125    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
126    where
127        S: Serializer,
128    {
129        serialize::<S, T>(&self.0, serializer)
130    }
131}
132
133impl<'de, T: Digest + Default> Deserialize<'de> for SerializableHash<T> {
134    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
135    where
136        D: Deserializer<'de>,
137    {
138        let hash_output: Output<T> = deserialize::<D, T>(deserializer)?;
139        Ok(SerializableHash(hash_output))
140    }
141}
142
143impl<T: Digest> From<Output<T>> for SerializableHash<T> {
144    fn from(output: Output<T>) -> Self {
145        SerializableHash(output)
146    }
147}
148
149impl<T: Digest> From<SerializableHash<T>> for Output<T> {
150    fn from(s: SerializableHash<T>) -> Self {
151        s.0
152    }
153}
154
155impl<T: Digest> Deref for SerializableHash<T> {
156    type Target = Output<T>;
157
158    fn deref(&self) -> &Self::Target {
159        &self.0
160    }
161}
162
163impl<T: Digest> SerializeAs<Output<T>> for SerializableHash<T>
164where
165    for<'a> &'a Output<T>: LowerHex,
166{
167    fn serialize_as<S>(source: &Output<T>, serializer: S) -> Result<S::Ok, S::Error>
168    where
169        S: Serializer,
170    {
171        serialize::<S, T>(source, serializer)
172    }
173}
174
175impl<'de, T: Digest + Default> DeserializeAs<'de, Output<T>> for SerializableHash<T> {
176    fn deserialize_as<D>(deserializer: D) -> Result<Output<T>, D::Error>
177    where
178        D: Deserializer<'de>,
179    {
180        deserialize::<D, T>(deserializer)
181    }
182}

219    #[test]
220    pub fn test_deserialize_messagepack_raw_bytes() {
221        use rmp::encode;
222
223        let hex_str = "fe51de6107f9edc7aa4f786a70f4a883943bc9d39b3bb7307c04c41410990726";
224        let expected_hash = SerializableHash::<sha2::Sha256>(
225            crate::parse_digest_from_hex::<sha2::Sha256>(hex_str).unwrap(),
226        );
227
228        // Get the raw bytes from the expected hash
229        let hash_bytes = expected_hash.0.as_slice();
230
231        // Test 1: MessagePack binary format (bin 8, bin 16, or bin 32)
232        let mut msgpack_bin = Vec::new();
233        encode::write_bin(&mut msgpack_bin, hash_bytes).unwrap();
234
235        let deserialized_bin: SerializableHash<sha2::Sha256> =
236            rmp_serde::from_slice(&msgpack_bin).unwrap();
237        assert_eq!(&expected_hash.0, &deserialized_bin.0);
238
239        // Test 2: MessagePack array format (fixarray or array 16/32)
240        let mut msgpack_array = Vec::new();
241        encode::write_array_len(&mut msgpack_array, hash_bytes.len() as u32).unwrap();
242        for byte in hash_bytes {
243            encode::write_uint(&mut msgpack_array, u64::from(*byte)).unwrap();
244        }
245
246        let deserialized_array: SerializableHash<sha2::Sha256> =
247            rmp_serde::from_slice(&msgpack_array).unwrap();
248        assert_eq!(&expected_hash.0, &deserialized_array.0);
249    }

300    #[test]
301    pub fn test_messagepack_serialized_size() {
302        let hex_str = "fe51de6107f9edc7aa4f786a70f4a883943bc9d39b3bb7307c04c41410990726";
303        let hash = SerializableHash::<sha2::Sha256>(
304            crate::parse_digest_from_hex::<sha2::Sha256>(hex_str).unwrap(),
305        );
306
307        // Serialize to MessagePack
308        let bytes = rmp_serde::to_vec(&hash).unwrap();
309
310        // SHA256 is 32 bytes, MessagePack bin8 format uses:
311        // - 1 byte for the format marker (0xc4 for bin8)
312        // - 1 byte for the length (32)
313        // - 32 bytes for the data
314        // Total: 34 bytes
315        assert_eq!(bytes.len(), 34);
316
317        // Verify the format is bin8
318        assert_eq!(bytes[0], 0xc4); // bin8 marker
319        assert_eq!(bytes[1], 32); // length
320    }
