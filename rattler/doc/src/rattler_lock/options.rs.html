1//! Defines [`SolveOptions`] and reexports from `rattler_solve` that are used.

3// Reexport these fields.
4pub use rattler_solve::{ChannelPriority, SolveStrategy};
5
6/// The prerelease mode used to resolve `PyPI` dependencies.
7///
8/// This controls how the resolver handles pre-release versions of packages.
9#[derive(
10    Debug, Clone, Copy, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize, Default,
11)]
12#[serde(rename_all = "kebab-case")]
13pub enum PypiPrereleaseMode {
14    /// Disallow all pre-release versions.
15    Disallow,
16    /// Allow all pre-release versions.
17    Allow,
18    /// Allow pre-release versions if necessary to satisfy the requirements.
19    IfNecessary,
20    /// Allow pre-release versions for packages explicitly requested.
21    Explicit,
22    /// Allow pre-release versions if necessary or explicitly requested.
23    /// This is the default mode.
24    #[default]
25    IfNecessaryOrExplicit,
26}
27
28/// Options that were used during the resolution of the packages stored in the
29/// lock-file. These options strongly influence the outcome of the solve and are
30/// therefore stored along with the locked packages.
31#[derive(Debug, Clone, PartialEq, Eq, Default, serde::Serialize, serde::Deserialize)]
32#[serde(rename_all = "kebab-case")]
33pub struct SolveOptions {
34    /// The strategy that was used to solve the dependencies.
35    #[serde(default, skip_serializing_if = "crate::utils::serde::is_default")]
36    pub strategy: SolveStrategy,
37
38    /// The channel priority that was used to solve the dependencies.
39    #[serde(default, skip_serializing_if = "crate::utils::serde::is_default")]
40    pub channel_priority: ChannelPriority,
41
42    /// Packages after this date have been excluded from the lock file.
43    #[serde(default, skip_serializing_if = "Option::is_none")]
44    pub exclude_newer: Option<chrono::DateTime<chrono::Utc>>,
45
46    /// The prerelease mode that was used to resolve `PyPI` dependencies.
47    #[serde(default, skip_serializing_if = "Option::is_none")]
48    pub pypi_prerelease_mode: Option<PypiPrereleaseMode>,
49}
