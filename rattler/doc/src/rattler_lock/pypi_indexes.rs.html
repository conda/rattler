6/// Defines the pypi indexes that were used during solving.
7#[serde_as]
8#[derive(Debug, Clone, PartialEq, Serialize, Eq, Deserialize, Default)]
9#[serde(rename_all = "kebab-case", deny_unknown_fields)]
10pub struct PypiIndexes {
11    /// The indexes used to resolve the dependencies.
12    pub indexes: Vec<Url>,
13
14    /// Flat indexes also called `--find-links` in pip
15    /// These are flat listings of distributions
16    #[serde(
17        default,
18        skip_serializing_if = "Vec::is_empty",
19        with = "serde_yaml::with::singleton_map_recursive"
20    )]
21    pub find_links: Vec<FindLinksUrlOrPath>,
22}
23
24/// A flat index is a static source of
25#[derive(Debug, Deserialize, Serialize, Clone, PartialEq, Eq, Hash)]
26#[serde(rename_all = "kebab-case", deny_unknown_fields)]
27pub enum FindLinksUrlOrPath {
28    /// Can be a path to a directory or a file containing the flat index
29    Path(PathBuf),
30
31    /// Can be a URL to a flat index
32    Url(Url),
33}
34
35impl FindLinksUrlOrPath {
36    /// Returns the URL if it is a URL
37    pub fn as_url(&self) -> Option<&Url> {
38        match self {
39            Self::Path(_) => None,
40            Self::Url(url) => Some(url),
41        }
42    }
43
44    /// Returns the path if it is a path
45    pub fn as_path(&self) -> Option<&Path> {
46        match self {
47            Self::Path(path) => Some(path),
48            Self::Url(_) => None,
49        }
50    }
51}
