15    // Test extension for comprehensive testing
16    #[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize, Default)]
17    pub struct TestExtension {
18        #[serde(default)]
19        pub custom_field: Option<String>,
20        #[serde(default)]
21        pub numeric_field: Option<u32>,
22        #[serde(default)]
23        pub bool_field: Option<bool>,
24        #[serde(default)]
25        pub array_field: Vec<String>,
26        #[serde(default)]
27        pub nested: HashMap<String, String>,
28    }
29
30    impl Config for TestExtension {
31        fn get_extension_name(&self) -> String {
32            "test".to_string()
33        }
34
35        fn merge_config(self, other: &Self) -> Result<Self, MergeError> {
36            Ok(Self {
37                custom_field: other.custom_field.clone().or(self.custom_field),
38                numeric_field: other.numeric_field.or(self.numeric_field),
39                bool_field: other.bool_field.or(self.bool_field),
40                array_field: if other.array_field.is_empty() {
41                    self.array_field
42                } else {
43                    other.array_field.clone()
44                },
45                nested: self
46                    .nested
47                    .into_iter()
48                    .chain(other.nested.iter().map(|(k, v)| (k.clone(), v.clone())))
49                    .collect(),
50            })
51        }
52
53        fn validate(&self) -> Result<(), ValidationError> {
54            if let Some(numeric) = self.numeric_field {
55                if numeric > 100 {
56                    return Err(ValidationError::InvalidValue(
57                        "numeric_field".to_string(),
58                        "must be <= 100".to_string(),
59                    ));
60                }
61            }
62            Ok(())
63        }
64
65        fn keys(&self) -> Vec<String> {
66            vec![
67                "custom_field".to_string(),
68                "numeric_field".to_string(),
69                "bool_field".to_string(),
70                "array_field".to_string(),
71                "nested".to_string(),
72            ]
73        }
74    }

289                array_field: vec!["item1".to_string(), "item2".to_string()],
290                ..Default::default()
291            },
292            ..Default::default()
293        };
294
295        let merged = config1.merge_config(&config2).unwrap();
296
297        // The second config should take priority for overlapping fields
298        assert_eq!(

300            Some(1)

305        );
306        assert_eq!(
307            merged.authentication_override_file,
308            Some(PathBuf::from("/new/auth"))
309        );
310        assert_eq!(merged.tls_no_verify, Some(false)); // from config1
311
312        // Extension fields should be merged properly
313        assert_eq!(merged.extensions.custom_field, Some("updated".to_string())); // from config2
314        assert_eq!(merged.extensions.numeric_field, Some(42)); // from config1
315        assert_eq!(merged.extensions.bool_field, Some(true)); // from config2

721    #[test]
722    fn test_merge_multiple_configs() {
723        let base_config = TestConfig {
724            default_channels: Some(vec!["defaults".parse().unwrap()]),
725            tls_no_verify: Some(false),
726            ..Default::default()
727        };
728
729        let user_config = TestConfig {
730            default_channels: Some(vec!["conda-forge".parse().unwrap()]),
731            authentication_override_file: Some(PathBuf::from("/home/user/.conda-auth")),
732            extensions: TestExtension {
733                custom_field: Some("user-value".to_string()),
734                ..Default::default()
735            },
736            ..Default::default()
737        };
738
739        let project_config = TestConfig {
740            concurrency: crate::config::concurrency::ConcurrencyConfig {
741                solves: 4,
742                downloads: 8,
743            },
744            extensions: TestExtension {
745                numeric_field: Some(50),
746                bool_field: Some(true),
747                ..Default::default()
748            },
749            ..Default::default()
750        };
751
752        // Merge configs in order: base -> user -> project
753        let merged = base_config
754            .merge_config(&user_config)
755            .unwrap()
756            .merge_config(&project_config)
757            .unwrap();
758
759        // Project config takes priority for overlapping fields
760        assert_eq!(merged.concurrency.solves, 4);
761        assert_eq!(merged.concurrency.downloads, 8);
762
763        // User config values are preserved where not overridden
764        assert_eq!(
765            merged.default_channels.as_ref().map(std::vec::Vec::len),
766            Some(1)
767        );
768        assert_eq!(
769            merged.authentication_override_file,
770            Some(PathBuf::from("/home/user/.conda-auth"))
771        );
772
773        // Base config values are preserved where not overridden
774        assert_eq!(merged.tls_no_verify, Some(false));
775
776        // Extension values are merged properly
777        assert_eq!(
778            merged.extensions.custom_field,
779            Some("user-value".to_string())
780        );
781        assert_eq!(merged.extensions.numeric_field, Some(50));
782        assert_eq!(merged.extensions.bool_field, Some(true));
783    }
