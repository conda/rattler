1//! Authentication methods for the conda ecosystem
2use std::str::FromStr;
3
4use serde::{Deserialize, Serialize};
5
6/// The different Authentication methods that are supported in the conda
7/// ecosystem
8#[derive(Clone, PartialEq, Eq, Serialize, Deserialize, Debug)]
9pub enum Authentication {
10    /// A bearer token is sent as a header of the form
11    /// `Authorization: Bearer {TOKEN}`
12    BearerToken(String),
13    /// A basic authentication token is sent as HTTP basic auth
14    BasicHTTP {
15        /// The username to use for basic auth
16        username: String,
17        /// The password to use for basic auth
18        password: String,
19    },
20    /// A conda token is sent in the URL as `/t/{TOKEN}/...`
21    CondaToken(String),
22    /// S3 credentials
23    S3Credentials {
24        /// The access key ID to use for S3 authentication
25        access_key_id: String,
26        /// The secret access key to use for S3 authentication
27        secret_access_key: String,
28        /// The session token to use for S3 authentication
29        session_token: Option<String>,
30    },
31    /// OAuth/OIDC credentials with automatic token refresh support
32    OAuth {
33        /// The OAuth access token
34        access_token: String,
35        /// The OAuth refresh token (if available)
36        refresh_token: Option<String>,
37        /// Seconds since UNIX epoch when `access_token` expires
38        expires_at: Option<i64>,
39        /// Token endpoint URL (cached from OIDC discovery for refresh without
40        /// openidconnect)
41        token_endpoint: String,
42        /// Revocation endpoint URL (RFC 7009, for logout)
43        revocation_endpoint: Option<String>,
44        /// OAuth client ID
45        client_id: String,
46    },
47}
48
49/// An error that can occur when parsing an authentication string
50#[derive(Debug)]
51pub enum AuthenticationParseError {
52    /// The scheme is not valid
53    InvalidScheme,
54    /// The token could not be parsed
55    InvalidToken,
56}
57
58impl FromStr for Authentication {
59    type Err = AuthenticationParseError;
60
61    /// Parse an authentication string into an Authentication struct
62    fn from_str(s: &str) -> std::result::Result<Self, Self::Err> {
63        serde_json::from_str(s).map_err(|_err| AuthenticationParseError::InvalidToken)
64    }
65}
66
67impl Authentication {
68    /// Get the scheme of the authentication method
69    pub fn method(&self) -> &str {
70        match self {
71            Authentication::BearerToken(_) => "BearerToken",
72            Authentication::BasicHTTP { .. } => "BasicHTTP",
73            Authentication::CondaToken(_) => "CondaToken",
74            Authentication::S3Credentials { .. } => "S3",
75            Authentication::OAuth { .. } => "OAuth",
76        }
77    }
78}
