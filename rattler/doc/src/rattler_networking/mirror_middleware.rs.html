13#[derive(Debug, Clone, PartialEq, Eq, Hash)]
14/// Settings for the specific mirror (e.g. no zstd or bz2 support)
15pub struct Mirror {
16    /// The url of this mirror
17    pub url: Url,
18    /// Disable zstd support (for repodata.json.zst files)
19    pub no_zstd: bool,
20    /// Disable bz2 support (for repodata.json.bz2 files)
21    pub no_bz2: bool,
22    /// Allowed number of failures before the mirror is considered dead
23    pub max_failures: Option<usize>,
24}
25
26#[allow(dead_code)]
27struct MirrorState {
28    failures: AtomicUsize,
29    mirror: Mirror,
30}
31
32impl MirrorState {
33    pub fn add_failure(&self) {
34        self.failures.fetch_add(1, atomic::Ordering::Relaxed);
35    }
36}
37
38/// Middleware to handle mirrors
39pub struct MirrorMiddleware {
40    mirror_map: HashMap<Url, Vec<MirrorState>>,
41    sorted_keys: Vec<(String, Url)>,
42}
43
44impl MirrorMiddleware {
45    /// Create a new `MirrorMiddleware` from a map of mirrors
46    pub fn from_map(mirror_map: HashMap<Url, Vec<Mirror>>) -> Self {
47        let mirror_map: HashMap<Url, Vec<MirrorState>> = mirror_map
48            .into_iter()
49            .map(|(url, mirrors)| {
50                let mirrors = mirrors
51                    .into_iter()
52                    .map(|mirror| MirrorState {
53                        failures: AtomicUsize::new(0),
54                        mirror,
55                    })
56                    .collect();
57                (url, mirrors)
58            })
59            .collect();
60
61        let sorted_keys = mirror_map
62            .keys()
63            .cloned()
64            .sorted_by(|a, b| b.path().len().cmp(&a.path().len()))
65            .map(|k| (k.to_string(), k.clone()))
66            .collect::<Vec<(String, Url)>>();
67
68        Self {
69            mirror_map,
70            sorted_keys,
71        }
72    }
73
74    /// Get sorted keys. The keys are sorted by length of the path,
75    /// so the longest path comes first.
76    pub fn keys(&self) -> &[(String, Url)] {
77        &self.sorted_keys
78    }
79}

81fn select_mirror(mirrors: &[MirrorState]) -> Option<&MirrorState> {
82    let mut min_failures = usize::MAX;
83    let mut min_failures_index = usize::MAX;
84
85    for (i, mirror) in mirrors.iter().enumerate() {
86        let failures = mirror.failures.load(atomic::Ordering::Relaxed);
87        if failures < min_failures && mirror.mirror.max_failures.is_none_or(|max| failures < max) {
88            min_failures = failures;
89            min_failures_index = i;
90        }
91    }
92    if min_failures_index == usize::MAX {
93        return None;
94    }
95    Some(&mirrors[min_failures_index])
96}

98#[cfg_attr(not(target_arch = "wasm32"), async_trait::async_trait)]
99#[cfg_attr(target_arch = "wasm32", async_trait::async_trait(?Send))]
100impl Middleware for MirrorMiddleware {
101    async fn handle(
102        &self,
103        mut req: Request,
104        extensions: &mut Extensions,
105        next: Next<'_>,
106    ) -> Result<Response> {
107        let url_str = req.url().to_string();
108
109        for (key, url) in self.keys() {
110            if let Some(url_rest) = url_str.strip_prefix(key) {
111                let url_rest = url_rest.trim_start_matches('/');
112                // replace the key with the mirror
113                let mirrors = self.mirror_map.get(url).unwrap();
114                let selected_mirror = select_mirror(mirrors);
115
116                let Some(selected_mirror) = selected_mirror else {
117                    return Ok(create_404_response(req.url(), "All mirrors are dead"));
118                };
119
120                let mirror = &selected_mirror.mirror;
121                let selected_url = mirror.url.join(url_rest).unwrap();
122
123                // Short-circuit if the mirror does not support the file type
124                if url_rest.ends_with(".json.zst") && mirror.no_zstd {
125                    return Ok(create_404_response(
126                        &selected_url,
127                        "Mirror does not support zstd",
128                    ));
129                }
130                if url_rest.ends_with(".json.bz2") && mirror.no_bz2 {
131                    return Ok(create_404_response(
132                        &selected_url,
133                        "Mirror does not support bz2",
134                    ));
135                }
136
137                *req.url_mut() = selected_url;
138                let res = next.run(req, extensions).await;
139
140                // record a failure if the request failed so we can avoid the mirror in the future
141                match res.as_ref() {
142                    Ok(res) if res.status().is_server_error() => selected_mirror.add_failure(),
143                    Err(_) => selected_mirror.add_failure(),
144                    _ => {}
145                }
146
147                return res;
148            }
149        }
150
151        // if we don't have a mirror, we don't need to do anything
152        next.run(req, extensions).await
153    }
154}

156#[cfg(not(target_arch = "wasm32"))]
157pub(crate) fn create_404_response(url: &Url, body: &str) -> Response {
158    use reqwest::ResponseBuilderExt;
159    Response::from(
160        http::response::Builder::new()
161            .status(http::StatusCode::NOT_FOUND)
162            .url(url.clone())
163            .body(body.to_string())
164            .unwrap(),
165    )
166}

193    async fn test_server(name: &str, broken: bool) -> Url {
194        let state = String::from(name);
195
196        // Construct a router that returns data from the static dir but fails the first try.
197        let router = if broken {
198            Router::new().route("/count", get(broken_return))
199        } else {
200            Router::new().route("/count", get(count)).with_state(state)
201        };
202
203        let addr = SocketAddr::new([127, 0, 0, 1].into(), 0);
204        let listener = tokio::net::TcpListener::bind(&addr).await.unwrap();
205        let addr = listener.local_addr().unwrap();
206
207        let service = router.into_make_service();
208        tokio::spawn(axum::serve(listener, service).into_future());
209        format!("http://{}:{}", addr.ip(), addr.port())
210            .parse()
211            .unwrap()
212    }

276        let client = reqwest_middleware::ClientBuilder::new(reqwest::Client::new())
277            // retry middleware has to come before the mirror middleware
278            .with(RetryTransientMiddleware::new_with_policy(retry_policy))
279            .with(middleware)
280            .build();
281
282        let res = client.get("http://bla.com/count").send().await.unwrap();
283        assert!(res.status().is_success());
