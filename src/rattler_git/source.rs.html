24/// A remote Git source that can be checked out locally.
25pub struct GitSource {
26    /// The Git reference from the manifest file.
27    git: GitUrl,
28    /// The HTTP client to use for fetching.
29    client: ClientWithMiddleware,
30    /// The path to the Git source database.
31    cache: PathBuf,
32    /// The reporter to use for this source.
33    reporter: Option<Arc<dyn Reporter>>,
34}
35
36impl GitSource {
37    /// Initialize a new Git source.
38    pub fn new(
39        git: GitUrl,
40        client: impl Into<ClientWithMiddleware>,
41        cache: impl Into<PathBuf>,
42    ) -> Self {
43        Self {
44            git,
45            client: client.into(),
46            cache: cache.into(),
47            reporter: None,
48        }
49    }
50
51    /// Set the [`Reporter`] to use for the [`GitSource`].
52    #[must_use]
53    pub fn with_reporter(self, reporter: Arc<dyn Reporter>) -> Self {
54        Self {
55            reporter: Some(reporter),
56            ..self
57        }
58    }
59
60    /// Fetch the underlying Git repository at the given revision.
61    #[instrument(skip(self), fields(repository = %self.git.repository, rev = self.git.precise.map(tracing::field::display)))]
62    pub fn fetch(self) -> Result<Fetch, GitError> {
63        // Compute the canonical URL for the repository.
64        let canonical = RepositoryUrl::new(&self.git.repository);
65
66        // The path to the repo, within the Git database.
67        let ident = cache_digest(&canonical);
68        let db_path = self.cache.join("db").join(&ident);
69
70        // Authenticate the URL, if necessary.
71        let remote = if let Some(credentials) = GIT_STORE.get(&canonical) {
72            Cow::Owned(credentials.apply(self.git.repository.clone()))
73        } else {
74            Cow::Borrowed(&self.git.repository)
75        };
76
77        let remote = GitRemote::new(&remote);
78
79        // Try to open the existing database, logging a warning if it's corrupted
80        let existing_db = match remote.db_at(&db_path) {
81            Ok(db) => Some(db),
82            Err(GitError::InvalidRepository(path)) => {
83                tracing::warn!(
84                    "Detected corrupted git cache at {} (not a valid git repository), removing and re-cloning",
85                    path.display()
86                );
87                None
88            }
89            Err(_) => None,
90        };
91
92        let (db, actual_rev, task) = match (self.git.precise, existing_db) {
93            // If we have a locked revision, and we have a preexisting database
94            // which has that revision, then no update needs to happen.
95            (Some(rev), Some(db)) if db.contains(rev.into()) => {
96                tracing::debug!(
97                    "Using existing Git source `{}` pointed at `{}`",
98                    self.git.repository,
99                    rev
100                );
101                (db, rev, None)
102            }
103
104            // ... otherwise we use this state to update the git database. Note
105            // that we still check for being offline here, for example in the
106            // situation that we have a locked revision but the database
107            // doesn't have it.
108            (locked_rev, db) => {
109                tracing::debug!("Updating Git source `{}`", self.git.repository);
110
111                // Report the checkout operation to the reporter.
112                let task = self.reporter.as_ref().map(|reporter| {
113                    reporter.on_checkout_start(remote.url(), self.git.reference.as_rev())
114                });
115
116                let (db, actual_rev) = remote.checkout(
117                    &db_path,
118                    db,
119                    &self.git.reference,
120                    locked_rev.map(GitOid::from),
121                    &self.client,
122                )?;
123
124                (db, GitSha::from(actual_rev), task)
125            }
126        };
127
128        // Don’t use the full hash, in order to contribute less to reaching the
129        // path length limit on Windows.
130        let short_id = db.to_short_id(actual_rev.into())?;
131
132        // Check out `actual_rev` from the database to a scoped location on the
133        // filesystem. This will use hard links and such to ideally make the
134        // checkout operation here pretty fast.
135        let checkout_path = self
136            .cache
137            .join("checkouts")
138            .join(&ident)
139            .join(short_id.as_str());
140
141        tracing::debug!(
142            "Copying git revision `{}` to path `{}`",
143            actual_rev,
144            checkout_path.display()
145        );
146        db.copy_to(actual_rev.into(), &checkout_path)?;
147
148        // Report the checkout operation to the reporter.
149        if let (Some(task), Some(reporter)) = (task, self.reporter.as_ref()) {
150            reporter.on_checkout_complete(remote.url(), short_id.as_str(), task);
151        }
152
153        tracing::trace!("Finished fetching Git source `{}`", self.git.repository);
154
155        Ok(Fetch {
156            repository: RepositoryReference {
157                url: canonical,
158                reference: self.git.reference.clone(),
159            },
160            commit: actual_rev,
161            path: checkout_path,
162        })
163    }
164}

166#[derive(Debug, Clone)]
167pub struct Fetch {
168    /// The [`RepositoryReference`] reference that was fetched.
169    repository: RepositoryReference,
170
171    /// The precise git checkout
172    commit: GitSha,
173
174    /// The path to the checked-out repository.
175    path: PathBuf,
176}
177
178impl Fetch {
179    pub fn repository(&self) -> &RepositoryReference {
180        &self.repository
181    }
182
183    pub fn commit(&self) -> GitSha {
184        self.commit
185    }
186
187    pub fn path(&self) -> &Path {
188        &self.path
189    }
190
191    pub fn into_path(self) -> PathBuf {
192        self.path
193    }
194}

196pub fn cache_digest(url: &RepositoryUrl) -> String {
197    let mut hasher = DefaultHasher::new();
198    url.hash(&mut hasher);
199    let hash = hasher.finish();
200    format!("{hash:x}")
201}
