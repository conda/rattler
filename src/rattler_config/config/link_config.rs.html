7/// Knobs for the link strategy used during package installation.
8///
9/// All fields are `Option<bool>` so that "unset" is distinguishable
10/// from an explicit `true`/`false`. When unset, the package installer
11/// uses its own default (typically: try every strategy, fall back as
12/// needed).
13///
14/// Currently exposed as a standalone module — **not** wired into
15/// `ConfigBase`. The intended embedding (`#[serde(flatten)]` so the
16/// keys live at the top level of the TOML) interacts poorly with
17/// `serde_ignored`: unknown top-level keys are no longer reported
18/// when any sibling field uses `flatten`. Downstream consumers that
19/// want typo detection on these keys should either embed under a
20/// dedicated table (`[link-config]`) or keep equivalent flat fields
21/// locally for now.
22#[derive(Clone, Debug, Default, Deserialize, Serialize, PartialEq, Eq)]
23#[serde(rename_all = "kebab-case")]
24pub struct LinkConfig {
25    /// If set to false, symbolic links will not be used during package
26    /// installation.
27    #[serde(default)]
28    #[serde(skip_serializing_if = "Option::is_none")]
29    pub allow_symbolic_links: Option<bool>,
30
31    /// If set to false, hard links will not be used during package
32    /// installation.
33    #[serde(default)]
34    #[serde(skip_serializing_if = "Option::is_none")]
35    pub allow_hard_links: Option<bool>,
36
37    /// If set to false, ref links (copy-on-write) will not be used
38    /// during package installation.
39    #[serde(default)]
40    #[serde(skip_serializing_if = "Option::is_none")]
41    pub allow_ref_links: Option<bool>,
42}
43
44impl LinkConfig {
45    /// Returns `true` when no link-strategy override is set. Useful for
46    /// `#[serde(skip_serializing_if = "...")]` on the field.
47    pub fn is_empty(&self) -> bool {
48        self.allow_symbolic_links.is_none()
49            && self.allow_hard_links.is_none()
50            && self.allow_ref_links.is_none()
51    }
52}
53
54impl Config for LinkConfig {
55    fn get_extension_name(&self) -> String {
56        "link".to_string()
57    }
58
59    fn merge_config(self, other: &Self) -> Result<Self, MergeError> {
60        Ok(Self {
61            allow_symbolic_links: other.allow_symbolic_links.or(self.allow_symbolic_links),
62            allow_hard_links: other.allow_hard_links.or(self.allow_hard_links),
63            allow_ref_links: other.allow_ref_links.or(self.allow_ref_links),
64        })
65    }
66
67    fn validate(&self) -> Result<(), ValidationError> {
68        Ok(())
69    }
70
71    fn keys(&self) -> Vec<String> {
72        vec![
73            "allow-symbolic-links".to_string(),
74            "allow-hard-links".to_string(),
75            "allow-ref-links".to_string(),
76        ]
77    }
78
79    #[cfg(feature = "edit")]
80    fn set(&mut self, key: &str, value: Option<String>) -> Result<(), ConfigEditError> {
81        let parse = |v: String| {
82            v.parse::<bool>()
83                .map_err(|e| ConfigEditError::BoolParseError {
84                    key: key.to_string(),
85                    source: e,
86                })
87        };
88        match key {
89            "allow-symbolic-links" => {
90                self.allow_symbolic_links = value.map(parse).transpose()?;
91            }
92            "allow-hard-links" => {
93                self.allow_hard_links = value.map(parse).transpose()?;
94            }
95            "allow-ref-links" => {
96                self.allow_ref_links = value.map(parse).transpose()?;
97            }
98            _ => {
99                return Err(ConfigEditError::UnknownKeyInner {
100                    key: key.to_string(),
101                });
102            }
103        }
104        Ok(())
105    }
106}

108#[cfg(test)]
109mod tests {
110    use super::*;
111
112    #[test]
113    fn default_is_empty() {
114        assert!(LinkConfig::default().is_empty());
115    }
116
117    #[test]
118    fn merge_other_wins() {
119        let base = LinkConfig {
120            allow_symbolic_links: Some(true),
121            allow_hard_links: None,
122            allow_ref_links: Some(false),
123        };
124        let override_ = LinkConfig {
125            allow_symbolic_links: Some(false),
126            allow_hard_links: Some(true),
127            allow_ref_links: None,
128        };
129        let merged = base.merge_config(&override_).unwrap();
130        assert_eq!(merged.allow_symbolic_links, Some(false));
131        assert_eq!(merged.allow_hard_links, Some(true));
132        assert_eq!(merged.allow_ref_links, Some(false));
133    }
134
135    #[test]
136    fn deserialize_kebab_case() {
137        let toml = r#"
138            allow-symbolic-links = true
139            allow-hard-links = false
140            allow-ref-links = true
141        "#;
142        let config: LinkConfig = toml::from_str(toml).unwrap();
143        assert_eq!(config.allow_symbolic_links, Some(true));
144        assert_eq!(config.allow_hard_links, Some(false));
145        assert_eq!(config.allow_ref_links, Some(true));
146    }
