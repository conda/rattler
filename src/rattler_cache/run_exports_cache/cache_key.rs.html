1use rattler_conda_types::{package::CondaArchiveIdentifier, PackageRecord};
2use rattler_digest::{Md5Hash, Sha256Hash};
3use std::fmt::{Display, Formatter};
4
5/// Provides a unique identifier for packages in the cache.
6#[derive(Debug, Hash, Clone, Eq, PartialEq)]
7pub struct CacheKey {
8    pub(crate) name: String,
9    pub(crate) version: String,
10    pub(crate) build_string: String,
11    pub(crate) sha256: Option<Sha256Hash>,
12    pub(crate) md5: Option<Md5Hash>,
13    pub(crate) extension: String,
14}
15
16impl CacheKey {
17    /// Potentially adds a sha256 hash of the archive.
18    pub fn with_opt_sha256(mut self, sha256: Option<Sha256Hash>) -> Self {
19        self.sha256 = sha256;
20        self
21    }
22
23    /// Potentially adds a md5 hash of the archive.
24    pub fn with_opt_md5(mut self, md5: Option<Md5Hash>) -> Self {
25        self.md5 = md5;
26        self
27    }
28}
29
30impl CacheKey {
31    /// Return the sha256 hash of the package if it is known.
32    pub fn sha256(&self) -> Option<Sha256Hash> {
33        self.sha256
34    }
35
36    /// Return the md5 hash of the package if it is known.
37    pub fn md5(&self) -> Option<Md5Hash> {
38        self.md5
39    }
40
41    /// Return the sha256 hash string of the package if it is known.
42    pub fn sha256_str(&self) -> String {
43        self.sha256()
44            .map(|hash| format!("{hash:x}"))
45            .unwrap_or_default()
46    }
47
48    /// Try to create a new cache key from a package record and a filename.
49    pub fn create(record: &PackageRecord, filename: &str) -> Result<Self, CacheKeyError> {
50        let archive_identifier = CondaArchiveIdentifier::try_from_filename(filename)
51            .ok_or_else(|| CacheKeyError::InvalidArchiveIdentifier(filename.to_string()))?;
52
53        Ok(Self {
54            name: record.name.as_normalized().to_string(),
55            version: record.version.to_string(),
56            build_string: record.build.clone(),
57            sha256: record.sha256,
58            md5: record.md5,
59            extension: archive_identifier.archive_type.extension().to_string(),
60        })
61    }
62}
63
64#[derive(Debug, thiserror::Error)]
65pub enum CacheKeyError {
66    #[error("could not identify the archive type from the name: {0}")]
67    InvalidArchiveIdentifier(String),
68}
69
70impl Display for CacheKey {
71    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
72        // we need to use either sha256 or md5 hash to display the key
73        // if both are none, we ignore them
74        let display_key = match (self.sha256(), self.md5()) {
75            (Some(sha256), _) => format!("-{sha256:x}"),
76            (_, Some(md5)) => format!("-{md5:x}"),
77            _ => "".to_string(),
78        };
79
80        write!(
81            f,
82            "{}-{}-{}{}{}",
83            &self.name, &self.version, &self.build_string, display_key, self.extension
84        )
85    }
86}
