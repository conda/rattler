5/// Provides a unique identifier for packages in the cache.
6#[derive(Debug, Hash, Clone, Eq, PartialEq)]
7pub struct CacheKey {
8    pub(crate) name: String,
9    pub(crate) version: String,
10    pub(crate) build_string: String,
11    pub(crate) sha256: Option<Sha256Hash>,
12    pub(crate) md5: Option<Md5Hash>,
13    pub(crate) extension: String,
14}
15
16impl CacheKey {
17    /// Potentially adds a sha256 hash of the archive.
18    pub fn with_opt_sha256(mut self, sha256: Option<Sha256Hash>) -> Self {
19        self.sha256 = sha256;
20        self
21    }
22
23    /// Potentially adds a md5 hash of the archive.
24    pub fn with_opt_md5(mut self, md5: Option<Md5Hash>) -> Self {
25        self.md5 = md5;
26        self
27    }
28}
29
30impl CacheKey {
31    /// Return the sha256 hash of the package if it is known.
32    pub fn sha256(&self) -> Option<Sha256Hash> {
33        self.sha256
34    }
35
36    /// Return the md5 hash of the package if it is known.
37    pub fn md5(&self) -> Option<Md5Hash> {
38        self.md5
39    }
40
41    /// Return the sha256 hash string of the package if it is known.
42    pub fn sha256_str(&self) -> String {
43        self.sha256().map(hex::encode).unwrap_or_default()
44    }
45
46    /// Renders the cache key as a file name, rejecting metadata-derived
47    /// components that could escape the cache root (GHSA-h672-p7h7-97v9).
48    ///
49    /// This is intentionally the only way to turn a key into a path: there is no
50    /// `Display`/`to_string` that could bypass the check.
51    pub(crate) fn to_path_segment(&self) -> Result<String, InvalidPathComponentError> {
52        // We use either the sha256 or md5 hash to disambiguate the key; if both
53        // are absent we omit it.
54        let hash = match (self.sha256(), self.md5()) {
55            (Some(sha256), _) => format!("-{}", hex::encode(sha256)),
56            (_, Some(md5)) => format!("-{}", hex::encode(md5)),
57            _ => String::new(),
58        };
59        let segment = format!(
60            "{}-{}-{}{}{}",
61            &self.name, &self.version, &self.build_string, hash, self.extension
62        );
63        ensure_safe_path_component(&segment)?;
64        Ok(segment)
65    }
66
67    /// Try to create a new cache key from a package record and a filename.
68    pub fn create(record: &PackageRecord, filename: &str) -> Result<Self, CacheKeyError> {
69        let archive_identifier = CondaArchiveIdentifier::try_from_filename(filename)
70            .ok_or_else(|| CacheKeyError::InvalidArchiveIdentifier(filename.to_string()))?;
71
72        Ok(Self {
73            name: record.name.as_normalized().to_string(),
74            version: record.version.to_string(),
75            build_string: record.build.clone(),
76            sha256: record.sha256,
77            md5: record.md5,
78            extension: archive_identifier.archive_type.extension().to_string(),
79        })
80    }
81}
82
83#[derive(Debug, thiserror::Error)]
84pub enum CacheKeyError {
85    #[error("could not identify the archive type from the name: {0}")]
86    InvalidArchiveIdentifier(String),
87}

90mod tests {
91    use super::CacheKey;
92    use rattler_conda_types::{PackageName, PackageRecord, VersionWithSource};

94    #[test]
95    fn to_path_segment_rejects_path_traversal() {
96        let record = PackageRecord::new(
97            PackageName::new_unchecked("demo"),
98            "1.0".parse::<VersionWithSource>().unwrap(),
99            r"x\..\..\..\project\.git\hooks".to_string(),
100        );
101        let key = CacheKey::create(&record, "demo-1.0-0.tar.bz2").unwrap();
102        assert!(key.to_path_segment().is_err());
103    }
104
105    #[test]
106    fn to_path_segment_accepts_well_formed_key() {
107        let record = PackageRecord::new(
108            PackageName::new_unchecked("demo"),
109            "1.0".parse::<VersionWithSource>().unwrap(),
110            "py39h6fdeb60_14".to_string(),
111        );
112        let key = CacheKey::create(&record, "demo-1.0-0.tar.bz2").unwrap();
113        assert!(key.to_path_segment().is_ok());
114    }
