30#[cfg(target_os = "netbsd")]
31/// NetBSD has `ptsname_r` in libc but apparently the `nix` crate does not expose it for NetBSD.
32/// Call `libc::ptsname_r` directly.
33fn ptsname_r(fd: &PtyMaster) -> nix::Result<String> {
34    use std::ffi::CStr;
35
36    let mut buf: [libc::c_char; 128] = [0; 128];
37
38    unsafe {
39        match libc::ptsname_r(fd.as_raw_fd(), buf.as_mut_ptr(), buf.len()) {
40            0 => Ok(CStr::from_ptr(buf.as_ptr()).to_string_lossy().into_owned()),
41            _ => Err(nix::Error::last()),
42        }
43    }
44}

56#[cfg(target_os = "macos")]
57/// `ptsname_r` is a linux extension but ptsname isn't thread-safe
58/// instead of using a static mutex this calls `ioctl` with `TIOCPTYGNAME` directly
59/// <https://blog.tarq.io/ptsname-on-osx-with-rust>/
60fn ptsname_r(fd: &PtyMaster) -> nix::Result<String> {
61    use nix::libc::{TIOCPTYGNAME, ioctl};
62    use std::ffi::CStr;
63
64    // the buffer size on OSX is 128, defined by sys/ttycom.h
65    let mut buf: [i8; 128] = [0; 128];
66
67    unsafe {
68        match ioctl(fd.as_raw_fd(), u64::from(TIOCPTYGNAME), &mut buf) {
69            0 => {
70                let res = CStr::from_ptr(buf.as_ptr()).to_string_lossy().into_owned();
71                Ok(res)
72            }
73            _ => Err(nix::Error::last()),
74        }
75    }
76}

78#[cfg(target_os = "freebsd")]
79/// FreeBSD implementation using FIODGNAME ioctl to get the slave pty name
80fn ptsname_r(fd: &PtyMaster) -> nix::Result<String> {
81    use nix::libc::ioctl;
82    use std::ffi::CStr;
83
84    // FreeBSD uses FIODGNAME ioctl with a fiodgname_arg struct
85    #[repr(C)]
86    struct fiodgname_arg {
87        len: libc::c_int,
88        buf: *mut libc::c_char,
89    }
90
91    // FIODGNAME is defined as _IOW('f', 120, struct fiodgname_arg) in FreeBSD
92    // _IOW(g,n,t) = (0x80000000 | ((sizeof(t) & 0x1fff) << 16) | ((g) << 8) | (n))
93    // sizeof(fiodgname_arg) = 16 on 64-bit (4 bytes int + 4 padding + 8 bytes pointer)
94    const FIODGNAME: libc::c_ulong = 0x80106678;
95
96    let mut buf: [libc::c_char; 128] = [0; 128];
97    let mut arg = fiodgname_arg {
98        len: buf.len() as libc::c_int,
99        buf: buf.as_mut_ptr(),
100    };
101
102    unsafe {
103        match ioctl(fd.as_raw_fd(), FIODGNAME, &mut arg) {
104            0 => {
105                let res = CStr::from_ptr(buf.as_ptr()).to_string_lossy().into_owned();
106                // FreeBSD returns just the device name (e.g., "pts/0"), prepend /dev/
107                Ok(format!("/dev/{}", res))
108            }
109            _ => Err(nix::Error::last()),
110        }
111    }
112}

120impl PtyProcess {
121    /// Start a process in a forked pty
122    pub fn new(mut command: Command, opts: PtyProcessOptions) -> nix::Result<Self> {
123        // Open a new PTY master
124        let master_fd = posix_openpt(OFlag::O_RDWR)?;
125
126        // Allow a slave to be generated for it
127        grantpt(&master_fd)?;
128        unlockpt(&master_fd)?;
129
130        let slave_name = ptsname_r(&master_fd)?;
131
132        // Get the current window size if it was not specified
133        let window_size = opts.window_size.unwrap_or_else(|| {
134            // find current window size with ioctl
135            let mut size: libc::winsize = unsafe { std::mem::zeroed() };
136            // Query the terminal dimensions
137            unsafe { libc::ioctl(io::stdout().as_raw_fd(), libc::TIOCGWINSZ, &mut size) };
138            size
139        });
140
141        match unsafe { fork()? } {
142            ForkResult::Child => {
143                // Avoid leaking master fd
144                close(master_fd.as_raw_fd())?;
145
146                setsid()?; // create new session with child as session leader
147                let slave_fd = open(
148                    std::path::Path::new(&slave_name),
149                    OFlag::O_RDWR,
150                    stat::Mode::empty(),
151                )?;
152
153                // assign stdin, stdout, stderr to the tty, just like a terminal does
154                dup2_stdin(&slave_fd)?;
155                dup2_stdout(&slave_fd)?;
156                dup2_stderr(&slave_fd)?;
157
158                // Avoid leaking slave fd
159                if slave_fd.as_raw_fd() > STDERR_FILENO {
160                    close(slave_fd)?;
161                }
162
163                // Set `echo` and `window_size` for the pty
164                set_echo(io::stdin(), opts.echo)?;
165                set_window_size(io::stdout().as_raw_fd(), window_size)?;
166
167                let _ = command.exec();
168                Err(nix::Error::last())
169            }
170            ForkResult::Parent { child: child_pid } => Ok(PtyProcess {
171                pty: master_fd,
172                child_pid,
173                kill_timeout: None,
174            }),
175        }
176    }

185    /// Get status of child process, non-blocking.
186    ///
187    /// This method runs waitpid on the process.
188    /// This means: If you ran `exit()` before or `status()` this method will

195    pub fn exit(&mut self) -> nix::Result<wait::WaitStatus> {

197    }

199    /// Kill the process with a specific signal. This method blocks, until the process is dead
200    ///
201    /// repeatedly sends SIGTERM to the process until it died,
202    /// the pty session is closed upon dropping `PtyMaster`,
203    /// so we don't need to explicitly do that here.
204    ///
205    /// if `kill_timeout` is set and a repeated sending of signal does not result in the process
206    /// being killed, then `kill -9` is sent after the `kill_timeout` duration has elapsed.
207    pub fn kill(&mut self, sig: signal::Signal) -> nix::Result<wait::WaitStatus> {
208        let start = time::Instant::now();
209        loop {
210            match signal::kill(self.child_pid, sig) {
211                Ok(_) => {}
212                // process was already killed before -> ignore
213                Err(nix::errno::Errno::ESRCH) => {
214                    return Ok(wait::WaitStatus::Exited(Pid::from_raw(0), 0));
215                }
216                Err(e) => return Err(e),
217            }
218
219            match self.status() {
220                Some(status) if status != wait::WaitStatus::StillAlive => return Ok(status),
221                Some(_) | None => thread::sleep(time::Duration::from_millis(100)),
222            }
223            // kill -9 if timeout is reached
224            if let Some(timeout) = self.kill_timeout
225                && start.elapsed() > timeout
226            {
227                signal::kill(self.child_pid, signal::Signal::SIGKILL)?;
228            }
229        }
230    }

232    /// Set raw mode on stdin and return the original mode
233    pub fn set_raw(&self) -> nix::Result<Termios> {
234        let original_mode = termios::tcgetattr(io::stdin())?;
235        let mut raw_mode = original_mode.clone();
236        raw_mode.input_flags.remove(
237            InputFlags::BRKINT
238                | InputFlags::ICRNL
239                | InputFlags::INPCK
240                | InputFlags::ISTRIP
241                | InputFlags::IXON,
242        );
243        raw_mode.output_flags.remove(termios::OutputFlags::OPOST);
244        raw_mode
245            .control_flags
246            .remove(termios::ControlFlags::CSIZE | termios::ControlFlags::PARENB);
247        raw_mode.control_flags.insert(termios::ControlFlags::CS8);
248        raw_mode.local_flags.remove(
249            termios::LocalFlags::ECHO
250                | termios::LocalFlags::ICANON
251                | termios::LocalFlags::IEXTEN
252                | termios::LocalFlags::ISIG,
253        );
254
255        raw_mode.control_chars[termios::SpecialCharacterIndices::VMIN as usize] = 1;
256        raw_mode.control_chars[termios::SpecialCharacterIndices::VTIME as usize] = 0;
257
258        termios::tcsetattr(io::stdin(), termios::SetArg::TCSAFLUSH, &raw_mode)?;
259
260        Ok(original_mode)
261    }

282    pub async fn async_read(&self, buf: &mut [u8]) -> io::Result<usize> {
283        let fd = dup(&self.pty).map_err(|e| io::Error::from_raw_os_error(e as i32))?;
284        let mut file = unsafe { tokio::fs::File::from_raw_fd(fd.into_raw_fd()) };
285        file.read(buf).await

294    pub async fn async_wait(&mut self) -> nix::Result<wait::WaitStatus> {
295        loop {
296            match self.status() {
297                Some(status) if status != wait::WaitStatus::StillAlive => return Ok(status),
298                Some(_) | None => sleep(Duration::from_millis(100)).await,
299            }
300        }
301    }

303    pub async fn async_exit(&mut self) -> nix::Result<wait::WaitStatus> {
304        self.async_kill(signal::SIGTERM).await

339pub fn set_echo<Fd: AsFd>(fd: Fd, echo: bool) -> nix::Result<()> {
340    let mut flags = termios::tcgetattr(&fd)?;
341    if echo {
342        flags.local_flags.insert(termios::LocalFlags::ECHO);
343    } else {
344        flags.local_flags.remove(termios::LocalFlags::ECHO);
345    }
346    termios::tcsetattr(&fd, termios::SetArg::TCSANOW, &flags)?;
347    Ok(())
348}

350impl Drop for PtyProcess {
351    fn drop(&mut self) {
352        if let Some(wait::WaitStatus::StillAlive) = self.status() {
353            self.exit().expect("cannot exit");
354        }
355    }
