41/// `ptsname_r` is a linux extension but ptsname isn't thread-safe
42/// instead of using a static mutex this calls `ioctl` with `TIOCPTYGNAME` directly
43/// <https://blog.tarq.io/ptsname-on-osx-with-rust>/
44fn ptsname_r(fd: &PtyMaster) -> nix::Result<String> {
45    use nix::libc::{ioctl, TIOCPTYGNAME};
46    use std::ffi::CStr;
47
48    // the buffer size on OSX is 128, defined by sys/ttycom.h
49    let mut buf: [i8; 128] = [0; 128];
50
51    unsafe {
52        match ioctl(fd.as_raw_fd(), u64::from(TIOCPTYGNAME), &mut buf) {
53            0 => {
54                let res = CStr::from_ptr(buf.as_ptr()).to_string_lossy().into_owned();
55                Ok(res)
56            }
57            _ => Err(nix::Error::last()),
58        }
59    }
60}

62#[cfg(target_os = "freebsd")]
63/// FreeBSD implementation using FIODGNAME ioctl to get the slave pty name
64fn ptsname_r(fd: &PtyMaster) -> nix::Result<String> {
65    use nix::libc::ioctl;
66    use std::ffi::CStr;
67
68    // FreeBSD uses FIODGNAME ioctl with a fiodgname_arg struct
69    #[repr(C)]
70    struct fiodgname_arg {
71        len: libc::c_int,
72        buf: *mut libc::c_char,
73    }
74
75    // FIODGNAME is defined as _IOW('f', 120, struct fiodgname_arg) in FreeBSD
76    // _IOW(g,n,t) = (0x80000000 | ((sizeof(t) & 0x1fff) << 16) | ((g) << 8) | (n))
77    // sizeof(fiodgname_arg) = 16 on 64-bit (4 bytes int + 4 padding + 8 bytes pointer)
78    const FIODGNAME: libc::c_ulong = 0x80106678;
79
80    let mut buf: [libc::c_char; 128] = [0; 128];
81    let mut arg = fiodgname_arg {
82        len: buf.len() as libc::c_int,
83        buf: buf.as_mut_ptr(),
84    };
85
86    unsafe {
87        match ioctl(fd.as_raw_fd(), FIODGNAME, &mut arg) {
88            0 => {
89                let res = CStr::from_ptr(buf.as_ptr()).to_string_lossy().into_owned();
90                // FreeBSD returns just the device name (e.g., "pts/0"), prepend /dev/
91                Ok(format!("/dev/{}", res))
92            }
93            _ => Err(nix::Error::last()),
94        }
95    }
96}

105    /// Start a process in a forked pty
106    pub fn new(mut command: Command, opts: PtyProcessOptions) -> nix::Result<Self> {
107        // Open a new PTY master
108        let master_fd = posix_openpt(OFlag::O_RDWR)?;
109
110        // Allow a slave to be generated for it
111        grantpt(&master_fd)?;
112        unlockpt(&master_fd)?;
113
114        // on Linux this is the libc function, on OSX this is our implementation of ptsname_r
115        let slave_name = ptsname_r(&master_fd)?;
116
117        // Get the current window size if it was not specified
118        let window_size = opts.window_size.unwrap_or_else(|| {
119            // find current window size with ioctl
120            let mut size: libc::winsize = unsafe { std::mem::zeroed() };
121            // Query the terminal dimensions
122            unsafe { libc::ioctl(io::stdout().as_raw_fd(), libc::TIOCGWINSZ, &mut size) };
123            size
124        });
125
126        match unsafe { fork()? } {
127            ForkResult::Child => {
128                // Avoid leaking master fd
129                close(master_fd.as_raw_fd())?;
130
131                setsid()?; // create new session with child as session leader
132                let slave_fd = open(
133                    std::path::Path::new(&slave_name),
134                    OFlag::O_RDWR,
135                    stat::Mode::empty(),
136                )?;
137
138                // assign stdin, stdout, stderr to the tty, just like a terminal does
139                dup2_stdin(&slave_fd)?;
140                dup2_stdout(&slave_fd)?;
141                dup2_stderr(&slave_fd)?;
142
143                // Avoid leaking slave fd
144                if slave_fd.as_raw_fd() > STDERR_FILENO {
145                    close(slave_fd)?;
146                }
147
148                // Set `echo` and `window_size` for the pty
149                set_echo(io::stdin(), opts.echo)?;
150                set_window_size(io::stdout().as_raw_fd(), window_size)?;
151
152                let _ = command.exec();
153                Err(nix::Error::last())
154            }
155            ForkResult::Parent { child: child_pid } => Ok(PtyProcess {
156                pty: master_fd,
157                child_pid,
158                kill_timeout: None,
159            }),
160        }
161    }

170    /// Get status of child process, non-blocking.
171    ///
172    /// This method runs waitpid on the process.
173    /// This means: If you ran `exit()` before or `status()` this method will
174    /// return `None`
175    pub fn status(&self) -> Option<wait::WaitStatus> {

177    }

179    /// Regularly exit the process, this method is blocking until the process is dead
180    pub fn exit(&mut self) -> nix::Result<wait::WaitStatus> {
181        self.kill(signal::SIGTERM)
182    }

184    /// Kill the process with a specific signal. This method blocks, until the process is dead
185    ///
186    /// repeatedly sends SIGTERM to the process until it died,
187    /// the pty session is closed upon dropping `PtyMaster`,
188    /// so we don't need to explicitly do that here.
189    ///
190    /// if `kill_timeout` is set and a repeated sending of signal does not result in the process
191    /// being killed, then `kill -9` is sent after the `kill_timeout` duration has elapsed.
192    pub fn kill(&mut self, sig: signal::Signal) -> nix::Result<wait::WaitStatus> {
193        let start = time::Instant::now();
194        loop {
195            match signal::kill(self.child_pid, sig) {
196                Ok(_) => {}
197                // process was already killed before -> ignore
198                Err(nix::errno::Errno::ESRCH) => {
199                    return Ok(wait::WaitStatus::Exited(Pid::from_raw(0), 0));
200                }
201                Err(e) => return Err(e),
202            }
203
204            match self.status() {
205                Some(status) if status != wait::WaitStatus::StillAlive => return Ok(status),
206                Some(_) | None => thread::sleep(time::Duration::from_millis(100)),
207            }
208            // kill -9 if timeout is reached
209            if let Some(timeout) = self.kill_timeout {
210                if start.elapsed() > timeout {
211                    signal::kill(self.child_pid, signal::Signal::SIGKILL)?;
212                }
213            }
214        }
215    }

217    /// Set raw mode on stdin and return the original mode
218    pub fn set_raw(&self) -> nix::Result<Termios> {
219        let original_mode = termios::tcgetattr(io::stdin())?;
220        let mut raw_mode = original_mode.clone();
221        raw_mode.input_flags.remove(
222            InputFlags::BRKINT
223                | InputFlags::ICRNL
224                | InputFlags::INPCK
225                | InputFlags::ISTRIP
226                | InputFlags::IXON,
227        );
228        raw_mode.output_flags.remove(termios::OutputFlags::OPOST);
229        raw_mode
230            .control_flags
231            .remove(termios::ControlFlags::CSIZE | termios::ControlFlags::PARENB);
232        raw_mode.control_flags.insert(termios::ControlFlags::CS8);
233        raw_mode.local_flags.remove(
234            termios::LocalFlags::ECHO
235                | termios::LocalFlags::ICANON
236                | termios::LocalFlags::IEXTEN
237                | termios::LocalFlags::ISIG,
238        );
239
240        raw_mode.control_chars[termios::SpecialCharacterIndices::VMIN as usize] = 1;
241        raw_mode.control_chars[termios::SpecialCharacterIndices::VTIME as usize] = 0;
242
243        termios::tcsetattr(io::stdin(), termios::SetArg::TCSAFLUSH, &raw_mode)?;
244
245        Ok(original_mode)
246    }

267    pub async fn async_read(&self, buf: &mut [u8]) -> io::Result<usize> {
268        let fd = dup(&self.pty).map_err(|e| io::Error::from_raw_os_error(e as i32))?;
269        let mut file = unsafe { tokio::fs::File::from_raw_fd(fd.into_raw_fd()) };
270        file.read(buf).await

279    pub async fn async_wait(&mut self) -> nix::Result<wait::WaitStatus> {
280        loop {
281            match self.status() {
282                Some(status) if status != wait::WaitStatus::StillAlive => return Ok(status),
283                Some(_) | None => sleep(Duration::from_millis(100)).await,
284            }
285        }
286    }

288    pub async fn async_exit(&mut self) -> nix::Result<wait::WaitStatus> {
289        self.async_kill(signal::SIGTERM).await
290    }

293        let start = Instant::now();
294        loop {
295            match signal::kill(self.child_pid, sig) {
296                Ok(_) => {}
297                // process was already killed before -> ignore
298                Err(nix::errno::Errno::ESRCH) => {
299                    return Ok(wait::WaitStatus::Exited(Pid::from_raw(0), 0));
300                }
301                Err(e) => return Err(e),
302            }
303
304            match self.status() {
305                Some(status) if status != wait::WaitStatus::StillAlive => return Ok(status),
306                Some(_) | None => sleep(Duration::from_millis(100)).await,
307            }
308
309            // kill -9 if timeout is reached
310            if let Some(timeout) = self.kill_timeout {

313                }

324pub fn set_echo<Fd: AsFd>(fd: Fd, echo: bool) -> nix::Result<()> {
325    let mut flags = termios::tcgetattr(&fd)?;
326    if echo {
327        flags.local_flags.insert(termios::LocalFlags::ECHO);
328    } else {
329        flags.local_flags.remove(termios::LocalFlags::ECHO);
330    }
331    termios::tcsetattr(&fd, termios::SetArg::TCSANOW, &flags)?;
332    Ok(())
333}

335impl Drop for PtyProcess {
336    fn drop(&mut self) {
337        if let Some(wait::WaitStatus::StillAlive) = self.status() {
338            self.exit().expect("cannot exit");
339        }
340    }
