3/// Log the output of a command to terminal
4pub fn log_output(cmd_info: &str, output: std::process::Output) {
5    tracing::info!("{}: status {}", cmd_info, output.status);
6    tracing::info!(
7        "stdout: {}\nstderr: {}",
8        String::from_utf8_lossy(&output.stdout),
9        String::from_utf8_lossy(&output.stderr)
10    );
11}

13#[cfg(target_family = "unix")]
14pub fn run_pre_create_command(pre_create_command: &str) -> Result<(), crate::MenuInstError> {
15    use std::os::unix::fs::PermissionsExt;
16    use std::{io::Write, process::Command};
17
18    use fs_err as fs;
19
20    use crate::MenuInstError;
21
22    let mut temp_file = tempfile::NamedTempFile::with_suffix(".sh")?;
23    temp_file.write_all(pre_create_command.as_bytes())?;
24    let temp_path = temp_file.into_temp_path();
25
26    // Mark the file as executable or run it with bash
27    let mut cmd = if pre_create_command.starts_with("!#") {
28        fs::set_permissions(&temp_path, std::fs::Permissions::from_mode(0o755))?;
29        Command::new(&temp_path)
30    } else {
31        let mut cmd = Command::new("bash");
32        cmd.arg(&temp_path);
33        cmd
34    };
35
36    let output = cmd.output()?;
37    if output.status.success() {
38        Ok(())
39    } else {
40        Err(MenuInstError::InstallError(format!(
41            "Failed to run pre-create command (status: {}): \nstdout: {}\nstderr: {}",
42            output.status,
43            String::from_utf8_lossy(&output.stdout),
44            String::from_utf8_lossy(&output.stderr)
45        )))
46    }
47}
