40/// Returns a [`Client`] that can be shared between all requests.
41fn reqwest_client() -> Client {
42    static CLIENT: OnceLock<Mutex<Client>> = OnceLock::new();
43    CLIENT
44        .get_or_init(|| Mutex::new(Client::new()))
45        .lock()
46        .unwrap()
47        .clone()
48}

67/// Downloads a file to a semi-temporary location that can be used for testing.
68pub fn download_and_cache_file(url: Url, expected_sha256: &str) -> Result<PathBuf, Error> {
69    // Acquire a lock to the cache directory
70    let cache_dir = cache_dir()?;
71
72    // Determine the extension of the file
73    let filename = url
74        .path_segments()
75        .into_iter()
76        .flatten()
77        .last()
78        .ok_or_else(|| Error::InvalidUrl(String::from("missing filename")))?;
79
80    // Determine the final location of the file
81    let final_parent_dir = cache_dir.join(expected_sha256);
82    let final_path = final_parent_dir.join(filename);
83
84    // Ensure the cache directory exists
85    std::fs::create_dir_all(&final_parent_dir)
86        .map_err(|e| Error::FailedToCreateCacheDir(final_parent_dir.display().to_string(), e))?;
87
88    // Acquire the lock on the cache directory
89    let mut lock = fslock::LockFile::open(&cache_dir.join(".lock"))
90        .map_err(Error::FailedToAcquireCacheLock)?;
91    lock.lock_with_pid()
92        .map_err(Error::FailedToAcquireCacheLock)?;
93
94    // Check if the file is already there
95    if final_path.is_file() {
96        return Ok(final_path);
97    }
98
99    eprintln!("Downloading {} to {}", url, final_path.display());
100    let start_download = Instant::now();
101
102    // Construct a temporary file to which we will write the file
103    let tempfile = tempfile::NamedTempFile::new_in(&final_parent_dir)
104        .map_err(Error::FailedToCreateTemporaryFile)?;
105
106    // Execute the download request
107    let mut response = reqwest_client().get(url).send()?.error_for_status()?;
108
109    // Compute the hash while downloading
110    let mut writer = rattler_digest::HashingWriter::<_, Sha256>::new(tempfile);
111    std::io::copy(&mut response, &mut writer)?;
112    let (tempfile, hash) = writer.finalize();
113
114    // Check if the hash matches
115    let actual_hash = format!("{hash:x}");
116    if actual_hash != expected_sha256 {
117        return Err(Error::HashMismatch(expected_sha256.to_owned(), actual_hash));
118    }
119
120    // Write the file to its final destination
121    tempfile.persist(&final_path).map_err(|e| e.error)?;
122
123    let end_download = Instant::now();
124    eprintln!(
125        "Finished download in {}s",
126        (end_download - start_download).as_secs_f32()
127    );
128
129    Ok(final_path)
130}

137/// Fetches conda-forge test repodata from rattler-test.pixi.run (blocking version).
138///
139/// This function downloads conda-forge repodata files from rattler-test.pixi.run if they
140/// don't already exist locally. It uses file locking to ensure thread-safety and avoid
141/// duplicate downloads.
142///
143/// # Arguments
144/// * `subdir` - The subdirectory (platform) to fetch, e.g., "linux-64", "noarch"
145///
146/// # Returns
147/// * `Ok(PathBuf)` - Path to the repodata.json file
148/// * `Err(reqwest::Error)` if the download failed
149pub fn fetch_test_conda_forge_repodata(subdir: &str) -> Result<PathBuf, reqwest::Error> {
150    let path = test_data_dir().join(format!("channels/conda-forge/{subdir}/repodata.json"));
151
152    // Early out if the file already exists
153    if path.exists() {
154        return Ok(path);
155    }
156
157    // Create the parent directory if it doesn't exist
158    let parent_dir = path.parent().unwrap();
159    std::fs::create_dir_all(parent_dir).unwrap();
160
161    // Acquire a lock on the file to ensure we don't download the file twice.
162    let mut lock = fslock::LockFile::open(&parent_dir.join(".lock")).unwrap();
163    loop {
164        if lock.try_lock_with_pid().unwrap() {
165            break;
166        }
167        std::thread::sleep(std::time::Duration::from_millis(100));
168    }
169
170    // Early out if the file was downloaded while we were waiting for the lock
171    if path.is_file() {
172        return Ok(path);
173    }
174
175    // Download the file and persist after download
176    let mut file = NamedTempFile::new_in(parent_dir).unwrap();
177    let mut response = reqwest_client()
178        .get(format!(
179            "https://rattler-test.pixi.run/test-data/channels/conda-forge/{subdir}/repodata.json"
180        ))
181        .send()?
182        .error_for_status()?;
183
184    std::io::copy(&mut response, &mut file).unwrap();
185    file.persist(&path).unwrap();
186
187    Ok(path)
188}
