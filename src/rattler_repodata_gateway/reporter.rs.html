9/// A trait that enables being notified of download progress.
10pub trait DownloadReporter: Send + Sync {
11    /// Called when a download of a file started.
12    ///
13    /// Returns an index that can be used to identify the download in subsequent
14    /// calls.
15    fn on_download_start(&self, _url: &Url) -> usize {
16        0
17    }
18
19    /// Called when the download of a file makes any progress.
20    ///
21    /// The `total_bytes` parameter is `None` if the total size of the file is
22    /// unknown.
23    ///
24    /// The `index` parameter is the index returned by `on_download_start`.
25    fn on_download_progress(
26        &self,
27        _url: &Url,
28        _index: usize,
29        _bytes_downloaded: usize,
30        _total_bytes: Option<usize>,
31    ) {
32    }
33
34    /// Called when the download of a file finished.
35    ///
36    /// The `index` parameter is the index returned by `on_download_start`.
37    fn on_download_complete(&self, _url: &Url, _index: usize) {}
38}

40/// A trait that enables being notified of repodata fetching progress.
41pub trait Reporter: Send + Sync {
42    /// Returns a reporter for downloading files.
43    fn download_reporter(&self) -> Option<&dyn DownloadReporter>;
44}

46#[allow(dead_code)]
47pub(crate) trait ResponseReporterExt {
48    /// Converts a response into a stream of bytes, notifying a reporter of the
49    /// progress.
50    fn byte_stream_with_progress(
51        self,
52        reporter: Option<(&dyn DownloadReporter, usize)>,
53    ) -> impl Stream<Item = reqwest::Result<Bytes>>;
54
55    /// Reads all the bytes from a stream and notifies a reporter of the
56    /// progress.
57    fn bytes_with_progress(
58        self,
59        reporter: Option<(&dyn DownloadReporter, usize)>,
60    ) -> impl Future<Output = reqwest::Result<Vec<u8>>>;
61
62    /// Reads all the bytes from a stream and convert it to text and notifies a
63    /// reporter of the progress.
64    fn text_with_progress(
65        self,
66        reporter: Option<(&dyn DownloadReporter, usize)>,
67    ) -> impl Future<Output = reqwest::Result<String>>;
68}
69
70impl ResponseReporterExt for reqwest::Response {
71    fn byte_stream_with_progress(
72        self,
73        reporter: Option<(&dyn DownloadReporter, usize)>,
74    ) -> impl Stream<Item = reqwest::Result<Bytes>> {
75        let total_size = self.content_length().map(|len| len as usize);
76        let url = self.url().clone();
77        let mut bytes_read = 0;
78        self.bytes_stream().inspect_ok(move |bytes| {
79            if let Some((reporter, index)) = reporter {
80                bytes_read += bytes.len();
81                reporter.on_download_progress(&url, index, bytes_read, total_size);
82            }
83        })
84    }
85
86    fn bytes_with_progress(
87        self,
88        reporter: Option<(&dyn DownloadReporter, usize)>,
89    ) -> impl Future<Output = reqwest::Result<Vec<u8>>> {
90        self.byte_stream_with_progress(reporter).bytes()
91    }
92
93    fn text_with_progress(
94        self,
95        reporter: Option<(&dyn DownloadReporter, usize)>,
96    ) -> impl Future<Output = reqwest::Result<String>> {
97        self.byte_stream_with_progress(reporter).text()
98    }
