15/// The newest repodata revision understood by this version of rattler.
16///
17/// Revision `3` is the current top-level `v3` map implemented by rattler.
18/// Newer revisions are intentionally ignored by older clients, but we still
19/// surface their metadata for user-facing warnings.
20#[cfg(feature = "sparse")]
21pub const SUPPORTED_REPODATA_REVISION: RepodataRevision = RepodataRevision::V3;
22
23/// A structured message indicating that a channel contains repodata newer than
24/// this client understands.
25#[cfg(feature = "sparse")]
26#[derive(Debug, Clone, PartialEq, Eq)]
27pub struct UnsupportedRepodataRevision {
28    /// Redacted channel base URL.
29    pub channel: String,
30
31    /// Channel subdirectory, for example `linux-64` or `noarch`.
32    pub subdir: String,
33
34    /// The newest revision supported by this client.
35    pub supported_revision: RepodataRevision,
36
37    /// Metadata for the unsupported revision advertised by the channel.
38    pub revision: RepodataRevisionInfo,
39}
40
41#[cfg(feature = "sparse")]
42impl std::fmt::Display for UnsupportedRepodataRevision {
43    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
44        write!(
45            f,
46            "{} contains repodata revision {}, but this client only supports up to {}",
47            self.channel, self.revision.revision, self.supported_revision
48        )?;
49
50        if let Some(n_packages) = self.revision.n_packages {
51            write!(f, " ({n_packages} packages may be unavailable)")?;
52        }
53
54        Ok(())
55    }
56}

58/// A trait that enables being notified of download progress.
59pub trait DownloadReporter: Send + Sync {
60    /// Called when a download of a file started.
61    ///
62    /// Returns an index that can be used to identify the download in subsequent
63    /// calls.
64    fn on_download_start(&self, _url: &Url) -> usize {
65        0
66    }
67
68    /// Called when the download of a file makes any progress.
69    ///
70    /// The `total_bytes` parameter is `None` if the total size of the file is
71    /// unknown.
72    ///
73    /// The `index` parameter is the index returned by `on_download_start`.
74    fn on_download_progress(
75        &self,
76        _url: &Url,
77        _index: usize,
78        _bytes_downloaded: usize,
79        _total_bytes: Option<usize>,
80    ) {
81    }
82
83    /// Called when the download of a file finished.
84    ///
85    /// The `index` parameter is the index returned by `on_download_start`.
86    fn on_download_complete(&self, _url: &Url, _index: usize) {}
87}

89/// A trait that enables being notified of repodata fetching progress.
90pub trait Reporter: Send + Sync {
91    /// Returns a reporter for downloading files.
92    fn download_reporter(&self) -> Option<&dyn DownloadReporter>;
93
94    /// Called when a channel advertises a repodata revision newer than this
95    /// client supports.
96    #[cfg(feature = "sparse")]
97    fn on_unsupported_repodata_revision(&self, _message: &UnsupportedRepodataRevision) {}
98}

100#[cfg(feature = "gateway")]
101pub(crate) fn report_unsupported_repodata_revisions<'a>(
102    reporter: Option<&dyn Reporter>,
103    channel: &Channel,
104    subdir: &str,
105    revisions: impl IntoIterator<Item = &'a RepodataRevisionInfo>,
106) {
107    let Some(reporter) = reporter else {
108        return;
109    };
110
111    let channel = channel.base_url.url().clone().redact().to_string();
112    for revision in revisions {
113        if revision.revision > SUPPORTED_REPODATA_REVISION {
114            reporter.on_unsupported_repodata_revision(&UnsupportedRepodataRevision {
115                channel: channel.clone(),
116                subdir: subdir.to_string(),
117                supported_revision: SUPPORTED_REPODATA_REVISION,
118                revision: revision.clone(),
119            });
120        }
121    }
122}

124#[allow(dead_code)]
125pub(crate) trait ResponseReporterExt {
126    /// Converts a response into a stream of bytes, notifying a reporter of the
127    /// progress.
128    fn byte_stream_with_progress(
129        self,
130        reporter: Option<(&dyn DownloadReporter, usize)>,
131    ) -> impl Stream<Item = reqwest::Result<Bytes>>;
132
133    /// Reads all the bytes from a stream and notifies a reporter of the
134    /// progress.
135    fn bytes_with_progress(
136        self,
137        reporter: Option<(&dyn DownloadReporter, usize)>,
138    ) -> impl Future<Output = reqwest::Result<Vec<u8>>>;
139
140    /// Reads all the bytes from a stream and convert it to text and notifies a
141    /// reporter of the progress.
142    fn text_with_progress(
143        self,
144        reporter: Option<(&dyn DownloadReporter, usize)>,
145    ) -> impl Future<Output = reqwest::Result<String>>;
146}
147
148impl ResponseReporterExt for reqwest::Response {
149    fn byte_stream_with_progress(
150        self,
151        reporter: Option<(&dyn DownloadReporter, usize)>,
152    ) -> impl Stream<Item = reqwest::Result<Bytes>> {
153        let total_size = self.content_length().map(|len| len as usize);
154        let url = self.url().clone();
155        let mut bytes_read = 0;
156        self.bytes_stream().inspect_ok(move |bytes| {
157            if let Some((reporter, index)) = reporter {
158                bytes_read += bytes.len();
159                reporter.on_download_progress(&url, index, bytes_read, total_size);
160            }
161        })
162    }
163
164    fn bytes_with_progress(
165        self,
166        reporter: Option<(&dyn DownloadReporter, usize)>,
167    ) -> impl Future<Output = reqwest::Result<Vec<u8>>> {
168        self.byte_stream_with_progress(reporter).bytes()
169    }
170
171    fn text_with_progress(
172        self,
173        reporter: Option<(&dyn DownloadReporter, usize)>,
174    ) -> impl Future<Output = reqwest::Result<String>> {
175        self.byte_stream_with_progress(reporter).text()
176    }
