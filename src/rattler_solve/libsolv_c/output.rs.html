15/// Result of extracting packages and extras from the solver transaction.
16pub struct SolverOutput {
17    pub packages: Vec<RepoDataRecord>,
18    pub extras: HashMap<PackageName, Vec<String>>,
19}
20
21/// Returns which packages should be installed in the environment and which extras are activated.
22///
23/// If the transaction contains libsolv operations that are not "install" an error is returned
24/// containing their ids.
25pub fn get_required_packages(
26    pool: &Pool,
27    repo_mapping: &HashMap<RepoId, usize>,
28    transaction: &Transaction<'_>,
29    repodata_records: &[Vec<&RepoDataRecord>],
30) -> Result<SolverOutput, Vec<ffi::Id>> {
31    let mut required_packages = Vec::new();
32    let mut extras: HashMap<PackageName, Vec<String>> = HashMap::new();
33    let mut unsupported_operations = Vec::new();
34
35    let solvable_index_id = pool
36        .find_interned_str(SOLVABLE_REPODATA_RECORD_INDEX)
37        .unwrap();
38
39    // Keys for retrieving extra info from synthetic solvables
40    let extra_package_id = pool.find_interned_str(SOLVABLE_EXTRA_PACKAGE);
41    let extra_name_id = pool.find_interned_str(SOLVABLE_EXTRA_NAME);
42
43    // Safe because `transaction.steps` is an active queue
44    let transaction_queue = transaction.get_steps();
45
46    for id in transaction_queue.iter() {
47        let transaction_type = transaction.transaction_type(id);
48
49        // Try to retrieve the repodata record corresponding to this solvable
50        if let Some((repo_index, solvable_index)) =
51            get_solvable_indexes(pool, repo_mapping, solvable_index_id, id)
52        {
53            let repodata_record = repodata_records[repo_index][solvable_index];
54
55            match transaction_type as u32 {
56                ffi::SOLVER_TRANSACTION_INSTALL => {
57                    required_packages.push(repodata_record.clone());
58                }
59                _ => {
60                    unsupported_operations.push(transaction_type);
61                }
62            }
63        } else if transaction_type as u32 == ffi::SOLVER_TRANSACTION_INSTALL {
64            // This might be a synthetic solvable for an extra
65            // Extract the extra information from stored attributes
66            if let Some((pkg_id, name_id)) = extra_package_id.zip(extra_name_id) {
67                if let Some((pkg_name, extra_name)) =
68                    extract_extra_from_solvable(pool, id, pkg_id, name_id)
69                {
70                    extras.entry(pkg_name).or_default().push(extra_name);
71                }
72            }
73        }
74    }
75
76    if !unsupported_operations.is_empty() {
77        return Err(unsupported_operations);
78    }
79
80    Ok(SolverOutput {
81        packages: required_packages,
82        extras,
83    })
84}

86/// Extracts the package name and extra name from a synthetic solvable using stored attributes.
87fn extract_extra_from_solvable(
88    pool: &Pool,
89    id: SolvableId,
90    extra_package_key: StringId,
91    extra_name_key: StringId,
92) -> Option<(PackageName, String)> {
93    let solvable = id.resolve_raw(pool);
94    let solvable_ptr = solvable.as_ptr();
95
96    let pkg_name_id = solvable::lookup_id(solvable_ptr, extra_package_key)?;
97    let extra_name_id = solvable::lookup_id(solvable_ptr, extra_name_key)?;
98
99    let pkg_name_str = pkg_name_id.resolve(pool)?;
100    let extra_name_str = extra_name_id.resolve(pool)?;
101
102    let pkg_name = pkg_name_str.parse::<PackageName>().ok()?;
103    Some((pkg_name, extra_name_str.to_string()))
104}
105
106fn get_solvable_indexes(
107    pool: &Pool,
108    repo_mapping: &HashMap<RepoId, usize>,
109    solvable_index_id: StringId,
110    id: SolvableId,
111) -> Option<(usize, usize)> {
112    let solvable = id.resolve_raw(pool);
113    let solvable_index = solvable::lookup_num(solvable.as_ptr(), solvable_index_id)? as usize;
114
115    // Safe because there are no active mutable borrows of any solvable at this stage
116    let repo_id = RepoId::from_ffi_solvable(unsafe { solvable.as_ref() });
117
118    let repo_index = repo_mapping[&repo_id];
119
120    Some((repo_index, solvable_index))
121}
