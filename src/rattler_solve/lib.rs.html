20/// Represents a solver implementation, capable of solving [`SolverTask`]s
21pub trait SolverImpl {
22    /// The repo data associated to a channel and platform combination
23    type RepoData<'a>: SolverRepoData<'a>;
24
25    /// Resolve the dependencies and return the [`RepoDataRecord`]s that should
26    /// be present in the environment.
27    fn solve<
28        'a,
29        R: IntoRepoData<'a, Self::RepoData<'a>>,
30        TAvailablePackagesIterator: IntoIterator<Item = R>,
31    >(
32        &mut self,
33        task: SolverTask<TAvailablePackagesIterator>,
34    ) -> Result<SolverResult, SolveError>;
35}

37/// Represents an error when solving the dependencies for a given environment
38#[derive(thiserror::Error, Debug)]
39pub enum SolveError {
40    /// There is no set of dependencies that satisfies the requirements
41    Unsolvable(Vec<String>),
42
43    /// The solver backend returned operations that we dont know how to install.
44    /// Each string is a somewhat user-friendly representation of which
45    /// operation was not recognized and can be used for error reporting
46    UnsupportedOperations(Vec<String>),
47
48    /// Error when converting matchspec
49    #[error(transparent)]
50    ParseMatchSpecError(#[from] rattler_conda_types::ParseMatchSpecError),
51
52    /// Encountered duplicate records in the available packages.
53    DuplicateRecords(String),
54
55    /// To support Resolvo cancellation
56    Cancelled,
57}
58
59impl fmt::Display for SolveError {
60    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
61        match self {
62            SolveError::Unsolvable(operations) => {
63                write!(
64                    f,
65                    "Cannot solve the request because of: {}",
66                    operations.join(", ")
67                )
68            }
69            SolveError::UnsupportedOperations(operations) => {
70                write!(f, "Unsupported operations: {}", operations.join(", "))
71            }
72            SolveError::ParseMatchSpecError(e) => {
73                write!(f, "Error parsing match spec: {e}")
74            }
75            SolveError::Cancelled => {
76                write!(f, "Solve operation has been cancelled")
77            }
78            SolveError::DuplicateRecords(filename) => {
79                write!(f, "encountered duplicate records for {filename}")
80            }
81        }
82    }
83}

85/// Configuration for filtering packages newer than a cutoff.
86///
87/// This feature helps reduce the risk of installing compromised packages by
88/// delaying the installation of newly published versions. In most cases,
89/// malicious releases are discovered and removed from channels within a short
90/// time window (often within an hour). By requiring packages to have been
91/// published for a minimum duration, you give the community time to identify
92/// and report malicious packages before they can be installed.
93///
94/// This is similar to pnpm's `minimumReleaseAge` feature.
95///
96/// # Example
97///
98/// ```
99/// use std::time::Duration;
100/// use rattler_solve::ExcludeNewer;
101///
102/// // Only allow packages that have been published for at least 1 hour
103/// let config = ExcludeNewer::from_duration(Duration::from_secs(60 * 60))
104///     // But allow "my-internal-package" to use a package-specific cutoff
105///     .with_package_duration("my-internal-package".parse().unwrap(), Duration::ZERO)
106///     // And allow a trusted internal channel to skip the delay entirely
107///     .with_channel_duration("my-internal-channel", Duration::ZERO);
108/// ```
109#[derive(Debug, Clone, PartialEq, Eq)]
110pub struct ExcludeNewer {
111    /// The default cutoff date. Packages uploaded after this date are excluded.
112    cutoff: DateTime<Utc>,
113
114    /// Channel-specific cutoff dates that override [`Self::cutoff`] for
115    /// records from matching channels.
116    ///
117    /// The key is matched against [`RepoDataRecord::channel`] exactly.
118    channel_cutoffs: HashMap<String, DateTime<Utc>>,
119
120    /// Package-specific cutoff dates that override both [`Self::cutoff`] and
121    /// [`Self::channel_cutoffs`] for matching package names.
122    package_cutoffs: HashMap<PackageName, DateTime<Utc>>,
123
124    /// Whether to include packages that don't have a timestamp.
125    include_unknown_timestamp: bool,
126}
127
128impl ExcludeNewer {
129    fn cutoff_from_duration(duration: std::time::Duration, now: DateTime<Utc>) -> DateTime<Utc> {
130        let duration =
131            chrono::Duration::from_std(duration).expect("exclude_newer duration is too large");
132        now - duration
133    }
134
135    /// Creates a new configuration from an absolute cutoff date.
136    pub fn from_datetime(cutoff: DateTime<Utc>) -> Self {
137        Self {
138            cutoff,
139            channel_cutoffs: HashMap::new(),
140            package_cutoffs: HashMap::new(),
141            include_unknown_timestamp: false,
142        }
143    }
144
145    /// Creates a new configuration from a relative duration.
146    pub fn from_duration(duration: std::time::Duration) -> Self {
147        Self::from_duration_with_now(duration, Utc::now())
148    }
149
150    /// Creates a new configuration from a relative duration and explicit
151    /// reference time.
152    pub fn from_duration_with_now(duration: std::time::Duration, now: DateTime<Utc>) -> Self {
153        Self {
154            cutoff: Self::cutoff_from_duration(duration, now),
155            channel_cutoffs: HashMap::new(),
156            package_cutoffs: HashMap::new(),
157            include_unknown_timestamp: false,
158        }
159    }
160
161    /// Sets the absolute cutoff override for a specific package.
162    pub fn with_package_cutoff(mut self, package: PackageName, cutoff: DateTime<Utc>) -> Self {
163        self.package_cutoffs.insert(package, cutoff);
164        self
165    }
166
167    /// Sets the duration override for a specific package.
168    pub fn with_package_duration(
169        mut self,
170        package: PackageName,
171        duration: std::time::Duration,
172    ) -> Self {
173        self.package_cutoffs
174            .insert(package, Self::cutoff_from_duration(duration, Utc::now()));
175        self
176    }
177
178    /// Sets the duration override for a specific package using an explicit
179    /// reference time.
180    pub fn with_package_duration_with_now(
181        mut self,
182        package: PackageName,
183        duration: std::time::Duration,
184        now: DateTime<Utc>,
185    ) -> Self {
186        self.package_cutoffs
187            .insert(package, Self::cutoff_from_duration(duration, now));
188        self
189    }
190
191    /// Sets the duration override for a specific channel.
192    pub fn with_channel_duration(
193        mut self,
194        channel: impl Into<String>,
195        duration: std::time::Duration,
196    ) -> Self {
197        self.channel_cutoffs.insert(
198            channel.into(),
199            Self::cutoff_from_duration(duration, Utc::now()),
200        );
201        self
202    }
203
204    /// Sets the duration override for a specific channel using an explicit
205    /// reference time.
206    pub fn with_channel_duration_with_now(
207        mut self,
208        channel: impl Into<String>,
209        duration: std::time::Duration,
210        now: DateTime<Utc>,
211    ) -> Self {
212        self.channel_cutoffs
213            .insert(channel.into(), Self::cutoff_from_duration(duration, now));
214        self
215    }
216
217    /// Sets the absolute cutoff override for a specific channel.
218    pub fn with_channel_cutoff(
219        mut self,
220        channel: impl Into<String>,
221        cutoff: DateTime<Utc>,
222    ) -> Self {
223        self.channel_cutoffs.insert(channel.into(), cutoff);
224        self
225    }
226
227    /// Sets whether packages without a timestamp should be included.
228    ///
229    /// Call this to override the constructor default.
230    pub fn with_include_unknown_timestamp(mut self, include: bool) -> Self {
231        self.include_unknown_timestamp = include;
232        self
233    }
234
235    /// Returns whether packages without a timestamp are included.
236    pub fn include_unknown_timestamp(&self) -> bool {
237        self.include_unknown_timestamp
238    }
239
240    /// Computes the cutoff time for the given package and channel.
241    pub fn cutoff_for_package(
242        &self,
243        package: &PackageName,
244        channel: Option<&str>,
245    ) -> DateTime<Utc> {
246        self.package_cutoffs
247            .get(package)
248            .copied()
249            .or_else(|| channel.and_then(|channel| self.channel_cutoffs.get(channel).copied()))
250            .unwrap_or(self.cutoff)
251    }
252
253    /// Returns whether a package should be excluded.
254    pub fn is_excluded(
255        &self,
256        package: &PackageName,
257        channel: Option<&str>,
258        timestamp: Option<&TimestampMs>,
259    ) -> bool {
260        match timestamp {
261            Some(timestamp) => *timestamp > self.cutoff_for_package(package, channel),
262            None => !self.include_unknown_timestamp(),
263        }
264    }
265}

267impl From<DateTime<Utc>> for ExcludeNewer {
268    fn from(value: DateTime<Utc>) -> Self {
269        Self::from_datetime(value)
270    }
271}

274    fn from(value: std::time::Duration) -> Self {

276    }

279/// Represents the channel priority option to use during solves.
280#[derive(Debug, Clone, Copy, PartialEq, Eq, Default)]
281#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
282#[cfg_attr(feature = "serde", serde(rename_all = "kebab-case"))]
283pub enum ChannelPriority {
284    /// The channel that the package is first found in will be used as the only
285    /// channel for that package.
286    #[default]
287    Strict,
288
289    // Conda also has "Flexible" as an option, where packages present in multiple channels
290    // are only taken from lower-priority channels when this prevents unsatisfiable environment
291    // errors, but this would need implementation in the solvers.
292    // Flexible,
293    /// Packages can be retrieved from any channel as package version takes
294    /// precedence.
295    Disabled,
296}

298/// Represents a dependency resolution task, to be solved by one of the backends
299pub struct SolverTask<TAvailablePackagesIterator> {
300    /// An iterator over all available packages
301    pub available_packages: TAvailablePackagesIterator,
302
303    /// Records of packages that are previously selected.
304    ///
305    /// If the solver encounters multiple variants of a single package
306    /// (identified by its name), it will sort the records and select the
307    /// best possible version. However, if there exists a locked version it
308    /// will prefer that variant instead. This is useful to reduce the number of
309    /// packages that are updated when installing new packages.
310    ///
311    /// Usually you add the currently installed packages or packages from a
312    /// lock-file here.
313    pub locked_packages: Vec<RepoDataRecord>,
314
315    /// Records of packages that are previously selected and CANNOT be changed.
316    ///
317    /// If the solver encounters multiple variants of a single package
318    /// (identified by its name), it will sort the records and select the
319    /// best possible version. However, if there is a variant available in
320    /// the `pinned_packages` field it will always select that version no matter
321    /// what even if that means other packages have to be downgraded.
322    pub pinned_packages: Vec<RepoDataRecord>,
323
324    /// Virtual packages considered active
325    pub virtual_packages: Vec<GenericVirtualPackage>,
326
327    /// The specs we want to solve
328    pub specs: Vec<MatchSpec>,
329
330    /// Additional constraints that should be satisfied by the solver.
331    /// Packages included in the `constraints` are not necessarily
332    /// installed, but they must be satisfied by the solution.
333    pub constraints: Vec<MatchSpec>,
334
335    /// The timeout after which the solver should stop
336    pub timeout: Option<std::time::Duration>,
337
338    /// The channel priority to solve with, either [`ChannelPriority::Strict`]
339    /// or [`ChannelPriority::Disabled`]
340    pub channel_priority: ChannelPriority,
341
342    /// Exclude packages newer than the configured cutoff.
343    ///
344    /// This can be either:
345    ///
346    /// - a fixed cutoff date, equivalent to the historical `exclude_newer`
347    ///   behavior; or
348    /// - a relative duration, equivalent to the historical `min_age`
349    ///   behavior.
350    pub exclude_newer: Option<ExcludeNewer>,
351
352    /// The solve strategy.
353    pub strategy: SolveStrategy,
354
355    /// Dependency overrides that replace dependencies of matching packages.
356    pub dependency_overrides: Vec<(MatchSpec, MatchSpec)>,
357}

359impl<'r, I: IntoIterator<Item = &'r RepoDataRecord>> FromIterator<I>
360    for SolverTask<Vec<RepoDataIter<I>>>
361{
362    fn from_iter<T: IntoIterator<Item = I>>(iter: T) -> Self {

366            pinned_packages: Vec::new(),
367            virtual_packages: Vec::new(),

374            dependency_overrides: Vec::new(),
375        }
376    }

379/// Represents the strategy to use when solving dependencies
380#[derive(Debug, Clone, Copy, Default, Eq, PartialEq)]
381#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
382#[cfg_attr(feature = "serde", serde(rename_all = "kebab-case"))]
383pub enum SolveStrategy {
384    /// Resolve the highest version of each package.
385    #[default]
386    Highest,
387
388    /// Resolve the lowest compatible version for each package.
389    ///
390    /// All candidates with the same version are still ordered the same as
391    /// with `Default`. This ensures that the candidate with the highest build
392    /// number is used and down-prioritization still works.
393    LowestVersion,
394
395    /// Resolve the lowest compatible version for direct dependencies but the
396    /// highest for transitive dependencies. This is similar to `LowestVersion`
397    /// but only for direct dependencies.
398    LowestVersionDirect,
399}

401/// A representation of a collection of [`RepoDataRecord`] usable by a
402/// [`SolverImpl`] implementation.
403///
404/// Some solvers might be able to cache the collection between different runs of
405/// the solver which could potentially eliminate some overhead. This trait
406/// enables creating a representation of the repodata that is most suitable for
407/// a specific backend.
408///
409/// Some solvers may add additional functionality to their specific
410/// implementation that enables caching the repodata to disk in an efficient way
411/// (see [`crate::libsolv_c::RepoData`] for an example).
412pub trait SolverRepoData<'a>: FromIterator<&'a RepoDataRecord> {}
413
414/// Defines the ability to convert a type into [`SolverRepoData`].
415pub trait IntoRepoData<'a, S: SolverRepoData<'a>> {
416    /// Converts this instance into an instance of [`SolverRepoData`] which is
417    /// consumable by a specific [`SolverImpl`] implementation.
418    fn into(self) -> S;
419}
420
421impl<'a, S: SolverRepoData<'a>> IntoRepoData<'a, S> for &'a Vec<RepoDataRecord> {
422    fn into(self) -> S {
423        self.iter().collect()
424    }
425}
426
427impl<'a, S: SolverRepoData<'a>> IntoRepoData<'a, S> for &'a [RepoDataRecord] {
428    fn into(self) -> S {
429        self.iter().collect()
430    }
431}
432
433impl<'a, S: SolverRepoData<'a>> IntoRepoData<'a, S> for S {
434    fn into(self) -> S {
435        self
436    }
437}
