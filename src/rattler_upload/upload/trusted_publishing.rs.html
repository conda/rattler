18/// Represents the CI provider being used for trusted publishing.
19#[derive(Debug, Clone, Copy, PartialEq, Eq)]
20pub enum CiProvider {
21    GitHubActions,
22    GitLabCI,
23    GoogleCloud,
24}
25
26impl std::fmt::Display for CiProvider {
27    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
28        match self {
29            CiProvider::GitHubActions => write!(f, "GitHub Actions"),
30            CiProvider::GitLabCI => write!(f, "GitLab CI"),
31            CiProvider::GoogleCloud => write!(f, "Google Cloud"),
32        }
33    }
34}

36/// Detects which CI provider is being used, if any.
37pub fn detect_ci_provider() -> Option<CiProvider> {
38    if github_action_runner() {
39        Some(CiProvider::GitHubActions)
40    } else if gitlab_ci_runner() {
41        Some(CiProvider::GitLabCI)
42    } else if google_cloud_runner() {
43        Some(CiProvider::GoogleCloud)
44    } else {
45        None
46    }
47}

49/// If applicable, attempt obtaining a token for trusted publishing.
50pub async fn check_trusted_publishing(
51    client: &ClientWithMiddleware,
52    prefix_url: &Url,
53) -> TrustedPublishResult {
54    // Check which CI provider we're running on
55    let provider = match detect_ci_provider() {
56        Some(p) => p,
57        None => return TrustedPublishResult::Skipped,
58    };
59
60    tracing::debug!(
61        "Running on {} without explicit credentials, checking for trusted publishing",
62        provider
63    );
64
65    match get_token(client, prefix_url, provider).await {
66        Ok(token) => TrustedPublishResult::Configured(token),
67        Err(err) => {
68            tracing::debug!("Could not obtain trusted publishing credentials, skipping: {err}");
69            TrustedPublishResult::Ignored(err)
70        }
71    }
72}

74pub enum TrustedPublishResult {
75    /// We didn't check for trusted publishing.
76    Skipped,
77    /// We checked for trusted publishing and found a token.
78    Configured(TrustedPublishingToken),
79    /// We checked for optional trusted publishing, but it didn't succeed.
80    Ignored(TrustedPublishingError),
81}
82
83#[derive(Debug, Error)]
84pub enum TrustedPublishingError {
85    #[error("Environment variable {0} not set, is the `id-token: write` permission missing?")]
86    MissingEnvVar(&'static str),
87    #[error("Environment variable {0} is not valid UTF-8: `{1:?}`")]
88    InvalidEnvVar(&'static str, OsString),
89    #[error(transparent)]
90    Url(#[from] url::ParseError),
91    #[error("Failed to fetch: `{0}`")]
92    Reqwest(Url, #[source] reqwest::Error),
93    #[error("Failed to fetch: `{0}`")]
94    ReqwestMiddleware(Url, #[source] reqwest_middleware::Error),
95    #[error(
96        "Prefix.dev returned error code {0}, is trusted publishing correctly configured?\nResponse: {1}"
97    )]
98    PrefixDev(StatusCode, String),
99    #[error("GitLab CI OIDC token not found. Make sure you have configured `id_tokens` in your .gitlab-ci.yml:\n\n\
100        job_name:\n  \
101          id_tokens:\n    \
102            PREFIX_ID_TOKEN:\n      \
103              aud: prefix.dev\n")]
104    GitLabOidcTokenNotFound,
105    #[error("Google Cloud OIDC token retrieval failed. Make sure you are running in a Google Cloud environment (Cloud Build, Cloud Run, GCE, or GKE) with a service account attached.")]
106    GoogleCloudOidcTokenNotFound,
107}
108
109impl TrustedPublishingError {
110    fn from_var_err(env_var: &'static str, err: VarError) -> Self {
111        match err {
112            VarError::NotPresent => Self::MissingEnvVar(env_var),
113            VarError::NotUnicode(os_string) => Self::InvalidEnvVar(env_var, os_string),
114        }
115    }
116}

128/// The response from querying `$ACTIONS_ID_TOKEN_REQUEST_URL&audience=prefix.dev`.

140/// Returns the short-lived token to use for uploading.
141pub async fn get_token(
142    client: &ClientWithMiddleware,
143    prefix_url: &Url,
144    provider: CiProvider,
145) -> Result<TrustedPublishingToken, TrustedPublishingError> {
146    // Get the OIDC token based on the CI provider
147    let oidc_token = match provider {
148        CiProvider::GitHubActions => {
149            // If this fails, we can skip the audience request.
150            let oidc_token_request_token = env::var(consts::ACTIONS_ID_TOKEN_REQUEST_TOKEN)
151                .map_err(|err| {
152                    TrustedPublishingError::from_var_err(
153                        consts::ACTIONS_ID_TOKEN_REQUEST_TOKEN,
154                        err,
155                    )
156                })?;
157
158            // Request 1: Get the OIDC token from GitHub.
159            get_github_oidc_token(&oidc_token_request_token, client).await?
160        }
161        CiProvider::GitLabCI => {
162            // Get the OIDC token from GitLab CI environment variable
163            get_gitlab_oidc_token()?
164        }
165        CiProvider::GoogleCloud => {
166            // Get the OIDC token from Google Cloud metadata server
167            get_google_cloud_oidc_token(client).await?
168        }
169    };
170
171    // Request 2: Get the publishing token from prefix.dev.
172    let publish_token = get_publish_token(&oidc_token, prefix_url, client).await?;
173
174    tracing::info!("Received token from {}, using trusted publishing", provider);
175
176    // Mask the token in CI logs
177    match provider {
178        CiProvider::GitHubActions => {
179            println!("::add-mask::{}", &publish_token.secret());
180        }
181        CiProvider::GitLabCI => {
182            // GitLab CI doesn't have a built-in mask mechanism like GitHub Actions,
183            // but the token should be short-lived anyway
184            tracing::debug!("Token obtained via GitLab CI trusted publishing");
185        }
186        CiProvider::GoogleCloud => {
187            // Google Cloud doesn't have a built-in mask mechanism,
188            // but the token should be short-lived anyway
189            tracing::debug!("Token obtained via Google Cloud trusted publishing");
190        }
191    }
192
193    Ok(publish_token)
194}

196/// Get the OIDC token from GitLab CI.
197/// GitLab CI provides the token via the `PREFIX_ID_TOKEN` environment variable
198/// when configured with `id_tokens` in the `.gitlab-ci.yml` file.
199fn get_gitlab_oidc_token() -> Result<String, TrustedPublishingError> {
200    // GitLab CI provides OIDC tokens via the `id_tokens` keyword in .gitlab-ci.yml
201    // The user should configure their job like this:
202    //
203    // job_name:
204    //   id_tokens:
205    //     PREFIX_ID_TOKEN:
206    //       aud: prefix.dev
207    //   script:
208    //     - rattler upload ...
209    //
210    // The token is then available as the PREFIX_ID_TOKEN environment variable.
211    match env::var(consts::PREFIX_ID_TOKEN) {
212        Ok(token) if !token.is_empty() => {
213            tracing::info!("Found GitLab CI OIDC token in PREFIX_ID_TOKEN");
214            Ok(token)
215        }
216        Ok(_) => {
217            tracing::warn!("PREFIX_ID_TOKEN is set but empty");
218            Err(TrustedPublishingError::GitLabOidcTokenNotFound)
219        }
220        Err(_) => {
221            tracing::debug!("PREFIX_ID_TOKEN not found in environment");
222            Err(TrustedPublishingError::GitLabOidcTokenNotFound)
223        }
224    }
225}

227/// Get the OIDC token from GitHub Actions.
228async fn get_github_oidc_token(
229    oidc_token_request_token: &str,
230    client: &ClientWithMiddleware,
231) -> Result<String, TrustedPublishingError> {
232    let oidc_token_url = env::var(consts::ACTIONS_ID_TOKEN_REQUEST_URL).map_err(|err| {
233        TrustedPublishingError::from_var_err(consts::ACTIONS_ID_TOKEN_REQUEST_URL, err)
234    })?;
235    let mut oidc_token_url = Url::parse(&oidc_token_url)?;
236    oidc_token_url
237        .query_pairs_mut()
238        .append_pair("audience", "prefix.dev");
239    tracing::info!("Querying the trusted publishing OIDC token from {oidc_token_url}");
240    let response = client
241        .get(oidc_token_url.clone())
242        .bearer_auth(oidc_token_request_token)
243        .send()
244        .await
245        .map_err(|err| TrustedPublishingError::ReqwestMiddleware(oidc_token_url.clone(), err))?;
246    let oidc_token: OidcToken = response
247        .error_for_status()
248        .map_err(|err| TrustedPublishingError::Reqwest(oidc_token_url.clone(), err))?
249        .json()
250        .await
251        .map_err(|err| TrustedPublishingError::Reqwest(oidc_token_url.clone(), err))?;
252    Ok(oidc_token.value)
253}

255/// Get the OIDC token from Google Cloud metadata server.
256/// Works in Cloud Build, Cloud Run, GCE, and GKE with Workload Identity.
257/// Respects the `GCE_METADATA_HOST` environment variable for custom metadata server hostnames.
258async fn get_google_cloud_oidc_token(
259    client: &ClientWithMiddleware,
260) -> Result<String, TrustedPublishingError> {
261    // Use GCE_METADATA_HOST if set, otherwise use the default hostname
262    let metadata_host = env::var(consts::GCE_METADATA_HOST)
263        .unwrap_or_else(|_| consts::GCP_METADATA_HOST_DEFAULT.to_string());
264
265    let metadata_url = format!(
266        "http://{}{}?audience=prefix.dev",
267        metadata_host,
268        consts::GCP_METADATA_IDENTITY_PATH
269    );
270    let url = Url::parse(&metadata_url)?;
271
272    tracing::info!(
273        "Querying the trusted publishing OIDC token from Google Cloud metadata server at {}",
274        metadata_host
275    );
276
277    let response = client
278        .get(url.clone())
279        .header("Metadata-Flavor", "Google")
280        .send()
281        .await
282        .map_err(|err| TrustedPublishingError::ReqwestMiddleware(url.clone(), err))?;
283
284    if !response.status().is_success() {
285        tracing::warn!(
286            "Google Cloud metadata server returned status {}",
287            response.status()
288        );
289        return Err(TrustedPublishingError::GoogleCloudOidcTokenNotFound);
290    }
291
292    let token = response
293        .text()
294        .await
295        .map_err(|err| TrustedPublishingError::Reqwest(url.clone(), err))?;
296
297    if token.is_empty() {
298        tracing::warn!("Google Cloud metadata server returned empty token");
299        return Err(TrustedPublishingError::GoogleCloudOidcTokenNotFound);
300    }
301
302    tracing::info!("Successfully obtained OIDC token from Google Cloud");
303    Ok(token)
304}

306async fn get_publish_token(
307    oidc_token: &str,
308    prefix_url: &Url,
309    client: &ClientWithMiddleware,
310) -> Result<TrustedPublishingToken, TrustedPublishingError> {
311    let mint_token_url = prefix_url.join("/api/oidc/mint_token")?;
312    tracing::info!("Querying the trusted publishing upload token from {mint_token_url}");
313    let mint_token_payload = MintTokenRequest {
314        token: oidc_token.to_string(),
315    };
316
317    let response = client
318        .post(mint_token_url.clone())
319        .json(&mint_token_payload)
320        .send()
321        .await
322        .map_err(|err| TrustedPublishingError::ReqwestMiddleware(mint_token_url.clone(), err))?;
323
324    // reqwest's implementation of `.json()` also goes through `.bytes()`
325    let status = response.status();
326    let body = response
327        .bytes()
328        .await
329        .map_err(|err| TrustedPublishingError::Reqwest(mint_token_url.clone(), err))?;
330
331    if status.is_success() {
332        let token = TrustedPublishingToken(String::from_utf8_lossy(&body).to_string());
333        Ok(token)
334    } else {
335        // An error here means that something is misconfigured,
336        // so we're showing the body for more context
337        Err(TrustedPublishingError::PrefixDev(
338            status,
339            String::from_utf8_lossy(&body).to_string(),
340        ))
341    }
