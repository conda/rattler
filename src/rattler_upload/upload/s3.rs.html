8use tokio::io::{AsyncReadExt, AsyncSeekExt};
9use tokio_util::bytes::BytesMut;
10use url::Url;

14const DESIRED_CHUNK_SIZE: usize = 1024 * 1024 * 10;
15
16/// Uploads a package to a channel in an S3 bucket.
17#[allow(clippy::too_many_arguments)]
18pub async fn upload_package_to_s3(
19    auth_storage: &AuthenticationStorage,
20    channel: Url,
21    credentials: Option<S3Credentials>,
22    package_files: &Vec<PathBuf>,
23    force: bool,
24) -> miette::Result<()> {
25    let bucket = channel
26        .host_str()
27        .ok_or(miette::miette!("No bucket in S3 URL"))?;
28
29    // Create the S3 configuration for opendal.
30    let mut s3_config = S3Config::default();
31    s3_config.root = Some(channel.path().to_string());
32    s3_config.bucket = bucket.to_string();
33
34    // Resolve the credentials to use.
35    let resolved_credentials = match credentials {
36        Some(credentials) => credentials
37            .resolve(&channel, auth_storage)
38            .ok_or_else(|| miette::miette!("Could not find S3 credentials in the authentication storage, and no credentials were provided via the command line."))?,
39        None => {
40            ResolvedS3Credentials::from_sdk().await.into_diagnostic()?
41        }
42    };
43
44    s3_config.endpoint = Some(resolved_credentials.endpoint_url.to_string());
45    s3_config.region = Some(resolved_credentials.region);
46    s3_config.access_key_id = Some(resolved_credentials.access_key_id);
47    s3_config.secret_access_key = Some(resolved_credentials.secret_access_key);
48    s3_config.session_token = resolved_credentials.session_token;
49    s3_config.enable_virtual_host_style =
50        resolved_credentials.addressing_style == rattler_s3::S3AddressingStyle::VirtualHost;
51
52    let builder = s3_config.into_builder();
53    let op = Operator::new(builder).into_diagnostic()?.finish();
54
55    for package_file in package_files {
56        let package = ExtractedPackage::from_package_file(package_file)?;
57        let subdir = package
58            .subdir()
59            .ok_or_else(|| miette::miette!("Failed to get subdir"))?;
60        let filename = package
61            .filename()
62            .ok_or_else(|| miette::miette!("Failed to get filename"))?;
63        let key = format!("{subdir}/{filename}");
64
65        // Compute the hash of the package by streaming its content.
66        let file = tokio::io::BufReader::new(
67            fs_err::tokio::File::open(package_file)
68                .await
69                .into_diagnostic()?,
70        );
71        let sha256_reader = HashingReader::<_, Sha256>::new(file);
72        let mut md5_reader = HashingReader::<_, Md5>::new(sha256_reader);
73        let size = tokio::io::copy(&mut md5_reader, &mut tokio::io::sink())
74            .await
75            .into_diagnostic()?;
76        let (sha256_reader, md5hash) = md5_reader.finalize();
77        let (mut file, sha256hash) = sha256_reader.finalize();
78
79        // Rewind the file to the beginning.
80        file.rewind().await.into_diagnostic()?;
81
82        // Construct a writer for the package.
83        let mut writer = match op
84            .writer_with(&key)
85            .content_disposition(&format!("attachment; filename={filename}"))
86            .if_not_exists(!force)
87            .user_metadata([
88                (String::from("package-sha256"), format!("{sha256hash:x}")),
89                (String::from("package-md5"), format!("{md5hash:x}")),
90            ])
91            .await
92        {
93            Err(e) if e.kind() == ErrorKind::ConditionNotMatch => {
94                miette::bail!(
95                    "Package s3://{bucket}{}/{key} already exists. Use --force to overwrite.",
96                    channel.path().to_string()
97                );
98            }
99            Ok(writer) => writer,
100            Err(e) => {
101                return Err(e).into_diagnostic();
102            }
103        };
104
105        // Write the contents to the writer. We do this in a more complex way than just
106        // using `io::copy` because some underlying storage providers expect to receive
107        // the data in specifically sized chunks. The code below guarantees chunks of
108        // equal size except for maybe the last chunk.
109        let mut remaining_size = size as usize;
110        loop {
111            // Allocate memory for this chunk
112            let chunk_size = remaining_size.min(DESIRED_CHUNK_SIZE);
113            let mut chunk = BytesMut::with_capacity(chunk_size);
114            // SAFE: because we do not care about the bytes that are currently in the buffer
115            unsafe { chunk.set_len(chunk_size) };
116
117            // Fill the chunk with data. This reads exactly the number of bytes we want. No
118            // more, no less.
119            let bytes_read = file.read_exact(&mut chunk[..]).await.into_diagnostic()?;
120            debug_assert_eq!(bytes_read, chunk.len());
121
122            // Write the writes directly to storage
123            writer.write(chunk.freeze()).await.into_diagnostic()?;
124
125            // Update the number of remaining bytes
126            remaining_size = remaining_size.saturating_sub(bytes_read);
127            if remaining_size == 0 {
128                break;
129            }
130        }
131
132        match writer.close().await {
133            Err(e) if e.kind() == ErrorKind::ConditionNotMatch => {
134                miette::bail!(
135                    "Package s3://{bucket}{}/{key} already exists. Use --force to overwrite.",
136                    channel.path().to_string()
137                );
138            }
139            Ok(_) => {
140                tracing::info!(
141                    "Uploaded package to s3://{bucket}{}/{key}",
142                    channel.path().to_string()
143                );
144            }
145            Err(e) => {
146                return Err(e).into_diagnostic();
147            }
148        }
149    }
150
151    Ok(())
152}
