11fn configure_default_store() -> Result<(), KeyringAuthenticationStorageError> {
12    if keyring_core::get_default_store().is_some() {
13        Ok(())
14    } else {
15        configure_platform_default_store()
16    }
17}

48/// Build the platform-specific [`CredentialStore::search`] spec that enumerates
49/// every entry written by this storage instance.
50///
51/// macOS and the dbus secret service filter on the `service` attribute
52/// directly. Windows has no notion of a "service" field — the keyring-core
53/// store encodes `service` into the credential target as `{user}.{service}`
54/// (default delimiters) and exposes a `pattern` (regex) filter, so we match on
55/// the suffix.
56#[cfg(any(
57    target_os = "macos",
58    all(unix, not(any(target_os = "macos", target_os = "ios")))
59))]
60fn search_spec(store_key: &str) -> HashMap<String, String> {
61    HashMap::from([("service".to_string(), store_key.to_string())])
62}
63
64#[cfg(target_os = "windows")]
65fn search_spec(store_key: &str) -> HashMap<String, String> {
66    // `\Q...\E` quotes the store_key as a literal so any future caller using a
67    // custom (possibly regex-meaningful) store key still gets the expected
68    // entries back.
69    HashMap::from([("pattern".to_string(), format!(r"\.\Q{store_key}\E\z"))])
70}
71
72#[cfg(not(any(
73    target_os = "macos",
74    target_os = "windows",
75    all(unix, not(any(target_os = "macos", target_os = "ios")))
76)))]
77fn search_spec(_store_key: &str) -> HashMap<String, String> {
78    HashMap::new()
79}

81#[derive(Clone, Debug)]
82/// A storage backend that stores credentials in the operating system's keyring
83pub struct KeyringAuthenticationStorage {
84    /// The `store_key` needs to be unique per program as it is stored
85    /// in a global dictionary in the operating system
86    pub store_key: String,
87}
88
89impl KeyringAuthenticationStorage {
90    /// Create a new authentication storage with the given store key
91    pub fn from_key(store_key: &str) -> Self {
92        Self {
93            store_key: store_key.to_string(),
94        }
95    }
96
97    fn entry(&self, host: &str) -> Result<Entry, KeyringAuthenticationStorageError> {
98        configure_default_store()?;
99        Entry::new(&self.store_key, host).map_err(KeyringAuthenticationStorageError::from)
100    }
101}
102
103fn credential_store() -> Result<Arc<CredentialStore>, KeyringAuthenticationStorageError> {
104    configure_default_store()?;
105    keyring_core::get_default_store().ok_or_else(|| {
106        KeyringAuthenticationStorageError::UnsupportedTarget {
107            target: std::env::consts::OS.to_string(),
108        }
109    })
110}

112/// An error that can occur when accessing the authentication storage
113#[derive(thiserror::Error, Debug)]
114pub enum KeyringAuthenticationStorageError {
115    // TODO: make this more fine-grained
116    /// An error occurred when accessing the authentication storage
117    #[error("Could not retrieve credentials from authentication storage: {0}")]
118    StorageError(#[from] keyring_core::Error),
119
120    /// The current target does not have a configured keyring-core store.
121    #[error("No keyring-core credential store is configured for {target}")]
122    UnsupportedTarget {
123        /// Target OS without a configured keyring-core store.
124        target: String,
125    },
126
127    /// An error occurred when serializing the credentials
128    #[error("Could not serialize credentials {0}")]
129    SerializeCredentialsError(#[from] serde_json::Error),
130
131    /// An error occurred when parsing the credentials
132    #[error("Could not parse credentials stored for {host}")]
133    ParseCredentialsError {
134        /// The host for which the credentials could not be parsed
135        host: String,
136    },
137}

139impl Default for KeyringAuthenticationStorage {
140    fn default() -> Self {
141        Self::from_key("rattler")
142    }

145impl StorageBackend for KeyringAuthenticationStorage {
146    fn name(&self) -> String {
147        #[cfg(target_os = "macos")]
148        {
149            "macOS keychain".to_string()
150        }
151        #[cfg(target_os = "windows")]
152        {
153            "Windows credential manager".to_string()
154        }
155        #[cfg(all(unix, not(any(target_os = "macos", target_os = "ios"))))]
156        {
157            "secret service (keyring)".to_string()
158        }
159        #[cfg(not(any(
160            target_os = "macos",
161            target_os = "windows",
162            all(unix, not(any(target_os = "macos", target_os = "ios")))
163        )))]
164        {
165            "keyring".to_string()
166        }
167    }
168
169    fn store(
170        &self,
171        host: &str,
172        authentication: &Authentication,
173    ) -> Result<(), AuthenticationStorageError> {
174        let password = serde_json::to_string(authentication)
175            .map_err(KeyringAuthenticationStorageError::from)?;
176        let entry = self.entry(host)?;
177
178        entry
179            .set_password(&password)
180            .map_err(KeyringAuthenticationStorageError::from)?;
181
182        Ok(())
183    }
184
185    fn get(&self, host: &str) -> Result<Option<Authentication>, AuthenticationStorageError> {
186        let entry = self.entry(host)?;
187        let password = entry.get_password();
188
189        let p_string = match password {
190            Ok(password) => password,
191            Err(keyring_core::Error::NoEntry) => return Ok(None),
192            Err(e) => return Err(KeyringAuthenticationStorageError::from(e))?,
193        };
194
195        match Authentication::from_str(&p_string) {
196            Ok(auth) => Ok(Some(auth)),
197            Err(err) => {
198                tracing::warn!("Error parsing credentials for {}: {:?}", host, err);
199                Err(KeyringAuthenticationStorageError::ParseCredentialsError {
200                    host: host.to_string(),
201                }
202                .into())
203            }
204        }
205    }
206
207    fn list(&self) -> Result<Vec<(String, Authentication)>, AuthenticationStorageError> {
208        let store = credential_store()?;
209        let spec = search_spec(&self.store_key);
210        let spec_refs: HashMap<&str, &str> =
211            spec.iter().map(|(k, v)| (k.as_str(), v.as_str())).collect();
212
213        let entries = store
214            .search(&spec_refs)
215            .map_err(KeyringAuthenticationStorageError::from)?;
216
217        let mut results = Vec::new();
218        for entry in entries {
219            let Some((service, account)) = entry.get_specifiers() else {
220                continue;
221            };
222            // Defensive: on Windows the regex may match credentials whose
223            // service component coincidentally ends in our store_key.
224            if service != self.store_key {
225                continue;
226            }
227
228            let password = match entry.get_password() {
229                Ok(password) => password,
230                Err(keyring_core::Error::NoEntry) => continue,
231                Err(err) => return Err(KeyringAuthenticationStorageError::from(err).into()),
232            };
233
234            match Authentication::from_str(&password) {
235                Ok(auth) => results.push((account, auth)),
236                Err(err) => {
237                    tracing::warn!("Error parsing credentials for {account}: {err:?}");
238                }
239            }
240        }
241
242        results.sort_by(|a, b| a.0.cmp(&b.0));
243        Ok(results)
244    }
245
246    /// Enumerate stored hosts without reading their passwords. On macOS this
247    /// avoids one keychain ACL prompt per entry — important for callers like
248    /// the `auth logout` interactive picker that only need host metadata to
249    /// build their UI.
250    fn list_keys(&self) -> Result<Vec<String>, AuthenticationStorageError> {
251        let store = credential_store()?;
252        let spec = search_spec(&self.store_key);
253        let spec_refs: HashMap<&str, &str> =
254            spec.iter().map(|(k, v)| (k.as_str(), v.as_str())).collect();
255
256        let entries = store
257            .search(&spec_refs)
258            .map_err(KeyringAuthenticationStorageError::from)?;
259
260        let mut hosts = Vec::new();
261        for entry in entries {
262            let Some((service, account)) = entry.get_specifiers() else {
263                continue;
264            };
265            if service != self.store_key {
266                continue;
267            }
268            hosts.push(account);
269        }
270        hosts.sort();
271        Ok(hosts)
272    }
273
274    fn delete(&self, host: &str) -> Result<(), AuthenticationStorageError> {
275        let entry = self.entry(host)?;
276
277        match entry.delete_credential() {
278            Ok(()) | Err(keyring_core::Error::NoEntry) => {}
279            Err(err) => return Err(KeyringAuthenticationStorageError::from(err).into()),
280        }
281
282        Ok(())
283    }
284}

289    use keyring_core::api::{CredentialApi, CredentialStoreApi};

291        Mutex,

295    /// Shared state of [`CountingStore`]: the stored secrets plus a counter
296    /// of how often a secret was actually read.
297    #[derive(Debug, Default)]
298    struct StoreState {
299        secrets: Mutex<HashMap<(String, String), Vec<u8>>>,
300        secret_reads: AtomicUsize,
301    }
302
303    /// In-memory keyring-core store that counts secret reads. Used to assert
304    /// that key enumeration never touches stored secrets — on macOS every
305    /// secret read of a foreign-owned item triggers a keychain ACL prompt,
306    /// so a regression here means one prompt per stored credential.
307    #[derive(Debug, Default)]
308    struct CountingStore {
309        state: Arc<StoreState>,
310    }
311
312    #[derive(Debug)]
313    struct CountingCred {
314        state: Arc<StoreState>,
315        service: String,
316        account: String,
317    }
318
319    impl CountingCred {
320        fn key(&self) -> (String, String) {
321            (self.service.clone(), self.account.clone())
322        }
323    }
324
325    impl CredentialApi for CountingCred {
326        fn set_secret(&self, secret: &[u8]) -> keyring_core::Result<()> {
327            self.state
328                .secrets
329                .lock()
330                .unwrap()
331                .insert(self.key(), secret.to_vec());
332            Ok(())
333        }
334
335        fn get_secret(&self) -> keyring_core::Result<Vec<u8>> {
336            self.state.secret_reads.fetch_add(1, Ordering::SeqCst);
337            self.state
338                .secrets
339                .lock()
340                .unwrap()
341                .get(&self.key())
342                .cloned()
343                .ok_or(keyring_core::Error::NoEntry)
344        }
345
346        fn delete_credential(&self) -> keyring_core::Result<()> {
347            self.state
348                .secrets
349                .lock()
350                .unwrap()
351                .remove(&self.key())
352                .map(|_| ())
353                .ok_or(keyring_core::Error::NoEntry)
354        }
355
356        fn get_credential(
357            &self,
358        ) -> keyring_core::Result<Option<Arc<keyring_core::api::Credential>>> {
359            Ok(None)
360        }
361
362        fn get_specifiers(&self) -> Option<(String, String)> {
363            Some((self.service.clone(), self.account.clone()))
364        }
365
366        fn as_any(&self) -> &dyn std::any::Any {
367            self
368        }
369    }
370
371    impl CountingStore {
372        fn entry(&self, service: &str, account: &str) -> Entry {
373            Entry::new_with_credential(Arc::new(CountingCred {
374                state: self.state.clone(),
375                service: service.to_string(),
376                account: account.to_string(),
377            }))
378        }
379    }
380
381    impl CredentialStoreApi for CountingStore {
382        fn vendor(&self) -> String {
383            "rattler-test".to_string()
384        }
385
386        fn id(&self) -> String {
387            "counting-store".to_string()
388        }
389
390        fn build(
391            &self,
392            service: &str,
393            user: &str,
394            _modifiers: Option<&HashMap<&str, &str>>,
395        ) -> keyring_core::Result<Entry> {
396            Ok(self.entry(service, user))
397        }
398
399        fn search(&self, spec: &HashMap<&str, &str>) -> keyring_core::Result<Vec<Entry>> {
400            // Honor the `service` filter (used on macOS/Linux); for any other
401            // spec (e.g. Windows' `pattern`) return everything and rely on
402            // the caller's defensive service check.
403            let service_filter = spec.get("service").map(ToString::to_string);
404            let entries = self
405                .state
406                .secrets
407                .lock()
408                .unwrap()
409                .keys()
410                .filter(|(service, _)| {
411                    service_filter
412                        .as_ref()
413                        .is_none_or(|filter| service == filter)
414                })
415                .map(|(service, account)| self.entry(service, account))
416                .collect();
417            Ok(entries)
418        }
419
420        fn as_any(&self) -> &dyn std::any::Any {
421            self
422        }
423    }

425    /// The whole point of `list_keys` is to enumerate hosts without reading
426    /// secrets (each read of a foreign-owned item prompts on macOS). Guard
427    /// against a regression to the `list()` fallback, which reads every one.
428    #[test]
429    fn list_keys_does_not_read_secrets() {
430        // The default store is process-global. This must stay the only test
431        // in this binary that installs one: it keeps the real OS keyring out
432        // of every test (configure_default_store never overrides an existing
433        // store) and avoids races between concurrent installers.
434        let store = Arc::new(CountingStore::default());
435        keyring_core::set_default_store(store.clone());
436
437        let backend = KeyringAuthenticationStorage::from_key("rattler-test-list-keys");
438        backend
439            .store(
440                "a.example.com",
441                &Authentication::BearerToken("token-a".into()),
442            )
443            .unwrap();
444        backend
445            .store(
446                "b.example.com",
447                &Authentication::BearerToken("token-b".into()),
448            )
449            .unwrap();
450
451        let keys = backend.list_keys().unwrap();
452        assert_eq!(
453            keys,
454            vec!["a.example.com".to_string(), "b.example.com".to_string()]
455        );
456        assert_eq!(
457            store.state.secret_reads.load(Ordering::SeqCst),
458            0,
459            "listing keys must not read stored secrets"
460        );
461
462        // Sanity-check that the counter counts: a credential lookup reads once.
463        let auth = backend.get("a.example.com").unwrap();
464        assert_eq!(auth, Some(Authentication::BearerToken("token-a".into())));
465        assert_eq!(store.state.secret_reads.load(Ordering::SeqCst), 1);
466    }
