8/// An error occurred when accessing the authentication storage
9#[derive(thiserror::Error, Debug)]
10pub enum AuthenticationStorageError {
11    /// An error occurred when accessing the file storage
12    #[error("FileStorageError")]
13    FileStorageError(#[from] crate::authentication_storage::backends::file::FileStorageError),
14    /// An error occurred when accessing the keyring storage
15    #[cfg(feature = "keyring")]
16    #[error("KeyringStorageError")]
17    KeyringStorageError(
18        #[from] crate::authentication_storage::backends::keyring::KeyringAuthenticationStorageError,
19    ),
20    /// An error occurred when accessing the netrc storage
21    #[cfg(feature = "netrc-rs")]
22    #[error("NetRcStorageError")]
23    NetRcStorageError(#[from] crate::authentication_storage::backends::netrc::NetRcStorageError),
24    /// An error occurred when accessing the memory storage
25    #[error("MemoryStorageError")]
26    MemoryStorageError(#[from] crate::authentication_storage::backends::memory::MemoryStorageError),
27}

29/// A trait that defines the interface for authentication storage backends
30pub trait StorageBackend: std::fmt::Debug {
31    /// A short human-readable description identifying this backend (and any
32    /// relevant location, like a file path). Surfaced to users by the
33    /// `auth status` CLI so they can tell where each credential lives.
34    fn name(&self) -> String;
35
36    /// Store the given authentication information for the given host
37    fn store(
38        &self,
39        host: &str,
40        authentication: &Authentication,
41    ) -> Result<(), AuthenticationStorageError>;
42
43    /// Retrieve the authentication information for the given host
44    fn get(&self, host: &str) -> Result<Option<Authentication>, AuthenticationStorageError>;
45
46    /// List all authentication entries known to this backend.
47    ///
48    /// Some backends, such as platform keyrings, cannot enumerate arbitrary
49    /// legacy entries. They may return only entries that were stored with an
50    /// index maintained by this crate.
51    fn list(&self) -> Result<Vec<(String, Authentication)>, AuthenticationStorageError> {
52        Ok(Vec::new())
53    }
54
55    /// List the host keys this backend holds, *without* decrypting or fetching
56    /// stored secrets.
57    ///
58    /// The default implementation falls back to [`list`](Self::list), which is
59    /// expensive on backends like the macOS keychain where reading each
60    /// credential triggers a per-item ACL prompt. Backends that can enumerate
61    /// metadata cheaply should override this — that lets callers (e.g. an
62    /// interactive picker) show entries to the user without prompting for
63    /// every stored password.
64    fn list_keys(&self) -> Result<Vec<String>, AuthenticationStorageError> {
65        Ok(self.list()?.into_iter().map(|(host, _)| host).collect())
66    }
67
68    /// Delete the authentication information for the given host
69    fn delete(&self, host: &str) -> Result<(), AuthenticationStorageError>;
70}
