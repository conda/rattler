8/// An error occurred when accessing the authentication storage
9#[derive(thiserror::Error, Debug)]
10pub enum AuthenticationStorageError {
11    /// An error occurred when accessing the file storage
12    #[error("FileStorageError")]
13    FileStorageError(#[from] crate::authentication_storage::backends::file::FileStorageError),
14    /// An error occurred when accessing the keyring storage
15    #[cfg(feature = "keyring")]
16    #[error("KeyringStorageError")]
17    KeyringStorageError(
18        #[from] crate::authentication_storage::backends::keyring::KeyringAuthenticationStorageError,
19    ),
20    /// An error occurred when accessing the netrc storage
21    #[cfg(feature = "netrc-rs")]
22    #[error("NetRcStorageError")]
23    NetRcStorageError(#[from] crate::authentication_storage::backends::netrc::NetRcStorageError),
24    /// An error occurred when accessing the memory storage
25    #[error("MemoryStorageError")]
26    MemoryStorageError(#[from] crate::authentication_storage::backends::memory::MemoryStorageError),
27}

29/// A trait that defines the interface for authentication storage backends
30pub trait StorageBackend: std::fmt::Debug {
31    /// Store the given authentication information for the given host
32    fn store(
33        &self,
34        host: &str,
35        authentication: &Authentication,
36    ) -> Result<(), AuthenticationStorageError>;
37
38    /// Retrieve the authentication information for the given host
39    fn get(&self, host: &str) -> Result<Option<Authentication>, AuthenticationStorageError>;
40
41    /// Delete the authentication information for the given host
42    fn delete(&self, host: &str) -> Result<(), AuthenticationStorageError>;
43}
