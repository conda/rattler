186/// Normalize a user-supplied host into its canonical hostname form.
187fn normalize_login_host(host: &str) -> String {
188    let host = host.trim_start_matches("*.");
189
190    // Try parsing as-is first (handles inputs like `https://prefix.dev`).
191    // Only accept the result if it actually yielded a hostname — not every
192    // parse-successful string contains a host component.
193    if let Some(h) = url::Url::parse(host)
194        .ok()
195        .and_then(|u| u.host_str().map(str::to_string))
196    {
197        return h;
198    }
199
200    // Fall back to prepending a scheme (handles bare `prefix.dev`,
201    // `prefix.dev/`, `localhost:8080`, etc.).
202    url::Url::parse(&format!("https://{host}"))
203        .ok()
204        .and_then(|u| u.host_str().map(str::to_string))
205        .unwrap_or_else(|| host.trim_end_matches('/').to_string())
206}

218/// Built-in OAuth defaults for a known host.
219///
220/// Returned by [`default_oauth_config_for_host`] for hosts where rattler
221/// ships an out-of-the-box OAuth configuration. Carries everything needed
222/// to start a login flow without the user passing any flags.
223#[cfg(feature = "oauth")]
224struct DefaultOAuthConfig {
225    issuer_url: String,
226    client_id: String,
227    scopes: Vec<String>,
228    redirect_uri: Option<String>,
229}
230
231/// Returns the built-in OAuth configuration for a host, if rattler ships one.
232#[cfg(feature = "oauth")]
233fn default_oauth_config_for_host(host: &str) -> Option<DefaultOAuthConfig> {
234    let normalized = normalize_login_host(host);
235
236    if !(normalized == "prefix.dev" || normalized.ends_with(".prefix.dev")) {
237        return None;
238    }
239
240    Some(DefaultOAuthConfig {
241        issuer_url: ensure_url_scheme(host),
242        client_id: "rattler".to_string(),
243        scopes: PREFIX_DEV_OAUTH_SCOPES
244            .iter()
245            .map(|&s| s.to_string())
246            .collect(),
247        redirect_uri: None,
248    })
249}

251/// Returns the built-in OAuth config for an implicit (flag-less) login —
252/// i.e. when the user passed no explicit auth method and the host ships
253/// an out-of-the-box OAuth configuration. The presence of `Some` is the
254/// signal that `login()` should fall back to OAuth.
255#[cfg(feature = "oauth")]
256fn default_oauth_for_login(args: &LoginArgs) -> Option<DefaultOAuthConfig> {
257    let no_explicit_method = args.token.is_none()
258        && args.username.is_none()
259        && args.password.is_none()
260        && args.conda_token.is_none()
261        && args.s3_access_key_id.is_none();
262
263    if !no_explicit_method {
264        return None;
265    }
266
267    default_oauth_config_for_host(&args.host)
268}

270fn get_url(url: &str) -> Result<String, AuthenticationCLIError> {
271    // parse as url and extract host without scheme or port
272    let host = if url.contains("://") {
273        url::Url::parse(url)?.host_str().unwrap().to_string()
274    } else {
275        url.to_string()
276    };
277
278    let host = if host.matches('.').count() == 1 {
279        // use wildcard for top-level domains
280        format!("*.{host}")
281    } else {
282        host
283    };
284
285    Ok(host)
286}

288/// Ensure a user-supplied `host` is a fully-qualified URL by prepending
289/// `https://` when it has no scheme.
290fn ensure_url_scheme(host: &str) -> String {
291    if host.contains("://") {
292        host.to_string()
293    } else {
294        format!("https://{host}")
295    }
296}

303    /// Token is invalid or unauthorized
304    Invalid,

307/// Authenticate with a host using the provided credentials.
308///
309/// This function validates the authentication method based on the host and
310/// stores the credentials if successful. For prefix.dev hosts, it validates the
311/// token by making a GraphQL API call.
312async fn login(
313    args: LoginArgs,
314    storage: AuthenticationStorage,
315) -> Result<(), AuthenticationCLIError> {
316    // explicit `--oauth` *or* no explicit method on an OAuth-capable host
317    #[cfg(feature = "oauth")]
318    {
319        let auto_default = default_oauth_for_login(&args);
320        if args.oauth || auto_default.is_some() {
321            if !args.oauth {
322                eprintln!(
323                    "No credentials provided; using OAuth browser login for {}.",
324                    args.host
325                );
326            }
327
328            // Reuse the implicit-default config when present; otherwise
329            // (`--oauth` was set explicitly) fall back to a fresh lookup.
330            let host_default = auto_default.or_else(|| default_oauth_config_for_host(&args.host));
331
332            let issuer_url = args
333                .oauth_issuer_url
334                .or_else(|| host_default.as_ref().map(|c| c.issuer_url.clone()))
335                .unwrap_or_else(|| ensure_url_scheme(&args.host));
336
337            let client_id = args
338                .oauth_client_id
339                .or_else(|| host_default.as_ref().map(|c| c.client_id.clone()))
340                .unwrap_or_else(|| "rattler".to_string());
341
342            let flow = match args.oauth_flow.as_deref() {
343                Some("auth-code") => oauth::OAuthFlow::AuthCode,
344                Some("device-code") => oauth::OAuthFlow::DeviceCode,
345                _ => oauth::OAuthFlow::Auto,
346            };
347
348            let redirect_uri = args
349                .oauth_redirect_uri
350                .or_else(|| host_default.as_ref().and_then(|c| c.redirect_uri.clone()));
351
352            let scopes: std::collections::HashSet<String> = if !args.oauth_scopes.is_empty() {
353                args.oauth_scopes.into_iter().collect()
354            } else if let Some(default) = host_default {
355                default.scopes.into_iter().collect()
356            } else {
357                oauth::DEFAULT_OAUTH_SCOPES
358                    .iter()
359                    .map(|&s| s.to_string())
360                    .collect()
361            };
362
363            let config = oauth::OAuthConfig {
364                issuer_url,
365                client_id,
366                client_secret: args.oauth_client_secret,
367                flow,
368                scopes,
369                redirect_uri,
370                user_agent: args.user_agent,
371            };
372
373            let auth = oauth::perform_oauth_login(config).await?;
374            // Normalize the host so that `prefix.dev` and `prefix.dev/` (and
375            // any `https://...` form) write to the same storage key
376            let host = normalize_login_host(&args.host);
377            storage.store(&host, &auth)?;
378            eprintln!("Credentials stored for {host}.");
379            return Ok(());
380        }
381    }
382
383    let auth = if let Some(conda_token) = args.conda_token {
384        Authentication::CondaToken(conda_token)
385    } else if let Some(username) = args.username {
386        if let Some(password) = args.password {
387            Authentication::BasicHTTP { username, password }
388        } else {
389            return Err(AuthenticationCLIError::MissingPassword);
390        }
391    } else if let Some(token) = args.token {
392        Authentication::BearerToken(token)
393    } else if let (Some(access_key_id), Some(secret_access_key)) =
394        (args.s3_access_key_id, args.s3_secret_access_key)
395    {
396        let session_token = args.s3_session_token;
397        Authentication::S3Credentials {
398            access_key_id,
399            secret_access_key,
400            session_token,
401        }
402    } else {
403        return Err(AuthenticationCLIError::NoAuthenticationMethod);
404    };
405
406    if args.host.contains("prefix.dev") && !matches!(auth, Authentication::BearerToken(_)) {
407        return Err(AuthenticationCLIError::PrefixDevBadMethod);
408    }
409
410    if args.host.contains("anaconda.org") && !matches!(auth, Authentication::CondaToken(_)) {
411        return Err(AuthenticationCLIError::AnacondaOrgBadMethod);
412    }
413
414    if args.host.contains("s3://") && !matches!(auth, Authentication::S3Credentials { .. })
415        || matches!(auth, Authentication::S3Credentials { .. }) && !args.host.contains("s3://")
416    {
417        return Err(AuthenticationCLIError::S3BadMethod);
418    }
419
420    let host = get_url(&args.host)?;
421    eprintln!("Authenticating with {host} using {} method", auth.method());
422
423    // Only validate token for prefix.dev
424    if args.host.contains("prefix.dev") {
425        // Extract the token from BearerToken
426        let token = match &auth {
427            Authentication::BearerToken(t) => t,
428            _ => return Err(AuthenticationCLIError::PrefixDevBadMethod),
429        };
430
431        // Validate the token using the extracted function
432        match validate_prefix_dev_token(token, &args.host, args.user_agent.as_deref()).await? {
433            ValidationResult::Valid(username, url) => {
434                println!(
435                    "✅ Token is valid. Logged into {url} as \"{username}\". Storing credentials..."
436                );
437                // Store the authentication
438                storage.store(&host, &auth)?;
439            }
440            ValidationResult::Invalid => {
441                return Err(AuthenticationCLIError::UnauthorizedToken);
442            }
443        }
444    } else {
445        // For non-prefix.dev hosts, store directly without validation
446        storage.store(&host, &auth)?;
447    }
448    Ok(())
449}

451/// Validates a token with prefix.dev by making a GraphQL API call
452///
453/// Returns `Ok(true)` if the token is valid, `Ok(false)` if invalid,
454/// or `Err` if there was a network/parsing error
455async fn validate_prefix_dev_token(
456    token: &str,
457    host: &str,
458    user_agent: Option<&str>,
459) -> Result<ValidationResult, AuthenticationCLIError> {
460    let prefix_url = if let Ok(env_var) = std::env::var("PREFIX_DEV_API_URL") {
461        // If env var is set, parse it as a full URL
462        Url::parse(&env_var).expect("PREFIX_DEV_API_URL must be a valid URL")
463    } else {
464        // Strip wildcard if given
465        let host = host.replace("*.", "");
466
467        let host_url = Url::parse(&ensure_url_scheme(&host))?;
468
469        let host_url = host_url.host_str().unwrap_or("prefix.dev");
470        // Strip "repo." prefix if present
471        let host_url = host_url.strip_prefix("repo.").unwrap_or(host_url);
472
473        Url::parse(&format!("https://{host_url}")).expect("constructed url must be valid")
474    };
475
476    let body = json!({
477        "query": "query { viewer { login } }"
478    });
479
480    let client = Client::builder()
481        .user_agent(user_agent.unwrap_or(DEFAULT_USER_AGENT))
482        .build()?;
483    let response = client
484        .post(prefix_url.join("api/graphql").expect("must be valid"))
485        .bearer_auth(token)
486        .header(CONTENT_TYPE, "application/json")
487        .json(&body)
488        .send()
489        .await?
490        .error_for_status()?;
491
492    let text = response.text().await?;
493
494    // Parse JSON
495    let json: serde_json::Value = serde_json::from_str(&text)
496        .map_err(|e| AuthenticationCLIError::JsonParseError(e.to_string()))?;
497
498    // Check if viewer is null (invalid token) or contains user data (valid token)
499    match &json["data"]["viewer"] {
500        serde_json::Value::Null => Ok(ValidationResult::Invalid),
501        viewer_data => {
502            if let Some(username) = viewer_data["login"].as_str() {
503                Ok(ValidationResult::Valid(username.to_string(), prefix_url))
504            } else {
505                Ok(ValidationResult::Invalid)
506            }
507        }
508    }
509}

511async fn logout(
512    args: LogoutArgs,
513    storage: AuthenticationStorage,
514) -> Result<(), AuthenticationCLIError> {
515    let host = get_url(&args.host)?;
516
517    // Revoke OAuth tokens before deleting credentials
518    #[cfg(feature = "oauth")]
519    if let Ok(Some(Authentication::OAuth {
520        ref access_token,
521        ref refresh_token,
522        revocation_endpoint: Some(ref revocation_endpoint),
523        ref client_id,
524        ..
525    })) = storage.get(&host)
526    {
527        eprintln!("Revoking OAuth tokens...");
528        oauth::revoke_tokens(
529            revocation_endpoint,
530            access_token,
531            refresh_token.as_deref(),
532            client_id,
533            None,
534        )
535        .await;
536    }
537
538    println!("Removing authentication for {host}");
539
540    storage.delete(&host)?;
541    Ok(())
542}

544/// CLI entrypoint for authentication
545pub async fn execute(args: Args) -> Result<(), AuthenticationCLIError> {
546    let storage = AuthenticationStorage::from_env_and_defaults()?;
547
548    match args.subcommand {
549        Subcommand::Login(args) => login(args, storage).await,
550        Subcommand::Logout(args) => logout(args, storage).await,
551    }
552}

810    #[cfg(feature = "oauth")]
811    #[test]
812    fn test_default_oauth_config_for_host() {
813        let has_default = |h: &str| default_oauth_config_for_host(h).is_some();
814
815        assert!(has_default("prefix.dev"));
816        assert!(has_default("repo.prefix.dev"));
817        assert!(has_default("https://prefix.dev"));
818        assert!(has_default("*.prefix.dev"));
819
820        // Normalization: trailing slash and full URLs should still match.
821        assert!(has_default("prefix.dev/"));
822        assert!(has_default("https://prefix.dev/"));
823        assert!(has_default("https://repo.prefix.dev/"));
824
825        // Loopback addresses are not auto-recognized: local dev servers
826        // could be running anything, so the user passes `--oauth` and
827        // their own `--oauth-scope` flags explicitly.
828        assert!(!has_default("localhost"));
829        assert!(!has_default("localhost:8080"));
830        assert!(!has_default("127.0.0.1"));
831
832        assert!(!has_default("example.com"));
833        // Suffix-injection guard: hostname containing "prefix.dev" must not match.
834        assert!(!has_default("evil-prefix.dev.attacker.com"));
835        assert!(!has_default("notprefix.dev"));
836
837        // Returned config carries the right scheme + client_id for prefix.dev.
838        let prefix = default_oauth_config_for_host("prefix.dev").unwrap();
839        assert_eq!(prefix.issuer_url, "https://prefix.dev");
840        assert_eq!(prefix.client_id, "rattler");
841        assert!(prefix.scopes.iter().any(|s| s == "channel:upload"));
842    }

845    #[test]
846    fn test_default_oauth_for_login() {
847        // No explicit method on prefix.dev → OAuth default kicks in
848        assert!(default_oauth_for_login(&create_login_args("prefix.dev")).is_some());
849
850        // Explicit method blocks the OAuth default, even on prefix.dev.
851        let mut args = create_login_args("prefix.dev");
852        args.token = Some("t".into());
853        assert!(default_oauth_for_login(&args).is_none());
854
855        // No explicit method on a non-OAuth host → still falls through to existing
856        // NoAuthenticationMethod error.
857        assert!(default_oauth_for_login(&create_login_args("example.com")).is_none());
