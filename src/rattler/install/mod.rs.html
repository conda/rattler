1//! This module contains the logic to install a package into a prefix. The main
2//! entry point is the [`link_package`] function.
3//!
4//! The [`link_package`] function takes a package directory and a target
5//! directory. The package directory is the directory that contains the
6//! extracted package archive. The target directory is the directory into which
7//! the package should be installed. The target directory is also called
8//! the "prefix".
9//!
10//! The [`link_package`] function will read the `paths.json` file from the
11//! package directory and link all files specified in that file into the target
12//! directory. The `paths.json` file contains a list of files that should be
13//! installed and how they should be installed. For example, the `paths.json`
14//! file might contain a file that should be copied into the target directory.
15//! Or it might contain a file that should be linked into the target directory.
16//! The `paths.json` file also contains a SHA256 hash for each file. This hash
17//! is used to verify that the file was not tampered with.

80/// An error that might occur when installing a package.
81#[derive(Debug, thiserror::Error)]
82pub enum InstallError {
83    /// The operation was cancelled.
84    #[error("the operation was cancelled")]
85    Cancelled,
86
87    /// The paths.json file could not be read.
88    #[error("failed to read 'paths.json'")]
89    FailedToReadPathsJson(#[source] std::io::Error),
90
91    /// The index.json file could not be read.
92    #[error("failed to read 'index.json'")]
93    FailedToReadIndexJson(#[source] std::io::Error),
94
95    /// The link.json file could not be read.
96    #[error("failed to read 'link.json'")]
97    FailedToReadLinkJson(#[source] std::io::Error),
98
99    /// A file could not be linked.
100    #[error("failed to link '{0}'")]
101    FailedToLink(PathBuf, #[source] LinkFileError),
102
103    /// A directory could not be created.
104    #[error("failed to create directory '{0}'")]
105    FailedToCreateDirectory(PathBuf, #[source] std::io::Error),
106
107    /// The target prefix is not UTF-8.
108    #[error("target prefix is not UTF-8")]
109    TargetPrefixIsNotUtf8,
110
111    /// Failed to create the target directory.
112    #[error("failed to create target directory")]
113    FailedToCreateTargetDirectory(#[source] std::io::Error),
114
115    /// A noarch package could not be installed because no python version was
116    /// specified.
117    #[error("cannot install noarch python package because there is no python version specified")]
118    MissingPythonInfo,
119
120    /// Failed to create a python entry point for a noarch package.
121    #[error("failed to create Python entry point")]
122    FailedToCreatePythonEntryPoint(#[source] std::io::Error),
123
124    /// When post-processing of the environment fails.
125    /// Post-processing involves removing clobbered paths.
126    #[error("failed to post process the environment (unclobbering)")]
127    PostProcessFailed(#[source] std::io::Error),
128}
129
130impl From<Cancelled> for InstallError {
131    fn from(_: Cancelled) -> Self {
132        InstallError::Cancelled
133    }
134}
135
136impl From<JoinError> for InstallError {
137    fn from(err: JoinError) -> Self {
138        if let Ok(panic) = err.try_into_panic() {
139            std::panic::resume_unwind(panic)
140        } else {
141            InstallError::Cancelled
142        }
143    }
144}

146/// Additional options to pass to [`link_package`] to modify the installation
147/// process. Using [`InstallOptions::default`] works in most cases unless you
148/// want specific control over the installation process.
149#[derive(Default, Clone)]
150pub struct InstallOptions {
151    /// When files are copied/linked to the target directory hardcoded paths in
152    /// these files are "patched". The hardcoded paths are replaced with the
153    /// full path of the target directory, also called the "prefix".
154    ///
155    /// However, in exceptional cases you might want to use a different prefix
156    /// than the one that is being installed to. This field allows you to do
157    /// that. When its set this is used instead of the target directory.
158    pub target_prefix: Option<PathBuf>,
159
160    /// Instead of reading the `paths.json` file from the package directory
161    /// itself, use the data specified here.
162    ///
163    /// This is sometimes useful to avoid reading the file twice or when you
164    /// want to modify installation process externally.
165    pub paths_json: Option<PathsJson>,
166
167    /// Instead of reading the `index.json` file from the package directory
168    /// itself, use the data specified here.
169    ///
170    /// This is sometimes useful to avoid reading the file twice or when you
171    /// want to modify installation process externally.
172    pub index_json: Option<IndexJson>,
173
174    /// Instead of reading the `link.json` file from the package directory
175    /// itself, use the data specified here.
176    ///
177    /// This is sometimes useful to avoid reading the file twice or when you
178    /// want to modify installation process externally.
179    ///
180    /// Because the the `link.json` file is optional this fields is using a
181    /// doubly wrapped Option. The first `Option` is to indicate whether or
182    /// not this value is set. The second Option is the [`LinkJson`] to use
183    /// or `None` if you want to force that there is no [`LinkJson`].
184    ///
185    /// This struct is only used if the package to be linked is a noarch Python
186    /// package.
187    pub link_json: Option<Option<LinkJson>>,
188
189    /// Whether or not to use symbolic links where possible. If this is set to
190    /// `Some(false)` symlinks are disabled, if set to `Some(true)` symbolic
191    /// links are always used when specified in the `info/paths.json` file
192    /// even if this is not supported. If the value is set to `None`
193    /// symbolic links are only used if they are supported.
194    ///
195    /// Windows only supports symbolic links in specific cases.
196    pub allow_symbolic_links: Option<bool>,
197
198    /// Whether or not to use hard links where possible. If this is set to
199    /// `Some(false)` the use of hard links is disabled, if set to
200    /// `Some(true)` hard links are always used when specified
201    /// in the `info/paths.json` file even if this is not supported. If the
202    /// value is set to `None` hard links are only used if they are
203    /// supported. A dummy hardlink is created to determine support.
204    ///
205    /// Hard links are supported by most OSes but often require that the hard
206    /// link and its content are on the same filesystem.
207    pub allow_hard_links: Option<bool>,
208
209    /// Whether or not to use ref links where possible. If this is set to
210    /// `Some(false)` the use of hard links is disabled, if set to
211    /// `Some(true)` ref links are always used when hard links are specified
212    /// in the `info/paths.json` file even if this is not supported. If the
213    /// value is set to `None` ref links are only used if they are
214    /// supported.
215    ///
216    /// Ref links are only support by a small number of OSes and filesystems. If
217    /// reflinking fails for whatever reason the files are hardlinked
218    /// instead (if allowed).
219    pub allow_ref_links: Option<bool>,
220
221    /// The platform for which the package is installed. Some operations like
222    /// signing require different behavior depending on the platform. If the
223    /// field is set to `None` the current platform is used.
224    pub platform: Option<Platform>,
225
226    /// Python version information of the python distribution installed within
227    /// the environment. This is only used when installing noarch Python
228    /// packages. Noarch python packages are python packages that contain
229    /// python source code that has to be installed in the correct
230    /// site-packages directory based on the version of python. This
231    /// site-packages directory depends on the version of python, therefor
232    /// it must be provided when linking.
233    ///
234    /// If you're installing a noarch python package and do not provide this
235    /// field, the [`link_package`] function will return
236    /// [`InstallError::MissingPythonInfo`].
237    pub python_info: Option<PythonInfo>,
238
239    /// For binaries on macOS (both Intel and Apple Silicon), binaries need to be signed
240    /// with an ad-hoc certificate to properly work when their signature has been invalidated
241    /// by prefix replacement (modifying binary content). This field controls whether or not to do that.
242    /// Code signing is executed when the target platform is macOS. By default, code signing
243    /// will fail the installation if it fails. This behavior can be changed by setting
244    /// this field to `AppleCodeSignBehavior::Ignore` or
245    /// `AppleCodeSignBehavior::DoNothing`.
246    ///
247    /// To sign the binaries, the `/usr/bin/codesign` executable is called with
248    /// `--force`, `--sign -`, and `--preserve-metadata=entitlements` arguments.
249    /// The `--force` argument is used to overwrite existing signatures, the `--sign -`
250    /// argument is used to sign with an ad-hoc certificate (which does not use an identity
251    /// at all), and `--preserve-metadata=entitlements` preserves the original entitlements
252    /// from the binary (required for programs that need specific permissions like virtualization).
253    pub apple_codesign_behavior: AppleCodeSignBehavior,
254
255    /// Controls how symlinks that point outside the target prefix are handled.
256    /// Some packages (e.g. driver packages) legitimately ship symlinks to paths
257    /// outside the environment.
258    pub external_symlink_policy: ExternalSymlinkPolicy,
259}

280/// Find a timestamp to put on all files we modify. Use `info/about.json`, as a base. This
281/// file is always written after all the packaged data is already stored.
282///
283/// We add 1s to make sure our modification time is strictly bigger than any timestamp
284/// seen in the package data, even on systems that do not have nanosecond timestamps.
285///
286/// Fall back to `now()` if we can not detect a file time.
287fn modification_time(package_dir: &Path) -> filetime::FileTime {
288    if let Ok(info_metadata) = fs_err::symlink_metadata(package_dir.join(AboutJson::package_path()))
289    {
290        let info_time = filetime::FileTime::from_last_modification_time(&info_metadata);
291
292        filetime::FileTime::from_unix_time(info_time.unix_seconds() + 1, info_time.nanoseconds())
293    } else {
294        filetime::FileTime::now()
295    }
296}

298/// Given an extracted package archive (`package_dir`), installs its files to
299/// the `target_dir`.
300///
301/// Returns a [`PathsEntry`] for every file that was linked into the target
302/// directory. The entries are ordered in the same order as they appear in the
303/// `paths.json` file of the package.
304#[instrument(skip_all, fields(package_dir = % package_dir.display()))]
305pub async fn link_package(
306    package_dir: &Path,
307    target_dir: &Prefix,
308    driver: &InstallDriver,
309    options: InstallOptions,
310) -> Result<Vec<prefix_record::PathsEntry>, InstallError> {
311    // Determine the target prefix for linking
312    let target_prefix = options
313        .target_prefix
314        .as_deref()
315        .unwrap_or(target_dir)
316        .to_str()
317        .ok_or(InstallError::TargetPrefixIsNotUtf8)?
318        .to_owned();
319
320    // Reuse or read the `paths.json` and `index.json` files from the package
321    // directory
322    let paths_json = read_paths_json(package_dir, driver, options.paths_json);
323    let index_json = read_index_json(package_dir, driver, options.index_json);
324    let (paths_json, index_json) = tokio::try_join!(paths_json, index_json)?;
325
326    let modification_time = modification_time(package_dir);
327
328    // Error out if this is a noarch python package but the python information is
329    // missing.
330    if index_json.noarch.is_python() && options.python_info.is_none() {
331        return Err(InstallError::MissingPythonInfo);
332    }
333
334    // Parse the `link.json` file and extract entry points from it.
335    let link_json = if index_json.noarch.is_python() {
336        read_link_json(package_dir, driver, options.link_json.flatten()).await?
337    } else {
338        None
339    };
340
341    // Determine whether or not we can use symbolic links
342    let (allow_symbolic_links, allow_hard_links) = tokio::join!(
343        // Determine if we can use symlinks
344        match options.allow_symbolic_links {
345            Some(value) => ready(value).left_future(),
346            None => can_create_symlinks(target_dir).right_future(),
347        },
348        // Determine if we can use hard links
349        match options.allow_hard_links {
350            Some(value) => ready(value).left_future(),
351            None => can_create_hardlinks(target_dir, package_dir).right_future(),
352        },
353    );
354    let allow_ref_links = options.allow_ref_links.unwrap_or_else(|| {
355        match reflink_copy::check_reflink_support(package_dir, target_dir) {
356            Ok(reflink_copy::ReflinkSupport::Supported) => true,
357            Ok(reflink_copy::ReflinkSupport::NotSupported) | Err(_) => false,
358            Ok(reflink_copy::ReflinkSupport::Unknown) => allow_hard_links,
359        }
360    });
361
362    // Determine the platform to use
363    let platform = options.platform.unwrap_or(Platform::current());
364
365    // compute all path renames
366    let final_paths = compute_paths(&index_json, &paths_json, options.python_info.as_ref());
367
368    // register all paths in the install driver path registry
369    let clobber_paths = Arc::new(
370        driver
371            .clobber_registry()
372            .register_paths(&index_json, &final_paths),
373    );
374
375    let final_paths: Vec<LinkPath> = final_paths
376        .into_iter()
377        .map(|el| {
378            let (entry, computed_path) = el;
379            let clobber_path = clobber_paths.get(&computed_path).cloned();
380            LinkPath {
381                entry,
382                computed_path,
383                clobber_path,
384            }
385        })
386        .collect();
387
388    // Figure out all the directories that we are going to need
389    let mut directories_to_construct = HashSet::new();
390    for link_path in &final_paths {
391        let mut current_path = link_path.computed_path.parent();
392        while let Some(path) = current_path {
393            if !path.as_os_str().is_empty() && directories_to_construct.insert(path.to_path_buf()) {
394                current_path = path.parent();
395            } else {
396                break;
397            }
398        }
399
400        // Since we store clobbers in the separate directory
401        // (`__clobbers__`) now we have to create all necessary
402        // directories for it as well.
403        let clobber_path = link_path.clobber_path.as_ref();
404        let mut current_clobber_path = clobber_path.and_then(|p| p.parent());
405        while let Some(path) = current_clobber_path {
406            if !path.as_os_str().is_empty() && directories_to_construct.insert(path.to_path_buf()) {
407                current_clobber_path = path.parent();
408            } else {
409                break;
410            }
411        }
412    }
413
414    let directories_target_dir = target_dir.path().to_path_buf();
415    driver
416        .run_blocking_io_task(move || {
417            for directory in directories_to_construct.into_iter().sorted() {
418                let full_path = directories_target_dir.join(directory);
419                match fs::create_dir(&full_path) {
420                    Ok(_) => (),
421                    Err(e) if e.kind() == ErrorKind::AlreadyExists => (),
422                    Err(e) => return Err(InstallError::FailedToCreateDirectory(full_path, e)),
423                }
424            }
425            Ok(())
426        })
427        .await?;
428
429    // Wrap the python info in an `Arc` so we can more easily share it with async
430    // tasks.
431    let python_info = options.python_info.map(Arc::new);
432
433    // Start linking all package files in parallel
434    let mut pending_futures = FuturesUnordered::new();
435    let mut number_of_paths_entries = 0;
436    for link_path in final_paths {
437        let entry = link_path.entry;
438        let package_dir = package_dir.to_owned();
439        let target_dir = target_dir.to_owned();
440        let target_prefix = target_prefix.clone();
441
442        let install_future = async move {
443            let _permit = driver.acquire_io_permit().await;
444
445            // Spawn a blocking task to link the specific file. We use a blocking task here
446            // because filesystem access is blocking anyway so its more
447            // efficient to group them together in a single blocking call.
448            let cloned_entry = entry.clone();
449            let is_clobber = link_path.clobber_path.is_some();
450            let result = match tokio::task::spawn_blocking(move || {
451                link_file(
452                    &cloned_entry,
453                    link_path.clobber_path.unwrap_or(link_path.computed_path),
454                    &package_dir,
455                    &target_dir,
456                    &target_prefix,
457                    allow_symbolic_links && !cloned_entry.no_link,
458                    allow_hard_links && !cloned_entry.no_link,
459                    allow_ref_links && !cloned_entry.no_link,
460                    platform,
461                    options.apple_codesign_behavior,
462                    modification_time,
463                    options.external_symlink_policy,
464                )
465            })
466            .await
467            .map_err(JoinError::try_into_panic)
468            {
469                Ok(Ok(linked_file)) => linked_file,
470                Ok(Err(e)) => {
471                    return Err(InstallError::FailedToLink(entry.relative_path.clone(), e));
472                }
473                Err(Ok(payload)) => std::panic::resume_unwind(payload),
474                Err(Err(_err)) => return Err(InstallError::Cancelled),
475            };
476
477            // Construct a `PathsEntry` from the result of the linking operation
478            let paths_entry = prefix_record::PathsEntry {
479                relative_path: result.relative_path,
480                original_path: if is_clobber {
481                    Some(entry.relative_path)
482                } else {
483                    None
484                },
485                path_type: entry.path_type.into(),
486                no_link: entry.no_link,
487                sha256: entry.sha256,
488                // Only set sha256_in_prefix if it differs from the original sha256
489                sha256_in_prefix: if Some(result.sha256) == entry.sha256 {
490                    None
491                } else {
492                    Some(result.sha256)
493                },
494                size_in_bytes: Some(result.file_size),
495                file_mode: match result.method {
496                    LinkMethod::Patched(file_mode) => Some(file_mode),
497                    _ => None,
498                },
499                prefix_placeholder: entry
500                    .prefix_placeholder
501                    .as_ref()
502                    .map(|p| p.placeholder.clone()),
503            };
504
505            Ok(vec![(number_of_paths_entries, paths_entry)])
506        };
507
508        pending_futures.push(install_future.boxed());
509        number_of_paths_entries += 1;
510    }
511
512    // If this package is a noarch python package we also have to create entry
513    // points.
514    //
515    // Be careful with the fact that this code is currently running in parallel with
516    // the linking of individual files.
517    if let Some(link_json) = link_json {
518        // Parse the `link.json` file and extract entry points from it.
519        let entry_points = match link_json.noarch {
520            NoArchLinks::Python(entry_points) => entry_points.entry_points,
521            NoArchLinks::Generic => {
522                unreachable!("we only use link.json for noarch: python packages")
523            }
524        };
525
526        // Get python info
527        let python_info = python_info
528            .clone()
529            .expect("should be safe because its checked above that this contains a value");
530
531        // Create entry points for each listed item. This is different between Windows
532        // and unix because on Windows, two PathEntry's are created whereas on
533        // Linux only one is created.
534        for entry_point in entry_points {
535            let python_info = python_info.clone();
536            let target_dir = target_dir.to_owned();
537            let target_prefix = target_prefix.clone();
538
539            let entry_point_fut = async move {
540                // Acquire an IO permit
541                let _permit = driver.acquire_io_permit().await;
542
543                let entries = if platform.is_windows() {
544                    match create_windows_python_entry_point(
545                        &target_dir,
546                        &target_prefix,
547                        &entry_point,
548                        &python_info,
549                        &platform,
550                    ) {
551                        Ok([a, b]) => vec![
552                            (number_of_paths_entries, a),
553                            (number_of_paths_entries + 1, b),
554                        ],
555                        Err(e) => return Err(InstallError::FailedToCreatePythonEntryPoint(e)),
556                    }
557                } else {
558                    match create_unix_python_entry_point(
559                        &target_dir,
560                        &target_prefix,
561                        &entry_point,
562                        &python_info,
563                    ) {
564                        Ok(a) => vec![(number_of_paths_entries, a)],
565                        Err(e) => return Err(InstallError::FailedToCreatePythonEntryPoint(e)),
566                    }
567                };
568
569                Ok(entries)
570            };
571
572            pending_futures.push(entry_point_fut.boxed());
573            number_of_paths_entries += if platform.is_windows() { 2 } else { 1 };
574        }
575    }
576
577    // Await the result of all the background tasks. The background tasks are
578    // scheduled in order, however, they can complete in any order. This means
579    // we have to reorder them back into their original order. This is achieved
580    // by waiting to add finished results to the result Vec, if the result
581    // before it has not yet finished. To that end we use a `BinaryHeap` as a
582    // priority queue which will buffer up finished results that finished before
583    // their predecessor.
584    //
585    // What makes this loop special is that it also aborts if any of the returned
586    // results indicate a failure.
587    let mut paths = Vec::with_capacity(number_of_paths_entries);
588    let mut out_of_order_queue =
589        BinaryHeap::<OrderWrapper<prefix_record::PathsEntry>>::with_capacity(100);
590    while let Some(link_result) = pending_futures.next().await {
591        for (index, data) in link_result? {
592            if index == paths.len() {
593                // If this is the next element expected in the sorted list, add it immediately.
594                // This basically means the future finished in order.
595                paths.push(data);
596
597                // By adding a finished future we have to check if there might also be another
598                // future that finished earlier and should also now be added to
599                // the result Vec.
600                while let Some(next_output) = out_of_order_queue.peek_mut() {
601                    if next_output.index == paths.len() {
602                        paths.push(PeekMut::pop(next_output).data);
603                    } else {
604                        break;
605                    }
606                }
607            } else {
608                // Otherwise add it to the out-of-order queue. This means that we still have to
609                // wait for another element before we can add the result to the
610                // ordered list.
611                out_of_order_queue.push(OrderWrapper { index, data });
612            }
613        }
614    }
615    debug_assert_eq!(
616        paths.len(),
617        paths.capacity(),
618        "some futures where not added to the result"
619    );
620
621    Ok(paths)
622}

624/// Given an extracted package archive (`package_dir`), installs its files to
625/// the `target_dir`.
626///
627/// Returns a [`PathsEntry`] for every file that was linked into the target
628/// directory. The entries are ordered in the same order as they appear in the
629/// `paths.json` file of the package.
630#[instrument(skip_all, fields(package_dir = % package_dir.display()))]
631pub fn link_package_sync(
632    package_dir: &Path,
633    target_dir: &Prefix,
634    clobber_registry: Arc<Mutex<ClobberRegistry>>,
635    options: InstallOptions,
636) -> Result<(Vec<prefix_record::PathsEntry>, LinkType), InstallError> {
637    // Determine the target prefix for linking
638    let target_prefix = options
639        .target_prefix
640        .as_deref()
641        .unwrap_or(target_dir)
642        .to_str()
643        .ok_or(InstallError::TargetPrefixIsNotUtf8)?
644        .to_owned();
645
646    // Reuse or read the `paths.json` and `index.json` files from the package
647    // directory
648    let paths_json = options.paths_json.map_or_else(
649        || {
650            PathsJson::from_package_directory_with_deprecated_fallback(package_dir)
651                .map_err(InstallError::FailedToReadPathsJson)
652        },
653        Ok,
654    )?;
655    let index_json = options.index_json.map_or_else(
656        || {
657            IndexJson::from_package_directory(package_dir)
658                .map_err(InstallError::FailedToReadIndexJson)
659        },
660        Ok,
661    )?;
662    let modification_time = modification_time(package_dir);
663
664    // Error out if this is a noarch python package but the python information is
665    // missing.
666    if index_json.noarch.is_python() && options.python_info.is_none() {
667        return Err(InstallError::MissingPythonInfo);
668    }
669
670    // Parse the `link.json` file and extract entry points from it.
671    let link_json = if index_json.noarch.is_python() {
672        options.link_json.flatten().map_or_else(
673            || {
674                LinkJson::from_package_directory(package_dir)
675                    .map_or_else(
676                        |e| {
677                            // Its ok if the file is not present.
678                            if e.kind() == ErrorKind::NotFound {
679                                Ok(None)
680                            } else {
681                                Err(e)
682                            }
683                        },
684                        |link_json| Ok(Some(link_json)),
685                    )
686                    .map_err(InstallError::FailedToReadLinkJson)
687            },
688            |value| Ok(Some(value)),
689        )?
690    } else {
691        None
692    };
693
694    // Determine whether or not we can use symbolic links
695    let allow_symbolic_links = options
696        .allow_symbolic_links
697        .unwrap_or_else(|| can_create_symlinks_sync(target_dir));
698    let allow_hard_links = options
699        .allow_hard_links
700        .unwrap_or_else(|| can_create_hardlinks_sync(target_dir, package_dir));
701    // Record the link type that will be used for this package. Hard links take
702    // priority
703    let link_type = if allow_hard_links {
704        LinkType::HardLink
705    } else {
706        LinkType::Copy
707    };
708    let mut allow_ref_links = options.allow_ref_links.unwrap_or_else(|| {
709        match reflink_copy::check_reflink_support(package_dir, target_dir) {
710            Ok(reflink_copy::ReflinkSupport::Supported) => true,
711            Ok(reflink_copy::ReflinkSupport::NotSupported) | Err(_) => false,
712            Ok(reflink_copy::ReflinkSupport::Unknown) => allow_hard_links,
713        }
714    });
715
716    // Determine the platform to use
717    let platform = options.platform.unwrap_or(Platform::current());
718
719    // compute all path renames
720    let final_paths = compute_paths(&index_json, &paths_json, options.python_info.as_ref());
721
722    // register all paths in the install driver path registry
723    let clobber_paths = clobber_registry
724        .lock()
725        .unwrap()
726        .register_paths(&index_json, &final_paths);
727
728    let final_paths = final_paths.into_iter().map(|el| {
729        let (entry, computed_path) = el;
730        let clobber_path = clobber_paths.get(&computed_path).cloned();
731        LinkPath {
732            entry,
733            computed_path,
734            clobber_path,
735        }
736    });
737
738    // Figure out all the directories that we are going to need
739    let mut directories_to_construct = HashSet::new();
740    let mut paths_by_directory = HashMap::new();
741    for link_path in final_paths {
742        let Some(entry_parent) = link_path.computed_path.parent() else {
743            continue;
744        };
745
746        // Iterate over all parent directories and create them if they do not exist.
747        let mut current_path = Some(entry_parent);
748        while let Some(path) = current_path {
749            if !path.as_os_str().is_empty() && directories_to_construct.insert(path.to_path_buf()) {
750                current_path = path.parent();
751            } else {
752                break;
753            }
754        }
755
756        // Since we store clobbers in the separate directory
757        // (`__clobbers__`) now we have to create all necessary
758        // directories for it as well.
759        let clobber_path = link_path.clobber_path.as_ref();
760        let mut current_path = clobber_path.and_then(|p| p.parent());
761        while let Some(path) = current_path {
762            if !path.as_os_str().is_empty() && directories_to_construct.insert(path.to_path_buf()) {
763                current_path = path.parent();
764            } else {
765                break;
766            }
767        }
768
769        // Store the path by directory so we can create them in parallel
770        paths_by_directory
771            .entry(entry_parent.to_path_buf())
772            .or_insert_with(Vec::new)
773            .push(link_path);
774    }
775
776    let mut created_directories = HashSet::new();
777    let mut reflinked_files = HashMap::new();
778    for directory in directories_to_construct
779        .into_iter()
780        .sorted_by(|a, b| a.components().count().cmp(&b.components().count()))
781    {
782        let full_path = target_dir.path().join(&directory);
783
784        // if we already (recursively) created the parent directory we can skip this
785        if created_directories
786            .iter()
787            .any(|dir| directory.starts_with(dir))
788        {
789            continue;
790        }
791
792        // can we lock this directory?
793        if full_path.exists() {
794            continue;
795        }
796
797        if allow_ref_links
798            && cfg!(target_os = "macos")
799            && !directory.starts_with(CLOBBERS_DIR_NAME)
800            && !index_json.noarch.is_python()
801        {
802            // reflink the whole directory if possible
803            // currently this does not handle noarch packages
804            match reflink_copy::reflink(package_dir.join(&directory), &full_path) {
805                Ok(_) => {
806                    created_directories.insert(directory.clone());
807                    // remove paths that we just reflinked (everything that starts with the directory)
808                    let (matching, non_matching): (HashMap<_, _>, HashMap<_, _>) =
809                        paths_by_directory
810                            .drain()
811                            .partition(|(k, _)| k.starts_with(&directory));
812
813                    // Store matching paths in reflinked_files
814                    reflinked_files.extend(matching);
815                    // Keep non-matching paths in paths_by_directory
816                    paths_by_directory = non_matching;
817                }
818                Err(e) if e.kind() == ErrorKind::AlreadyExists => (),
819                Err(_) => {
820                    allow_ref_links = false;
821                    match fs::create_dir(&full_path) {
822                        Ok(_) => (),
823                        Err(e) if e.kind() == ErrorKind::AlreadyExists => (),
824                        Err(e) => return Err(InstallError::FailedToCreateDirectory(full_path, e)),
825                    }
826                }
827            }
828        } else {
829            match fs::create_dir(&full_path) {
830                Ok(_) => (),
831                Err(e) if e.kind() == ErrorKind::AlreadyExists => (),
832                Err(e) => return Err(InstallError::FailedToCreateDirectory(full_path, e)),
833            }
834        }
835    }
836
837    // Take care of all the reflinked files (macos only)
838    //  - Add them to the paths.json
839    //  - Fix any occurrences of the prefix in the files
840    //  - Rename files that need clobber-renames
841    let mut reflinked_paths_entries = Vec::new();
842    for (parent_dir, files) in reflinked_files {
843        // files that are either in the clobber map or contain a placeholder,
844        // we defer to the regular linking that comes after this block
845        // and re-add them to the paths_by_directory map
846        for link_path in files {
847            if link_path.clobber_path.is_some() || link_path.entry.prefix_placeholder.is_some() {
848                paths_by_directory
849                    .entry(parent_dir.clone())
850                    .or_insert_with(Vec::new)
851                    .push(link_path);
852            } else {
853                let entry = link_path.entry;
854                reflinked_paths_entries.push(prefix_record::PathsEntry {
855                    relative_path: entry.relative_path,
856                    path_type: entry.path_type.into(),
857                    no_link: entry.no_link,
858                    sha256: entry.sha256,
859                    size_in_bytes: entry.size_in_bytes,
860                    // No placeholder, no clobbering, so these are none for sure
861                    original_path: None,
862                    sha256_in_prefix: None,
863                    file_mode: None,
864                    prefix_placeholder: None,
865                });
866            }
867        }
868    }
869
870    // Wrap the python info in an `Arc` so we can more easily share it with async
871    // tasks.
872    let python_info = options.python_info;
873
874    // Link the individual files in parallel
875    let link_target_prefix = target_prefix.clone();
876    let package_dir = package_dir.to_path_buf();
877    let mut paths = paths_by_directory
878        .into_values()
879        .collect_vec()
880        .into_par_iter()
881        .with_min_len(100)
882        .flat_map(move |entries_in_subdir| {
883            let mut path_entries = Vec::with_capacity(entries_in_subdir.len());
884            for link_path in entries_in_subdir {
885                let entry = link_path.entry;
886
887                let is_clobber = link_path.clobber_path.is_some();
888                let link_result = link_file(
889                    &entry,
890                    link_path
891                        .clobber_path
892                        .unwrap_or(link_path.computed_path.clone()),
893                    &package_dir,
894                    target_dir,
895                    &link_target_prefix,
896                    allow_symbolic_links && !entry.no_link,
897                    allow_hard_links && !entry.no_link,
898                    allow_ref_links && !entry.no_link,
899                    platform,
900                    options.apple_codesign_behavior,
901                    modification_time,
902                    options.external_symlink_policy,
903                );
904
905                let result = match link_result {
906                    Ok(linked_file) => linked_file,
907                    Err(e) => {
908                        return vec![Err(InstallError::FailedToLink(
909                            entry.relative_path.clone(),
910                            e,
911                        ))];
912                    }
913                };
914
915                // Construct a `PathsEntry` from the result of the linking operation
916                path_entries.push(Ok(prefix_record::PathsEntry {
917                    relative_path: result.relative_path,
918                    original_path: if is_clobber {
919                        Some(link_path.computed_path)
920                    } else {
921                        None
922                    },
923                    path_type: entry.path_type.into(),
924                    no_link: entry.no_link,
925                    sha256: entry.sha256,
926                    // Only set sha256_in_prefix if it differs from the original sha256
927                    sha256_in_prefix: if Some(result.sha256) == entry.sha256 {
928                        None
929                    } else {
930                        Some(result.sha256)
931                    },
932                    size_in_bytes: Some(result.file_size),
933                    file_mode: match result.method {
934                        LinkMethod::Patched(file_mode) => Some(file_mode),
935                        _ => None,
936                    },
937                    prefix_placeholder: entry
938                        .prefix_placeholder
939                        .as_ref()
940                        .map(|p| p.placeholder.clone()),
941                }));
942            }
943
944            path_entries
945        })
946        .collect::<Result<Vec<_>, _>>()?;
947
948    paths.extend(reflinked_paths_entries);
949
950    // If this package is a noarch python package we also have to create entry
951    // points.
952    //
953    // Be careful with the fact that this code is currently running in parallel with
954    // the linking of individual files.
955    if let Some(link_json) = link_json {
956        // Parse the `link.json` file and extract entry points from it.
957        let entry_points = match link_json.noarch {
958            NoArchLinks::Python(entry_points) => entry_points.entry_points,
959            NoArchLinks::Generic => {
960                unreachable!("we only use link.json for noarch: python packages")
961            }
962        };
963
964        // Get python info
965        let python_info = python_info
966            .clone()
967            .expect("should be safe because its checked above that this contains a value");
968
969        let target_prefix = target_prefix.clone();
970
971        // Create entry points for each listed item. This is different between Windows
972        // and unix because on Windows, two PathEntry's are created whereas on
973        // Linux only one is created.
974        let mut entry_point_paths = if platform.is_windows() {
975            entry_points
976                .into_iter()
977                // .into_par_iter()
978                // .with_min_len(100)
979                .flat_map(move |entry_point| {
980                    match create_windows_python_entry_point(
981                        target_dir,
982                        &target_prefix,
983                        &entry_point,
984                        &python_info,
985                        &platform,
986                    ) {
987                        Ok([a, b]) => Either::Left([Ok(a), Ok(b)].into_iter()),
988                        Err(e) => Either::Right(std::iter::once(Err(
989                            InstallError::FailedToCreatePythonEntryPoint(e),
990                        ))),
991                    }
992                })
993                .collect::<Result<Vec<_>, _>>()?
994        } else {
995            entry_points
996                .into_iter()
997                // .into_par_iter()
998                // .with_min_len(100)
999                .map(move |entry_point| {
1000                    match create_unix_python_entry_point(
1001                        target_dir,
1002                        &target_prefix,
1003                        &entry_point,
1004                        &python_info,
1005                    ) {
1006                        Ok(a) => Ok(a),
1007                        Err(e) => Err(InstallError::FailedToCreatePythonEntryPoint(e)),
1008                    }
1009                })
1010                .collect::<Result<_, _>>()?
1011        };
1012
1013        paths.append(&mut entry_point_paths);
1014    };
1015
1016    Ok((paths, link_type))
1017}

1019fn compute_paths(
1020    index_json: &IndexJson,
1021    paths_json: &PathsJson,
1022    python_info: Option<&PythonInfo>,
1023) -> Vec<(rattler_conda_types::package::PathsEntry, PathBuf)> {
1024    let mut final_paths = Vec::with_capacity(paths_json.paths.len());
1025    for entry in &paths_json.paths {
1026        let path = if index_json.noarch.is_python() {
1027            python_info
1028                .unwrap()
1029                .get_python_noarch_target_path(&entry.relative_path)
1030                .to_path_buf()
1031        } else {
1032            entry.relative_path.clone()
1033        };
1034
1035        final_paths.push((entry.clone(), path));
1036    }
1037    final_paths
1038}

1114    let symlink_path = target_dir.path().join(format!("symtest_{uuid}"));
1115    #[cfg(windows)]

1117    #[cfg(unix)]

1119    match result {

1128        }
1129        Err(e) => {
1130            tracing::debug!(

1132            );
1133            false
1134        }

1138/// A helper struct for a `BinaryHeap` to provides ordering to items that are
1139/// otherwise unordered.
1140struct OrderWrapper<T> {
1141    index: usize,
1142    data: T,
1143}
1144
1145impl<T> PartialEq for OrderWrapper<T> {
1146    fn eq(&self, other: &Self) -> bool {
1147        self.index == other.index
1148    }
1149}
1150
1151impl<T> Eq for OrderWrapper<T> {}
1152
1153impl<T> PartialOrd for OrderWrapper<T> {
1154    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
1155        Some(self.cmp(other))
1156    }
1157}
1158
1159impl<T> Ord for OrderWrapper<T> {
1160    fn cmp(&self, other: &Self) -> Ordering {
1161        // BinaryHeap is a max heap, so compare backwards here.
1162        other.index.cmp(&self.index)
1163    }
1164}

1205/// Returns true if two paths share the same filesystem
1206#[cfg(unix)]
1207async fn paths_have_same_filesystem(a: &Prefix, b: &Path) -> bool {
1208    use std::os::unix::fs::MetadataExt;
1209    match tokio::join!(tokio_fs::metadata(a.path()), tokio_fs::metadata(b)) {
1210        (Ok(a), Ok(b)) => a.dev() == b.dev(),
1211        _ => false,
1212    }
1213}

1215/// Returns true if two paths share the same filesystem
1216#[cfg(unix)]
1217fn paths_have_same_filesystem_sync(a: &Path, b: &Path) -> bool {
1218    use std::os::unix::fs::MetadataExt;
1219    let a = std::fs::metadata(a);
1220    let b = std::fs::metadata(b);
1221    match (a, b) {
1222        (Ok(a), Ok(b)) => a.dev() == b.dev(),
1223        _ => false,
1224    }
1225}

1228#[cfg(not(unix))]
1229async fn paths_have_same_filesystem(a: &Path, b: &Path) -> bool {
1230    match (a.canonicalize(), b.canonicalize()) {
1231        (Ok(a), Ok(b)) => a.components().next() == b.components().next(),
1232        _ => false,
1233    }

1312    pub async fn test_install_python(
1313        urls: impl Iterator<Item = Url>,
1314        cache_name: &str,
1315        platform: Platform,
1316    ) {
1317        // Open a package cache in the systems temporary directory with a specific name.
1318        // This allows us to reuse a package cache across multiple invocations
1319        // of this test. Useful if you're debugging.
1320        let package_cache = PackageCache::new(temp_dir().join("rattler").join(cache_name));
1321
1322        // Create an HTTP client we can use to download packages
1323        let client = LazyClient::default();
1324
1325        // Specify python version
1326        let python_version =
1327            PythonInfo::from_version(&Version::from_str("3.11.0").unwrap(), None, platform)
1328                .unwrap();
1329
1330        // Download and install each layer into an environment.
1331        let install_driver = InstallDriver::default();
1332        let target_dir = tempdir().unwrap();
1333        let prefix_path = Prefix::create(target_dir.path()).unwrap();
1334        stream::iter(urls)
1335            .for_each_concurrent(Some(50), |package_url| {
1336                let client = client.clone();
1337                let package_cache = &package_cache;
1338                let install_driver = &install_driver;
1339                let python_version = &python_version;
1340                let prefix_path = prefix_path.clone();
1341                async move {
1342                    // Populate the cache
1343                    let package_info = CondaArchiveIdentifier::try_from_url(&package_url).unwrap();
1344                    let package_cache_lock = package_cache
1345                        .get_or_fetch_from_url(
1346                            package_info,
1347                            package_url.clone(),
1348                            client.clone(),
1349                            None,
1350                        )
1351                        .await
1352                        .unwrap();
1353
1354                    // Install the package to the prefix
1355                    link_package(
1356                        package_cache_lock.path(),
1357                        &prefix_path,
1358                        install_driver,
1359                        InstallOptions {
1360                            python_info: Some(python_version.clone()),
1361                            ..InstallOptions::default()
1362                        },
1363                    )
1364                    .await
1365                    .unwrap();
1366                }
1367            })
1368            .await;
1369
1370        // Run the python command and validate the version it outputs
1371        let python_path = if Platform::current().is_windows() {
1372            "python.exe"
1373        } else {
1374            "bin/python"
1375        };
1376        let python_version = Command::new(target_dir.path().join(python_path))
1377            .arg("--version")
1378            .output()
1379            .unwrap();
1380
1381        assert!(python_version.status.success());
1382        assert_eq!(
1383            String::from_utf8_lossy(&python_version.stdout).trim(),
1384            "Python 3.11.0"
1385        );
1386    }
