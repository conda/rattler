1use chrono::{DateTime, Utc};
2use serde::de::Error;
3use serde::{Deserialize, Deserializer, Serialize, Serializer};

6pub(crate) struct Timestamp;

8impl<'de> DeserializeAs<'de, chrono::DateTime<chrono::Utc>> for Timestamp {
9    fn deserialize_as<D>(deserializer: D) -> Result<chrono::DateTime<chrono::Utc>, D::Error>
10    where
11        D: Deserializer<'de>,
12    {
13        let timestamp = i64::deserialize(deserializer)?;
14
15        // Convert from milliseconds to seconds
16        let microseconds = if timestamp > 253_402_300_799 {
17            timestamp * 1_000
18        } else {
19            timestamp * 1_000_000
20        };
21
22        // Convert the timestamp to a UTC timestamp
23        chrono::DateTime::from_timestamp_micros(microseconds)
24            .ok_or_else(|| D::Error::custom("got invalid timestamp, timestamp out of range"))
25    }
26}
27
28impl SerializeAs<chrono::DateTime<chrono::Utc>> for Timestamp {
29    fn serialize_as<S>(source: &DateTime<Utc>, serializer: S) -> Result<S::Ok, S::Error>
30    where
31        S: Serializer,
32    {
33        // Convert the date to a timestamp
34        let timestamp: i64 = source.timestamp_millis();
35
36        // Determine the precision of the timestamp.
37        let timestamp = if timestamp % 1000 == 0 {
38            timestamp / 1000
39        } else {
40            timestamp
41        };
42
43        // Serialize the timestamp
44        timestamp.serialize(serializer)
45    }
46}
