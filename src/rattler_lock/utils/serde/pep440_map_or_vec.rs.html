1use std::hash::BuildHasherDefault;
2
3use indexmap::IndexMap;
4use pep508_rs::{Requirement, VersionOrUrl};
5use serde::{Deserialize, Deserializer};
6use serde_with::{serde_as, DeserializeAs, DisplayFromStr};
7
8pub(crate) struct Pep440MapOrVec;
9
10impl<'de> DeserializeAs<'de, Vec<Requirement>> for Pep440MapOrVec {
11    fn deserialize_as<D>(deserializer: D) -> Result<Vec<Requirement>, D::Error>
12    where
13        D: Deserializer<'de>,
14    {
15        #[serde_as]
16        #[derive(Deserialize)]
17        #[serde(untagged)]
18        enum MapOrVec {
19            Vec(Vec<Requirement>),
20            Map(
21                #[serde_as(as = "IndexMap<_, DisplayFromStr, BuildHasherDefault<ahash::AHasher>>")]
22                IndexMap<String, pep440_rs::VersionSpecifiers, BuildHasherDefault<ahash::AHasher>>,
23            ),
24        }
25
26        Ok(match MapOrVec::deserialize(deserializer)? {
27            MapOrVec::Vec(v) => v,
28            MapOrVec::Map(m) => m
29                .into_iter()
30                .map(|(name, spec)| {
31                    Ok::<_, pep508_rs::InvalidNameError>(pep508_rs::Requirement {
32                        name: pep508_rs::PackageName::new(name)?,
33                        extras: Vec::new(),
34                        version_or_url: if spec.is_empty() {
35                            None
36                        } else {
37                            Some(VersionOrUrl::VersionSpecifier(spec))
38                        },
39                        #[allow(clippy::default_trait_access)]
40                        marker: Default::default(),
41                        origin: None,
42                    })
43                })
44                .collect::<Result<Vec<_>, _>>()
45                .map_err(serde::de::Error::custom)?,
46        })
47    }
48}
