1//! This module contains functions to derive certain fields from other fields in
2//! a [`CondaPackageData`].
3//!
4//! For instance if we know the location of a package, and it contains a valid
5//! archive identifier filename, then we can derive the name, version, build,
6//! subdir and channel from it (see [`LocationDerivedFields`]).
7//!
8//! Care must be taken when changing the logic in this module, as it encodes how
9//! we resolve empty fields in a lock-file. This is important for
10//! reproducibility.
11
12use std::fmt::{Debug, Formatter};
13use std::str::FromStr;
14
15use crate::UrlOrPath;
16use rattler_conda_types::package::DistArchiveIdentifier;
17use rattler_conda_types::{
18    package::CondaArchiveIdentifier, BuildNumber, ChannelUrl, NoArchType, PackageName, Platform,
19    VersionWithSource,
20};
21use url::Url;

23/// A helper struct that wraps all fields of a [`CondaPackageData`] that can be
24/// derived just from the location of the package.
25pub(crate) struct LocationDerivedFields {
26    pub identifier: Option<DistArchiveIdentifier>,
27    pub name: Option<PackageName>,
28    pub version: Option<VersionWithSource>,
29    pub build: Option<String>,
30    pub subdir: Option<String>,
31    pub channel: Option<ChannelUrl>,
32}
33
34impl Debug for LocationDerivedFields {
35    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
36        f.debug_struct("LocationDerivedFields")
37            .field("identifier", &self.identifier)
38            .field("name", &self.name.as_ref().map(PackageName::as_source))
39            .field("version", &self.version.as_ref().map(|s| s.as_str()))
40            .field("build", &self.build)
41            .field("subdir", &self.subdir)
42            .field("channel", &self.channel.as_ref().map(ChannelUrl::as_str))
43            .finish()
44    }
45}
46
47impl LocationDerivedFields {
48    /// Constructs a new instance by deriving all fields from the given
49    /// location.
50    pub fn new(location: &UrlOrPath) -> Self {
51        let identifier = location
52            .file_name()
53            .and_then(DistArchiveIdentifier::try_from_filename);
54        let (name, version, build) = identifier
55            .as_ref()
56            .and_then(|a| {
57                Some((
58                    PackageName::new_unchecked(a.identifier.name.clone()),
59                    VersionWithSource::from_str(&a.identifier.version).ok()?,
60                    a.identifier.build_string.clone(),
61                ))
62            })
63            .map_or((None, None, None), |(name, version, build_string)| {
64                (Some(name), Some(version), Some(build_string))
65            });
66        let subdir = derive_subdir_from_location(location).map(ToString::to_string);
67        let channel = derive_channel_from_location(location);
68        Self {
69            identifier,
70            name,
71            version,
72            build,
73            subdir,
74            channel,
75        }
76    }
77}

79/// Try to derive the build number from the build string. It is common to append
80/// the build number to the end of the build string.
81pub(crate) fn derive_build_number_from_build(build: &str) -> Option<BuildNumber> {
82    let (_, trailing_number_str) = build
83        .rsplit_once(|c: char| !c.is_ascii_digit())
84        .unwrap_or(("", build));
85
86    trailing_number_str.parse().ok()
87}

89/// Try to derive the subdir from a common conda URL. This assumes that the URL
90/// is formatted as `scheme://../subdir/name-version-build.ext`.
91pub(crate) fn derive_subdir_from_location(location: &UrlOrPath) -> Option<&str> {
92    match location {
93        UrlOrPath::Url(url) => derive_subdir_from_url(url),
94        UrlOrPath::Path(_) => None,
95    }
96}
97
98/// Try to derive the subdir from a common conda URL. This assumes that the URL
99/// is formatted as `scheme://../subdir/name-version-build.ext`.
100pub fn derive_subdir_from_url(url: &Url) -> Option<&str> {
101    if !(url.scheme() == "file" || url.scheme() == "http" || url.scheme() == "https") {
102        return None;
103    }
104
105    let mut path_iter = url.path_segments()?.rev();
106    let archive_str = path_iter.next()?;
107    let subdir_str = path_iter.next()?;
108
109    // Try to parse the archive string as an archive identifier. If it fails we
110    // can't derive the subdir.
111    let _ = CondaArchiveIdentifier::try_from_filename(archive_str)?;
112
113    // Parse the subdir as a platform, if it fails we can't derive the subdir.
114    Platform::from_str(subdir_str).is_ok().then_some(subdir_str)
115}

117/// Channel from url, this is everything before the filename and the subdir
118/// So for example: <https://conda.anaconda.org/conda-forge/> is a channel name
119/// that we parse from something like: <https://conda.anaconda.org/conda-forge/osx-64/python-3.11.0-h4150a38_1_cpython.conda>
120pub(crate) fn derive_channel_from_url(url: &Url) -> Option<ChannelUrl> {
121    let mut result = url.clone();
122
123    // Strip the last two path segments. We assume the first one contains the
124    // file_name, and the other the subdirectory.
125    result.path_segments_mut().ok()?.pop().pop();
126
127    Some(result.into())
128}
129
130/// Returns the channel when deriving it from the location if possible.
131pub(crate) fn derive_channel_from_location(url: &UrlOrPath) -> Option<ChannelUrl> {
132    match url {
133        UrlOrPath::Url(url) => derive_channel_from_url(url),
134        UrlOrPath::Path(_) => None,
135    }
136}

138pub(crate) fn derive_arch_and_platform(subdir: &str) -> (Option<String>, Option<String>) {
139    let platform = Platform::from_str(subdir).ok();
140    platform.map_or((None, None), |p| {
141        (
142            p.arch().map(|arch| arch.to_string()),
143            p.only_platform().map(ToString::to_string),
144        )
145    })
146}
147
148/// Derive the noarch type from the subdir and the build string.
149///
150/// By default, the noarch type is set to `None`. However, if the channel is
151/// `noarch`, then we assume that the package is a noarch package but it does
152/// not tell us whether this is a generic or python noarch package. As a
153/// heuristic we then look at the build string, if it contains the word `py` we
154/// assume that this is a python noarch package.
155pub(crate) fn derive_noarch_type(subdir: &str, build_string: &str) -> NoArchType {
156    let is_python_build_string = build_string.contains("py");
157    match subdir {
158        "noarch" => {
159            if is_python_build_string {
160                NoArchType::python()
161            } else {
162                NoArchType::generic()
163            }
164        }
165        _ => NoArchType::none(),
166    }
167}

174    #[test]
175    fn test_derive_build_number_from_build() {
176        assert_eq!(derive_build_number_from_build("2"), Some(2));
177        assert_eq!(derive_build_number_from_build("1.2.3"), Some(3));
178        assert_eq!(derive_build_number_from_build("1.2.3-4"), Some(4));
179        assert_eq!(derive_build_number_from_build("py313hb6a6212_1"), Some(1));
180        assert_eq!(derive_build_number_from_build("foobar"), None);
181        assert_eq!(derive_build_number_from_build("123_"), None);
182        assert_eq!(derive_build_number_from_build("123foo"), None);
183        assert_eq!(derive_build_number_from_build("pyhd8ed1ab_100"), Some(100));
184        assert_eq!(derive_build_number_from_build("h803f02a_3"), Some(3));
185    }
