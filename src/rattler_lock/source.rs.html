1//! Provides data types that are used to describe the location of a source
2//! package.
3
4use rattler_digest::{Md5Hash, Sha256Hash};
5use typed_path::Utf8TypedPathBuf;
6use url::Url;
7
8/// Describes the source location of a package.
9#[derive(Clone, Debug, PartialEq, Eq, Hash)]
10pub enum SourceLocation {
11    /// The source is stored as a downloadable archive
12    Url(UrlSourceLocation),
13
14    /// The source is stored in a git repository
15    Git(GitSourceLocation),
16
17    /// The source is stored on the local filesystem
18    Path(PathSourceLocation),
19}
20
21/// A specification of a source archive from a URL.
22#[derive(Clone, Debug, PartialEq, Eq, Hash)]
23pub struct UrlSourceLocation {
24    /// The URL of the package contents
25    pub url: Url,
26
27    /// The md5 hash of the archive
28    pub md5: Option<Md5Hash>,
29
30    /// The sha256 hash of the archive
31    pub sha256: Option<Sha256Hash>,
32}
33
34/// A specification of source from a git repository.
35#[derive(Clone, Debug, PartialEq, Eq, Hash)]
36pub struct GitSourceLocation {
37    /// The git url of the package which can contain git+ prefixes.
38    pub git: Url,
39
40    /// The git revision of the package
41    pub rev: Option<GitReference>,
42
43    /// The git subdirectory of the package
44    pub subdirectory: Option<String>,
45}
46
47/// A reference to a specific commit in a git repository.
48#[derive(Clone, Debug, PartialEq, Eq, Hash)]
49pub enum GitReference {
50    /// The HEAD commit of a branch.
51    Branch(String),
52
53    /// A specific tag.
54    Tag(String),
55
56    /// A specific commit.
57    Rev(String),
58}
59
60/// Source located somewhere on the filesystem.
61#[derive(Clone, Debug, PartialEq, Eq, Hash)]
62pub struct PathSourceLocation {
63    /// The path to the package. Either a directory or an archive.
64    pub path: Utf8TypedPathBuf,
65}
