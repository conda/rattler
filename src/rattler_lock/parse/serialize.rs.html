33#[derive(Serialize)]

35struct SerializablePlatform<'a> {

37    #[serde(default, skip_serializing_if = "Option::is_none")]

43impl<'a> SerializablePlatform<'a> {
44    fn from_platform(platform: &'a PlatformData) -> Self {
45        let subdir = (platform.subdir.as_str() != platform.name.as_str())
46            .then_some(platform.subdir.as_str());
47        Self {
48            name: platform.name.as_str(),
49            subdir,
50            virtual_packages: &platform.virtual_packages,
51        }
52    }

60    #[serde(default, skip_serializing_if = "crate::utils::serde::is_default")]

65impl<'a> SerializableEnvironment<'a> {
66    fn from_environment(
67        inner: &'a LockFileInner,
68        env_data: &'a EnvironmentData,
69        used_pypi_packages: &HashSet<usize>,
70    ) -> Self {
71        SerializableEnvironment {
72            channels: &env_data.channels,
73            indexes: env_data.indexes.as_ref(),
74            options: env_data.options.clone(),
75            packages: env_data
76                .packages
77                .iter()
78                .map(|(platform, packages)| {
79                    let platform_name = inner
80                        .platforms
81                        .get(*platform)
82                        .expect("Platform indices are valid")
83                        .name
84                        .to_string();
85                    (
86                        platform_name,
87                        packages
88                            .iter()
89                            .map(|&package_data| {
90                                SerializablePackageSelector::from_lock_file(
91                                    inner,
92                                    package_data,
93                                    used_pypi_packages,
94                                )
95                            })
96                            .sorted()
97                            .collect(),
98                    )
99                })
100                .collect(),
101        }
102    }
103}

128/// Package selector for V7+ environments.
129///
130/// For V7+, binary conda packages are uniquely identified by their URL (which includes the
131/// filename), and source packages use `SourceIdentifier` with an embedded hash.
132#[derive(Serialize, Eq, PartialEq)]
133#[serde(untagged, rename_all = "snake_case")]
134enum SerializablePackageSelector<'a> {
135    /// Binary conda packages are uniquely identified by their URL.
136    Conda {
137        conda: &'a UrlOrPath,
138    },
139    /// Source packages use `SourceIdentifier` which uniquely identifies the package
140    /// via the format `name[hash] @ location`. No additional disambiguation fields needed.
141    CondaSource {
142        conda_source: SourceIdentifier,
143    },
144    Pypi {
145        pypi: &'a Verbatim<UrlOrPath>,
146    },
147}
148
149impl<'a> SerializablePackageSelector<'a> {
150    fn from_lock_file(
151        inner: &'a LockFileInner,
152        package: EnvironmentPackageData,
153        used_pypi_packages: &HashSet<usize>,
154    ) -> Self {
155        match package {
156            EnvironmentPackageData::Conda(idx) => Self::from_conda(&inner.conda_packages[idx]),
157            EnvironmentPackageData::Pypi(pkg_data_idx) => Self::from_pypi(
158                inner,
159                &inner.pypi_packages[pkg_data_idx],
160                used_pypi_packages,
161            ),
162        }
163    }
164
165    fn from_conda(package: &'a CondaPackageData) -> Self {
166        match package {
167            // Source packages use SourceIdentifier with an embedded hash
168            CondaPackageData::Source(source_data) => Self::CondaSource {
169                conda_source: SourceIdentifier::from_source_data(source_data),
170            },
171            // Binary packages are uniquely identified by their URL
172            CondaPackageData::Binary(binary_data) => Self::Conda {
173                conda: &binary_data.location,
174            },
175        }
176    }
177
178    fn from_pypi(
179        _inner: &'a LockFileInner,
180        package: &'a PypiPackageData,
181        _used_pypi_packages: &HashSet<usize>,
182    ) -> Self {
183        Self::Pypi {
184            pypi: package.location(),
185        }
186    }
187}

195impl Ord for SerializablePackageSelector<'_> {
196    fn cmp(&self, other: &Self) -> Ordering {
197        // Helper to get package type ordering: Conda (0) < Source (1) < Pypi (2)
198        fn type_order(selector: &SerializablePackageSelector<'_>) -> u8 {
199            match selector {
200                SerializablePackageSelector::Conda { .. } => 0,
201                SerializablePackageSelector::CondaSource { .. } => 1,
202                SerializablePackageSelector::Pypi { .. } => 2,
203            }
204        }
205
206        // First compare by type
207        let type_cmp = type_order(self).cmp(&type_order(other));
208        if type_cmp != Ordering::Equal {
209            return type_cmp;
210        }
211
212        // Same type, compare by content
213        match (self, other) {
214            (
215                SerializablePackageSelector::CondaSource { conda_source: a },
216                SerializablePackageSelector::CondaSource { conda_source: b },
217            ) => {
218                // Compare by name first, then by hash, then by location
219                a.name()
220                    .cmp(b.name())
221                    .then_with(|| a.hash().cmp(b.hash()))
222                    .then_with(|| compare_url_by_location(a.location(), b.location()))
223            }
224            // Conda and Pypi both compare by location
225            (
226                SerializablePackageSelector::Conda { conda: a },
227                SerializablePackageSelector::Conda { conda: b },
228            ) => compare_url_by_location(a, b),
229            (
230                SerializablePackageSelector::Pypi { pypi: a },
231                SerializablePackageSelector::Pypi { pypi: b },
232            ) => compare_url_by_location(a, b),
233            // Different types are already handled by type_cmp above
234            _ => unreachable!(),
235        }
236    }
237}

239/// First sort packages just by their filename. Since most of the time the urls
240/// end in the packages filename this causes the urls to be sorted by package
241/// name.
242fn compare_url_by_filename(a: &Url, b: &Url) -> Ordering {
243    if let (Some(a), Some(b)) = (
244        a.path_segments()
245            .and_then(Iterator::last)
246            .map(str::to_lowercase),
247        b.path_segments()
248            .and_then(Iterator::last)
249            .map(str::to_lowercase),
250    ) {
251        match a.cmp(&b) {
252            Ordering::Equal => {}
253            ordering => return ordering,
254        }
255    }
256
257    // Otherwise just sort by their full URL
258    a.cmp(b)
259}
260
261fn compare_url_by_location(a: &UrlOrPath, b: &UrlOrPath) -> Ordering {
262    match (a, b) {
263        (UrlOrPath::Url(a), UrlOrPath::Url(b)) => compare_url_by_filename(a, b),
264        (UrlOrPath::Url(_), UrlOrPath::Path(_)) => Ordering::Less,
265        (UrlOrPath::Path(_), UrlOrPath::Url(_)) => Ordering::Greater,
266        (UrlOrPath::Path(a), UrlOrPath::Path(b)) => a.as_str().cmp(b.as_str()),
267    }
268}

279impl Serialize for LockFile {
280    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
281    where
282        S: Serializer,
283    {
284        let inner = self.inner.as_ref();
285
286        // Determine the package indexes that are used in the lock-file.
287        let mut used_conda_packages = HashSet::new();
288        let mut used_pypi_packages = HashSet::new();
289        for env in inner.environments.iter() {
290            for packages in env.packages.values() {
291                for package in packages {
292                    match package {
293                        EnvironmentPackageData::Conda(idx) => {
294                            used_conda_packages.insert(*idx);
295                        }
296                        EnvironmentPackageData::Pypi(pkg_idx) => {
297                            used_pypi_packages.insert(*pkg_idx);
298                        }
299                    }
300                }
301            }
302        }
303
304        // Collect all environments
305        let environments = inner
306            .environment_lookup
307            .iter()
308            .map(|(name, env_idx)| {
309                (
310                    name,
311                    SerializableEnvironment::from_environment(
312                        inner,
313                        &inner.environments[*env_idx],
314                        &used_pypi_packages,
315                    ),
316                )
317            })
318            .collect::<BTreeMap<_, _>>();
319
320        // Get all packages, deduplicating binary packages by location.
321        // V7 identifies binary packages by URL uniquely, so we deduplicate here
322        // to handle older formats that may have had duplicate entries.
323        // Source packages are NOT deduplicated because they use SourceIdentifier
324        // which includes a hash to distinguish different configurations at the same location.
325        let mut seen_binary_locations = HashSet::new();
326        let conda_packages = inner
327            .conda_packages
328            .iter()
329            .enumerate()
330            .filter(|(idx, _)| used_conda_packages.contains(idx))
331            .filter(|(_, p)| {
332                match p {
333                    // Deduplicate binary packages by location
334                    CondaPackageData::Binary(binary) => {
335                        seen_binary_locations.insert(binary.location.clone())
336                    }
337                    // Don't deduplicate source packages
338                    CondaPackageData::Source(_) => true,
339                }
340            })
341            .map(|(_, p)| PackageData::Conda(p));
342
343        let pypi_packages = inner
344            .pypi_packages
345            .iter()
346            .enumerate()
347            .filter(|(idx, _)| used_pypi_packages.contains(idx))
348            .map(|(_, p)| PackageData::Pypi(p));
349
350        // Sort the packages in a deterministic order. See [`SerializablePackageData`]
351        // for more information.
352        let packages = itertools::chain!(conda_packages, pypi_packages).sorted();
353
354        let platforms = {
355            let mut tmp: Vec<_> = inner
356                .platforms
357                .iter()
358                .map(SerializablePlatform::from_platform)
359                .collect();
360            tmp.sort_by_key(|p| p.name);
361            tmp
362        };
363
364        let raw = SerializableLockFile {
365            version: FileFormatVersion::LATEST,
366            platforms,
367            environments,
368            packages: packages.collect(),
369            _version: PhantomData::<V7>,
370        };
371
372        raw.serialize(serializer)
373    }
374}

376#[derive(Debug, Eq, PartialEq, Copy, Clone)]
377pub enum PackageData<'a> {
378    Conda(&'a CondaPackageData),
379    Pypi(&'a PypiPackageData),
380}
381
382impl PackageData<'_> {
383    fn source_name(&self) -> &str {
384        match self {
385            PackageData::Conda(p) => p.name().as_source(),
386            PackageData::Pypi(p) => p.name().as_ref(),
387        }
388    }
389}
390
391impl PartialOrd<Self> for PackageData<'_> {
392    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
393        Some(self.cmp(other))
394    }
395}
396
397impl Ord for PackageData<'_> {
398    fn cmp(&self, other: &Self) -> Ordering {
399        use PackageData::{Conda, Pypi};
400        self.source_name()
401            .cmp(other.source_name())
402            .then_with(|| match (self, other) {
403                (Conda(a), Conda(b)) => a.cmp(b),
404                (Pypi(a), Pypi(b)) => a.cmp(b),
405                (Pypi(_), _) => Ordering::Less,
406                (_, Pypi(_)) => Ordering::Greater,
407            })
408    }
409}
