199impl<'de> DeserializeAs<'de, LegacyPackageData> for V6 {
200    fn deserialize_as<D>(deserializer: D) -> Result<LegacyPackageData, D::Error>
201    where
202        D: Deserializer<'de>,
203    {
204        #[derive(Deserialize)]
205        #[serde(untagged)]
206        enum Discriminant {
207            Conda {
208                #[serde(rename = "conda")]
209                _conda: String,
210            },
211            Pypi {
212                #[serde(rename = "pypi")]
213                _pypi: String,
214            },
215        }
216
217        let value = serde_value::Value::deserialize(deserializer)?;
218        let Ok(discriminant) = Discriminant::deserialize(
219            serde_value::ValueDeserializer::<D::Error>::new(value.clone()),
220        ) else {
221            return Err(D::Error::custom(
222                "expected at least `conda` or `pypi` field",
223            ));
224        };
225
226        let deserializer = serde_value::ValueDeserializer::<D::Error>::new(value);
227        Ok(match discriminant {
228            Discriminant::Conda { .. } => LegacyPackageData::Conda(
229                v6::CondaPackageDataModel::deserialize(deserializer)?
230                    .try_into()
231                    .map_err(D::Error::custom)?,
232            ),
233            Discriminant::Pypi { .. } => {
234                LegacyPackageData::Pypi(v6::PypiPackageDataModel::deserialize(deserializer)?.into())
235            }
236        })
237    }

364/// Returns `true` if the location refers to an immutable wheel artifact
365/// rather than a mutable source directory.
366fn is_wheel_or_archive_location(location: &UrlOrPath) -> bool {
367    let archives = [
368        ".whl", ".tar.gz", ".tar.bz2", ".tar.xz", ".tar.Z", ".tgz", ".tbz", ".tar", ".zip",
369    ];
370    match location {
371        UrlOrPath::Url(url) => {
372            if url.scheme() == "file" {
373                let path = url.path();
374                archives.iter().any(|a| path.ends_with(a))
375            } else {
376                true
377            }
378        }
379        UrlOrPath::Path(path) => archives.iter().any(|a| path.to_string().ends_with(a)),
380    }
381}

383fn convert_raw_pypi_package(
384    file_version: FileFormatVersion,
385    raw_package: PypiPackageDataRaw,
386    base_dir: Option<&Path>,
387) -> Result<PypiPackageData, ParseCondaLockError> {
388    let requires_dist = raw_package
389        .requires_dist
390        .iter()
391        .map(|input| {
392            if let Some(base_dir) = base_dir {
393                pep508_rs::Requirement::parse(input, base_dir)
394            } else {
395                use std::str::FromStr as _;
396
397                pep508_rs::Requirement::from_str(input)
398            }
399        })
400        .collect::<Result<Vec<_>, _>>()?;
401
402    let location = if file_version < FileFormatVersion::V7 {
403        Verbatim::new(raw_package.location.take())
404    } else {
405        raw_package.location
406    };
407
408    // In v7+ the version field is authoritative: present → Distribution,
409    // absent → Source.  For older formats we fall back to the location
410    // heuristic.
411    let is_distribution = if file_version >= FileFormatVersion::V7 {
412        raw_package.hash.is_some()
413            || raw_package.version.is_some()
414            || raw_package.index_url.is_some()
415    } else {
416        is_wheel_or_archive_location(location.inner())
417    };
418
419    if is_distribution {
420        Ok(PypiPackageData::Distribution(Box::new(
421            PypiDistributionData {
422                name: raw_package.name,
423                version: raw_package
424                    .version
425                    .unwrap_or_else(|| pep440_rs::MIN_VERSION.clone()),
426                location,
427                index_url: raw_package.index_url,
428                hash: raw_package.hash,
429                requires_dist,
430                requires_python: raw_package.requires_python,
431            },
432        )))
433    } else {
434        Ok(PypiPackageData::Source(Box::new(PypiSourceData {
435            name: raw_package.name,
436            location,
437            requires_dist,
438            requires_python: raw_package.requires_python,
439        })))
440    }
441}

443/// Checks if a legacy conda package matches the given selector fields.
444fn legacy_package_matches_selector(
445    package: &LegacyCondaPackageData,
446    name: Option<&PackageName>,
447    version: Option<&VersionWithSource>,
448    build: Option<&str>,
449    subdir: Option<&str>,
450) -> bool {
451    let record = package.record();
452    name.is_none_or(|n| n == &record.name)
453        && version.is_none_or(|v| v == &record.version)
454        && build.is_none_or(|b| b == record.build)
455        && subdir.is_none_or(|s| s == record.subdir)
456}

458/// Resolves a conda package index for legacy lock file formats (V1-V6).
459///
460/// For V6, uses exact field matching (name, version, build, subdir).
461/// For V5 and earlier, matches by platform/subdir and merges duplicate packages
462/// that share the same URL to combine optional fields like purls and hashes.
463#[allow(clippy::too_many_arguments)]
464fn resolve_legacy_conda_package(
465    file_version: FileFormatVersion,
466    platform_name: &str,
467    name: Option<&PackageName>,
468    version: Option<&VersionWithSource>,
469    build: Option<&str>,
470    subdir: Option<&str>,
471    candidate_indices: &[usize],
472    packages: &mut Vec<LegacyCondaPackageData>,
473) -> Option<usize> {
474    if file_version >= FileFormatVersion::V6 {
475        // V6+: Find the package that matches all selector fields exactly.
476        return candidate_indices
477            .iter()
478            .find(|&&idx| {
479                legacy_package_matches_selector(&packages[idx], name, version, build, subdir)
480            })
481            .copied();
482    }
483
484    // V5 and earlier: Match by platform subdir and merge duplicates.
485    // This handles a historical quirk where multiple packages could share
486    // the same URL but have different metadata.
487    let packages_for_platform: Vec<_> = candidate_indices
488        .iter()
489        .filter(|&&idx| packages[idx].record().subdir.as_str() == platform_name)
490        .copied()
491        .collect();
492
493    // Fall back to all candidates if none match the platform
494    let matching_indices = if packages_for_platform.is_empty() {
495        candidate_indices
496    } else {
497        &packages_for_platform
498    };
499
500    let mut iter = matching_indices.iter().copied();
501    let first_idx = iter.next()?;
502
503    // Merge all matching packages into one
504    let merged = iter.fold(
505        Cow::Borrowed(&packages[first_idx]),
506        |acc, next_idx| match acc.merge(&packages[next_idx]) {
507            Cow::Owned(merged) => Cow::Owned(merged),
508            Cow::Borrowed(_) => acc,
509        },
510    );
511
512    Some(match merged {
513        Cow::Borrowed(_) => first_idx,
514        Cow::Owned(merged_package) => {
515            packages.push(merged_package);
516            packages.len() - 1
517        }
518    })
519}

521/// Parses a legacy lock file (V4-V6) using `LegacyCondaPackageData` as the
522/// intermediate type, then converts to `CondaPackageData` at the end.
523///
524/// This separation allows `CondaPackageData` to evolve for newer versions (V7+)
525/// without breaking backward compatibility with older lock files.
526fn parse_from_lock_legacy<P>(
527    file_version: FileFormatVersion,
528    raw: DeserializableLockFileLegacy<P>,
529    base_dir: Option<&Path>,
530) -> Result<LockFile, ParseCondaLockError> {
531    let platforms = create_legacy_platforms(&raw)?;
532
533    // Split the packages into conda and pypi packages using legacy intermediate types.
534    let mut legacy_conda_packages: Vec<LegacyCondaPackageData> = Vec::new();
535    let mut pypi_packages = Vec::new();
536
537    for package in raw.packages {
538        match package {
539            LegacyPackageData::Conda(p) => legacy_conda_packages.push(p),
540            LegacyPackageData::Pypi(p) => {
541                pypi_packages.push(convert_raw_pypi_package(file_version, p, base_dir)?);
542            }
543        }
544    }
545
546    // Determine the indices of the packages by url
547    let mut conda_url_lookup: ahash::HashMap<UrlOrPath, Vec<_>> = ahash::HashMap::default();
548    for (idx, conda_package) in legacy_conda_packages.iter().enumerate() {
549        conda_url_lookup
550            .entry(conda_package.location().clone())
551            .or_default()
552            .push(idx);
553    }
554
555    let pypi_url_lookup = pypi_packages
556        .iter()
557        .enumerate()
558        .map(|(idx, p)| (p.location(), idx))
559        .collect::<ahash::HashMap<_, _>>();
560
561    let environments = raw
562        .environments
563        .into_iter()
564        .map(|(env_name, env)| {
565            Ok((
566                env_name.clone(),
567                EnvironmentData {
568                    channels: env.channels,
569                    indexes: env.indexes,
570                    options: env.options,
571                    packages: env
572                        .packages
573                        .into_iter()
574                        .map(|(platform, packages)| {
575                            let platform_name = platform.to_string();
576                            let Some((platform_index, _)) = platforms
577                                .iter()
578                                .enumerate()
579                                .find(|(_, p)| p.name.as_str() == platform_name.as_str())
580                            else {
581                                return Err(ParseCondaLockError::UnknownPlatform {
582                                    environment: env_name.clone(),
583                                    platform: platform_name,
584                                });
585                            };
586
587                            let packages = packages
588                                .into_iter()
589                                .map(|p| {
590                                    Ok(match p {
591                                        LegacyPackageSelector::Conda {
592                                            conda,
593                                            name,
594                                            version,
595                                            build,
596                                            subdir,
597                                        } => {
598                                            let candidate_indices = conda_url_lookup
599                                                .get(&conda)
600                                                .map_or(&[] as &[usize], Vec::as_slice);
601
602                                            let package_index = resolve_legacy_conda_package(
603                                                file_version,
604                                                platform.as_str(),
605                                                name.as_ref(),
606                                                version.as_ref(),
607                                                build.as_deref(),
608                                                subdir.as_deref(),
609                                                candidate_indices,
610                                                &mut legacy_conda_packages,
611                                            );
612
613                                            EnvironmentPackageData::Conda(
614                                                package_index.ok_or_else(|| {
615                                                    ParseCondaLockError::MissingPackage {
616                                                        environment: env_name.clone(),
617                                                        platform: platform_name.clone(),
618                                                        location: conda.to_string(),
619                                                    }
620                                                })?,
621                                            )
622                                        }
623                                        LegacyPackageSelector::Pypi { pypi } => {
624                                            EnvironmentPackageData::Pypi(
625                                                *pypi_url_lookup.get(&pypi).ok_or_else(|| {
626                                                    ParseCondaLockError::MissingPackage {
627                                                        environment: env_name.clone(),
628                                                        platform: platform_name.clone(),
629                                                        location: pypi.inner().to_string(),
630                                                    }
631                                                })?,
632                                            )
633                                        }
634                                    })
635                                })
636                                .collect::<Result<_, ParseCondaLockError>>()?;
637
638                            Ok((platform_index, packages))
639                        })
640                        .collect::<Result<_, ParseCondaLockError>>()?,
641                },
642            ))
643        })
644        .collect::<Result<BTreeMap<_, _>, ParseCondaLockError>>()?;
645
646    let (environment_lookup, environments) = environments
647        .into_iter()
648        .enumerate()
649        .map(|(idx, (name, env))| ((name, idx), env))
650        .unzip();
651
652    // Convert legacy conda packages to final CondaPackageData
653    let conda_packages: Vec<CondaPackageData> =
654        legacy_conda_packages.into_iter().map(Into::into).collect();
655
656    Ok(LockFile {
657        inner: Arc::new(LockFileInner {
658            version: file_version,
659            platforms,
660            environments,
661            environment_lookup,
662            conda_packages,
663            pypi_packages,
664        }),
665    })
666}

668/// Create a new `Vec<Platform>` from a legacy lock file.
669///
670/// Iterate over the environments and take the `rattler_conda_types::Platform`
671/// listed there and turn those into `Platform`.
672fn create_legacy_platforms<P>(
673    raw: &DeserializableLockFileLegacy<P>,
674) -> Result<Vec<crate::platform::PlatformData>, ParseCondaLockError> {
675    let mut unique_platforms = ahash::HashSet::default();
676    raw.environments
677        .iter()
678        .flat_map(|(env_name, env)| {
679            env.packages
680                .keys()
681                .map(move |platform| (env_name, platform))
682        })
683        .filter(move |(_, platform)| unique_platforms.insert(*platform))
684        .map(|(_, subdir)| -> Result<_, ParseCondaLockError> {
685            let name = crate::platform::PlatformName::try_from(subdir.as_str())?;
686
687            Ok(crate::platform::PlatformData {
688                name,
689                subdir: *subdir,
690                virtual_packages: Vec::new(),
691            })
692        })
693        .collect::<Result<Vec<_>, _>>()
694}
695
696/// Extract a `Vec<Platform>` from the lock file.
697fn read_platforms<P>(
698    raw: &DeserializableLockFile<P>,
699) -> Result<Vec<crate::platform::PlatformData>, ParseCondaLockError> {
700    let mut unique_platforms = ahash::HashSet::default();
701    raw.platforms
702        .iter()
703        .map(crate::platform::PlatformData::try_from)
704        .map(move |platform| match platform {
705            Ok(platform) => {
706                if unique_platforms.insert(platform.name.clone()) {
707                    Ok(platform)
708                } else {
709                    Err(ParseCondaLockError::DuplicatePlatformName(
710                        platform.name.to_string(),
711                    ))
712                }
713            }
714            Err(e) => Err(e.into()),
715        })
716        .collect::<Result<Vec<_>, _>>()
717}

719/// Parses a V7+ lock file directly to `CondaPackageData`.
720fn parse_from_lock<P>(
721    file_version: FileFormatVersion,
722    raw: DeserializableLockFile<P>,
723    base_dir: Option<&Path>,
724) -> Result<LockFile, ParseCondaLockError> {
725    let platforms = read_platforms(&raw)?;
726
727    // Split the packages into conda and pypi packages, building lookups as we go.
728    // Binary packages are uniquely identified by URL.
729    // Source packages are uniquely identified by SourceIdentifier (name[hash] @ location).
730    let num_packages = raw.packages.len();
731    let mut conda_packages: Vec<CondaPackageData> = Vec::with_capacity(num_packages);
732    let mut pypi_packages = Vec::with_capacity(num_packages);
733    let mut binary_url_lookup: ahash::HashMap<UrlOrPath, usize> =
734        ahash::HashMap::with_capacity(num_packages);
735    let mut source_identifier_lookup: ahash::HashMap<SourceIdentifier, usize> =
736        ahash::HashMap::with_capacity(num_packages);
737
738    for package in raw.packages {
739        match package {
740            PackageData::Conda(binary) => {
741                let idx = conda_packages.len();
742                binary_url_lookup.insert(binary.location.clone(), idx);
743                conda_packages.push(CondaPackageData::Binary(Box::new(binary)));
744            }
745            PackageData::CondaSource(identifier, source_data) => {
746                let idx = conda_packages.len();
747                source_identifier_lookup.insert(identifier, idx);
748                conda_packages.push(CondaPackageData::Source(Box::new(source_data)));
749            }
750            PackageData::Pypi(p) => {
751                pypi_packages.push(convert_raw_pypi_package(file_version, p, base_dir)?);
752            }
753        }
754    }
755
756    let pypi_url_lookup: ahash::HashMap<Verbatim<UrlOrPath>, _> = pypi_packages
757        .iter()
758        .enumerate()
759        .map(|(idx, p)| (p.location().clone(), idx))
760        .collect();
761
762    // Parse environments
763    let num_environments = raw.environments.len();
764    let mut environments = Vec::with_capacity(num_environments);
765    let mut environment_lookup: ahash::HashMap<String, usize> =
766        ahash::HashMap::with_capacity(num_environments);
767
768    for (env_name, env) in raw.environments {
769        let mut env_packages: ahash::HashMap<usize, IndexSet<EnvironmentPackageData>> =
770            ahash::HashMap::with_capacity(env.packages.len());
771
772        for (platform_name, selectors) in env.packages {
773            let Some((platform_index, platform_data)) = platforms
774                .iter()
775                .enumerate()
776                .find(|(_, p)| p.name.as_str() == platform_name)
777            else {
778                return Err(ParseCondaLockError::UnknownPlatform {
779                    environment: env_name.clone(),
780                    platform: platform_name,
781                });
782            };
783
784            let mut resolved = IndexSet::with_capacity(selectors.len());
785
786            for selector in selectors {
787                let package_data = resolve_package_selector(
788                    selector,
789                    &env_name,
790                    platform_data.subdir,
791                    &binary_url_lookup,
792                    &source_identifier_lookup,
793                    &pypi_url_lookup,
794                )?;
795                resolved.insert(package_data);
796            }
797
798            // For v7+ lockfiles, fill in the default index URL for pypi
799            // distribution packages that don't have an explicit one. The
800            // default is the first entry in the environment's `indexes`
801            // list, falling back to https://pypi.org/simple.
802            if file_version >= FileFormatVersion::V7 {
803                let default_index = env
804                    .indexes
805                    .as_ref()
806                    .and_then(|i| i.indexes.first())
807                    .cloned()
808                    .unwrap_or_else(|| {
809                        url::Url::parse("https://pypi.org/simple").expect("valid hard-coded URL")
810                    });
811                for pkg in &resolved {
812                    if let EnvironmentPackageData::Pypi(idx) = pkg {
813                        if let Some(w) = pypi_packages[*idx].as_wheel_mut() {
814                            if w.index_url.is_none() {
815                                w.index_url = Some(default_index.clone());
816                            }
817                        }
818                    }
819                }
820            }
821
822            env_packages.insert(platform_index, resolved);
823        }
824
825        environment_lookup.insert(env_name, environments.len());
826        environments.push(EnvironmentData {
827            channels: env.channels,
828            indexes: env.indexes,
829            options: env.options,
830            packages: env_packages,
831        });
832    }
833
834    Ok(LockFile {
835        inner: Arc::new(LockFileInner {
836            version: file_version,
837            platforms,
838            environments,
839            environment_lookup,
840            conda_packages,
841            pypi_packages,
842        }),
843    })
844}

846/// Resolves a package selector to an `EnvironmentPackageData`.
847fn resolve_package_selector(
848    selector: DeserializablePackageSelector,
849    env_name: &str,
850    platform: rattler_conda_types::Platform,
851    binary_url_lookup: &ahash::HashMap<UrlOrPath, usize>,
852    source_identifier_lookup: &ahash::HashMap<SourceIdentifier, usize>,
853    pypi_url_lookup: &ahash::HashMap<Verbatim<UrlOrPath>, usize>,
854) -> Result<EnvironmentPackageData, ParseCondaLockError> {
855    match selector {
856        DeserializablePackageSelector::Conda { conda } => {
857            let idx = binary_url_lookup.get(&conda).ok_or_else(|| {
858                ParseCondaLockError::MissingPackage {
859                    environment: env_name.to_owned(),
860                    platform: platform.to_string(),
861                    location: conda.to_string(),
862                }
863            })?;
864            Ok(EnvironmentPackageData::Conda(*idx))
865        }
866        DeserializablePackageSelector::CondaSource {
867            conda_source: source,
868        } => {
869            let idx = source_identifier_lookup.get(&source).ok_or_else(|| {
870                ParseCondaLockError::MissingPackage {
871                    environment: env_name.to_owned(),
872                    platform: platform.to_string(),
873                    location: source.to_string(),
874                }
875            })?;
876            Ok(EnvironmentPackageData::Conda(*idx))
877        }
878        DeserializablePackageSelector::Pypi { pypi } => {
879            let idx =
880                pypi_url_lookup
881                    .get(&pypi)
882                    .ok_or_else(|| ParseCondaLockError::MissingPackage {
883                        environment: env_name.to_owned(),
884                        platform: platform.to_string(),
885                        location: pypi.inner().to_string(),
886                    })?;
887            Ok(EnvironmentPackageData::Pypi(*idx))
888        }
889    }
