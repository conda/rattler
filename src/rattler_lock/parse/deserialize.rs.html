210impl<'de> DeserializeAs<'de, LegacyPackageData> for V6 {
211    fn deserialize_as<D>(deserializer: D) -> Result<LegacyPackageData, D::Error>
212    where
213        D: Deserializer<'de>,
214    {
215        #[derive(Deserialize)]
216        #[serde(untagged)]
217        enum Discriminant {
218            Conda {
219                #[serde(rename = "conda")]
220                _conda: String,
221            },
222            Pypi {
223                #[serde(rename = "pypi")]
224                _pypi: String,
225            },
226        }
227
228        let value = serde_value::Value::deserialize(deserializer)?;
229        let Ok(discriminant) = Discriminant::deserialize(
230            serde_value::ValueDeserializer::<D::Error>::new(value.clone()),
231        ) else {
232            return Err(D::Error::custom(
233                "expected at least `conda` or `pypi` field",
234            ));
235        };
236
237        let deserializer = serde_value::ValueDeserializer::<D::Error>::new(value);
238        Ok(match discriminant {
239            Discriminant::Conda { .. } => LegacyPackageData::Conda(
240                v6::CondaPackageDataModel::deserialize(deserializer)?
241                    .try_into()
242                    .map_err(D::Error::custom)?,
243            ),
244            Discriminant::Pypi { .. } => {
245                LegacyPackageData::Pypi(v6::PypiPackageDataModel::deserialize(deserializer)?.into())
246            }
247        })
248    }
249}

251impl<'de> DeserializeAs<'de, PackageData> for V7 {
252    fn deserialize_as<D>(deserializer: D) -> Result<PackageData, D::Error>
253    where
254        D: Deserializer<'de>,
255    {
256        #[derive(Deserialize)]
257        #[serde(untagged)]
258        enum Discriminant {
259            Conda {
260                #[serde(rename = "conda")]
261                _conda: String,
262            },
263            Source {
264                #[serde(rename = "conda_source")]
265                _source: String,
266            },
267            Pypi {
268                #[serde(rename = "pypi")]
269                _pypi: String,
270            },
271        }
272
273        let value = serde_value::Value::deserialize(deserializer)?;
274        let Ok(discriminant) = Discriminant::deserialize(
275            serde_value::ValueDeserializer::<D::Error>::new(value.clone()),
276        ) else {
277            return Err(D::Error::custom(
278                "expected at least `conda`, `conda_source`, or `pypi` field",
279            ));
280        };
281
282        let deserializer = serde_value::ValueDeserializer::<D::Error>::new(value);
283        Ok(match discriminant {
284            Discriminant::Conda { .. } => PackageData::Conda(
285                v7::CondaPackageDataModel::deserialize(deserializer)?
286                    .try_into()
287                    .map_err(D::Error::custom)?,
288            ),
289            Discriminant::Source { .. } => {
290                let mut model = v7::SourcePackageDataModel::deserialize(deserializer)?;
291                let build_packages = std::mem::take(&mut model.build_packages);
292                let host_packages = std::mem::take(&mut model.host_packages);
293                let (_identifier, data) = model.into_parts().map_err(D::Error::custom)?;
294                PackageData::CondaSource {
295                    data,
296                    build_packages,
297                    host_packages,
298                }
299            }
300            Discriminant::Pypi { .. } => {
301                let mut model = v7::PypiPackageDataModel::deserialize(deserializer)?;
302                let build_packages = std::mem::take(&mut model.build_packages);
303                let host_packages = std::mem::take(&mut model.host_packages);
304                PackageData::Pypi {
305                    data: model.into(),
306                    build_packages,
307                    host_packages,
308                }
309            }
310        })
311    }

386/// Returns `true` if the location refers to an immutable wheel artifact
387/// rather than a mutable source directory.
388fn is_wheel_or_archive_location(location: &UrlOrPath) -> bool {
389    let archives = [
390        ".whl", ".tar.gz", ".tar.bz2", ".tar.xz", ".tar.Z", ".tgz", ".tbz", ".tar", ".zip",
391    ];
392    match location {
393        UrlOrPath::Url(url) => {
394            if url.scheme() == "file" {
395                let path = url.path();
396                archives.iter().any(|a| path.ends_with(a))
397            } else {
398                true
399            }
400        }
401        UrlOrPath::Path(path) => archives.iter().any(|a| path.to_string().ends_with(a)),
402    }
403}

405fn convert_raw_pypi_package(
406    file_version: FileFormatVersion,
407    raw_package: PypiPackageDataRaw,
408    base_dir: Option<&Path>,
409) -> Result<PypiPackageData, ParseCondaLockError> {
410    let requires_dist = raw_package
411        .requires_dist
412        .iter()
413        .map(|input| {
414            if let Some(base_dir) = base_dir {
415                pep508_rs::Requirement::parse(input, base_dir)
416            } else {
417                use std::str::FromStr as _;
418
419                pep508_rs::Requirement::from_str(input)
420            }
421        })
422        .collect::<Result<Vec<_>, _>>()?;
423
424    let location = if file_version < FileFormatVersion::V7 {
425        Verbatim::new(raw_package.location.take())
426    } else {
427        raw_package.location
428    };
429
430    // In v7+ the version field is authoritative: present → Distribution,
431    // absent → Source.  For older formats we fall back to the location
432    // heuristic.
433    let is_distribution = if file_version >= FileFormatVersion::V7 {
434        raw_package.hash.is_some()
435            || raw_package.version.is_some()
436            || raw_package.index_url.is_some()
437    } else {
438        is_wheel_or_archive_location(location.inner())
439    };
440
441    if is_distribution {
442        Ok(PypiPackageData::Distribution(Box::new(
443            PypiDistributionData {
444                name: raw_package.name,
445                version: raw_package
446                    .version
447                    .unwrap_or_else(|| pep440_rs::MIN_VERSION.clone()),
448                location,
449                index_url: raw_package.index_url,
450                hash: raw_package.hash,
451                requires_dist,
452                requires_python: raw_package.requires_python,
453            },
454        )))
455    } else {
456        Ok(PypiPackageData::Source(Box::new(PypiSourceData {
457            name: raw_package.name,
458            location,
459            requires_dist,
460            requires_python: raw_package.requires_python,
461            source_data: SourceData::default(),
462        })))
463    }
464}

466/// Checks if a legacy conda package matches the given selector fields.
467fn legacy_package_matches_selector(
468    package: &LegacyPackageData,
469    name: Option<&PackageName>,
470    version: Option<&VersionWithSource>,
471    build: Option<&str>,
472    subdir: Option<&str>,
473) -> bool {
474    let LegacyPackageData::Conda(package) = package else {
475        return false;
476    };
477
478    let record = package.record();
479    name.is_none_or(|n| n == &record.name)
480        && version.is_none_or(|v| v == &record.version)
481        && build.is_none_or(|b| b == record.build)
482        && subdir.is_none_or(|s| s == record.subdir)
483}

485/// Resolves a conda package index for legacy lock file formats (V1-V6).
486///
487/// For V6, uses exact field matching (name, version, build, subdir).
488/// For V5 and earlier, matches by platform/subdir and merges duplicate packages
489/// that share the same URL to combine optional fields like purls and hashes.
490#[allow(clippy::too_many_arguments)]
491fn resolve_legacy_conda_package(
492    file_version: FileFormatVersion,
493    platform_name: &str,
494    name: Option<&PackageName>,
495    version: Option<&VersionWithSource>,
496    build: Option<&str>,
497    subdir: Option<&str>,
498    candidate_indices: &[PackageIndex],
499    packages: &mut Vec<LegacyPackageData>,
500) -> Option<PackageIndex> {
501    if file_version >= FileFormatVersion::V6 {
502        // V6+: Find the package that matches all selector fields exactly.
503        return candidate_indices
504            .iter()
505            .find(|&&index| {
506                legacy_package_matches_selector(&packages[index.0], name, version, build, subdir)
507            })
508            .copied();
509    }
510
511    // V5 and earlier: Match by platform subdir and merge duplicates.
512    // This handles a historical quirk where multiple packages could share
513    // the same URL but have different metadata.
514    let packages_for_platform: Vec<_> = candidate_indices
515        .iter()
516        .filter(|&&index| {
517            if let LegacyPackageData::Conda(p) = &packages[index.0] {
518                p.record().subdir.as_str() == platform_name
519            } else {
520                false
521            }
522        })
523        .copied()
524        .collect();
525
526    // Fall back to all candidates if none match the platform
527    let matching_indices = if packages_for_platform.is_empty() {
528        candidate_indices
529    } else {
530        &packages_for_platform
531    };
532
533    let mut iter = matching_indices.iter().copied();
534    let first_index = iter.next()?;
535
536    let acc = {
537        let p = &packages[first_index.0];
538        let LegacyPackageData::Conda(p) = &p else {
539            panic!("Internal error: Conda package expected, but not found");
540        };
541        Cow::Borrowed(p)
542    };
543
544    // Merge all matching packages into one
545    let merged = iter.fold(acc, |acc, next_idx| {
546        if let LegacyPackageData::Conda(p) = &packages[next_idx.0] {
547            match acc.merge(p) {
548                Cow::Owned(merged) => Cow::Owned(merged),
549                Cow::Borrowed(_) => acc,
550            }
551        } else {
552            acc
553        }
554    });
555
556    Some(match merged {
557        Cow::Borrowed(_) => first_index,
558        Cow::Owned(merged_package) => {
559            let index = PackageIndex(packages.len());
560            packages.push(LegacyPackageData::Conda(merged_package));
561            index
562        }
563    })
564}

566/// Parses a legacy lock file (V4-V6) using `LegacyCondaPackageData` as the
567/// intermediate type, then converts to `CondaPackageData` at the end.
568///
569/// This separation allows `CondaPackageData` to evolve for newer versions (V7+)
570/// without breaking backward compatibility with older lock files.
571fn parse_from_lock_legacy<P>(
572    file_version: FileFormatVersion,
573    mut raw: DeserializableLockFileLegacy<P>,
574    base_dir: Option<&Path>,
575) -> Result<LockFile, ParseCondaLockError> {
576    let platforms = create_legacy_platforms(&raw)?;
577
578    // Determine the indices of the packages by url
579    let (conda_url_lookup, pypi_url_lookup) = {
580        let mut tmp_conda: ahash::HashMap<UrlOrPath, Vec<_>> = ahash::HashMap::default();
581        let mut tmp_pypi: ahash::HashMap<UrlOrPath, PackageIndex> = ahash::HashMap::default();
582        for (index, package) in raw.packages.iter().enumerate() {
583            match package {
584                LegacyPackageData::Conda(p) => {
585                    tmp_conda
586                        .entry(p.location().clone())
587                        .or_default()
588                        .push(PackageIndex(index));
589                }
590                LegacyPackageData::Pypi(p) => {
591                    tmp_pypi.insert((*p.location).clone(), PackageIndex(index));
592                }
593            }
594        }
595        (tmp_conda, tmp_pypi)
596    };
597
598    // First pass: resolve each environment's selectors to raw PackageIndex
599    // sets. `resolve_legacy_conda_package` may mutate `raw.packages` (for
600    // merging), so we can't build the final package vec until this is done.
601    let environments_raw = raw
602        .environments
603        .into_iter()
604        .map(|(env_name, env)| {
605            Ok((
606                env_name.clone(),
607                env.channels,
608                env.indexes,
609                env.options,
610                env.packages
611                    .into_iter()
612                    .map(|(platform, packages)| {
613                        let platform_name = platform.to_string();
614                        let Some((platform_index, _)) = platforms
615                            .iter()
616                            .enumerate()
617                            .find(|(_, p)| p.name.as_str() == platform_name.as_str())
618                        else {
619                            return Err(ParseCondaLockError::UnknownPlatform {
620                                environment: env_name.clone(),
621                                platform: platform_name,
622                            });
623                        };
624
625                        let platform_index = PlatformIndex(platform_index);
626
627                        let indices = packages
628                            .into_iter()
629                            .map(|p| {
630                                Ok::<PackageIndex, ParseCondaLockError>(match p {
631                                    LegacyPackageSelector::Conda {
632                                        conda,
633                                        name,
634                                        version,
635                                        build,
636                                        subdir,
637                                    } => {
638                                        let candidate_indices = conda_url_lookup
639                                            .get(&conda)
640                                            .map_or(&[] as &[PackageIndex], Vec::as_slice);
641
642                                        let package_index = resolve_legacy_conda_package(
643                                            file_version,
644                                            platform.as_str(),
645                                            name.as_ref(),
646                                            version.as_ref(),
647                                            build.as_deref(),
648                                            subdir.as_deref(),
649                                            candidate_indices,
650                                            &mut raw.packages,
651                                        );
652
653                                        package_index.ok_or_else(|| {
654                                            ParseCondaLockError::MissingPackage {
655                                                environment: env_name.clone(),
656                                                platform: platform_name.clone(),
657                                                location: conda.to_string(),
658                                            }
659                                        })?
660                                    }
661                                    LegacyPackageSelector::Pypi { pypi } => *pypi_url_lookup
662                                        .get(&pypi)
663                                        .ok_or_else(|| ParseCondaLockError::MissingPackage {
664                                            environment: env_name.clone(),
665                                            platform: platform_name.clone(),
666                                            location: pypi.inner().to_string(),
667                                        })?,
668                                })
669                            })
670                            .collect::<Result<IndexSet<PackageIndex>, ParseCondaLockError>>()?;
671
672                        Ok((platform_index, indices))
673                    })
674                    .collect::<Result<Vec<_>, ParseCondaLockError>>()?,
675            ))
676        })
677        .collect::<Result<Vec<_>, ParseCondaLockError>>()?;
678
679    // Now that `raw.packages` is stable, build the final package vec.
680    let built_packages = raw
681        .packages
682        .into_iter()
683        .map(|p| match p {
684            LegacyPackageData::Conda(p) => Ok(LockedPackage::Conda(p.into())),
685            LegacyPackageData::Pypi(p) => {
686                convert_raw_pypi_package(file_version, p, base_dir).map(LockedPackage::Pypi)
687            }
688        })
689        .collect::<Result<Vec<_>, _>>()?;
690
691    // Second pass: wrap each environment's index sets into `EnvironmentPackages`
692    // (which looks up selector ids from `built_packages`).
693    let environments = environments_raw
694        .into_iter()
695        .map(
696            |(env_name, channels, indexes, options, packages_per_platform)| {
697                let packages = packages_per_platform
698                    .into_iter()
699                    .map(|(platform_index, indices)| {
700                        Ok((
701                            platform_index,
702                            EnvironmentPackages::from_indices(indices, &built_packages)?,
703                        ))
704                    })
705                    .collect::<Result<_, ParseCondaLockError>>()?;
706                Ok::<_, ParseCondaLockError>((
707                    env_name,
708                    EnvironmentData {
709                        channels,
710                        indexes,
711                        options,
712                        packages,
713                    },
714                ))
715            },
716        )
717        .collect::<Result<BTreeMap<_, _>, ParseCondaLockError>>()?;
718
719    let (environment_lookup, environments) = environments
720        .into_iter()
721        .enumerate()
722        .map(|(index, (name, env))| ((name, EnvironmentIndex(index)), env))
723        .unzip();
724
725    let packages = built_packages;
726
727    Ok(LockFile {
728        inner: Arc::new(LockFileInner {
729            version: file_version,
730            platforms,
731            environments,
732            environment_lookup,
733            packages,
734        }),
735    })
736}

738/// Create a new `Vec<Platform>` from a legacy lock file.
739///
740/// Iterate over the environments and take the `rattler_conda_types::Platform`
741/// listed there and turn those into `Platform`.
742fn create_legacy_platforms<P>(
743    raw: &DeserializableLockFileLegacy<P>,
744) -> Result<Vec<PlatformData>, ParseCondaLockError> {
745    let mut unique_platforms = ahash::HashSet::default();
746    raw.environments
747        .iter()
748        .flat_map(|(env_name, env)| {
749            env.packages
750                .keys()
751                .map(move |platform| (env_name, platform))
752        })
753        .filter(move |(_, platform)| unique_platforms.insert(*platform))
754        .map(|(_, subdir)| -> Result<_, ParseCondaLockError> {
755            let name = PlatformName::try_from(subdir.as_str())?;
756
757            Ok(PlatformData {
758                name,
759                subdir: *subdir,
760                virtual_packages: Vec::new(),
761            })
762        })
763        .collect::<Result<Vec<_>, _>>()
764}

766/// Extract a `Vec<Platform>` from the lock file.
767fn read_platforms<P>(
768    raw: &DeserializableLockFile<P>,
769) -> Result<Vec<PlatformData>, ParseCondaLockError> {
770    let mut unique_platforms = ahash::HashSet::default();
771    raw.platforms
772        .iter()
773        .map(PlatformData::try_from)
774        .map(move |platform| match platform {
775            Ok(platform) => {
776                if unique_platforms.insert(platform.name.clone()) {
777                    Ok(platform)
778                } else {
779                    Err(ParseCondaLockError::DuplicatePlatformName(
780                        platform.name.to_string(),
781                    ))
782                }
783            }
784            Err(e) => Err(e.into()),
785        })
786        .collect::<Result<Vec<_>, _>>()
787}

789/// Parses a V7+ lock file directly to `CondaPackageData`.
790fn parse_from_lock<P>(
791    file_version: FileFormatVersion,
792    raw: DeserializableLockFile<P>,
793    base_dir: Option<&Path>,
794) -> Result<LockFile, ParseCondaLockError> {
795    let platforms = read_platforms(&raw)?;
796
797    // Split the packages into conda and pypi packages.
798    //
799    // Source packages' `build_packages` / `host_packages` selector strings are
800    // collected here and resolved to [`PackageIndex`] values in a second pass
801    // after all packages have been indexed.
802    let (mut packages, pending_source_data) = {
803        let num_packages = raw.packages.len();
804        let mut packages = Vec::with_capacity(num_packages);
805        let mut pending_source_data: Vec<(
806            PackageIndex,
807            Vec<PackageSelector>,
808            Vec<PackageSelector>,
809        )> = Vec::new();
810
811        for (index, package) in raw.packages.into_iter().enumerate() {
812            let index = PackageIndex(index);
813            let package = match package {
814                PackageData::Conda(binary) => Ok(LockedPackage::Conda(CondaPackageData::Binary(
815                    Box::new(binary),
816                ))),
817                PackageData::CondaSource {
818                    data,
819                    build_packages,
820                    host_packages,
821                } => {
822                    if !build_packages.is_empty() || !host_packages.is_empty() {
823                        pending_source_data.push((index, build_packages, host_packages));
824                    }
825                    Ok(LockedPackage::Conda(CondaPackageData::Source(Box::new(
826                        data,
827                    ))))
828                }
829                PackageData::Pypi {
830                    data,
831                    build_packages,
832                    host_packages,
833                } => {
834                    if !build_packages.is_empty() || !host_packages.is_empty() {
835                        pending_source_data.push((index, build_packages, host_packages));
836                    }
837                    convert_raw_pypi_package(file_version, data, base_dir).map(LockedPackage::Pypi)
838                }
839            }?;
840            packages.push(package);
841        }
842        Ok::<_, ParseCondaLockError>((packages, pending_source_data))
843    }?;
844
845    // Build a sorted index of (SelectorId, PackageIndex) pairs. All lookups
846    // (source-package build/host references and per-environment package
847    // selectors) go through this via binary search on the full selector id.
848    let mut selector_index: Vec<(SelectorId, PackageIndex)> = packages
849        .iter()
850        .enumerate()
851        .map(|(i, pkg)| (SelectorId::new(pkg), PackageIndex(i)))
852        .collect();
853    selector_index.sort_by(|a, b| a.0.cmp(&b.0));
854
855    // Resolve a `PackageSelector` (as read from the lockfile) to a
856    // `PackageIndex` by rebuilding its full `SelectorId` and binary-searching
857    // `selector_index`.
858    let resolve_index = |selector: &PackageSelector| -> Result<PackageIndex, ParseCondaLockError> {
859        let id = selector.to_selector_id();
860        selector_index
861            .binary_search_by(|(sel, _)| sel.cmp(&id))
862            .map(|pos| selector_index[pos].1)
863            .map_err(|_not_found| ParseCondaLockError::MissingPackage {
864                environment: String::new(),
865                platform: String::new(),
866                location: selector.id().to_owned(),
867            })
868    };
869
870    for (pkg_idx, build_selectors, host_selectors) in pending_source_data {
871        // Resolve in a scoped block so the immutable borrow of `packages`
872        // that the closure takes is dropped before we mutate `packages`.
873        let (build_packages, host_packages) = {
874            let resolve =
875                |selector: &PackageSelector| -> Result<PackageHandle, ParseCondaLockError> {
876                    let index = resolve_index(selector)?;
877                    Ok(PackageHandle::new(index, &packages[index.0]))
878                };
879            let build = EnvironmentPackages::from_selector_ids(build_selectors, &resolve)?;
880            let host = EnvironmentPackages::from_selector_ids(host_selectors, &resolve)?;
881            (build, host)
882        };
883
884        let source_data = SourceData {
885            build_packages,
886            host_packages,
887        };
888        match &mut packages[pkg_idx.0] {
889            LockedPackage::Conda(CondaPackageData::Source(data)) => {
890                data.source_data = source_data;
891            }
892            LockedPackage::Pypi(PypiPackageData::Source(data)) => {
893                data.source_data = source_data;
894            }
895            _ => {
896                // Binary/wheel packages don't have source_data; silently ignore.
897            }
898        }
899    }
900
901    // Parse environments
902    let num_environments = raw.environments.len();
903    let mut environments = Vec::with_capacity(num_environments);
904    let mut environment_lookup: ahash::HashMap<String, EnvironmentIndex> =
905        ahash::HashMap::with_capacity(num_environments);
906
907    for (env_name, env) in raw.environments {
908        let mut env_packages: ahash::HashMap<PlatformIndex, EnvironmentPackages> =
909            ahash::HashMap::with_capacity(env.packages.len());
910
911        for (platform_name, selectors) in env.packages {
912            let Some((platform_index, platform_data)) = platforms
913                .iter()
914                .enumerate()
915                .find(|(_, p)| p.name.as_str() == platform_name)
916            else {
917                return Err(ParseCondaLockError::UnknownPlatform {
918                    environment: env_name.clone(),
919                    platform: platform_name,
920                });
921            };
922
923            let platform_index = PlatformIndex(platform_index);
924
925            let mut resolved = IndexSet::with_capacity(selectors.len());
926
927            for selector in selectors {
928                let package_data = resolve_package_selector(
929                    selector,
930                    &env_name,
931                    platform_data.subdir,
932                    &selector_index,
933                )?;
934                resolved.insert(package_data);
935            }
936
937            // For v7+ lockfiles, fill in the default index URL for pypi
938            // distribution packages that don't have an explicit one. The
939            // default is the first entry in the environment's `indexes`
940            // list, falling back to https://pypi.org/simple.
941            if file_version >= FileFormatVersion::V7 {
942                let default_index = env
943                    .indexes
944                    .as_ref()
945                    .and_then(|i| i.indexes.first())
946                    .cloned()
947                    .unwrap_or_else(|| {
948                        url::Url::parse("https://pypi.org/simple").expect("valid hard-coded URL")
949                    });
950                for pkg in &resolved {
951                    if let LockedPackage::Pypi(p) = &mut packages[pkg.0]
952                        && let Some(w) = p.as_wheel_mut()
953                        && w.index_url.is_none()
954                    {
955                        w.index_url = Some(default_index.clone());
956                    }
957                }
958            }
959
960            env_packages.insert(
961                platform_index,
962                EnvironmentPackages::from_indices(resolved, &packages)?,
963            );
964        }
965
966        environment_lookup.insert(env_name, EnvironmentIndex(environments.len()));
967        environments.push(EnvironmentData {
968            channels: env.channels,
969            indexes: env.indexes,
970            options: env.options,
971            packages: env_packages,
972        });
973    }
974
975    Ok(LockFile {
976        inner: Arc::new(LockFileInner {
977            version: file_version,
978            platforms,
979            environments,
980            environment_lookup,
981            packages,
982        }),
983    })
984}

986/// Resolves a package selector to an `EnvironmentPackageData`.
987///
988/// Builds the full (kind-prefixed) selector id from the YAML-tagged selector
989/// variant and binary-searches `selector_index`. The pypi `given`-over-`inner`
990/// preference must match [`SelectorId::new`] (see `crates/rattler_lock/src/lib.rs`).
991fn resolve_package_selector(
992    selector: DeserializablePackageSelector,
993    env_name: &str,
994    platform: rattler_conda_types::Platform,
995    selector_index: &[(SelectorId, PackageIndex)],
996) -> Result<PackageIndex, ParseCondaLockError> {
997    let (kind, id) = match selector {
998        DeserializablePackageSelector::Conda { conda } => {
999            (SelectorKind::CondaBinary, conda.as_str().to_owned())
1000        }
1001        DeserializablePackageSelector::CondaSource {
1002            conda_source: source,
1003        } => (SelectorKind::CondaSource, source.to_string()),
1004        DeserializablePackageSelector::Pypi { pypi } => {
1005            let id = pypi
1006                .given()
1007                .map_or_else(|| pypi.inner().as_str().to_owned(), str::to_owned);
1008            (SelectorKind::Pypi, id)
1009        }
1010    };
1011    let full_id = SelectorId::from_parts(kind, &id);
1012
1013    selector_index
1014        .binary_search_by(|(sel, _)| sel.cmp(&full_id))
1015        .map(|pos| selector_index[pos].1)
1016        .map_err(|_not_found| ParseCondaLockError::MissingPackage {
1017            environment: env_name.to_owned(),
1018            platform: platform.to_string(),
1019            location: id,
1020        })
1021}
