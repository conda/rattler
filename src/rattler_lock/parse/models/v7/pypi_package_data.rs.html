13/// A helper struct that wraps all fields of a [`crate::PypiPackageData`] and
14/// allows for easy conversion between the two.
15///
16/// This type provides full control over the order of the fields when
17/// serializing. This is important because one of the design goals is that it
18/// should be easy to read the lock file. A [`PackageRecord`] is serialized in
19/// alphabetic order which might not be the most readable. This type instead
20/// puts the "most important" fields at the top followed by more detailed ones.
21///
22/// So note that for reproducibility the order of these fields should not change
23/// or should be reflected in a version change.
24///
25/// The complexity with `Cow<_>` types is introduced to allow both efficient
26/// deserialization and serialization without requiring all data to be cloned
27/// when serializing. We want to be able to use the same type of both
28/// serialization and deserialization to ensure that when any of the
29/// types involved change we are forced to update this struct as well.
30#[serde_as]
31#[skip_serializing_none]
32#[derive(Serialize, Deserialize, Eq, PartialEq, Clone, Debug, Hash)]
33pub(crate) struct PypiPackageDataModel<'a> {
34    #[serde(rename = "pypi")]
35    pub location: Cow<'a, Verbatim<UrlOrPath>>,
36    pub name: Cow<'a, PackageName>,
37    pub version: Option<Cow<'a, pep440_rs::Version>>,
38    #[serde(default, skip_serializing_if = "Option::is_none", flatten)]
39    pub hash: Cow<'a, Option<PackageHashes>>,
40    #[serde(default, skip_serializing_if = "Option::is_none")]
41    pub index: Option<Cow<'a, url::Url>>,
42    #[serde(default, skip_serializing_if = "<[String]>::is_empty")]
43    pub requires_dist: Cow<'a, [String]>,
44    #[serde(default, skip_serializing_if = "Option::is_none")]
45    pub requires_python: Cow<'a, Option<VersionSpecifiers>>,
46
47    /// Selectors for packages in the build environment (pypi source
48    /// packages only — empty for wheel distributions).
49    /// Populated at lockfile serialization time; empty for standalone package
50    /// serialization. Resolved to indices after deserialization.
51    #[serde(default, skip_serializing_if = "Vec::is_empty")]
52    pub build_packages: Vec<PackageSelector>,
53    /// Selectors for packages in the host environment.
54    #[serde(default, skip_serializing_if = "Vec::is_empty")]
55    pub host_packages: Vec<PackageSelector>,
56}

58impl<'a> From<PypiPackageDataModel<'a>> for PypiPackageDataRaw {
59    fn from(value: PypiPackageDataModel<'a>) -> Self {
60        let index_url = match value.location.inner() {
61            UrlOrPath::Url(_) => value.index.map(std::borrow::Cow::into_owned),
62            UrlOrPath::Path(_) => None,
63        };
64
65        Self {
66            name: value.name.into_owned(),
67            version: value.version.map(std::borrow::Cow::into_owned),
68            location: value.location.into_owned(),
69            hash: value.hash.into_owned(),
70            index_url,
71            requires_dist: value.requires_dist.into_owned(),
72            requires_python: value.requires_python.into_owned(),
73        }
74    }
75}

77static PYPI_URL: LazyLock<url::Url> =
78    LazyLock::new(|| url::Url::parse("https://pypi.org/simple").expect("Valid, hard-coded URL"));
79
80impl<'a> From<&'a PypiPackageData> for PypiPackageDataModel<'a> {
81    fn from(value: &'a PypiPackageData) -> Self {
82        match value {
83            PypiPackageData::Distribution(w) => {
84                let requires_dist = w
85                    .requires_dist
86                    .iter()
87                    .map(requirement_to_string)
88                    .collect::<Vec<_>>();
89                let index_url = w.index_url.as_ref().and_then(|i| {
90                    if *i == *PYPI_URL {
91                        None
92                    } else {
93                        Some(Cow::Borrowed(i))
94                    }
95                });
96                Self {
97                    name: Cow::Borrowed(&w.name),
98                    version: Some(Cow::Borrowed(&w.version)),
99                    location: Cow::Borrowed(&w.location),
100                    hash: Cow::Borrowed(&w.hash),
101                    index: index_url,
102                    requires_dist: requires_dist.into(),
103                    requires_python: Cow::Borrowed(&w.requires_python),
104                    build_packages: Vec::new(),
105                    host_packages: Vec::new(),
106                }
107            }
108            PypiPackageData::Source(s) => {
109                let requires_dist = s
110                    .requires_dist
111                    .iter()
112                    .map(requirement_to_string)
113                    .collect::<Vec<_>>();
114                Self {
115                    name: Cow::Borrowed(&s.name),
116                    version: None,
117                    location: Cow::Borrowed(&s.location),
118                    hash: Cow::Owned(None),
119                    index: None,
120                    requires_dist: requires_dist.into(),
121                    requires_python: Cow::Borrowed(&s.requires_python),
122                    build_packages: Vec::new(),
123                    host_packages: Vec::new(),
124                }
125            }
126        }
127    }
128}

130/// Serialize a requirement as a string, preserving relative paths for file
131/// dependencies. Delegates to `pep508_rs`'s own [`Display`](std::fmt::Display)
132/// impl via the [`GivenVerbatimUrl`] wrapper.
133fn requirement_to_string(req: &pep508_rs::Requirement) -> String {
134    GivenVerbatimUrl::wrap_requirement(req).to_string()
135}

149    fn url_location(url: &str) -> Verbatim<UrlOrPath> {
150        Verbatim::new(UrlOrPath::Url(url::Url::parse(url).unwrap()))
151    }
152
153    fn path_location(path: &str) -> Verbatim<UrlOrPath> {
154        Verbatim::new(UrlOrPath::Path(Utf8TypedPathBuf::from(path)))
155    }
156
157    fn make_model(
158        location: Verbatim<UrlOrPath>,
159        index_url: Option<url::Url>,
160    ) -> PypiPackageDataModel<'static> {
161        PypiPackageDataModel {
162            location: Cow::Owned(location),
163            name: Cow::Owned("test-pkg".parse::<PackageName>().unwrap()),
164            version: None,
165            hash: Cow::Owned(None),
166            index: index_url.map(Cow::Owned),
167            requires_dist: Cow::Owned(vec![]),
168            requires_python: Cow::Owned(None),
169            build_packages: Vec::new(),
170            host_packages: Vec::new(),
171        }
172    }
173
174    #[test]
175    fn url_location_without_index_url_is_none() {
176        let model = make_model(
177            url_location("https://files.pythonhosted.org/pkg-1.0.whl"),
178            None,
179        );
180        let raw: PypiPackageDataRaw = model.into();
181        assert!(
182            raw.index_url.is_none(),
183            "default index is applied per-environment, not at model level"
184        );
185    }
186
187    #[test]
188    fn url_location_with_custom_index_url_preserved() {
189        let custom = url::Url::parse("https://custom.index/simple").unwrap();
190        let model = make_model(
191            url_location("https://custom.index/packages/pkg-1.0.whl"),
192            Some(custom.clone()),
193        );
194        let raw: PypiPackageDataRaw = model.into();
195        assert_eq!(raw.index_url.as_ref(), Some(&custom));
196    }
197
198    #[test]
199    fn path_location_has_no_index_url() {
200        let model = make_model(path_location("./my-pkg"), None);
201        let raw: PypiPackageDataRaw = model.into();
202        assert!(raw.index_url.is_none());
203    }
