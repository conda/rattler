23/// This struct is similar to [`crate::parse::models::v6::CondaPackageData`] but
24/// used for the V5 version of the lock file format.
25#[serde_as]
26#[derive(Serialize, Deserialize, Eq, PartialEq)]
27pub(crate) struct CondaPackageDataModel<'a> {
28    // Unique identifiers go to the top
29    pub name: Cow<'a, PackageName>,
30    pub version: Cow<'a, VersionWithSource>,
31    #[serde(default, skip_serializing_if = "is_default")]
32    pub build: Cow<'a, str>,
33    #[serde(default, skip_serializing_if = "is_default")]
34    pub build_number: BuildNumber,
35    #[serde(default)]
36    pub subdir: Cow<'a, str>,
37    #[serde(skip_serializing_if = "NoArchType::is_none")]
38    pub noarch: Cow<'a, NoArchType>,
39
40    // Followed by the URL of the package
41    pub url: Cow<'a, Url>,
42
43    // Then the hashes
44    #[serde(default, skip_serializing_if = "Option::is_none")]
45    #[serde_as(as = "Option<SerializableHash::<rattler_digest::Sha256>>")]
46    pub sha256: Option<Sha256Hash>,
47    #[serde(default, skip_serializing_if = "Option::is_none")]
48    #[serde_as(as = "Option<SerializableHash::<rattler_digest::Md5>>")]
49    pub md5: Option<Md5Hash>,
50    #[serde(default, skip_serializing_if = "Option::is_none")]
51    #[serde_as(as = "Option<SerializableHash::<rattler_digest::Md5>>")]
52    pub legacy_bz2_md5: Option<Md5Hash>,
53
54    // Dependencies
55    #[serde(default, skip_serializing_if = "<[String]>::is_empty")]
56    pub depends: Cow<'a, [String]>,
57    #[serde(default, skip_serializing_if = "<[String]>::is_empty")]
58    pub constrains: Cow<'a, [String]>,
59
60    // Additional properties (in semi alphabetic order but grouped by commonality)
61    #[serde(default, skip_serializing_if = "Option::is_none")]
62    pub arch: Cow<'a, Option<String>>,
63    #[serde(default, skip_serializing_if = "Option::is_none")]
64    pub platform: Cow<'a, Option<String>>,
65
66    #[serde(default, skip_serializing_if = "Option::is_none")]
67    pub channel: Cow<'a, Option<Url>>,
68
69    #[serde(default, skip_serializing_if = "Option::is_none")]
70    pub features: Cow<'a, Option<String>>,
71    #[serde(default, skip_serializing_if = "<[String]>::is_empty")]
72    pub track_features: Cow<'a, [String]>,
73
74    #[serde(default, skip_serializing_if = "Option::is_none")]
75    pub file_name: Cow<'a, Option<DistArchiveIdentifier>>,
76
77    #[serde(default, skip_serializing_if = "Option::is_none")]
78    pub python_site_packages_path: Cow<'a, Option<String>>,
79
80    #[serde(default, skip_serializing_if = "Option::is_none")]
81    pub license: Cow<'a, Option<String>>,
82    #[serde(default, skip_serializing_if = "Option::is_none")]
83    pub license_family: Cow<'a, Option<String>>,
84    #[serde(default, skip_serializing_if = "Option::is_none")]
85    pub purls: Cow<'a, Option<BTreeSet<PackageUrl>>>,
86
87    #[serde(default, skip_serializing_if = "Option::is_none")]
88    pub size: Cow<'a, Option<u64>>,
89    #[serde(default, skip_serializing_if = "Option::is_none")]
90    pub legacy_bz2_size: Cow<'a, Option<u64>>,
91
92    #[serde(default, skip_serializing_if = "Option::is_none")]
93    #[serde_as(as = "Option<crate::utils::serde::Timestamp>")]
94    pub timestamp: Option<chrono::DateTime<chrono::Utc>>,
95}

97impl<'a> From<CondaPackageDataModel<'a>> for CondaPackageData {
98    fn from(value: CondaPackageDataModel<'a>) -> Self {
99        let location = UrlOrPath::Url(value.url.into_owned());
100        let subdir = value.subdir.into_owned();
101        let (derived_arch, derived_platform) = derive_arch_and_platform(&subdir);
102
103        let file_name = value
104            .file_name
105            .into_owned()
106            .or_else(|| location.file_name().and_then(|f| f.parse().ok()))
107            .unwrap_or_else(|| DistArchiveIdentifier {
108                identifier: ArchiveIdentifier {
109                    name: value.name.as_normalized().to_owned(),
110                    version: value.version.to_string(),
111                    build_string: value.build.to_string(),
112                },
113                archive_type: DistArchiveType::Conda(CondaArchiveType::Conda),
114            });
115
116        Self::Binary(CondaBinaryData {
117            package_record: PackageRecord {
118                build: value.build.into_owned(),
119                build_number: value.build_number,
120                constrains: value.constrains.into_owned(),
121                depends: value.depends.into_owned(),
122                experimental_extra_depends: std::collections::BTreeMap::new(),
123                features: value.features.into_owned(),
124                legacy_bz2_md5: value.legacy_bz2_md5,
125                legacy_bz2_size: value.legacy_bz2_size.into_owned(),
126                license: value.license.into_owned(),
127                license_family: value.license_family.into_owned(),
128                md5: value.md5,
129                name: value.name.into_owned(),
130                noarch: value.noarch.into_owned(),
131                purls: value.purls.into_owned(),
132                sha256: value.sha256,
133                size: value.size.into_owned(),
134                subdir,
135                timestamp: value.timestamp.map(Into::into),
136                track_features: value.track_features.into_owned(),
137                version: value.version.into_owned(),
138                run_exports: None,
139                // Polyfill the arch and platform fields if they are not present in the lock-file.
140                arch: value.arch.into_owned().or(derived_arch),
141                platform: value.platform.into_owned().or(derived_platform),
142                python_site_packages_path: value.python_site_packages_path.into_owned(),
143            },
144            channel: value
145                .channel
146                .into_owned()
147                .map(ChannelUrl::from)
148                .or_else(|| derive_channel_from_location(&location)),
149            file_name,
150            location,
151        })
152    }
153}
