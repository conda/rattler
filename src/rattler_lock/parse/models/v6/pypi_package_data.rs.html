10/// A helper struct that wraps all fields of a [`crate::PypiPackageData`] and
11/// allows for easy conversion between the two.
12///
13/// This type provides full control over the order of the fields when
14/// serializing. This is important because one of the design goals is that it
15/// should be easy to read the lock file. A [`PackageRecord`] is serialized in
16/// alphabetic order which might not be the most readable. This type instead
17/// puts the "most important" fields at the top followed by more detailed ones.
18///
19/// So note that for reproducibility the order of these fields should not change
20/// or should be reflected in a version change.
21///
22/// The complexity with `Cow<_>` types is introduced to allow both efficient
23/// deserialization and serialization without requiring all data to be cloned
24/// when serializing. We want to be able to use the same type of both
25/// serialization and deserialization to ensure that when any of the
26/// types involved change we are forced to update this struct as well.
27#[serde_as]
28#[skip_serializing_none]
29#[derive(Serialize, Deserialize, Eq, PartialEq, Clone, Debug, Hash)]
30pub(crate) struct PypiPackageDataModel<'a> {
31    #[serde(rename = "pypi")]
32    pub location: Cow<'a, UrlOrPath>,
33    pub name: Cow<'a, PackageName>,
34    pub version: Cow<'a, pep440_rs::Version>,
35    #[serde(default, skip_serializing_if = "Option::is_none", flatten)]
36    pub hash: Cow<'a, Option<PackageHashes>>,
37    #[serde(default, skip_serializing_if = "<[Requirement]>::is_empty")]
38    pub requires_dist: Cow<'a, [Requirement]>,
39    #[serde(default, skip_serializing_if = "Option::is_none")]
40    pub requires_python: Cow<'a, Option<VersionSpecifiers>>,
41    #[serde(default, skip_serializing_if = "should_skip_serializing_editable")]
42    pub editable: bool,
43}
44
45/// Used in `skip_serializing_if` to skip serializing the `editable` field if it
46/// is `false`.
47fn should_skip_serializing_editable(editable: &bool) -> bool {
48    !*editable
49}
50
51impl<'a> From<PypiPackageDataModel<'a>> for PypiPackageData {
52    fn from(value: PypiPackageDataModel<'a>) -> Self {
53        Self {
54            name: value.name.into_owned(),
55            version: value.version.into_owned(),
56            location: value.location.into_owned(),
57            hash: value.hash.into_owned(),
58            requires_dist: value.requires_dist.into_owned(),
59            requires_python: value.requires_python.into_owned(),
60            editable: value.editable,
61        }
62    }
63}

65impl<'a> From<&'a PypiPackageData> for PypiPackageDataModel<'a> {
66    fn from(value: &'a PypiPackageData) -> Self {
67        Self {
68            name: Cow::Borrowed(&value.name),
69            version: Cow::Borrowed(&value.version),
70            location: Cow::Borrowed(&value.location),
71            hash: Cow::Borrowed(&value.hash),
72            requires_dist: Cow::Borrowed(&value.requires_dist),
73            requires_python: Cow::Borrowed(&value.requires_python),
74            editable: value.editable,
75        }
76    }
