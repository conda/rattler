25/// A helper struct that wraps all fields of a [`crate::CondaPackageData`] and
26/// allows for easy conversion between the two.
27///
28/// This type provides full control over the order of the fields when
29/// serializing. This is important because one of the design goals is that it
30/// should be easy to read the lock file. A [`PackageRecord`] is serialized in
31/// alphabetic order which might not be the most readable. This type instead
32/// puts the "most important" fields at the top followed by more detailed ones.
33///
34/// So note that for reproducibility the order of these fields should not change
35/// or should be reflected in a version change.
36//
37/// This type also adds more default values (e.g. for `build_number` and
38/// `build_string`).
39///
40/// The complexity with `Cow<_>` types is introduced to allow both efficient
41/// deserialization and serialization without requiring all data to be cloned
42/// when serializing. We want to be able to use the same type of both
43/// serialization and deserialization to ensure that when any of the
44/// types involved change we are forced to update this struct as well.
45#[serde_as]
46#[derive(Deserialize, Eq, PartialEq)]
47pub(crate) struct CondaPackageDataModel<'a> {
48    /// The location of the package. This can be a URL or a path.
49    #[serde(rename = "conda")]
50    pub location: UrlOrPath,
51
52    // Unique identifiers go to the top
53    #[serde(default, skip_serializing_if = "Option::is_none")]
54    pub name: Option<Cow<'a, PackageName>>,
55    #[serde(default, skip_serializing_if = "Option::is_none")]
56    pub version: Option<Cow<'a, VersionWithSource>>,
57    #[serde(default, skip_serializing_if = "Option::is_none")]
58    pub build: Option<Cow<'a, str>>,
59    #[serde(default, skip_serializing_if = "Option::is_none")]
60    pub build_number: Option<BuildNumber>,
61    #[serde(default, skip_serializing_if = "Option::is_none")]
62    pub subdir: Option<Cow<'a, str>>,
63    #[serde(default, skip_serializing_if = "Option::is_none")]
64    pub noarch: Option<Cow<'a, NoArchType>>,
65
66    // Conda-build variants for source packages (optional in V6, required in V7)
67    #[serde(default, skip_serializing_if = "Option::is_none")]
68    pub variants: Option<Cow<'a, BTreeMap<String, VariantValue>>>,
69
70    // Then the hashes
71    #[serde(default, skip_serializing_if = "Option::is_none")]
72    #[serde_as(as = "Option<SerializableHash::<rattler_digest::Sha256>>")]
73    pub sha256: Option<Sha256Hash>,
74    #[serde(default, skip_serializing_if = "Option::is_none")]
75    #[serde_as(as = "Option<SerializableHash::<rattler_digest::Md5>>")]
76    pub md5: Option<Md5Hash>,
77    #[serde(default, skip_serializing_if = "Option::is_none")]
78    #[serde_as(as = "Option<SerializableHash::<rattler_digest::Md5>>")]
79    pub legacy_bz2_md5: Option<Md5Hash>,
80
81    // Dependencies
82    #[serde(default, skip_serializing_if = "<[String]>::is_empty")]
83    pub depends: Cow<'a, [String]>,
84    #[serde(default, skip_serializing_if = "<[String]>::is_empty")]
85    pub constrains: Cow<'a, [String]>,
86    #[serde(default, skip_serializing_if = "BTreeMap::is_empty")]
87    #[serde(rename = "extra_depends")]
88    pub experimental_extra_depends: Cow<'a, BTreeMap<String, Vec<String>>>,
89
90    // Additional properties (in semi alphabetic order but grouped by commonality)
91    #[serde(default, skip_serializing_if = "Option::is_none")]
92    pub channel: Option<Cow<'a, Option<Url>>>,
93
94    #[serde(default, skip_serializing_if = "Option::is_none")]
95    pub features: Cow<'a, Option<String>>,
96    #[serde(default, skip_serializing_if = "<[Flag]>::is_empty")]
97    pub flags: Cow<'a, [Flag]>,
98    #[serde(default, skip_serializing_if = "<[String]>::is_empty")]
99    pub track_features: Cow<'a, [String]>,
100
101    #[serde(default, skip_serializing_if = "Option::is_none")]
102    pub file_name: Option<Cow<'a, Option<DistArchiveIdentifier>>>,
103
104    #[serde(default, skip_serializing_if = "Option::is_none")]
105    pub license: Cow<'a, Option<String>>,
106    #[serde(default, skip_serializing_if = "Option::is_none")]
107    pub license_family: Cow<'a, Option<String>>,
108    #[serde(default, skip_serializing_if = "Option::is_none")]
109    pub purls: Cow<'a, Option<BTreeSet<PackageUrl>>>,
110
111    #[serde(default, skip_serializing_if = "Option::is_none")]
112    pub size: Cow<'a, Option<u64>>,
113    #[serde(default, skip_serializing_if = "Option::is_none")]
114    pub legacy_bz2_size: Cow<'a, Option<u64>>,
115
116    #[serde(default, skip_serializing_if = "Option::is_none")]
117    #[serde_as(as = "Option<crate::utils::serde::Timestamp>")]
118    pub timestamp: Option<chrono::DateTime<chrono::Utc>>,
119
120    #[serde(default, skip_serializing_if = "Option::is_none")]
121    pub input: Option<InputHash<'a>>,
122
123    #[serde(default, skip_serializing_if = "Option::is_none")]
124    #[serde_as(as = "Option<PackageBuildSourceSerializer>")]
125    pub package_build_source: Option<PackageBuildSource>,
126
127    #[serde(default, skip_serializing_if = "BTreeMap::is_empty")]
128    #[serde_as(as = "BTreeMap<_, SourceLocationSerializer>")]
129    pub sources: BTreeMap<String, SourceLocation>,
130
131    #[serde(default, skip_serializing_if = "Option::is_none")]
132    pub python_site_packages_path: Cow<'a, Option<String>>,
133}

143impl<'a> TryFrom<CondaPackageDataModel<'a>> for LegacyCondaPackageData {
144    type Error = ConversionError;
145
146    fn try_from(value: CondaPackageDataModel<'a>) -> Result<Self, Self::Error> {
147        let derived = LocationDerivedFields::new(&value.location);
148        let build = value
149            .build
150            .map(Cow::into_owned)
151            .or_else(|| derived.build.clone())
152            .unwrap_or_default();
153        let build_number = value
154            .build_number
155            .or_else(|| derived_fields::derive_build_number_from_build(&build))
156            .unwrap_or(0);
157        let subdir = value
158            .subdir
159            .map(Cow::into_owned)
160            .or_else(|| derived.subdir.clone())
161            .ok_or_else(|| ConversionError::Missing("subdir".to_string()))?;
162        let noarch = value.noarch.map_or_else(
163            || {
164                derived_fields::derive_noarch_type(
165                    derived.subdir.as_deref().unwrap_or(&subdir),
166                    derived.build.as_deref().unwrap_or(&build),
167                )
168            },
169            Cow::into_owned,
170        );
171        let (derived_arch, derived_platform) = derived_fields::derive_arch_and_platform(&subdir);
172
173        let package_record = PackageRecord {
174            build,
175            build_number,
176            constrains: value.constrains.into_owned(),
177            depends: value.depends.into_owned(),
178            experimental_extra_depends: value.experimental_extra_depends.into_owned(),
179            features: value.features.into_owned(),
180            flags: value.flags.into_owned(),
181            legacy_bz2_md5: value.legacy_bz2_md5,
182            legacy_bz2_size: value.legacy_bz2_size.into_owned(),
183            license: value.license.into_owned(),
184            license_family: value.license_family.into_owned(),
185            md5: value.md5,
186            name: value
187                .name
188                .map(Cow::into_owned)
189                .or(derived.name)
190                .ok_or_else(|| ConversionError::Missing("name".to_string()))?,
191            noarch,
192            arch: derived_arch,
193            platform: derived_platform,
194            purls: value.purls.into_owned(),
195            sha256: value.sha256,
196            size: value.size.into_owned(),
197            subdir,
198            timestamp: value.timestamp.map(Into::into),
199            track_features: value.track_features.into_owned(),
200            version: value
201                .version
202                .map(Cow::into_owned)
203                .or(derived.version)
204                .ok_or_else(|| ConversionError::Missing("version".to_string()))?,
205            run_exports: None,
206            python_site_packages_path: value.python_site_packages_path.into_owned(),
207        };
208
209        if value
210            .location
211            .file_name()
212            .is_some_and(|name| CondaArchiveIdentifier::try_from_filename(name).is_some())
213        {
214            let file_name = value
215                .location
216                .file_name()
217                .expect("if checked this")
218                .to_owned();
219            Ok(LegacyCondaPackageData::Binary(LegacyCondaBinaryData {
220                location: value.location,
221                file_name,
222                channel: value
223                    .channel
224                    .map(Cow::into_owned)
225                    .map(|m| m.map(ChannelUrl::from))
226                    .unwrap_or(derived.channel),
227                package_record,
228            }))
229        } else {
230            Ok(LegacyCondaPackageData::Source(LegacyCondaSourceData {
231                package_record,
232                location: value.location,
233                variants: value.variants.map(Cow::into_owned).unwrap_or_default(),
234                package_build_source: value.package_build_source,
235                sources: value.sources,
236            }))
237        }
238    }
239}
