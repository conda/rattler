26/// A helper struct that wraps all fields of a [`crate::CondaPackageData`] and
27/// allows for easy conversion between the two.
28///
29/// This type provides full control over the order of the fields when
30/// serializing. This is important because one of the design goals is that it
31/// should be easy to read the lock file. A [`PackageRecord`] is serialized in
32/// alphabetic order which might not be the most readable. This type instead
33/// puts the "most important" fields at the top followed by more detailed ones.
34///
35/// So note that for reproducibility the order of these fields should not change
36/// or should be reflected in a version change.
37//
38/// This type also adds more default values (e.g. for `build_number` and
39/// `build_string`).
40///
41/// The complexity with `Cow<_>` types is introduced to allow both efficient
42/// deserialization and serialization without requiring all data to be cloned
43/// when serializing. We want to be able to use the same type of both
44/// serialization and deserialization to ensure that when any of the
45/// types involved change we are forced to update this struct as well.
46#[serde_as]
47#[derive(Serialize, Deserialize, Eq, PartialEq)]
48pub(crate) struct CondaPackageDataModel<'a> {
49    /// The location of the package. This can be a URL or a path.
50    #[serde(rename = "conda")]
51    pub location: UrlOrPath,
52
53    // Unique identifiers go to the top
54    #[serde(default, skip_serializing_if = "Option::is_none")]
55    pub name: Option<Cow<'a, PackageName>>,
56    #[serde(default, skip_serializing_if = "Option::is_none")]
57    pub version: Option<Cow<'a, VersionWithSource>>,
58    #[serde(default, skip_serializing_if = "Option::is_none")]
59    pub build: Option<Cow<'a, str>>,
60    #[serde(default, skip_serializing_if = "Option::is_none")]
61    pub build_number: Option<BuildNumber>,
62    #[serde(default, skip_serializing_if = "Option::is_none")]
63    pub subdir: Option<Cow<'a, str>>,
64    #[serde(default, skip_serializing_if = "Option::is_none")]
65    pub noarch: Option<Cow<'a, NoArchType>>,
66
67    // Conda-build variants for source packages (optional in V6, required in V7)
68    #[serde(default, skip_serializing_if = "Option::is_none")]
69    pub variants: Option<Cow<'a, BTreeMap<String, VariantValue>>>,
70
71    // Then the hashes
72    #[serde(default, skip_serializing_if = "Option::is_none")]
73    #[serde_as(as = "Option<SerializableHash::<rattler_digest::Sha256>>")]
74    pub sha256: Option<Sha256Hash>,
75    #[serde(default, skip_serializing_if = "Option::is_none")]
76    #[serde_as(as = "Option<SerializableHash::<rattler_digest::Md5>>")]
77    pub md5: Option<Md5Hash>,
78    #[serde(default, skip_serializing_if = "Option::is_none")]
79    #[serde_as(as = "Option<SerializableHash::<rattler_digest::Md5>>")]
80    pub legacy_bz2_md5: Option<Md5Hash>,
81
82    // Dependencies
83    #[serde(default, skip_serializing_if = "<[String]>::is_empty")]
84    pub depends: Cow<'a, [String]>,
85    #[serde(default, skip_serializing_if = "<[String]>::is_empty")]
86    pub constrains: Cow<'a, [String]>,
87    #[serde(default, skip_serializing_if = "BTreeMap::is_empty")]
88    #[serde(rename = "extra_depends")]
89    pub experimental_extra_depends: Cow<'a, BTreeMap<String, Vec<String>>>,
90
91    // Additional properties (in semi alphabetic order but grouped by commonality)
92    #[serde(default, skip_serializing_if = "Option::is_none")]
93    pub channel: Option<Cow<'a, Option<Url>>>,
94
95    #[serde(default, skip_serializing_if = "Option::is_none")]
96    pub features: Cow<'a, Option<String>>,
97    #[serde(default, skip_serializing_if = "<[String]>::is_empty")]
98    pub track_features: Cow<'a, [String]>,
99
100    #[serde(default, skip_serializing_if = "Option::is_none")]
101    pub file_name: Option<Cow<'a, Option<DistArchiveIdentifier>>>,
102
103    #[serde(default, skip_serializing_if = "Option::is_none")]
104    pub license: Cow<'a, Option<String>>,
105    #[serde(default, skip_serializing_if = "Option::is_none")]
106    pub license_family: Cow<'a, Option<String>>,
107    #[serde(default, skip_serializing_if = "Option::is_none")]
108    pub purls: Cow<'a, Option<BTreeSet<PackageUrl>>>,
109
110    #[serde(default, skip_serializing_if = "Option::is_none")]
111    pub size: Cow<'a, Option<u64>>,
112    #[serde(default, skip_serializing_if = "Option::is_none")]
113    pub legacy_bz2_size: Cow<'a, Option<u64>>,
114
115    #[serde(default, skip_serializing_if = "Option::is_none")]
116    #[serde_as(as = "Option<crate::utils::serde::Timestamp>")]
117    pub timestamp: Option<chrono::DateTime<chrono::Utc>>,
118
119    #[serde(default, skip_serializing_if = "Option::is_none")]
120    pub input: Option<InputHash<'a>>,
121
122    #[serde(default, skip_serializing_if = "Option::is_none")]
123    #[serde_as(as = "Option<PackageBuildSourceSerializer>")]
124    pub package_build_source: Option<PackageBuildSource>,
125
126    #[serde(default, skip_serializing_if = "BTreeMap::is_empty")]
127    #[serde_as(as = "BTreeMap<_, SourceLocationSerializer>")]
128    pub sources: BTreeMap<String, SourceLocation>,
129
130    #[serde(default, skip_serializing_if = "Option::is_none")]
131    pub python_site_packages_path: Cow<'a, Option<String>>,
132}

142impl<'a> TryFrom<CondaPackageDataModel<'a>> for CondaPackageData {
143    type Error = ConversionError;
144
145    fn try_from(value: CondaPackageDataModel<'a>) -> Result<Self, Self::Error> {
146        let derived = LocationDerivedFields::new(&value.location);
147        let build = value
148            .build
149            .map(Cow::into_owned)
150            .or_else(|| derived.build.clone())
151            .unwrap_or_default();
152        let build_number = value
153            .build_number
154            .or_else(|| derived_fields::derive_build_number_from_build(&build))
155            .unwrap_or(0);
156        let subdir = value
157            .subdir
158            .map(Cow::into_owned)
159            .or_else(|| derived.subdir.clone())
160            .ok_or_else(|| ConversionError::Missing("subdir".to_string()))?;
161        let noarch = value.noarch.map_or_else(
162            || {
163                derived_fields::derive_noarch_type(
164                    derived.subdir.as_deref().unwrap_or(&subdir),
165                    derived.build.as_deref().unwrap_or(&build),
166                )
167            },
168            Cow::into_owned,
169        );
170        let (derived_arch, derived_platform) = derived_fields::derive_arch_and_platform(&subdir);
171
172        let package_record = PackageRecord {
173            build,
174            build_number,
175            constrains: value.constrains.into_owned(),
176            depends: value.depends.into_owned(),
177            experimental_extra_depends: value.experimental_extra_depends.into_owned(),
178            features: value.features.into_owned(),
179            legacy_bz2_md5: value.legacy_bz2_md5,
180            legacy_bz2_size: value.legacy_bz2_size.into_owned(),
181            license: value.license.into_owned(),
182            license_family: value.license_family.into_owned(),
183            md5: value.md5,
184            name: value
185                .name
186                .map(Cow::into_owned)
187                .or(derived.name)
188                .ok_or_else(|| ConversionError::Missing("name".to_string()))?,
189            noarch,
190            arch: derived_arch,
191            platform: derived_platform,
192            purls: value.purls.into_owned(),
193            sha256: value.sha256,
194            size: value.size.into_owned(),
195            subdir,
196            timestamp: value.timestamp.map(Into::into),
197            track_features: value.track_features.into_owned(),
198            version: value
199                .version
200                .map(Cow::into_owned)
201                .or(derived.version)
202                .ok_or_else(|| ConversionError::Missing("version".to_string()))?,
203            run_exports: None,
204            python_site_packages_path: value.python_site_packages_path.into_owned(),
205        };
206
207        if value
208            .location
209            .file_name()
210            .is_some_and(|name| CondaArchiveIdentifier::try_from_filename(name).is_some())
211        {
212            Ok(CondaPackageData::Binary(CondaBinaryData {
213                location: value.location,
214                file_name: value
215                    .file_name
216                    .map(Cow::into_owned)
217                    .unwrap_or(derived.identifier)
218                    .unwrap_or_else(|| DistArchiveIdentifier {
219                        identifier: ArchiveIdentifier {
220                            name: package_record.name.as_normalized().to_owned(),
221                            version: package_record.version.to_string(),
222                            build_string: package_record.build.clone(),
223                        },
224                        archive_type: DistArchiveType::Conda(CondaArchiveType::Conda),
225                    }),
226                channel: value
227                    .channel
228                    .map(Cow::into_owned)
229                    .map(|m| m.map(ChannelUrl::from))
230                    .unwrap_or(derived.channel),
231                package_record,
232            }))
233        } else {
234            Ok(CondaPackageData::Source(CondaSourceData {
235                package_record,
236                location: value.location,
237                variants: value.variants.map(Cow::into_owned),
238                package_build_source: value.package_build_source,
239                input: value.input.map(|input| conda::InputHash {
240                    hash: input.hash,
241                    globs: input.globs.into_owned(),
242                }),
243                sources: value.sources,
244            }))
245        }
246    }
247}

249impl<'a> From<&'a CondaPackageData> for CondaPackageDataModel<'a> {
250    fn from(value: &'a CondaPackageData) -> Self {
251        let package_record = value.record();
252        let derived = LocationDerivedFields::new(value.location());
253        let derived_build_number =
254            derived_fields::derive_build_number_from_build(&package_record.build).unwrap_or(0);
255        let derived_noarch = derived_fields::derive_noarch_type(
256            derived.subdir.as_deref().unwrap_or(&package_record.subdir),
257            derived.build.as_deref().unwrap_or(&package_record.build),
258        );
259
260        let channel = value.as_binary().and_then(|binary| binary.channel.as_ref());
261        let file_name = value.as_binary().map(|binary| &binary.file_name);
262        let variants = value
263            .as_source()
264            .and_then(|source| source.variants.as_ref());
265        let input = value.as_source().and_then(|source| source.input.as_ref());
266        let package_build_source = value
267            .as_source()
268            .and_then(|source| source.package_build_source.as_ref());
269        let sources = value
270            .as_source()
271            .map_or_else(BTreeMap::new, |source| source.sources.clone());
272
273        let normalized_channel = channel
274            .map(|channel| strip_trailing_slash(channel.as_ref()))
275            .map(Cow::into_owned);
276
277        Self {
278            location: value.location().clone(),
279            name: (Some(package_record.name.as_source())
280                != derived.name.as_ref().map(PackageName::as_source))
281            .then_some(Cow::Borrowed(&package_record.name)),
282            version: (Some(package_record.version.as_str())
283                != derived.version.as_ref().map(VersionWithSource::as_str))
284            .then_some(Cow::Borrowed(&package_record.version)),
285            build: (package_record.build.as_str()
286                != derived.build.as_ref().map_or("", |s| s.as_str()))
287            .then_some(Cow::Borrowed(&package_record.build)),
288            build_number: (package_record.build_number != derived_build_number)
289                .then_some(package_record.build_number),
290            subdir: (Some(package_record.subdir.as_str()) != derived.subdir.as_deref())
291                .then_some(Cow::Borrowed(&package_record.subdir)),
292            noarch: (package_record.noarch != derived_noarch)
293                .then_some(Cow::Borrowed(&package_record.noarch)),
294            variants: variants.map(Cow::Borrowed),
295            channel: (channel != derived.channel.as_ref())
296                .then_some(Cow::Owned(normalized_channel)),
297            file_name: (file_name != derived.identifier.as_ref())
298                .then_some(Cow::Owned(file_name.cloned())),
299            purls: Cow::Borrowed(&package_record.purls),
300            depends: Cow::Borrowed(&package_record.depends),
301            constrains: Cow::Borrowed(&package_record.constrains),
302            experimental_extra_depends: Cow::Borrowed(&package_record.experimental_extra_depends),
303            md5: package_record.md5,
304            legacy_bz2_md5: package_record.legacy_bz2_md5,
305            sha256: package_record.sha256,
306            size: Cow::Borrowed(&package_record.size),
307            legacy_bz2_size: Cow::Borrowed(&package_record.legacy_bz2_size),
308            timestamp: package_record.timestamp.map(TimestampMs::into_datetime),
309            features: Cow::Borrowed(&package_record.features),
310            track_features: Cow::Borrowed(&package_record.track_features),
311            license: Cow::Borrowed(&package_record.license),
312            license_family: Cow::Borrowed(&package_record.license_family),
313            python_site_packages_path: Cow::Borrowed(&package_record.python_site_packages_path),
314            input: input.map(|input| InputHash {
315                hash: input.hash,
316                globs: Cow::Borrowed(&input.globs),
317            }),
318            package_build_source: package_build_source.cloned(),
319            sources,
320        }
321    }
322}

324fn strip_trailing_slash(url: &Url) -> Cow<'_, Url> {
325    let path = url.path();
326    if path.ends_with("/") {
327        let mut updated_url = url.clone();
328        updated_url.set_path(path.trim_end_matches('/'));
329        Cow::Owned(updated_url)
330    } else {
331        Cow::Borrowed(url)
332    }
