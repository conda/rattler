1//! Intermediate types for legacy lock file deserialization (V1-V6).
2//!
3//! These types are used during lock file parsing for versions 1 through 6 to
4//! preserve all fields needed for package disambiguation. After environment
5//! resolution is complete, they are converted to the final `CondaPackageData`
6//! type.
7//!
8//! This intermediate layer allows `CondaPackageData` to evolve for newer
9//! versions (V7+) without breaking backward compatibility with older lock
10//! files.
11//!
12//! # Why merging exists
13//!
14//! In lock file versions before V6, multiple packages could share the same URL
15//! but differ in metadata (e.g., different subdirs for the same package). When
16//! resolving environment references, these packages needed to be merged to
17//! combine optional fields like `purls`, `run_exports`, and hashes that might
18//! only be present in one of the duplicates.
19//!
20//! V6 introduced disambiguating fields (name, version, build, subdir) in
21//! environment references, making exact matching possible. However, the merging
22//! logic is retained here for parsing older format versions.
23
24use crate::{
25    conda::{CondaBinaryData, CondaSourceData, PackageBuildSource, SourceMetadata, VariantValue},
26    source::SourceLocation,
27    CondaPackageData, UrlOrPath,

29use rattler_conda_types::{package::DistArchiveIdentifier, ChannelUrl, PackageRecord};

32/// Intermediate representation of a conda package during legacy deserialization.
33///
34/// This type preserves all fields from legacy lock file formats (V1-V6) that
35/// are needed for package disambiguation (matching environment references to
36/// package data). After resolution, it is converted to `CondaPackageData`.
37#[derive(Clone, Debug, Hash, Eq, PartialEq)]
38pub(crate) enum LegacyCondaPackageData {
39    /// A binary package.
40    Binary(LegacyCondaBinaryData),
41    /// A source package.
42    Source(LegacyCondaSourceData),
43}
44
45/// Intermediate binary package data for legacy deserialization.
46#[derive(Clone, Debug, Hash, Eq, PartialEq)]
47pub(crate) struct LegacyCondaBinaryData {
48    /// The package record with all metadata.
49    pub package_record: PackageRecord,
50    /// The location of the package (URL or path).
51    pub location: UrlOrPath,
52    /// The filename of the package.
53    pub file_name: String,
54    /// The channel of the package.
55    pub channel: Option<ChannelUrl>,
56}
57
58/// Intermediate source package data for legacy deserialization.
59#[derive(Clone, Debug, Hash, Eq, PartialEq)]
60pub(crate) struct LegacyCondaSourceData {
61    /// The package record with all metadata.
62    pub package_record: PackageRecord,
63    /// The location of the package (URL or path).
64    pub location: UrlOrPath,
65    /// Conda-build variants for disambiguation.
66    pub variants: BTreeMap<String, VariantValue>,
67    /// Package build source location.
68    pub package_build_source: Option<PackageBuildSource>,
69    /// Source locations for packages built from source.
70    pub sources: BTreeMap<String, SourceLocation>,
71}
72
73impl LegacyCondaPackageData {
74    /// Returns the location of the package.
75    pub fn location(&self) -> &UrlOrPath {
76        match self {
77            Self::Binary(data) => &data.location,
78            Self::Source(data) => &data.location,
79        }
80    }
81
82    /// Returns the package record.
83    pub fn record(&self) -> &PackageRecord {
84        match self {
85            Self::Binary(data) => &data.package_record,
86            Self::Source(data) => &data.package_record,
87        }
88    }
89
90    /// Performs best-effort merge of two packages with the same location.
91    ///
92    /// This is needed for lock file versions before V6, where multiple packages
93    /// could share the same URL but have different metadata. Optional fields
94    /// like `purls`, `run_exports`, `md5`, and `sha256` are merged from the
95    /// second package if they are missing in the first.
96    ///
97    /// For V6 and later, this method is typically not called because packages
98    /// are matched exactly using name/version/build/subdir fields.
99    pub fn merge(&self, other: &Self) -> Cow<'_, Self> {
100        match (self, other) {
101            (Self::Binary(left), Self::Binary(right)) if left.location == right.location => {
102                if let Cow::Owned(merged) =
103                    merge_package_record(&left.package_record, &right.package_record)
104                {
105                    return Cow::Owned(Self::Binary(LegacyCondaBinaryData {
106                        package_record: merged,
107                        location: left.location.clone(),
108                        file_name: left.file_name.clone(),
109                        channel: left.channel.clone(),
110                    }));
111                }
112            }
113            (Self::Source(left), Self::Source(right)) if left.location == right.location => {
114                let record_merge =
115                    merge_package_record(&left.package_record, &right.package_record);
116                let build_source_merge = merge_package_build_source(
117                    &left.package_build_source,
118                    &right.package_build_source,
119                );
120
121                if matches!(record_merge, Cow::Owned(_))
122                    || matches!(build_source_merge, Cow::Owned(_))
123                {
124                    return Cow::Owned(Self::Source(LegacyCondaSourceData {
125                        package_record: record_merge.into_owned(),
126                        location: left.location.clone(),
127                        variants: left.variants.clone(),
128                        package_build_source: build_source_merge.into_owned(),
129                        sources: left.sources.clone(),
130                    }));
131                }
132            }
133            _ => {}
134        }
135        Cow::Borrowed(self)
136    }
137}

139/// Convert `LegacyCondaPackageData` to the final `CondaPackageData` type.
140///
141/// Currently this is a 1:1 conversion. When `CondaPackageData` is updated for
142/// V7 (e.g., source packages use hash instead of full `PackageRecord`), this
143/// conversion will compute the hash from the legacy data.
144impl From<LegacyCondaPackageData> for CondaPackageData {
145    fn from(value: LegacyCondaPackageData) -> Self {
146        match value {
147            LegacyCondaPackageData::Binary(data) => {
148                let file_id = DistArchiveIdentifier::try_from_filename(&data.file_name)
149                    .expect("This file name must be valid here");
150                CondaPackageData::Binary(Box::new(CondaBinaryData {
151                    package_record: data.package_record,
152                    location: data.location,
153                    file_name: file_id,
154                    channel: data.channel,
155                }))
156            }
157            LegacyCondaPackageData::Source(data) => {
158                CondaPackageData::Source(Box::new(CondaSourceData {
159                    location: data.location,
160                    package_build_source: data.package_build_source,
161                    variants: data.variants,
162                    identifier_hash: None,
163                    sources: data.sources,
164                    source_data: crate::SourceData::default(),
165                    metadata: SourceMetadata::Full(Box::new(data.package_record)),
166                }))
167            }
168        }
169    }
170}

172/// Merge optional fields from `right` into `left` if they are missing.
173fn merge_package_record<'a>(
174    left: &'a PackageRecord,
175    right: &PackageRecord,
176) -> Cow<'a, PackageRecord> {
177    // Check which fields need to be merged
178    let needs_purls = left.purls.is_none() && right.purls.is_some();
179    let needs_run_exports = left.run_exports.is_none() && right.run_exports.is_some();
180    let needs_md5 = left.md5.is_none() && right.md5.is_some();
181    let needs_sha256 = left.sha256.is_none() && right.sha256.is_some();
182
183    // Return borrowed if no merging needed
184    if !needs_purls && !needs_run_exports && !needs_md5 && !needs_sha256 {
185        return Cow::Borrowed(left);
186    }
187
188    // Clone once and update only the fields that need merging
189    let mut merged = left.clone();
190    if needs_purls {
191        merged.purls.clone_from(&right.purls);
192    }
193    if needs_run_exports {
194        merged.run_exports.clone_from(&right.run_exports);
195    }
196    if needs_md5 {
197        merged.md5 = right.md5;
198    }
199    if needs_sha256 {
200        merged.sha256 = right.sha256;
201    }
202    Cow::Owned(merged)
203}

205/// Merge package build source, preferring the right value if left is None.
206fn merge_package_build_source<'a>(
207    left: &'a Option<PackageBuildSource>,
208    right: &Option<PackageBuildSource>,
209) -> Cow<'a, Option<PackageBuildSource>> {
210    if left == right {
211        Cow::Borrowed(left)
212    } else if let Some(right_source) = right {
213        // New data takes precedence
214        Cow::Owned(Some(right_source.clone()))
215    } else {
216        Cow::Borrowed(left)
217    }
