129/// A function that enables parsing of lock files version 3 or lower.
130pub fn parse_v3_or_lower(
131    document: serde_yaml::Value,
132    version: FileFormatVersion,
133) -> Result<LockFile, ParseCondaLockError> {
134    let lock_file: LockFileV3 =
135        serde_yaml::from_value(document).map_err(ParseCondaLockError::ParseError)?;
136
137    // Iterate over all packages, deduplicate them and store the list of packages
138    // per platform. There might be duplicates for noarch packages.
139    let mut conda_packages = IndexSet::with_capacity(lock_file.package.len());
140    let mut pypi_packages = IndexSet::with_capacity(lock_file.package.len());
141    let mut pypi_runtime_configs = IndexSet::with_capacity(lock_file.package.len());
142    let mut per_platform: ahash::HashMap<Platform, IndexSet<EnvironmentPackageData>> =
143        ahash::HashMap::default();
144    for package in lock_file.package {
145        let LockedPackageV3 { platform, kind } = package;
146
147        let pkg: EnvironmentPackageData = match kind {
148            LockedPackageKindV3::Conda(value) => {
149                let md5 = match value.hash {
150                    PackageHashes::Md5(md5) | PackageHashes::Md5Sha256(md5, _) => Some(md5),
151                    PackageHashes::Sha256(_) => None,
152                };
153                let sha256 = match value.hash {
154                    PackageHashes::Sha256(sha256) | PackageHashes::Md5Sha256(_, sha256) => {
155                        Some(sha256)
156                    }
157                    PackageHashes::Md5(_) => None,
158                };
159
160                // Guess the subdir from the url.
161                let subdir = value
162                    .subdir
163                    .as_deref()
164                    .or_else(|| {
165                        value
166                            .url
167                            .path_segments()
168                            .and_then(|split| split.rev().nth(1))
169                    })
170                    .and_then(|subdir_str| Platform::from_str(subdir_str).ok())
171                    .unwrap_or(platform);
172
173                let location = UrlOrPath::Url(value.url).normalize().into_owned();
174                let derived = LocationDerivedFields::new(&location);
175
176                let build = value
177                    .build
178                    .or_else(|| derived.build.clone())
179                    .unwrap_or_default();
180                let build_number = value
181                    .build_number
182                    .or_else(|| derive_build_number_from_build(&build))
183                    .unwrap_or(0);
184                let derived_noarch = derive_noarch_type(
185                    derived.subdir.as_deref().unwrap_or(subdir.as_str()),
186                    derived.build.as_deref().unwrap_or(&build),
187                );
188                let (derived_arch, derived_platform) =
189                    derive_arch_and_platform(derived.subdir.as_deref().unwrap_or(subdir.as_str()));
190
191                let deduplicated_idx = conda_packages
192                    .insert_full(CondaPackageData::Binary(CondaBinaryData {
193                        channel: derived
194                            .channel
195                            .unwrap_or_else(|| Url::parse("https://example.com").unwrap().into())
196                            .into(),
197                        file_name: derived.identifier.unwrap_or_else(|| DistArchiveIdentifier {
198                            identifier: ArchiveIdentifier {
199                                name: value.name.clone(),
200                                version: value.version.to_string(),
201                                build_string: build.clone(),
202                            },
203                            archive_type: DistArchiveType::Conda(CondaArchiveType::Conda),
204                        }),
205                        package_record: PackageRecord {
206                            arch: value.arch.or(derived_arch),
207                            build,
208                            build_number,
209                            constrains: value.constrains,
210                            depends: value.dependencies,
211                            experimental_extra_depends: std::collections::BTreeMap::new(),
212                            features: value.features,
213                            legacy_bz2_md5: None,
214                            legacy_bz2_size: None,
215                            license: value.license,
216                            license_family: value.license_family,
217                            md5,
218                            name: PackageName::new_unchecked(value.name),
219                            noarch: value.noarch.unwrap_or(derived_noarch),
220                            platform: derived_platform,
221                            sha256,
222                            size: value.size,
223                            subdir: subdir.to_string(),
224                            timestamp: value.timestamp.map(Into::into),
225                            track_features: value.track_features,
226                            version: value.version,
227                            purls: value.purls.is_empty().not().then_some(value.purls),
228                            python_site_packages_path: value.python_site_packages_path,
229                            run_exports: None,
230                        },
231                        location,
232                    }))
233                    .0;
234
235                EnvironmentPackageData::Conda(deduplicated_idx)
236            }
237            LockedPackageKindV3::Pypi(pkg) => {
238                let deduplicated_index = pypi_packages
239                    .insert_full(PypiPackageData {
240                        name: pep508_rs::PackageName::new(pkg.name)?,
241                        version: pkg.version,
242                        requires_dist: pkg.requires_dist,
243                        requires_python: pkg.requires_python,
244                        location: UrlOrPath::Url(pkg.url),
245                        hash: pkg.hash,
246                        editable: false,
247                    })
248                    .0;
249                EnvironmentPackageData::Pypi(
250                    deduplicated_index,
251                    pypi_runtime_configs.insert_full(pkg.runtime).0,
252                )
253            }
254        };
255
256        per_platform
257            .entry(package.platform)
258            .or_default()
259            .insert(pkg);
260    }
261
262    // Construct the default environment
263    let default_environment = EnvironmentData {
264        channels: lock_file.metadata.channels,
265        indexes: None,
266        packages: per_platform,
267        options: SolveOptions::default(),
268    };
269
270    Ok(LockFile {
271        inner: Arc::new(LockFileInner {
272            version,
273            conda_packages: conda_packages.into_iter().collect(),
274            pypi_packages: pypi_packages.into_iter().collect(),
275            pypi_environment_package_data: pypi_runtime_configs
276                .into_iter()
277                .map(Into::into)
278                .collect(),
279
280            environment_lookup: [(DEFAULT_ENVIRONMENT_NAME.to_string(), 0)]
281                .into_iter()
282                .collect(),
283            environments: vec![default_environment],
284        }),
285    })
