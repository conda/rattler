15/// The minimum safe timestamp (1980-01-01T00:00:00 UTC) for filesystems like exFAT
16/// that do not support timestamps before 1980.
17const SAFE_MTIME_FLOOR: u64 = 315_532_800;

19/// Returns the `.tar.bz2` as a decompressed `tar::Archive`. The `tar::Archive` can be used to
20/// extract the files from it, or perform introspection.
21pub fn stream_tar_bz2(reader: impl Read) -> tar::Archive<impl Read + Sized> {
22    tar::Archive::new(bzip2::read::BzDecoder::new(reader))
23}
24
25/// Returns the `.tar.zst` as a decompressed `tar` archive. The `tar::Archive` can be used to
26/// extract the files from it, or perform introspection.
27pub(crate) fn stream_tar_zst(
28    reader: impl Read,
29) -> Result<tar::Archive<impl Read + Sized>, ExtractError> {
30    Ok(tar::Archive::new(zstd::stream::read::Decoder::new(reader)?))
31}

33/// Extracts the contents a `.tar.bz2` package archive.
34pub fn extract_tar_bz2(
35    reader: impl Read,
36    destination: &Path,
37) -> Result<ExtractResult, ExtractError> {
38    std::fs::create_dir_all(destination).map_err(ExtractError::CouldNotCreateDestination)?;
39
40    process_with_hashing(reader, |reader| {
41        let mut archive = stream_tar_bz2(reader);
42        unpack_tar_archive_sync(&mut archive, destination)?;
43        Ok(())
44    })
45}

47/// Extracts the contents of a `.conda` package archive.
48pub fn extract_conda_via_streaming(
49    reader: impl Read,
50    destination: &Path,
51) -> Result<ExtractResult, ExtractError> {
52    // Construct the destination path if it doesn't exist yet
53    std::fs::create_dir_all(destination).map_err(ExtractError::CouldNotCreateDestination)?;
54
55    process_with_hashing(reader, |reader| {
56        while let Some(file) = read_zipfile_from_stream(reader)? {
57            extract_zipfile(file, destination)?;
58        }
59        Ok(())
60    })
61}

63/// Extracts the contents of a .conda package archive by fully reading the stream and then decompressing
64pub fn extract_conda_via_buffering(
65    reader: impl Read,
66    destination: &Path,
67) -> Result<ExtractResult, ExtractError> {
68    // delete destination first, as this method is usually used as a fallback from a failed streaming decompression
69    if destination.exists() {
70        std::fs::remove_dir_all(destination).map_err(ExtractError::CouldNotCreateDestination)?;
71    }
72    std::fs::create_dir_all(destination).map_err(ExtractError::CouldNotCreateDestination)?;
73
74    process_with_hashing(reader, |reader| {
75        // Create a SpooledTempFile with a 5MB limit
76        let mut temp_file = SpooledTempFile::new(5 * 1024 * 1024);
77        copy(reader, &mut temp_file)?;
78        temp_file.seek(SeekFrom::Start(0))?;
79        let mut archive = ZipArchive::new(temp_file)?;
80
81        for i in 0..archive.len() {
82            let file = archive.by_index(i)?;
83            extract_zipfile(file, destination)?;
84        }
85        Ok(())
86    })
87}

89fn extract_zipfile<R: std::io::Read>(
90    zip_file: ZipFile<'_, R>,
91    destination: &Path,
92) -> Result<(), ExtractError> {
93    // If an error occurs while we are reading the contents of the zip we don't want to
94    // seek to the end of the file. Using [`ManuallyDrop`] we prevent `drop` to be called on
95    // the `file` in case the stack unwinds.
96    let mut file = ManuallyDrop::new(zip_file);
97
98    if file
99        .mangled_name()
100        .file_name()
101        .map(OsStr::to_string_lossy)
102        .is_some_and(|file_name| file_name.ends_with(".tar.zst"))
103    {
104        let mut archive = stream_tar_zst(&mut *file)?;
105        unpack_tar_archive_sync(&mut archive, destination)?;
106    } else {
107        // Manually read to the end of the stream if that didn't happen.
108        std::io::copy(&mut *file, &mut std::io::sink())?;
109    }
110
111    // Take the file out of the [`ManuallyDrop`] to properly drop it.
112    let _ = ManuallyDrop::into_inner(file);
113
114    Ok(())
115}

117/// Unpacks a tar archive while handling mtime-setting failures gracefully.
118///
119/// Disables the tar crate's automatic mtime preservation and instead sets
120/// mtimes manually with clamping (to `SAFE_MTIME_FLOOR`) and error handling.
121/// This prevents fatal extraction failures on filesystems like exFAT that
122/// do not support timestamps before 1980-01-01.
123fn unpack_tar_archive_sync<R: Read>(
124    archive: &mut tar::Archive<R>,
125    destination: &Path,
126) -> Result<(), ExtractError> {
127    archive.set_preserve_mtime(false);
128
129    for entry in archive.entries().map_err(ExtractError::IoError)? {
130        let mut entry = entry.map_err(ExtractError::IoError)?;
131        let mtime = entry.header().mtime().unwrap_or(0);
132        let is_symlink = entry.header().entry_type().is_symlink();
133        let entry_path = entry.path().map_err(ExtractError::IoError)?.into_owned();
134
135        let unpacked = entry
136            .unpack_in(destination)
137            .map_err(ExtractError::IoError)?;
138
139        // Set mtime on the path the entry was actually written to, recomputed
140        // with the same sanitization `unpack_in` applies. Joining the raw
141        // header path onto `destination` would let an absolute or `..` entry
142        // point the mtime write at a file outside the extraction directory.
143        if unpacked && let Some(full_path) = unpacked_destination_path(destination, &entry_path) {
144            set_mtime_safe(&full_path, mtime, is_symlink);
145        }
146    }
147
148    Ok(())
149}
150
151/// Resolves the on-disk path a tar entry is unpacked to, mirroring the
152/// sanitization in [`tar::Entry::unpack_in`]: absolute-path roots and `.`
153/// components are stripped and a `..` component makes the entry unsafe.
154///
155/// Returns `None` when the entry would not map to a distinct path inside
156/// `destination`, so callers never set metadata on a path that could resolve
157/// outside the extraction directory.
158fn unpacked_destination_path(destination: &Path, entry_path: &Path) -> Option<PathBuf> {
159    let mut full_path = destination.to_path_buf();
160    for component in entry_path.components() {
161        match component {
162            Component::Prefix(_) | Component::RootDir | Component::CurDir => {}
163            Component::ParentDir => return None,
164            Component::Normal(part) => full_path.push(part),
165        }
166    }
167
168    if full_path == destination {
169        return None;
170    }
171
172    Some(full_path)
173}

175/// Sets the modification time on a file, clamping to a safe minimum and
176/// logging a warning on failure instead of propagating the error.
177fn set_mtime_safe(path: &Path, mtime: u64, is_symlink: bool) {
178    let clamped = std::cmp::max(mtime, SAFE_MTIME_FLOOR);
179    let file_time = filetime::FileTime::from_unix_time(clamped as i64, 0);
180
181    let result = if is_symlink {
182        filetime::set_symlink_file_times(path, file_time, file_time)
183    } else {
184        filetime::set_file_mtime(path, file_time)
185    };
186
187    if let Err(e) = result {
188        tracing::warn!(
189            "Failed to set mtime for '{}': {}. \
190             The target filesystem may not support this timestamp. \
191             This does not affect package integrity.",
192            path.display(),
193            e
194        );
195    }
196}

198// Define a custom reader to track file size
199pub(crate) struct SizeCountingReader<R> {
200    inner: R,
201    size: u64,
202}
203
204impl<R> SizeCountingReader<R> {
205    pub(crate) fn new(inner: R) -> Self {
206        Self { inner, size: 0 }
207    }
208
209    pub(crate) fn finalize(self) -> (R, u64) {
210        (self.inner, self.size)
211    }
212}
213
214impl<R: Read> Read for SizeCountingReader<R> {
215    fn read(&mut self, buf: &mut [u8]) -> std::io::Result<usize> {
216        let bytes_read = self.inner.read(buf)?;
217        self.size += bytes_read as u64;
218        Ok(bytes_read)
219    }
220}
221
222// AsyncRead implementation for use with tokio
223impl<R: tokio::io::AsyncRead + Unpin> tokio::io::AsyncRead for SizeCountingReader<R> {
224    fn poll_read(
225        mut self: std::pin::Pin<&mut Self>,
226        cx: &mut std::task::Context<'_>,
227        buf: &mut tokio::io::ReadBuf<'_>,
228    ) -> std::task::Poll<std::io::Result<()>> {
229        let previously_filled = buf.filled().len();
230
231        // Since R: Unpin, we can safely use get_mut
232        let this = self.as_mut().get_mut();
233        let reader = std::pin::Pin::new(&mut this.inner);
234
235        match reader.poll_read(cx, buf) {
236            std::task::Poll::Ready(Ok(())) => {
237                let bytes_read = buf.filled().len() - previously_filled;
238                this.size += bytes_read as u64;
239                std::task::Poll::Ready(Ok(()))
240            }
241            other => other,
242        }
243    }
244}

246/// Helper function to compute hashes and size while processing a tar archive
247fn process_with_hashing<E, R, F>(reader: R, processor: F) -> Result<ExtractResult, E>
248where
249    R: Read,
250    E: From<std::io::Error>,
251    F: FnOnce(
252        &mut SizeCountingReader<
253            &mut rattler_digest::HashingReader<
254                rattler_digest::HashingReader<R, rattler_digest::Sha256>,
255                rattler_digest::Md5,
256            >,
257        >,
258    ) -> Result<(), E>,
259{
260    // Wrap the reading in additional readers that will compute the hashes of the file while its
261    // being read, and count the total size.
262    let sha256_reader = rattler_digest::HashingReader::<_, rattler_digest::Sha256>::new(reader);
263    let mut md5_reader =
264        rattler_digest::HashingReader::<_, rattler_digest::Md5>::new(sha256_reader);
265    let mut size_reader = SizeCountingReader::new(&mut md5_reader);
266
267    processor(&mut size_reader)?;
268
269    // Read the file to the end to make sure the hash is properly computed
270    std::io::copy(&mut size_reader, &mut std::io::sink())?;
271
272    // Get the size and hashes
273    let (_, total_size) = size_reader.finalize();
274    let (sha256_reader, md5) = md5_reader.finalize();
275    let (_, sha256) = sha256_reader.finalize();
276
277    Ok(ExtractResult {
278        sha256,
279        md5,
280        total_size,
281    })
