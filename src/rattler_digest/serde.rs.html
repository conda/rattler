1//! Provides custom serialization/deserialization functions for [`Output`] of a [`Digest`].
2//!
3//! Use the struct [`SerializableHash`] to easily serialize digest outputs. This type
4//! automatically handles both human-readable formats (like JSON) and binary formats
5//! (like MessagePack). See [`SerializableHash`] for detailed documentation on the
6//! serialization behavior.
7
8#![allow(clippy::doc_markdown)]

10use digest::{Digest, Output, OutputSizeUser};
11use serde::de::Error;
12use serde::{Deserialize, Deserializer, Serialize, Serializer};
13use serde_with::{DeserializeAs, SerializeAs};
14use std::ops::Deref;

16/// Deserialize the [`Output`] of a [`Digest`].
17///
18/// Accepts either a hex-encoded string (for human-readable formats like JSON)
19/// or raw bytes (for binary formats like MessagePack). The format is detected
20/// per-value rather than by branching on `deserializer.is_human_readable()`,
21/// because serde's internal `Content` deserializer (used for
22/// `#[serde(flatten)]` and friends) hardcodes `is_human_readable() == true`
23/// regardless of the underlying format, so a flattened digest field reached
24/// via a binary deserializer would otherwise be asked for a hex string.
25pub fn deserialize<'de, D, Dig: Digest>(deserializer: D) -> Result<Output<Dig>, D::Error>
26where
27    D: Deserializer<'de>,
28{
29    let expected_len = <Dig as OutputSizeUser>::output_size();
30    let check_len = |bytes: &[u8]| -> Result<Output<Dig>, String> {
31        Output::<Dig>::try_from(bytes).map_err(|_e| {
32            format!(
33                "invalid length, expected {expected_len}, got {}",
34                bytes.len()
35            )
36        })
37    };
38
39    serde_untagged::UntaggedEnumVisitor::new()
40        .expecting("a hex-encoded digest string or raw digest bytes")
41        .string(|s| {
42            super::parse_digest_from_hex::<Dig>(s)
43                .ok_or_else(|| Error::custom("failed to parse digest"))
44        })
45        .bytes(|b| check_len(b).map_err(Error::custom))
46        .seq(|seq| {
47            let bytes: Vec<u8> = seq.deserialize()?;
48            check_len(&bytes).map_err(Error::custom)
49        })
50        .deserialize(deserializer)
51}

53/// Serializes the [`Output`] of a [`Digest`].
54///
55/// If the serializer is human-readable, it will write the digest as a hex
56/// string. Otherwise, it will deserialize raw bytes.
57pub fn serialize<S: Serializer, Dig: Digest>(
58    digest: &Output<Dig>,
59    s: S,
60) -> Result<S::Ok, S::Error> {
61    if s.is_human_readable() {
62        hex::encode::<&[u8]>(digest.as_ref()).serialize(s)
63    } else {
64        // Without specialization, Rust forces Serde to treat &[u8] just like any other slice and
65        // Vec<u8> just like any other vector. In reality this particular slice and vector can
66        // often be serialized and deserialized in a more efficient, compact representation in many
67        // formats.
68        //
69        // Using the `serde_bytes` crate, we can opt into specialized handling.
70        serde_bytes::Bytes::new(digest.as_ref()).serialize(s)
71    }
72}

74/// Wrapper type for easily serializing and deserializing cryptographic hash digests.
75///
76/// This type provides automatic format-aware serialization:
77/// - **Human-readable formats** (JSON, YAML, TOML, etc.): Serializes as a lowercase hexadecimal string
78/// - **Binary formats** (MessagePack, Bincode, etc.): Serializes as raw bytes using `serde_bytes`
79///
80/// # Deserialization Behavior
81///
82/// ## Human-Readable Formats
83/// Accepts a hexadecimal string (case-insensitive) and parses it into the digest output.
84/// Returns an error if the string is not valid hexadecimal or has incorrect length.
85///
86/// ## Binary Formats
87/// Accepts raw bytes in multiple MessagePack encoding formats:
88/// - **Binary format** (`bin 8/16/32`): Most efficient, preferred format
89/// - **Array format** (`fixarray/array 16/32`): Array of integers, for compatibility
90///
91/// Returns an error if the byte length doesn't match the expected digest size.
92///
93/// # Examples
94///
95/// ## JSON (Human-Readable)
96/// ```
97/// use rattler_digest::serde::SerializableHash;
98///
99/// let hash = SerializableHash::<sha2::Sha256>(
100///     rattler_digest::parse_digest_from_hex::<sha2::Sha256>(
101///         "fe51de6107f9edc7aa4f786a70f4a883943bc9d39b3bb7307c04c41410990726"
102///     ).unwrap()
103/// );
104///
105/// // Serializes as: "fe51de6107f9edc7aa4f786a70f4a883943bc9d39b3bb7307c04c41410990726"
106/// let json = serde_json::to_string(&hash).unwrap();
107/// let deserialized: SerializableHash<sha2::Sha256> = serde_json::from_str(&json).unwrap();
108/// ```
109///
110/// ## MessagePack (Binary)
111/// ```
112/// use rattler_digest::serde::SerializableHash;
113///
114/// let hash = SerializableHash::<sha2::Sha256>(
115///     rattler_digest::parse_digest_from_hex::<sha2::Sha256>(
116///         "fe51de6107f9edc7aa4f786a70f4a883943bc9d39b3bb7307c04c41410990726"
117///     ).unwrap()
118/// );
119///
120/// // Serializes as raw 32 bytes using MessagePack binary format
121/// let bytes = rmp_serde::to_vec(&hash).unwrap();
122/// let deserialized: SerializableHash<sha2::Sha256> = rmp_serde::from_slice(&bytes).unwrap();
123/// ```
124#[derive(Debug, Clone)]
125pub struct SerializableHash<T: Digest>(pub Output<T>);
126
127impl<T: Digest> Serialize for SerializableHash<T> {
128    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
129    where
130        S: Serializer,
131    {
132        serialize::<S, T>(&self.0, serializer)
133    }
134}
135
136impl<'de, T: Digest + Default> Deserialize<'de> for SerializableHash<T> {
137    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
138    where
139        D: Deserializer<'de>,
140    {
141        let hash_output: Output<T> = deserialize::<D, T>(deserializer)?;
142        Ok(SerializableHash(hash_output))
143    }
144}
145
146impl<T: Digest> From<Output<T>> for SerializableHash<T> {
147    fn from(output: Output<T>) -> Self {
148        SerializableHash(output)
149    }
150}
151
152impl<T: Digest> From<SerializableHash<T>> for Output<T> {
153    fn from(s: SerializableHash<T>) -> Self {
154        s.0
155    }
156}
157
158impl<T: Digest> Deref for SerializableHash<T> {
159    type Target = Output<T>;
160
161    fn deref(&self) -> &Self::Target {
162        &self.0
163    }
164}
165
166impl<T: Digest> SerializeAs<Output<T>> for SerializableHash<T> {
167    fn serialize_as<S>(source: &Output<T>, serializer: S) -> Result<S::Ok, S::Error>
168    where
169        S: Serializer,
170    {
171        serialize::<S, T>(source, serializer)
172    }
173}
174
175impl<'de, T: Digest + Default> DeserializeAs<'de, Output<T>> for SerializableHash<T> {
176    fn deserialize_as<D>(deserializer: D) -> Result<Output<T>, D::Error>
177    where
178        D: Deserializer<'de>,
179    {
180        deserialize::<D, T>(deserializer)
181    }
182}

220    pub fn test_deserialize_messagepack_raw_bytes() {
221        use rmp::encode;
222
223        let hex_str = "fe51de6107f9edc7aa4f786a70f4a883943bc9d39b3bb7307c04c41410990726";
224        let expected_hash = SerializableHash::<sha2::Sha256>(
225            crate::parse_digest_from_hex::<sha2::Sha256>(hex_str).unwrap(),
226        );
227
228        // Get the raw bytes from the expected hash
229        let hash_bytes = expected_hash.0.as_slice();
230
231        // Test 1: MessagePack binary format (bin 8, bin 16, or bin 32)
232        let mut msgpack_bin = Vec::new();
233        encode::write_bin(&mut msgpack_bin, hash_bytes).unwrap();
234
235        let deserialized_bin: SerializableHash<sha2::Sha256> =
236            rmp_serde::from_slice(&msgpack_bin).unwrap();
237        assert_eq!(&expected_hash.0, &deserialized_bin.0);
238
239        // Test 2: MessagePack array format (fixarray or array 16/32)
240        let mut msgpack_array = Vec::new();
241        encode::write_array_len(&mut msgpack_array, hash_bytes.len() as u32).unwrap();
242        for byte in hash_bytes {
243            encode::write_uint(&mut msgpack_array, u64::from(*byte)).unwrap();
244        }
245
246        let deserialized_array: SerializableHash<sha2::Sha256> =
247            rmp_serde::from_slice(&msgpack_array).unwrap();
248        assert_eq!(&expected_hash.0, &deserialized_array.0);
249    }

251    #[test]
252    pub fn test_deserialize_messagepack_under_flatten() {
253        // Regression test: when SerializableHash is inside a struct that uses
254        // `#[serde(flatten)]`, serde buffers fields through its internal Content
255        // representation. That Content deserializer hardcodes
256        // `is_human_readable() == true`, which previously broke the bin/hex
257        // dispatch and surfaced as "byte array, expected a string" for shards
258        // produced for conda-pypi-style channels.
259        use serde::{Deserialize, Serialize};
260
261        #[serde_with::serde_as]
262        #[derive(Deserialize, Serialize)]
263        struct Inner {
264            #[serde_as(as = "SerializableHash::<sha2::Sha256>")]
265            sha256: sha2::digest::Output<sha2::Sha256>,
266        }
267
268        #[derive(Deserialize, Serialize)]
269        struct Outer {
270            #[serde(flatten)]
271            inner: Inner,
272            extra: String,
273        }
274
275        let expected = crate::parse_digest_from_hex::<sha2::Sha256>(
276            "fe51de6107f9edc7aa4f786a70f4a883943bc9d39b3bb7307c04c41410990726",
277        )
278        .unwrap();
279        let outer = Outer {
280            inner: Inner { sha256: expected },
281            extra: "hi".into(),
282        };
283        let bytes = rmp_serde::to_vec(&outer).unwrap();
284        let parsed: Outer = rmp_serde::from_slice(&bytes).unwrap();
285        assert_eq!(parsed.inner.sha256, expected);
286    }

338    pub fn test_messagepack_serialized_size() {
339        let hex_str = "fe51de6107f9edc7aa4f786a70f4a883943bc9d39b3bb7307c04c41410990726";
340        let hash = SerializableHash::<sha2::Sha256>(
341            crate::parse_digest_from_hex::<sha2::Sha256>(hex_str).unwrap(),
342        );
343
344        // Serialize to MessagePack
345        let bytes = rmp_serde::to_vec(&hash).unwrap();
346
347        // SHA256 is 32 bytes, MessagePack bin8 format uses:
348        // - 1 byte for the format marker (0xc4 for bin8)
349        // - 1 byte for the length (32)
350        // - 32 bytes for the data
351        // Total: 34 bytes
352        assert_eq!(bytes.len(), 34);
353
354        // Verify the format is bin8
355        assert_eq!(bytes[0], 0xc4); // bin8 marker
356        assert_eq!(bytes[1], 32); // length
357    }
