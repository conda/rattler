1//! Lazily populated view of the 'conda-meta' directory
2use std::collections::HashMap;
3use std::fs;
4use std::path::{Path, PathBuf};
5use std::sync::Arc;
6use std::sync::OnceLock;
7
8use crate::package::ArchiveIdentifier;
9use crate::package_name::PackageName;
10use crate::PrefixRecord;
11
12/// An error that can occur when loading a prefix record
13#[derive(Debug, Clone, thiserror::Error)]
14#[error("io error: {0}")]
15pub struct PrefixDataError(pub Arc<std::io::Error>);
16
17impl From<std::io::Error> for PrefixDataError {
18    fn from(err: std::io::Error) -> Self {
19        PrefixDataError(Arc::new(err))
20    }
21}
22
23/// Internal state for a lazily loaded package record
24struct LazyRecordEntry {
25    path: PathBuf,
26    record: OnceLock<Result<PrefixRecord, PrefixDataError>>,
27}
28
29/// A lazily populated view of the `conda-meta` directory in a prefix.
30pub struct PrefixData {
31    /// The path to the environment prefix
32    prefix_path: PathBuf,
33    /// A map of package names to their file path and a lazy lock for the parsed record
34    records: HashMap<PackageName, LazyRecordEntry>,
35}
36
37impl PrefixData {
38    /// Returns the path to the environment prefix
39    pub fn prefix_path(&self) -> &Path {
40        &self.prefix_path
41    }
42
43    /// Discovers all packages in the `conda-meta` directory but does not parse the JSON yet.
44    pub fn new(prefix_path: impl Into<PathBuf>) -> Result<Self, std::io::Error> {
45        let prefix_path = prefix_path.into();
46        let meta_dir = prefix_path.join("conda-meta");
47        let mut records = HashMap::new();
48
49        if !meta_dir.exists() {
50            return Ok(Self {
51                prefix_path,
52                records,
53            });
54        }
55
56        for entry in fs::read_dir(meta_dir)? {
57            let entry = entry?;
58            let path = entry.path();
59
60            if path.extension().and_then(|s| s.to_str()) != Some("json") {
61                continue;
62            }
63
64            if let Some(filename) = path.file_name().and_then(|s| s.to_str()) {
65                if let Some(base_name) = filename.strip_suffix(".json") {
66                    if let Ok(archive_id) = base_name.parse::<ArchiveIdentifier>() {
67                        if let Ok(package_name) = PackageName::try_from(archive_id.name) {
68                            records.insert(
69                                package_name,
70                                LazyRecordEntry {
71                                    path,
72                                    record: OnceLock::new(),
73                                },
74                            );
75                        }
76                    }
77                }
78            }
79        }
80        Ok(Self {
81            prefix_path,
82            records,
83        })
84    }
85
86    /// Retrieves a record lazily. Parses the JSON only on the first call.
87    pub fn get(
88        &self,
89        package_name: &PackageName,
90    ) -> Option<Result<&PrefixRecord, &PrefixDataError>> {
91        // 1. Check if the file path exists in our initial scan
92        let entry = self.records.get(package_name)?;
93
94        // 2. Parse the file if we haven't already.
95        let record_result = entry
96            .record
97            .get_or_init(|| PrefixRecord::from_path(&entry.path).map_err(PrefixDataError::from));
98
99        // 3. .as_ref() elegantly converts &Result<T, E> into Result<&T, &E>
100        Some(record_result.as_ref())
101    }
102}

104#[cfg(test)]
105mod tests {
106    use super::*;
107    use std::fs;
108    use tempfile::tempdir;
109
110    #[test]
111    fn test_prefix_data_lazy_loading() {
112        let dir = tempdir().unwrap();
113        let meta_dir = dir.path().join("conda-meta");
114        fs::create_dir_all(&meta_dir).unwrap();
115
116        let fake_json_path = meta_dir.join("numpy-1.24.3-py311h_0.json");
117        fs::write(&fake_json_path, "{}").unwrap();
118
119        let prefix_data = PrefixData::new(dir.path()).unwrap();
120        let numpy_name = PackageName::try_from("numpy").unwrap();
121        assert!(
122            prefix_data.records.contains_key(&numpy_name),
123            "Did not extract package name correctly!"
124        );
125    }
126
127    #[test]
128    fn test_prefix_data_strict() {
129        let dir = tempdir().unwrap();
130        let meta_dir = dir.path().join("conda-meta");
131        fs::create_dir_all(&meta_dir).unwrap();
132
133        fs::write(meta_dir.join("numpy-1.24.3-py311h_0.json"), "{}").unwrap();
134        fs::write(meta_dir.join("scikit-learn-1.2.2-py311_1.json"), "{}").unwrap();
135
136        fs::write(meta_dir.join("ignore_me.txt"), "some text").unwrap();
137        fs::write(meta_dir.join("not_a_json.yaml"), "{}").unwrap();
138
139        let prefix_data = PrefixData::new(dir.path()).unwrap();
140
141        let numpy_name = PackageName::try_from("numpy").unwrap();
142        let scikit_name = PackageName::try_from("scikit-learn").unwrap();
143        let does_not_exist_name = PackageName::try_from("does-not-exist").unwrap();
144
145        assert_eq!(
146            prefix_data.records.len(),
147            2,
148            "Should only load the 2 valid .json files"
149        );
150        assert!(
151            prefix_data.records.contains_key(&numpy_name),
152            "Failed to extract simple package name"
153        );
154        assert!(
155            prefix_data.records.contains_key(&scikit_name),
156            "Failed to extract package name with hyphens"
157        );
158
159        let numpy_record = prefix_data.get(&numpy_name);
160        assert!(
161            numpy_record.unwrap().is_err(),
162            "Expected Some(Err) because the JSON is malformed/empty"
163        );
164
165        let entry = prefix_data.records.get(&numpy_name).unwrap();
166        assert!(
167            entry.record.get().is_some(),
168            "The OnceLock should be populated (with an Err value) after the first get() call"
169        );
170
171        assert!(
172            prefix_data.get(&does_not_exist_name).is_none(),
173            "Expected None because the package is not in the directory at all"
174        );
175    }
