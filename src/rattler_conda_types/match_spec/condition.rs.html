14/// Represents a condition in a match spec, which can be a match spec itself or a logical combination
15#[derive(Debug, Clone, PartialEq, Serialize, Deserialize, Eq, Hash)]
16pub enum MatchSpecCondition {
17    /// A condition on a certain match spec (e.g. `python >=3.12`)
18    MatchSpec(Box<crate::MatchSpec>),
19    /// A logical AND condition combining two conditions
20    And(Box<MatchSpecCondition>, Box<MatchSpecCondition>),
21    /// A logical OR condition combining two conditions
22    Or(Box<MatchSpecCondition>, Box<MatchSpecCondition>),
23}
24
25impl Display for MatchSpecCondition {
26    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
27        match self {
28            MatchSpecCondition::MatchSpec(ms) => write!(f, "{ms}"),
29            MatchSpecCondition::And(lhs, rhs) => write!(f, "({lhs} and {rhs})"),
30            MatchSpecCondition::Or(lhs, rhs) => write!(f, "({lhs} or {rhs})"),
31        }
32    }
33}

35// Parse whitespace
36fn ws(input: &str) -> IResult<&str, &str> {
37    multispace0(input)
38}

40/// Check if the word-boundary delimiter `word` starts at byte position `pos` in `input`.
41/// A word boundary means the character before (if any) and after (if any) must be
42/// whitespace or a parenthesis.
43fn check_word_delimiter(input: &str, pos: usize, word: &str) -> bool {
44    let bytes = input.as_bytes();
45    // Check that the word actually matches at this position
46    if !input[pos..].starts_with(word) {
47        return false;
48    }
49    let before_ok = pos == 0 || {
50        let b = bytes[pos - 1];
51        b.is_ascii_whitespace() || b == b'(' || b == b')'
52    };
53    let after_pos = pos + word.len();
54    let after_ok = after_pos >= bytes.len() || {
55        let b = bytes[after_pos];
56        b.is_ascii_whitespace() || b == b'(' || b == b')'
57    };
58    before_ok && after_ok
59}

61// Parse a matchspec by consuming until we hit a delimiter.
62// Correctly skips delimiters inside quoted strings and brackets.
63fn matchspec_token(input: &str) -> IResult<&str, &str> {
64    let bytes = input.as_bytes();
65    let mut i = 0;
66    let mut bracket_depth: u32 = 0;
67    let mut in_double_quote = false;
68    let mut in_single_quote = false;
69
70    while i < bytes.len() {
71        let b = bytes[i];
72        match b {
73            b'\\' if in_double_quote || in_single_quote => {
74                // Skip the escaped character
75                i += 2;
76                continue;
77            }
78            b'"' if !in_single_quote => {
79                in_double_quote = !in_double_quote;
80            }
81            b'\'' if !in_double_quote => {
82                in_single_quote = !in_single_quote;
83            }
84            b'[' if !in_double_quote && !in_single_quote => {
85                bracket_depth += 1;
86            }
87            b']' if !in_double_quote && !in_single_quote => {
88                bracket_depth = bracket_depth.saturating_sub(1);
89            }
90            b'(' | b')' if !in_double_quote && !in_single_quote && bracket_depth == 0 => {
91                break;
92            }
93            _ if !in_double_quote && !in_single_quote && bracket_depth == 0 => {
94                if check_word_delimiter(input, i, "and") || check_word_delimiter(input, i, "or") {
95                    break;
96                }
97            }
98            _ => {}
99        }
100        i += 1;
101    }
102
103    if i == 0 {
104        return Err(nom::Err::Error(nom::error::Error::new(
105            input,
106            nom::error::ErrorKind::TakeUntil,
107        )));
108    }
109
110    let token = input[..i].trim();
111    if token.is_empty() {
112        return Err(nom::Err::Error(nom::error::Error::new(
113            input,
114            nom::error::ErrorKind::TakeUntil,
115        )));
116    }
117
118    Ok((&input[i..], token))
119}

121// Parse a matchspec
122fn matchspec(input: &str) -> IResult<&str, MatchSpecCondition> {
123    let (remaining, matchspec_str) = matchspec_token(input)?;
124
125    match matchspec_parser(matchspec_str, crate::ParseStrictness::Strict.into()) {
126        Ok(parsed_matchspec) => Ok((
127            remaining,
128            MatchSpecCondition::MatchSpec(Box::new(parsed_matchspec)),
129        )),
130        Err(_) => Err(nom::Err::Error(nom::error::Error::new(
131            input,
132            nom::error::ErrorKind::MapRes,
133        ))),
134    }
135}

137// Parse parenthesized condition
138fn parenthesized_condition(input: &str) -> IResult<&str, MatchSpecCondition> {
139    delimited((char('('), ws), parse_condition, (ws, char(')'))).parse(input)
140}
141
142// Parse primary condition (matchspec or parenthesized)
143fn primary_condition(input: &str) -> IResult<&str, MatchSpecCondition> {
144    alt((parenthesized_condition, matchspec)).parse(input)
145}
146
147// Parse AND expressions (higher precedence)
148fn and_condition(input: &str) -> IResult<&str, MatchSpecCondition> {
149    let (input, first) = primary_condition(input)?;
150    let (input, rest) =
151        nom::multi::many0(preceded((ws, tag("and"), ws), primary_condition)).parse(input)?;
152
153    Ok((
154        input,
155        rest.into_iter().fold(first, |acc, next| {
156            MatchSpecCondition::And(Box::new(acc), Box::new(next))
157        }),
158    ))
159}
160
161// Parse OR expressions (lower precedence)
162fn or_condition(input: &str) -> IResult<&str, MatchSpecCondition> {
163    let (input, first) = and_condition(input)?;
164    let (input, rest) =
165        nom::multi::many0(preceded((ws, tag("or"), ws), and_condition)).parse(input)?;
166
167    Ok((
168        input,
169        rest.into_iter().fold(first, |acc, next| {
170            MatchSpecCondition::Or(Box::new(acc), Box::new(next))
171        }),
172    ))
173}
174
175// Parse the main condition
176pub(crate) fn parse_condition(input: &str) -> IResult<&str, MatchSpecCondition> {
177    or_condition(input)
178}

261    fn test_matchspec_token_with_quoted_and_or() {
262        // "and" inside double-quoted bracket value should NOT be treated as a delimiter
263        let (rem, token) = matchspec_token(r#"python[build="fast and slow"] and linux"#).unwrap();
264        assert_eq!(token, r#"python[build="fast and slow"]"#);

269        assert_eq!(token, "foo[version='1 or 2']");
270        assert_eq!(rem, "or bar");
