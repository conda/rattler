20/// Representation of an entry in `info/has_prefix`.
21#[derive(Debug, Clone, PartialEq, Eq)]
22pub struct HasPrefixEntry {
23    /// The prefix placeholder in the file
24    pub prefix: Cow<'static, str>,
25    /// The file's mode
26    pub file_mode: FileMode,
27    /// The file's relative path respective to the environment's prefix
28    pub relative_path: PathBuf,
29}

31/// Representation of the `info/has_prefix` file in older package archives.
32#[derive(Debug, Clone, PartialEq, Eq)]
33pub struct HasPrefix {
34    /// A list of files in the package that contain the `prefix` (and need
35    /// prefix replacement).
36    pub files: Vec<HasPrefixEntry>,
37}
38
39impl PackageFile for HasPrefix {
40    fn package_path() -> &'static Path {
41        Path::new("info/has_prefix")
42    }
43
44    fn from_str(str: &str) -> Result<Self, std::io::Error> {
45        Ok(Self {
46            files: str
47                .lines()
48                .map(HasPrefixEntry::from_str)
49                .collect::<Result<_, _>>()?,
50        })
51    }
52}

54/// Returns the default placeholder path. Although this is just a constant it is
55/// constructed at runtime. This ensures that the string itself is not present
56/// in the binary when compiled. The reason we want that is that conda-build
57/// (and friends) tries to replace this placeholder in the binary to point to
58/// the actual path in the installed conda environment. In this case we don't
59/// want to that so we deliberately break up the string and reconstruct it at
60/// runtime.
61fn placeholder_string() -> &'static str {
62    static PLACEHOLDER: OnceLock<String> = OnceLock::new();
63    PLACEHOLDER
64        .get_or_init(|| {
65            let mut result = black_box(String::from("/opt/"));
66            for i in 1..=3 {
67                result.push_str(&format!("anaconda{i}"));
68            }
69            result
70        })
71        .as_str()
72}

74impl FromStr for HasPrefixEntry {
75    type Err = std::io::Error;
76
77    fn from_str(s: &str) -> Result<Self, Self::Err> {
78        /// Parses `<prefix> <file_mode> <path>` and fails if there is more
79        /// input.
80        fn prefix_file_mode_path(buf: &str) -> IResult<&str, HasPrefixEntry> {
81            all_consuming(map(
82                (
83                    possibly_quoted_string,
84                    multispace1,
85                    file_mode,
86                    multispace1,
87                    possibly_quoted_string,
88                ),
89                |(prefix, _, file_mode, _, path)| HasPrefixEntry {
90                    prefix: Cow::Owned(prefix.into_owned()),
91                    file_mode,
92                    relative_path: PathBuf::from(&*path),
93                },
94            ))
95            .parse(buf)
96        }
97
98        /// Parses "<path>" and fails if there is more input.
99        fn only_path(buf: &str) -> IResult<&str, HasPrefixEntry> {
100            all_consuming(map(possibly_quoted_string, |path| HasPrefixEntry {
101                prefix: Cow::Borrowed(placeholder_string()),
102                file_mode: FileMode::Text,
103                relative_path: PathBuf::from(&*path),
104            }))
105            .parse(buf)
106        }
107
108        /// Parses "text|binary" as a [`FileMode`]
109        fn file_mode(buf: &str) -> IResult<&str, FileMode> {
110            alt((
111                value(FileMode::Text, tag_no_case("text")),
112                value(FileMode::Binary, tag_no_case("binary")),
113            ))
114            .parse(buf)
115        }
116
117        /// Parses either a quoted or an unquoted string.
118        fn possibly_quoted_string(buf: &str) -> IResult<&str, Cow<'_, str>> {
119            alt((
120                map(quoted_string, Cow::Owned),
121                map(take_till1(char::is_whitespace), Cow::Borrowed),
122            ))
123            .parse(buf)
124        }
125
126        /// Parses a quoted string and delimited '\"'
127        fn quoted_string(buf: &str) -> IResult<&str, String> {
128            fn in_quotes(buf: &str) -> IResult<&str, String> {
129                let mut ret = String::new();
130                let mut skip_delimiter = false;
131                for (i, ch) in buf.char_indices() {
132                    if ch == '\\' && !skip_delimiter {
133                        skip_delimiter = true;
134                    } else if ch == '"' && !skip_delimiter {
135                        return Ok((&buf[i..], ret));
136                    } else {
137                        ret.push(ch);
138                        skip_delimiter = false;
139                    }
140                }
141                Err(nom::Err::Incomplete(nom::Needed::Unknown))
142            }
143
144            let qs = preceded(tag("\""), in_quotes);
145            terminated(qs, tag("\"")).parse(buf)
146        }
147
148        alt((prefix_file_mode_path, only_path))
149            .parse(s)
150            .map(|(_, res)| res)
151            .map_err(|e| std::io::Error::new(std::io::ErrorKind::InvalidData, e.to_string()))
152    }
153}

155#[cfg(test)]
156mod test {
157    use std::{borrow::Cow, path::PathBuf, str::FromStr};
158
159    use super::*;
160    use crate::package::FileMode;
161
162    #[test]
163    fn test_placeholder() {
164        assert_eq!(placeholder_string(), "/opt/anaconda1anaconda2anaconda3");
165    }
166
167    #[test]
168    pub fn test_parse_has_prefix() {
169        let parsed =
170            HasPrefixEntry::from_str("/opt/anaconda1anaconda2anaconda3 text lib/pkgconfig/zlib.pc")
171                .unwrap();
172        assert_eq!(
173            parsed,
174            HasPrefixEntry {
175                prefix: Cow::Borrowed("/opt/anaconda1anaconda2anaconda3"),
176                file_mode: FileMode::Text,
177                relative_path: PathBuf::from("lib/pkgconfig/zlib.pc"),
178            }
179        );
180
181        let parsed = HasPrefixEntry::from_str(
182            "\"/opt/anaconda1 anaconda2anaconda3\" binary \"lib/pkg config/zlib.pc\"",
183        )
184        .unwrap();
185        assert_eq!(
186            parsed,
187            HasPrefixEntry {
188                prefix: Cow::Borrowed("/opt/anaconda1 anaconda2anaconda3"),
189                file_mode: FileMode::Binary,
190                relative_path: PathBuf::from("lib/pkg config/zlib.pc"),
191            }
192        );
193
194        let parsed = HasPrefixEntry::from_str("lib/pkgconfig/zlib.pc").unwrap();
195        assert_eq!(
196            parsed,
197            HasPrefixEntry {
198                prefix: Cow::Borrowed("/opt/anaconda1anaconda2anaconda3"),
199                file_mode: FileMode::Text,
200                relative_path: PathBuf::from("lib/pkgconfig/zlib.pc"),
201            }
202        );
203    }
