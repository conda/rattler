5/// Describes the type of a conda package archive.
6#[derive(Copy, Clone, Debug, Ord, PartialOrd, Eq, PartialEq, Hash, Serialize, Deserialize)]
7#[serde(rename_all = "snake_case")]
8pub enum CondaArchiveType {
9    /// A file with the `.conda` extension.
10    Conda,
11
12    /// A file with the `.tar.bz2` extension.
13    TarBz2,
14}
15
16impl CondaArchiveType {
17    /// Returns the file extension for this archive type.
18    pub fn extension(self) -> &'static str {
19        match self {
20            CondaArchiveType::TarBz2 => ".tar.bz2",
21            CondaArchiveType::Conda => ".conda",
22        }
23    }
24
25    /// Tries to determine the type of a Conda archive from its magic bytes.
26    pub fn try_from_magic_bytes<T: AsRef<[u8]>>(bytes: T) -> Option<CondaArchiveType> {
27        // https://en.wikipedia.org/wiki/List_of_file_signatures
28        let bytes = bytes.as_ref();
29        if bytes.len() >= 4 {
30            match bytes[0..4] {
31                // zip magic number
32                [0x50, 0x4B, 0x03, 0x04] | [0x50, 0x4B, 0x05, 0x06] | [0x50, 0x4B, 0x07, 0x08] => {
33                    Some(CondaArchiveType::Conda)
34                }
35                // bz2 magic number
36                [0x42, 0x5a, 0x68, _] => Some(CondaArchiveType::TarBz2),
37                _ => None,
38            }
39        } else {
40            None
41        }
42    }
43
44    /// Tries to determine the type of conda archive from its filename.
45    pub fn try_from(path: impl AsRef<Path>) -> Option<CondaArchiveType> {
46        Self::split_str(path.as_ref().to_string_lossy().as_ref())
47            .map(|(_, archive_type)| archive_type)
48    }
49
50    /// Split the given string into its filename and archive type, removing the
51    /// extension. Only recognizes conda package extensions.
52    #[allow(clippy::manual_map)]
53    pub fn split_str(path: &str) -> Option<(&str, CondaArchiveType)> {
54        if let Some(path) = path.strip_suffix(".conda") {
55            Some((path, CondaArchiveType::Conda))
56        } else if let Some(path) = path.strip_suffix(".tar.bz2") {
57            Some((path, CondaArchiveType::TarBz2))
58        } else {
59            None
60        }
61    }
62}

64/// Describes any type of distributable package archive (conda packages or
65/// wheels).
66#[derive(Copy, Clone, Debug, Ord, PartialOrd, Eq, PartialEq, Hash, Serialize, Deserialize)]
67#[serde(untagged)]
68pub enum DistArchiveType {
69    /// A conda package archive (`.tar.bz2` or `.conda`)
70    Conda(CondaArchiveType),
71
72    /// A Python wheel package (`.whl`)
73    Wheel(WheelArchiveType),
74}
75
76impl DistArchiveType {
77    /// Compares this archive type against another returning which one is
78    /// preferred over another if there are two archive types that represent the
79    /// same package.
80    ///
81    /// The order returned by this function is that `.conda` packages are
82    /// preferred over all others and that `.tar.bz2` packages are preferred
83    /// over `.whl` packages.
84    pub fn cmp_preference(self, other: DistArchiveType) -> std::cmp::Ordering {
85        match (self, other) {
86            (a, b) if a == b => Ordering::Equal,
87            (DistArchiveType::Conda(CondaArchiveType::Conda), _) => Ordering::Greater,
88            (_, DistArchiveType::Conda(CondaArchiveType::Conda)) => Ordering::Less,
89            (DistArchiveType::Conda(CondaArchiveType::TarBz2), _) => Ordering::Greater,
90            (_, DistArchiveType::Conda(CondaArchiveType::TarBz2)) => Ordering::Less,
91            (DistArchiveType::Wheel(WheelArchiveType::Whl), _) => Ordering::Greater,
92        }
93    }
94
95    /// Tries to determine the type of archive from its filename.
96    pub fn try_from(path: impl AsRef<Path>) -> Option<DistArchiveType> {
97        Self::split_str(path.as_ref().to_string_lossy().as_ref())
98            .map(|(_, archive_type)| archive_type)
99    }
100
101    /// Returns the file extension for this archive type.
102    pub fn extension(self) -> &'static str {
103        match self {
104            DistArchiveType::Conda(conda_type) => conda_type.extension(),
105            DistArchiveType::Wheel(wheel_type) => wheel_type.extension(),
106        }
107    }
108
109    /// Split the given string into its filename and archive type, removing the
110    /// extension.
111    #[allow(clippy::manual_map)]
112    pub fn split_str(path: &str) -> Option<(&str, DistArchiveType)> {
113        if let Some((path, conda_type)) = CondaArchiveType::split_str(path) {
114            Some((path, DistArchiveType::Conda(conda_type)))
115        } else if let Some((path, wheel_type)) = WheelArchiveType::split_str(path) {
116            Some((path, DistArchiveType::Wheel(wheel_type)))
117        } else {
118            None
119        }
120    }
121}
122
123impl From<CondaArchiveType> for DistArchiveType {
124    fn from(conda_type: CondaArchiveType) -> Self {
125        DistArchiveType::Conda(conda_type)
126    }
127}
128
129impl From<WheelArchiveType> for DistArchiveType {
130    fn from(wheel_type: WheelArchiveType) -> Self {
131        DistArchiveType::Wheel(wheel_type)
132    }
133}
