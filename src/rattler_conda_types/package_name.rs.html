13/// A representation of a conda package name. This struct both stores the source string from which
14/// this instance was created as well as a normalized name that can be used to compare different
15/// names. The normalized name is guaranteed to be a valid conda package name.
16///
17/// Conda package names are always lowercase and can only contain ascii characters.
18///
19/// This struct explicitly does not implement [`std::fmt::Display`] because its ambiguous if that
20/// would display the source or the normalized version. Simply call `as_source` or `as_normalized`
21/// to make the distinction.
22#[derive(Debug, Clone, Eq, DeserializeFromStr)]
23pub struct PackageName {
24    normalized: Option<Box<str>>,
25    source: Box<str>,
26}
27
28impl PackageName {
29    /// Constructs a new `PackageName` from a string without checking if the string is actually a
30    /// valid or normalized conda package name. This should only be used if you are sure that the
31    /// input string is valid, otherwise use the `TryFrom` implementations.
32    pub fn new_unchecked<S: Into<String>>(normalized: S) -> Self {
33        Self {
34            normalized: None,
35            source: normalized.into().into_boxed_str(),
36        }
37    }

39    /// Returns the source representation of the package name. This is the string from which this
40    /// instance was created.
41    pub fn as_source(&self) -> &str {
42        &self.source
43    }

45    /// Returns the normalized version of the package name. The normalized string is guaranteed to
46    /// be a valid conda package name.
47    pub fn as_normalized(&self) -> &str {
48        self.normalized.as_deref().unwrap_or(&self.source)
49    }

51    /// Parses the package name part from a matchspec string without parsing
52    /// the entire matchspec.
53    ///
54    /// This extracts the package name by splitting on whitespace or version
55    /// constraint characters (`>`, `<`, `=`, `!`, `~`, `;`).
56    ///
57    /// # Examples

59    /// ```
60    /// use rattler_conda_types::PackageName;

72    /// Parses the package name part from a matchspec string without parsing
73    /// the entire matchspec. This function assumes the matchspec string is a
74    /// valid matchspec.
75    ///
76    /// This extracts the package name by splitting on whitespace or version
77    /// constraint characters (`>`, `<`, `=`, `!`, `~`, `;`). The original
78    /// capitalization is preserved in the source, while the normalized version
79    /// is lowercase.
80    ///
81    /// # Safety
82    ///
83    /// This function does not validate the package name. If the package name
84    /// is not valid, the returned `PackageName` may not behave correctly.
85    /// Use [`Self::from_matchspec_str`] for a fallible version.
86    ///
87    /// # Examples
88    ///
89    /// ```
90    /// use rattler_conda_types::PackageName;
91    ///
92    /// let name = PackageName::from_matchspec_str_unchecked("Pillow >=10");
93    /// assert_eq!(name.as_source(), "Pillow");
94    /// assert_eq!(name.as_normalized(), "pillow");
95    /// ```
96    pub fn from_matchspec_str_unchecked(spec: &str) -> Self {
97        let (name, has_upper) = scan_matchspec_name(spec);
98        let normalized = if has_upper {
99            Some(name.to_ascii_lowercase().into_boxed_str())
100        } else {
101            None
102        };
103        Self {
104            normalized,
105            source: name.to_string().into_boxed_str(),
106        }
107    }

109    /// Extracts and normalizes the package name part from a matchspec string
110    /// without constructing a full `PackageName` instance.
111    ///
112    /// This is a lightweight alternative to [`Self::from_matchspec_str_unchecked`]
113    /// that avoids allocation when the package name is already lowercase.
114    /// Returns a borrowed string slice when no normalization is needed, or an
115    /// owned string when the name contains uppercase characters.
116    ///
117    /// # Examples
118    ///
119    /// ```
120    /// use rattler_conda_types::PackageName;
121    /// use std::borrow::Cow;
122    ///
123    /// // Lowercase names are borrowed (no allocation)
124    /// let name = PackageName::normalized_name_from_matchspec_str("numpy>=1.0");
125    /// assert!(matches!(name, Cow::Borrowed("numpy")));
126    ///
127    /// // Uppercase names are normalized and owned (allocation required)
128    /// let name = PackageName::normalized_name_from_matchspec_str("Pillow >=10");
129    /// assert!(matches!(name, Cow::Owned(_)));
130    /// assert_eq!(name, "pillow");
131    /// ```
132    pub fn normalized_name_from_matchspec_str(spec: &str) -> Cow<'_, str> {
133        let (name, has_upper) = scan_matchspec_name(spec);
134        if has_upper {
135            Cow::Owned(name.to_ascii_lowercase())
136        } else {
137            Cow::Borrowed(name)
138        }
139    }
140
141    /// Extracts the normalized package name and the optional features (extras)
142    /// from a matchspec string.
143    ///
144    /// When the spec contains no `[` bracket section, this is the fast path:
145    /// equivalent to [`Self::normalized_name_from_matchspec_str`] with no
146    /// allocation for the extras `Vec`.
147    ///
148    /// When brackets are present, the full matchspec is parsed (with
149    /// experimental extras enabled) to correctly distinguish extras from other
150    /// bracket keys like `build`, `version` or `when`. If parsing fails, the
151    /// name is still extracted with the cheap scan and the extras list is
152    /// empty.
153    ///
154    /// # Examples
155    ///
156    /// ```
157    /// use rattler_conda_types::PackageName;
158    ///
159    /// let (name, extras) = PackageName::name_and_extras_from_matchspec_str("numpy >=1.0");
160    /// assert_eq!(name, "numpy");
161    /// assert!(extras.is_empty());
162    ///
163    /// let (name, extras) =
164    ///     PackageName::name_and_extras_from_matchspec_str("numpy[extras=[gpu, mkl]]");
165    /// assert_eq!(name, "numpy");
166    /// assert_eq!(extras, vec!["gpu".to_string(), "mkl".to_string()]);
167    /// ```
168    pub fn name_and_extras_from_matchspec_str(spec: &str) -> (Cow<'_, str>, Vec<String>) {
169        if !spec.contains('[') {
170            return (Self::normalized_name_from_matchspec_str(spec), Vec::new());
171        }
172
173        let name = Self::normalized_name_from_matchspec_str(spec);
174
175        let options = crate::ParseMatchSpecOptions::default().with_extras(true);
176        match crate::MatchSpec::from_str(spec, options) {
177            Ok(parsed) => (name, parsed.extras.unwrap_or_default()),
178            Err(err) => {
179                // Failure mode is silent under-fetching of an extra's deps;
180                // log so it shows up under RUST_LOG=debug when investigating.
181                tracing::debug!("failed to parse matchspec '{spec}' for extras extraction: {err}",);
182                (name, Vec::new())
183            }
184        }
185    }
186}

188/// Returns `true` if the byte is a matchspec delimiter (whitespace or version
189/// constraint character: `>`, `<`, `=`, `!`, `~`, `;`).
190fn is_matchspec_delimiter(b: u8) -> bool {
191    b.is_ascii_whitespace() || matches!(b, b'>' | b'<' | b'=' | b'!' | b'~' | b';' | b'[')
192}
193
194/// Scans a matchspec string to find the package name boundary and whether it
195/// contains uppercase characters. Single-pass over the bytes.
196fn scan_matchspec_name(spec: &str) -> (&str, bool) {
197    let bytes = spec.as_bytes();
198    let mut has_upper = false;
199    let mut end = bytes.len();
200    for (i, &b) in bytes.iter().enumerate() {
201        if is_matchspec_delimiter(b) {
202            end = i;
203            break;
204        }
205        has_upper |= b.is_ascii_uppercase();
206    }
207    (&spec[..end], has_upper)
208}

210/// Extracts the package name part from a matchspec string by splitting on
211/// whitespace, version constraint characters (`>`, `<`, `=`, `!`, `~`, `;`),
212/// or bracket `[` (used for bracket syntax like `pkg[when="..."]`).
213fn name_from_matchspec_str(spec: &str) -> &str {
214    scan_matchspec_name(spec).0
215}

243impl TryFrom<String> for PackageName {
244    type Error = InvalidPackageNameError;
245
246    fn try_from(source: String) -> Result<Self, Self::Error> {
247        // Ensure that the string only contains valid characters
248        if !source
249            .bytes()
250            .all(|b| matches!(b, b'a'..=b'z'|b'A'..=b'Z'|b'0'..=b'9'|b'-'|b'_'|b'.'))
251        {
252            return Err(InvalidPackageNameError::InvalidCharacters(source));
253        }
254
255        // Convert all characters to lowercase but only if it actually contains uppercase. This way
256        // we dont allocate the memory of the string if it is already lowercase.
257        let normalized = if source.bytes().any(|b| b.is_ascii_uppercase()) {
258            Some(source.to_ascii_lowercase().into_boxed_str())
259        } else {
260            None
261        };
262
263        Ok(Self {
264            normalized,
265            source: source.into_boxed_str(),
266        })
267    }
268}
